90#define strprintf(__n_var, ...) \
91 ({ \
92 char* __strprintf_ptr = NULL; \
93 if (!(__n_var)) { \
94 int __needed = snprintf(NULL, 0, __VA_ARGS__); \
95 if (__needed > 0) { \
96 Malloc(__n_var, char, (size_t)__needed + 1); \
97 if (__n_var) { \
98 snprintf(__n_var, (size_t)__needed + 1, __VA_ARGS__); \
99 __strprintf_ptr = __n_var; \
100 } else { \
101 n_log(LOG_ERR, "couldn't allocate %s with %d bytes", \
102 #__n_var, __needed + 1); \
103 } \
104 } \
105 } else { \
106 n_log(LOG_ERR, "%s is already allocated.", #__n_var); \
107 } \
108 __strprintf_ptr; \
109 })
