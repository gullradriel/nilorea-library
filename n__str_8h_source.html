78#define local_strdup(__src_) \

82 Malloc(__local_strdup_str, char, __local_strdup_len + 1); \

94#define strprintf(__n_var, ...) \
95 ({ \
96 char* __strprintf_ptr = NULL; \
97 if (!(__n_var)) { \
98 int __needed = snprintf(NULL, 0, __VA_ARGS__); \
99 if (__needed > 0) { \
100 Malloc(__n_var, char, (size_t)__needed + 1); \
101 if (__n_var) { \
102 snprintf(__n_var, (size_t)__needed + 1, __VA_ARGS__); \
103 __strprintf_ptr = __n_var; \
104 } else { \
105 n_log(LOG_ERR, "couldn't allocate %s with %d bytes", \
106 #__n_var, __needed + 1); \
107 } \
108 } \
109 } else { \
110 n_log(LOG_ERR, "%s is already allocated.", #__n_var); \
111 } \
112 __strprintf_ptr; \
113 })
