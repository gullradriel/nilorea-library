28/* Compile-time availability gate. Linux + Android both ship epoll
29 * + eventfd. Solaris has /dev/poll and event ports (different
30 * primitive); Windows has IOCP (different architecture). For now
31 * the reactor is Linux/Android-only and falls back to the existing
32 * thread engine on every other target.
33 *
34 * Wrapped in `#ifndef` so the build system can force the gate off
35 * via `-DN_REACTOR_AVAILABLE=0` (e.g. on Linux when the user
36 * deliberately opts out of reactor mode to drop the n_reactor.o
37 * link dependency). The Makefile sets this when HAVE_REACTOR=0. */
38#ifndef N_REACTOR_AVAILABLE
39#if defined(__linux__) || defined(__ANDROID__)
40#define N_REACTOR_AVAILABLE 1
41#else
42#define N_REACTOR_AVAILABLE 0
43#endif
44#endif
