241 for (int i = 0; i < NUM_WINDOWS; i++) {
242 if (win_ids[i] < 0) continue;
243 char buf[N_GUI_ID_MAX];
244 int is_open = n_gui_window_is_open(gui, win_ids[i]);
245 snprintf(buf, sizeof(buf), "[%s] %s", is_open ? "X" : " ", win_names[i]);
246 n_gui_dropmenu_add_dynamic_entry(gui, widget_id, buf, win_ids[i],
247 on_window_toggle_click, gui);

288 switch (getoptret) {
289 case 'h':
290 n_log(LOG_NOTICE, "\n %s -h help -v version -V DEBUGLEVEL -L logfile -t theme.json", argv[0]);
291 exit(TRUE);
292 case 'v':

296 case 'V':
297 if (!strncmp("NOTICE", optarg, 6)) {
298 log_level = LOG_INFO;
299 } else if (!strncmp("VERBOSE", optarg, 7)) {
300 log_level = LOG_NOTICE;
301 } else if (!strncmp("ERROR", optarg, 5)) {
302 log_level = LOG_ERR;
303 } else if (!strncmp("DEBUG", optarg, 5)) {
304 log_level = LOG_DEBUG;
305 } else {
306 n_log(LOG_ERR, "%s is not a valid log level", optarg);
307 exit(FALSE);
308 }
309 set_log_level(log_level);
310 break;
311 case 'L':
312 set_log_file(optarg);
313 break;
314 case 't':
315 strncpy(theme_file, optarg, sizeof(theme_file) - 1);
316 theme_file[sizeof(theme_file) - 1] = '\0';
317 break;
318 default:
319 n_log(LOG_ERR, "\n %s -h help -v version -V DEBUGLEVEL -L logfile -t theme.json", argv[0]);
