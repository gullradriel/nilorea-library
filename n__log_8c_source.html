122void set_log_level(const int log_level) {
123 if (log_level == LOG_FILE) {
124 LOG_TYPE = LOG_FILE;
125 return;
126 }
127 if (log_level == LOG_STDERR) {
128 LOG_TYPE = LOG_STDERR;
129 return;
130 }
131 if (log_level == LOG_SYSJRNL) {
132 LOG_TYPE = LOG_SYSJRNL;
133 return;
134 }
135 LOG_LEVEL = log_level;
136#ifdef __windows__
137 HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
138 if (hOut != INVALID_HANDLE_VALUE) {
139 DWORD dwMode = 0;
140 if (GetConsoleMode(hOut, &dwMode)) {
141 dwMode |= ENABLE_VIRTUAL_TERMINAL_PROCESSING;
142 if (SetConsoleMode(hOut, dwMode)) {
143 // color support is enabled
144 terminal_support_colors = 1;
145 }
146 }
147 }
148#else
149 // Probably a terminal, may support colors
150 if (isatty(fileno(stdout))) {
151 terminal_support_colors = 1;
152 }
153#endif
154} /* set_log_level() */

169int set_log_file(char* file) {
170 __n_assert(file, return FALSE);
171
172 if (log_file) {
173 fclose(log_file);
174 log_file = NULL;
175 }
176
177 int fd = open(file, O_CREAT | O_APPEND | O_WRONLY, 0600); // rw------- permissions
178 if (fd == -1) {
179 n_log(LOG_ERR, "couldn't create %s with 0600 permission", _str(file));
180 return FALSE;
181 }
182
183 log_file = fdopen(fd, "a");
184 if (!log_file) {
185 close(fd);
186 n_log(LOG_ERR, "fdopen returned an invalid file descriptor pointer for %s:%d", _str(file), fd);
187 return FALSE;
188 }
189
190 set_log_level(LOG_FILE);
191
192 return TRUE;
193} /* set_log_file */

220#ifndef vasprintf
228int vasprintf(char** strp, const char* fmt, va_list ap) {
229 long long int len = _vscprintf_so(fmt, ap);
230 if (len == -1)
231 return -1;
232 char* str = NULL;
233 Malloc(str, char, (size_t)len + 1 + sizeof(void*)); // len + EndOfString + padding
234 if (!str)
235 return -1;
236 int r = vsnprintf(str, (size_t)(len + 1), fmt, ap); /* "secure" version of vsprintf */
237 if (r == -1)
238 return free(str), -1;
239 *strp = str;
240 return r;
241}
242#endif

269void _n_log(int level, const char* file, const char* func, int line, const char* format, ...) {
270 FILE* out = NULL;
271
272 if (level == LOG_NULL)
273 return;
274
275 if (!log_file)
276 out = stderr;
277 else
278 out = log_file;
279
280 int log_level = get_log_level();
281
282 if (level <= log_level) {
283 va_list args;
284 char* syslogbuffer = NULL;
285 char* eventbuffer = NULL;
286 char* syslogsafebuffer = NULL;
287#ifdef __windows__
288 size_t needed = 0;
289 char* name = "NULL";
290 if (proc_name)
291 name = proc_name;
292#endif
293
294 const char* color = "";
295 const char* color_reset = "";
296
297 if (terminal_support_colors) {
298 color_reset = COLOR_RESET;
299 switch (level) {
300 case LOG_EMERG:
301 color = COLOR_RED;
302 break;
303 case LOG_ALERT:
304 color = COLOR_MAGENTA;
305 break;
306 case LOG_CRIT:
307 color = COLOR_RED;
308 break;
309 case LOG_ERR:
310 color = COLOR_RED;
311 break;
312 case LOG_WARNING:
313 color = COLOR_YELLOW;
314 break;
315 case LOG_NOTICE:
316 color = COLOR_GREEN;
317 break;
318 case LOG_INFO:
319 color = COLOR_CYAN;
320 break;
321 case LOG_DEBUG:
322 color = COLOR_BLUE;
323 break;
324 default:
325 color = COLOR_WHITE;
326 break;
327 }
328 }
329 switch (LOG_TYPE) {
330 case LOG_SYSJRNL:
331 va_start(args, format);
332 if (vasprintf(&syslogbuffer, format, args) == -1) {
333 syslogbuffer = NULL;
334 }
335 va_end(args);
336 // Fallback if vasprintf is NULL */
337 syslogsafebuffer = syslogbuffer ? syslogbuffer : "<log: vasprintf failed>";
338#ifdef __windows__
339 /* Sanitize syslog content to prevent command injection via system().
340 * Only mutate when syslogbuffer is non-NULL: syslogsafebuffer then
341 * points to heap memory. When syslogbuffer is NULL it points to a
342 * read-only string literal and must not be written to. */
343 if (syslogbuffer) {
344 for (char* p = syslogsafebuffer; *p; p++) {
345 if (*p == '"' || *p == '&' || *p == '|' || *p == '<' || *p == '>' || *p == '^') *p = '_';
346 }
347 }
348 int snprintf_ret = snprintf(NULL, 0, "start /B EventCreate /t %s /id 666 /l APPLICATION /so %s /d \"%s\" > NUL 2>&1", prioritynames[level].w_name, name, syslogsafebuffer);
349 if (snprintf_ret > 0) {
350 needed = (unsigned long long)snprintf_ret;
351 Malloc(eventbuffer, char, needed + 4);
352 if (eventbuffer) {
353 snprintf(eventbuffer, needed + 4, "start /B EventCreate /t %s /id 666 /l APPLICATION /so %s /d \"%s\" > NUL 2>&1", prioritynames[level].w_name, name, syslogsafebuffer);
354 system(eventbuffer);
355 }
356 }
357#else
358 syslog(level, "%s->%s:%d %s", _str(file), _str(func), line, syslogsafebuffer);
359#endif
360 FreeNoLog(syslogbuffer);
361 FreeNoLog(eventbuffer);
362 break;
363 default:
364 fprintf(out, "%s%s:%jd:%s->%s:%d ", color, prioritynames[level].c_name, (intmax_t)time(NULL), _str(file), _str(func), line);
365 if (format) {
366 va_start(args, format);
367 vfprintf(out, format, args);
368 va_end(args);
369 }
370 fprintf(out, "%s\n", color_reset);
371 break;
372 }
373 fflush(out);
374 }
375} /* _n_log( ... ) */

384int open_safe_logging(TS_LOG** log, char* pathname, char* opt) {
385 if (!log) {
386 n_log(LOG_ERR, "Invalid log pointer (NULL)");
387 return FALSE;
388 }
389
390 if (*log) {
391 if ((*log)->file) {
392 n_log(LOG_ERR, "Log file '%s' already open", _str(pathname));
393 return -1000; // already open
394 }
395 n_log(LOG_ERR, "Log struct already allocated without file");
396 return FALSE;
397 }
398
399 if (!pathname || !opt) {
400 n_log(LOG_ERR, "Invalid pathname or mode (pathname=%s, opt=%s)", _str(pathname), _str(opt));
401 return FALSE;
402 }
403
404 Malloc((*log), TS_LOG, 1);
405 if (!(*log)) {
406 n_log(LOG_ERR, "Failed to allocate TS_LOG structure");
407 return FALSE;
408 }
409
410 pthread_mutex_init(&(*log)->LOG_MUTEX, NULL);
411
412 int flags = 0;
413 if (strcmp(opt, "a") == 0 || strcmp(opt, "a+") == 0) {
414 flags = O_CREAT | O_APPEND | O_WRONLY;
415 } else if (strcmp(opt, "w") == 0 || strcmp(opt, "w+") == 0) {
416 flags = O_CREAT | O_TRUNC | O_WRONLY;
417 } else {
418 n_log(LOG_ERR, "Unsupported open mode '%s' for pathname '%s'", _str(opt), _str(pathname));
419 pthread_mutex_destroy(&(*log)->LOG_MUTEX);
420 Free(*log);
421 *log = NULL;
422 return FALSE;
423 }
424
425 int fd = open(pathname, flags, 0600);
426 if (fd == -1) {
427 n_log(LOG_ERR, "Failed to open '%s' with 0600 permissions", _str(pathname));
428 pthread_mutex_destroy(&(*log)->LOG_MUTEX);
429 Free(*log);
430 *log = NULL;
431 return FALSE;
432 }
433
434 (*log)->file = fdopen(fd, opt);
435 if (!(*log)->file) {
436 n_log(LOG_ERR, "fdopen failed for '%s' (fd=%d, mode='%s')", _str(pathname), fd, _str(opt));
437 close(fd);
438 pthread_mutex_destroy(&(*log)->LOG_MUTEX);
439 Free(*log);
440 *log = NULL;
441 return FALSE;
442 }
443
444 return TRUE;
445} /* open_safe_logging */

453int write_safe_log(TS_LOG* log, char* pat, ...) {
454 /* argument list */
455 va_list arg;
456 char str[2048] = "";
457
458 if (!log)
459 return FALSE;

461 va_start(arg, pat);

466
467 pthread_mutex_lock(&log->LOG_MUTEX);
468 fprintf(log->file, "%s", str);

472 return TRUE;
