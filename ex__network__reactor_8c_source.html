93static int run_server(const char* addr, const char* port, int target) {
94 NETWORK* listener = NULL;
95 if (netw_make_listening(&listener, (char*)((addr && addr[0]) ? addr : NULL),
96 (char*)port, 64, NETWORK_IPALL) == FALSE) {
97 n_log(LOG_ERR, "netw_make_listening failed on %s:%s", addr ? addr : "*", port);
98 return 1;
99 }
100 n_log(LOG_NOTICE, "reactor server listening on %s:%s (target %d connections)",
101 addr && addr[0] ? addr : "*", port, target);
102
103 /* Reactor + dedicated I/O thread. n_reactor_new returns NULL on
104 * non-Linux platforms (the public function is still safe to call,
105 * it just yields a polite no-op so callers don't need #ifdefs). */
106 n_reactor* reactor = n_reactor_new(0);
107 if (!reactor) {
108 n_log(LOG_NOTICE, "n_reactor unavailable on this platform, skipping (exit 0)");
109 netw_close(&listener);
110 netw_unload();
111 return 0;
112 }
113
114 pthread_t reactor_thr;
115 if (pthread_create(&reactor_thr, NULL, &n_reactor_run_thread_entry, reactor) != 0) {
116 n_log(LOG_ERR, "pthread_create(reactor): %s", strerror(errno));
117 n_reactor_destroy(&reactor);
118 netw_close(&listener);
119 netw_unload();
120 return 2;
121 }
122
123 /* Tracking list of active reactor-registered clients. The reactor
124 * owns I/O; we own the per-connection lifecycle (accept -> echo ->
125 * close). */
126 LIST* active = new_generic_list(MAX_LIST_ITEMS);
127 if (!active) {
128 n_log(LOG_ERR, "new_generic_list failed");
129 n_reactor_stop(reactor);
130 pthread_join(reactor_thr, NULL);
131 n_reactor_destroy(&reactor);
132 netw_close(&listener);
133 netw_unload();
134 return 3;
135 }
136
137 int handled = 0;
138 while (g_running && handled < target) {
139 /* Step 1: try to accept (500 ms select timeout, short enough
140 * that we keep draining recv queues responsively). */
141 int retval = 0;
142 NETWORK* client = netw_accept_into_reactor(listener, 0, 0, 500, reactor, &retval);
143 if (client) {
144 n_log(LOG_INFO, "accepted client fd=%d (now %d active)",
145 client->link.sock, (int)(active->nb_items + 1));
146 list_push(active, client, NULL);
147 }
148
149 /* Step 2: drain any messages the reactor posted onto active
150 * clients' recv_buf. Echo them back via netw_add_msg, that
151 * calls n_reactor_notify_send under the hood, waking the
152 * reactor so the send queue gets flushed. */
153 LIST_NODE* node = active->start;
154 while (node) {
155 LIST_NODE* next = node->next;
156 NETWORK* c = (NETWORK*)node->ptr;
157 N_STR* msg = netw_get_msg(c);
158 if (msg) {
159 n_log(LOG_INFO, "echoing %zu bytes back to fd=%d",
160 msg->length, c->link.sock);
161 if (netw_add_msg(c, msg) != TRUE) {
162 free_nstr(&msg);
163 }
164 /* Give the reactor a brief window to flush the echo
165 * before the close handshake severs SHUT_WR. */
166 u_sleep(20000);
167 /* netw_close calls n_reactor_close_netw_sync internally
168 * because c->reactor_mode is set, no manual unregister
169 * needed. */
170 netw_close(&c);
171 /* remove_list_node_f unlinks `node`, frees the
172 * LIST_NODE struct, and returns the void* it held
173 * (which we already netw_close'd). */
174 (void)remove_list_node_f(active, node);
175 handled++;
176 n_log(LOG_NOTICE, "handled %d/%d connections", handled, target);
177 }
178 node = next;
179 }
180 }
181
182 /* Anything still registered didn't get its echo before the target
183 * was reached or SIGINT fired. Close them cleanly. */
184 LIST_NODE* node = active->start;
185 while (node) {
186 LIST_NODE* next = node->next;
187 NETWORK* c = (NETWORK*)node->ptr;
188 netw_close(&c);
189 (void)remove_list_node_f(active, node);
190 node = next;
191 }
192 list_destroy(&active);
193
194 n_reactor_stats stats;
195 n_reactor_get_stats(reactor, &stats);
196 n_log(LOG_NOTICE,
197 "reactor stats: events=%lld registered=%lld unregistered=%lld "
198 "wake=%lld writes_partial=%lld reads_partial=%lld",
199 stats.events_processed, stats.fds_registered, stats.fds_unregistered,
200 stats.wake_signals, stats.writes_partial, stats.reads_partial);
201
202 n_reactor_stop(reactor);
203 pthread_join(reactor_thr, NULL);
204 n_reactor_destroy(&reactor);
205
206 netw_close(&listener);
207 netw_unload();
208 n_log(LOG_NOTICE, "reactor server done (%d connections handled)", handled);
209 return 0;
210}

221static int run_client(const char* host, const char* port, int attempts) {
222 int rc = 0;
223 for (int i = 0; g_running && i < attempts; i++) {
224 NETWORK* netw = NULL;
225 if (netw_connect(&netw, (char*)host, (char*)port, NETWORK_IPALL) != TRUE) {
226 n_log(LOG_ERR, "client connect %d/%d to %s:%s failed", i + 1, attempts, host, port);
227 rc = 4;
228 continue;
229 }
230 netw_start_thr_engine(netw);
231
232 char payload[64];
233 snprintf(payload, sizeof(payload), "hello-from-client-%d", i + 1);
234 N_STR* out = char_to_nstr(payload);
235 if (netw_add_msg(netw, out) != TRUE) {
236 free_nstr(&out);
237 }
238
239 N_STR* in = netw_wait_msg(netw, 25000, 5000000);
240 if (in) {
241 n_log(LOG_NOTICE, "client %d: echo received (%zu bytes)", i + 1, in->length);
242 free_nstr(&in);
243 } else {
244 n_log(LOG_ERR, "client %d: no echo within timeout", i + 1);
245 rc = 5;
246 }
247 netw_close(&netw);
248 }
249 netw_unload();
250 return rc;
251}

253int main(int argc, char** argv) {
254 char* addr = NULL;
255 char* host = NULL;
256 char* port = NULL;
257 int mode = MODE_SERVER;
258 int count = 5;
259 int log_level = LOG_NOTICE;
260 int explicit_server = 0;
261 int opt;
262
263 while ((opt = getopt(argc, argv, "ha:s:p:n:V:")) != -1) {
264 switch (opt) {
265 case 'a':
266 mode = MODE_SERVER;
267 explicit_server = 1;
268 addr = strdup(optarg);
269 break;
270 case 's':
271 mode = MODE_CLIENT;
272 host = strdup(optarg);
273 break;
274 case 'p':
275 port = strdup(optarg);
276 break;
277 case 'n':
278 count = atoi(optarg);
279 if (count <= 0) count = 1;
280 break;
281 case 'V':
282 if (!strcmp(optarg, "LOG_DEBUG"))
283 log_level = LOG_DEBUG;
284 else if (!strcmp(optarg, "LOG_INFO"))
285 log_level = LOG_INFO;
286 else if (!strcmp(optarg, "LOG_NOTICE"))
287 log_level = LOG_NOTICE;
288 else if (!strcmp(optarg, "LOG_ERR"))
289 log_level = LOG_ERR;
290 break;
291 case 'h':
292 default:
293 usage();
294 FreeNoLog(addr);
295 FreeNoLog(host);
296 FreeNoLog(port);
297 return 1;
298 }
299 }
300 (void)explicit_server;
301
302 if (!port) {
303 fprintf(stderr, "ex_network_reactor: -p PORT is required\n");
304 usage();
305 FreeNoLog(addr);
306 FreeNoLog(host);
307 return 1;
308 }
309
310 set_log_level(log_level);
311 signal(SIGINT, sighandler);
312 signal(SIGTERM, sighandler);
313#ifdef __linux__
314 signal(SIGPIPE, SIG_IGN);
315#endif
316
317 int rc;
318 if (mode == MODE_CLIENT) {
319 rc = run_client(host, port, count);
320 } else {
321 rc = run_server(addr, port, count);
322 }
323
324 FreeNoLog(addr);
325 FreeNoLog(host);
326 FreeNoLog(port);
327 return rc;
328}
