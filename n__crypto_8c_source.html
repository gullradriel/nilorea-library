68N_STR* n_vigenere_get_rootkey(size_t rootkey_size) {
69 if ((rootkey_size % 2) != 0) {
70 n_log(LOG_ERR, "key size %zu is not a multiple of 2", rootkey_size);
71 return NULL;
72 }
73 time_t t = time(NULL);
74 if (t == -1) {
75 n_log(LOG_ERR, "Error: Unable to get the current time");
76 return NULL;
77 }
78 N_STR* output = new_nstr(rootkey_size + 1);
79 /* NOTE: rand() seeded with time() is not cryptographically secure.
80 * This key is suitable only for obfuscation, not for strong protection. */
81 srand((unsigned int)t);
82 for (size_t it = 0; it < rootkey_size; it++) {
83 output->data[it] = (char)('A' + rand() % 26);
84 }
85 output->written = rootkey_size;
86 return output;
87}

93N_STR* n_get_current_dir_hd_serial() {
94 N_STR* output = NULL;
95
96#ifdef __windows__
97 DWORD dwVolSerial;
98 BOOL bIsRetrieved;
99 bIsRetrieved = GetVolumeInformation(NULL, NULL, 0, &dwVolSerial, NULL, NULL, NULL, 0);
100
101 if (bIsRetrieved) {
102 nstrprintf(output, "%X", (unsigned int)dwVolSerial);
103 } else {
104 n_log(LOG_ERR, "Could not retrieve current directory disk serial id");
105 return NULL;
106 }
107 return output;
108#elif __linux__
109 char* curdir = getcwd(NULL, 0);
110
111 N_STR* cmd = NULL;
112 nstrprintf(cmd, "df -T \"%s\" | awk '/^\\/dev/ {print $1}'", curdir);
113 Free(curdir);
114
115 N_STR* cur_disk = NULL;
116 int ret = -1;
117 n_popen(_nstr(cmd), 1024, (void**)&cur_disk, &ret);
118 free_nstr(&cmd);
119
120 nstrprintf(cmd, "udevadm info --query=all --name=\"%s\" | egrep '(ID_SERIAL=|ID_FS_UUID=)' | awk -F'=' '{print $2}'", _nstr(cur_disk));
121 n_popen(_nstr(cmd), 1024, (void**)&output, &ret);
122 free_nstr(&cmd);
123
124 N_STR* hd_serial = NULL;
125 nstrprintf(hd_serial, "%s=>%s", _nstr(output), _nstr(cur_disk));
126
127 free_nstr(&output);
128 free_nstr(&cur_disk);
129
130 for (unsigned int it = 0; it < hd_serial->written; it++) {
131 if (isspace(hd_serial->data[it])) {
132 hd_serial->data[it] = '-';
133 }
134 }
135 return hd_serial;
136
137#else /* SOLARIS */
138
139 char* curdir = getcwd(NULL, 0);
140 N_STR* hostid = NULL;
141 int ret = -1;
142 n_popen("hostid", 1024, (void**)&hostid, &ret);
143 /* build output before freeing curdir */
144 nstrprintf(output, "%s:%s", _nstr(hostid), _str(curdir));
145 free_nstr(&hostid);
146 Free(curdir);
147 return output;
148
149#endif
150}

156N_STR* n_get_cpu_id() {
157 N_STR* cpu_id = NULL;
158
159 /* x86 / x86_64 */
160#if defined(__i386__) || defined(__x86_64__)
161
162 unsigned int eax, ebx, ecx, edx;
163
164 if (__get_cpuid_max(0, NULL) >= 1 &&
165 __get_cpuid(1, &eax, &ebx, &ecx, &edx)) {
166 unsigned int stepping = eax & 0xF;
167 unsigned int model = (eax >> 4) & 0xF;
168 unsigned int family = (eax >> 8) & 0xF;
169 unsigned int processorType = (eax >> 12) & 0x3;
170
171 nstrprintf(cpu_id, "%u%u%u%u-%u%u%u%u-%u%u%u%u-%u%u",
172 (stepping >> 3) & 1,
173 (stepping >> 2) & 1,
174 (stepping >> 1) & 1,
175 stepping & 1,
176
177 (model >> 3) & 1,
178 (model >> 2) & 1,
179 (model >> 1) & 1,
180 model & 1,
181
182 (family >> 3) & 1,
183 (family >> 2) & 1,
184 (family >> 1) & 1,
185 family & 1,
186
187 (processorType >> 1) & 1,
188 processorType & 1);
189
190 return cpu_id;
191 }
192#endif
193
194 /* Solaris fallback */
195#ifdef __solaris__
196 static const char devname[] = "/dev/cpu/self/cpuid";
197
198 struct {
199 uint32_t r_eax, r_ebx, r_ecx, r_edx;
200 } regs;
201
202 int d;
203
204 if ((d = open(devname, O_RDONLY)) != -1) {
205 if (pread(d, &regs, sizeof(regs), 0) == sizeof(regs)) {
206 nstrprintf(cpu_id, "%08x", regs.r_eax);
207 close(d);
208 return cpu_id;
209 }
210 close(d);
211 }
212#endif
213
214 /* Linux / Android fallback */
215#if defined(__linux__) || defined(__ANDROID__)
216 FILE* f = fopen("/proc/cpuinfo", "r");
217 if (f) {
218 char line[256];
219
220 while (fgets(line, sizeof(line), f)) {
221 if (strncmp(line, "model name", 10) == 0 ||
222 strncmp(line, "Hardware", 8) == 0) {
223 char* colon = strchr(line, ':');
224 if (colon) {
225 colon += 2; /* skip ": " */
226 colon[strcspn(colon, "\n")] = 0;
227 nstrprintf(cpu_id, "%s", colon);
228 fclose(f);
229 return cpu_id;
230 }
231 }
232 }
233
234 fclose(f);
235 }
236#endif
237
238 /* Ultimate fallback */
239 n_log(LOG_ERR, "unable to determine CPU id");
240 nstrprintf(cpu_id, "unknown");
241 return cpu_id;
242}

265void* n_vigenere_cypher_thread(void* ptr) {
266 __n_assert(ptr, return NULL);
267 CYPHER_PARAM* params = (CYPHER_PARAM*)ptr;
268 __n_assert(params, return NULL);
269
270 N_STR* input = params->input;
271 N_STR* dest = params->dest;
272 N_STR* key = params->key;
273
274 __n_assert(input, return NULL);
275 __n_assert(input->data, return NULL);
276 __n_assert(key, return NULL);
277 __n_assert(key->data, return NULL);
278
279 if (key->written == 0) return NULL;
280
281 bool encipher = params->encipher;
282
283 size_t start_pos = params->start_pos;
284 size_t end_pos = params->end_pos;
285
286 bool cIsUpper = 0;
287 size_t keyIndexMod = 0;
288 if (encipher) {
289 for (size_t i = start_pos; i < end_pos; i++) {
290 if ((cIsUpper = n_isupper(input->data[i])) || (n_islower(input->data[i]))) {
291 char offset = cIsUpper ? 'A' : 'a';
292 keyIndexMod = i % key->written;
293 int k = (cIsUpper ? n_toupper(key->data[keyIndexMod]) : n_tolower(key->data[keyIndexMod])) - offset;
294 dest->data[i] = (char)((n_mathmod(((input->data[i] + k) - offset), 26)) + offset);
295 }
296 }
297 } else {
298 for (size_t i = start_pos; i < end_pos; i++) {
299 if ((cIsUpper = n_isupper(input->data[i])) || (n_islower(input->data[i]))) {
300 char offset = cIsUpper ? 'A' : 'a';
301 keyIndexMod = i % key->written;
302 int k = (cIsUpper ? n_toupper(key->data[keyIndexMod]) : n_tolower(key->data[keyIndexMod])) - offset;
303 dest->data[i] = (char)((n_mathmod(((input->data[i] - k) - offset), 26)) + offset);
304 }
305 }
306 }
307 Free(params);
308 return NULL;
309}

319N_STR* n_vigenere_cypher(N_STR* input, N_STR* key, bool encipher, bool in_place) {
320 __n_assert(input, return NULL);
321 __n_assert(input->data, return NULL);
322 __n_assert(key, return NULL);
323 __n_assert(key->data, return NULL);
324
325 if (input->written == 0) {
326 n_log(LOG_ERR, "empty input, written=0");
327 return NULL;
328 }
329
330 if ((key->written % 2) != 0) {
331 n_log(LOG_ERR, "odd number of elements in root key %s, %zu", _nstr(key), key->written);
332 return NULL;
333 }
334
335 for (size_t i = 0; i < key->written; ++i) {
336 if (!isalpha(key->data[i])) {
337 n_log(LOG_ERR, "key contain bad char '%c'", key->data[i]);
338 return NULL; // Error
339 }
340 }
341
342 N_STR* dest = NULL;
343 if (in_place) {
344 dest = input;
345 } else {
346 dest = nstrdup(input);
347 }
348
349#ifdef __windows__
350 SYSTEM_INFO sysinfo;
351 GetSystemInfo(&sysinfo);
352 size_t nb_procs = (size_t)sysinfo.dwNumberOfProcessors;
353#else
354 size_t nb_procs = (size_t)get_nprocs();
355#endif
356
357 if (nb_procs < 2)
358 nb_procs = 2;
359
360 THREAD_POOL* thread_pool = new_thread_pool(nb_procs, 0);
361 if (!thread_pool) {
362 n_log(LOG_ERR, "Could not create thread pool for vigenere cypher");
363 if (!in_place) free_nstr(&dest);
364 return NULL;
365 }
366
367 size_t bytesteps = input->written / nb_procs;
368 if (bytesteps == 0) bytesteps = input->written;
369
370 size_t end_pos = 0;
371 for (size_t it = 0; it < input->written; it += bytesteps) {
372 end_pos = it + bytesteps;
373
374 if (end_pos >= input->written)
375 end_pos = input->written;
376
377 CYPHER_PARAM* params = NULL;
378 Malloc(params, CYPHER_PARAM, 1);
379 params->input = input;
380 params->dest = dest;
381 params->key = key;
382 params->encipher = encipher;
383 params->start_pos = it;
384 params->end_pos = end_pos;
385
386 if (add_threaded_process(thread_pool, &n_vigenere_cypher_thread, (void*)(intptr_t)params, DIRECT_PROC) == FALSE) {
387 n_log(LOG_ERR, "Error adding n_vigenere_cypher_thread to thread pool");
388 Free(params);
389 }
390 refresh_thread_pool(thread_pool);
391 usleep(1);
392 }
393 if (end_pos < input->written) {
394 CYPHER_PARAM* params = NULL;
395 Malloc(params, CYPHER_PARAM, 1);
396 params->input = input;
397 params->dest = dest;
398 params->key = key;
399 params->encipher = encipher;
400 params->start_pos = end_pos;
401 params->end_pos = input->written;
402
403 if (add_threaded_process(thread_pool, &n_vigenere_cypher_thread, (void*)(intptr_t)params, DIRECT_PROC) == FALSE) {
404 n_log(LOG_ERR, "Error adding n_vigenere_cypher_thread to thread pool");
405 Free(params);
406 }
407 refresh_thread_pool(thread_pool);
408 usleep(1);
409 }
410 wait_for_threaded_pool(thread_pool);
411 destroy_threaded_pool(&thread_pool, 100000);
412
413 /*
414 uint32_t nonAlphaCharCount = 0;
415 for( uint32_t i = 0 ; i < input -> written ; i++ )
416 {
417 char ch = input -> data[ i ];
418 if( isalpha( input -> data[ i ] ) )
419 {
420 bool cIsUpper = isupper( input -> data[ i ] );
421 char offset = cIsUpper ? 'A' : 'a';
422 int keyIndex = ( i - nonAlphaCharCount ) % key -> written ;
423 int k = ( cIsUpper ? toupper( key -> data[ keyIndex % key -> written ] ) : tolower( key -> data[ keyIndex % key -> written ] ) ) - offset ;
424 k = encipher ? k : -k;
425 ch = (char)( ( n_mathmod( ( ( input -> data[ i ] + k ) - offset ) , 26 ) ) + offset );
426 }
427 else
428 {
429 ++nonAlphaCharCount;
430 }
431 dest -> data[ i ] = ch;
432 }
433 */
434
435 return dest;
436}

515N_STR* n_vigenere_get_question(size_t question_size) {
516 size_t it = 0;
517
518 if ((question_size % 2) != 0) {
519 n_log(LOG_ERR, "odd number of elements for question size, %zu", question_size);
520 return NULL;
521 }
522
523 N_STR* output = NULL;
524 output = n_get_current_dir_hd_serial();
525
526 N_STR* cpu_id = n_get_cpu_id();
527 nstrprintf_cat(output, "@%s", _nstr(cpu_id));
528 free_nstr(&cpu_id);
529
530 if (output) {
531 N_STR* tmpoutput = n_base64_encode(output);
532 free_nstr(&output);
533 output = new_nstr(question_size + 1);
534 for (it = 0; it < question_size; it++) {
535 int val = tmpoutput->data[it % tmpoutput->written];
536 while (val < 'A') val += 26;
537 while (val > 'Z') val -= 26;
538 output->data[it] = (char)val;
539 }
540 output->written = question_size;
541 free_nstr(&tmpoutput);
542 } else {
543 // generate random
544 output = new_nstr(question_size + 1);
545 for (it = 0; it < question_size; it++) {
546 int val = rand() % 256;
547 while (val < 'A') val += 26;
548 while (val > 'Z') val -= 26;
549 output->data[it] = (char)val;
550 }
551 }
552
553 N_STR* encoded_data = n_vigenere_quick_encode(output);
554 free_nstr(&output);
555
556 if (!encoded_data) {
557 return NULL;
558 }
559
560 it = 0;
561 while (it < encoded_data->written) {
562 if (encoded_data->data[it] == '=') {
563 encoded_data->data[it] = '\0';
564 }
565 it++;
566 }
567 return encoded_data;
568}

576N_STR* n_vigenere_get_answer(N_STR* root_key, N_STR* question) {
577 __n_assert(root_key, return NULL);
578 __n_assert(question, return NULL);
579
580 N_STR* decoded_question = n_vigenere_quick_decode(question);
581
582 if (!decoded_question) {
583 return NULL;
584 }
585
586 if (root_key->written < decoded_question->written) {
587 n_log(LOG_ERR, "root key of size %zu is lower than question key of size %zu", root_key->written, decoded_question->written);
588 free_nstr(&decoded_question);
589 return NULL;
590 }
591 if ((root_key->written % decoded_question->written) != 0) {
592 n_log(LOG_ERR, "question key of size %zu is not a multiple of root key of size %zu", decoded_question->written, root_key->written);
593 free_nstr(&decoded_question);
594 return NULL;
595 }
596
597 N_STR* answer = new_nstr(root_key->length + 1);
598 for (uint32_t it = 0; it < root_key->written; it++) {
599 int32_t val = (root_key->data[it % root_key->written] - 'A') + (decoded_question->data[it % decoded_question->written] - 'A');
600 val = val % 26;
601 val = val + 'A';
602 answer->data[it] = (char)val;
603 }
604 free_nstr(&decoded_question);
605 answer->written = root_key->written;
606 return answer;
607}

745N_STR* n_vigenere_decode_qa(N_STR* input_data, N_STR* question, N_STR* answer) {
746 __n_assert(input_data, return NULL);
747 __n_assert(input_data->data, return NULL);
748 __n_assert(question, return NULL);
749 __n_assert(question->data, return NULL);
750 __n_assert(answer, return NULL);
751 __n_assert(answer->data, return NULL);
752
753 N_STR* decoded_question = n_vigenere_quick_decode(question);
754
755 N_STR* encoded_data = n_vigenere_encode(input_data, decoded_question);
756 free_nstr(&decoded_question);
757 if (!encoded_data) {
758 n_log(LOG_ERR, "could not encode input_data %p", input_data);
759 return NULL;
760 }
761 n_log(LOG_DEBUG, "encoded_data: %zu/%zu", encoded_data->written, encoded_data->length);
762
763 N_STR* encoded_with_answer_data = n_vigenere_decode_in_place(encoded_data, answer);
764 if (!encoded_with_answer_data) {
765 n_log(LOG_ERR, "could not decode: input_data %p", input_data);
766 return NULL;
767 }
768 n_log(LOG_DEBUG, "encoded_with_answer_data: %zu/%zu", encoded_with_answer_data->written, encoded_with_answer_data->length);
769
770 N_STR* b64_decoded = n_base64_decode(encoded_with_answer_data);
