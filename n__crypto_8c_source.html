236void* n_vigenere_cypher_thread(void* ptr) {
237 __n_assert(ptr, return NULL);
238 CYPHER_PARAM* params = (CYPHER_PARAM*)ptr;
239 __n_assert(params, return NULL);
240
241 N_STR* input = params->input;
242 N_STR* dest = params->dest;
243 N_STR* key = params->key;
244
245 __n_assert(input, return NULL);
246 __n_assert(input->data, return NULL);
247 __n_assert(key, return NULL);
248 __n_assert(key->data, return NULL);
249
250 bool encipher = params->encipher;
251
252 size_t start_pos = params->start_pos;
253 size_t end_pos = params->end_pos;
254
255 bool cIsUpper = 0;
256 size_t keyIndexMod = 0;
257 if (encipher) {
258 for (size_t i = start_pos; i < end_pos; i++) {
259 if ((cIsUpper = n_isupper(input->data[i])) || (n_islower(input->data[i]))) {
260 char offset = cIsUpper ? 'A' : 'a';
261 keyIndexMod = i % key->written;
262 int k = (cIsUpper ? n_toupper(key->data[keyIndexMod]) : n_tolower(key->data[keyIndexMod])) - offset;
263 dest->data[i] = (char)((n_mathmod(((input->data[i] + k) - offset), 26)) + offset);
264 }
265 }
266 } else {
267 for (size_t i = start_pos; i < end_pos; i++) {
268 if ((cIsUpper = n_isupper(input->data[i])) || (n_islower(input->data[i]))) {
269 char offset = cIsUpper ? 'A' : 'a';
270 keyIndexMod = i % key->written;

272 dest->data[i] = (char)((n_mathmod(((input->data[i] - k) - offset), 26)) + offset);
273 }
274 }
275 }
276 Free(params);
277 return NULL;

477N_STR* n_vigenere_get_question(size_t question_size) {
478 size_t it = 0;
479
480 if ((question_size % 2) != 0) {
481 n_log(LOG_ERR, "odd number of elements for question size, %d", question_size);
482 return NULL;
483 }
484
485 N_STR* output = NULL;
486 output = n_get_current_dir_hd_serial();
487
488 N_STR* cpu_id = n_get_cpu_id();
489 nstrprintf_cat(output, "@%s", _nstr(cpu_id));
490 free_nstr(&cpu_id);
491
492 if (output) {
493 N_STR* tmpoutput = n_base64_encode(output);
494 free_nstr(&output);
495 output = new_nstr(question_size + 1);
496 for (it = 0; it < question_size; it++) {
497 output->data[it] = tmpoutput->data[it % tmpoutput->written];
498 while (output->data[it] < 'A') output->data[it] += 26;
499 while (output->data[it] > 'Z') output->data[it] -= 26;
500 }
501 output->written = question_size;
502 free_nstr(&tmpoutput);
503 } else {
504 // generate random
505 output = new_nstr(question_size + 1);
506 for (it = 0; it < question_size; it++) {
507 output->data[it] = (char)(rand() % 256);
508 while (output->data[it] < 'A') output->data[it] += 26;
509 while (output->data[it] > 'Z') output->data[it] -= 26;
510 }
511 }
512
513 N_STR* encoded_data = n_vigenere_quick_encode(output);
514 free_nstr(&output);
515
516 it = 0;
517 while (it < encoded_data->written) {
518 if (encoded_data->data[it] == '=') {
519 encoded_data->data[it] = '\0';
520 }
521 it++;
522 }
523 return encoded_data;
524}

532N_STR* n_vigenere_get_answer(N_STR* root_key, N_STR* question) {
533 __n_assert(root_key, return NULL);
534 __n_assert(question, return NULL);
535
536 N_STR* decoded_question = n_vigenere_quick_decode(question);
537
538 if (root_key->written < decoded_question->written) {
539 n_log(LOG_ERR, "root key of size %d is lower than question key of size %d", root_key->written, decoded_question->written);
540 free_nstr(&decoded_question);
541 return NULL;
542 }
543 if ((root_key->written % decoded_question->written) != 0) {
544 n_log(LOG_ERR, "question key of size %d is not a multiple of root key of size %d", decoded_question->written, root_key->written);
545 free_nstr(&decoded_question);
546 return NULL;
547 }
548
549 N_STR* answer = new_nstr(root_key->length + 1);
550 for (uint32_t it = 0; it < root_key->written; it++) {
551 int32_t val = (root_key->data[it % root_key->written] - 'A') + (decoded_question->data[it % decoded_question->written] - 'A');
552 val = val % 26;
553 val = val + 'A';
554 answer->data[it] = (char)val;
555 }
556 free_nstr(&decoded_question);
557 answer->written = root_key->written;
558 return answer;
559}
