54N_STR* zip4_nstr(N_STR* src) {
55 __n_assert(src, return NULL);
56 __n_assert(src->data, return NULL);
57
58 if (src->length == 0) {
59 n_log(LOG_ERR, "length of src == 0");
60 return NULL;
61 }
62 if (src->written == 0) {
63 n_log(LOG_ERR, "written of src == 0");
64 return NULL;
65 }
66 /* LZ4's API is int-sized. Reject anything that would overflow. */
67 if (src->written > (size_t)INT_MAX) {
68 n_log(LOG_ERR, "written of src > INT_MAX");
69 return NULL;
70 }
71
72 int src_len = (int)src->written;
73 int max_compressed = LZ4_compressBound(src_len);
74 if (max_compressed <= 0) {
75 n_log(LOG_ERR, "LZ4_compressBound(%d) failed", src_len);
76 return NULL;
77 }
78
79 N_STR* zipped = new_nstr((size_t)(4 + max_compressed));
80 __n_assert(zipped, return NULL);
81 __n_assert(zipped->data, return NULL);
82
83 /* Original length header (network byte order). */
84 uint32_t src_length_be = htonl((uint32_t)src->written);
85 memcpy(zipped->data, &src_length_be, sizeof(uint32_t));
86 char* dataptr = zipped->data + 4;
87
88 int compressed = LZ4_compress_default(src->data, dataptr, src_len, max_compressed);
89 if (compressed <= 0) {
90 n_log(LOG_ERR, "LZ4_compress_default failed (src=%d bytes)", src_len);
91 free_nstr(&zipped);
92 return NULL;
93 }
94 zipped->written = (size_t)(4 + compressed);
95 n_log(LOG_DEBUG, "lz4 zip: %d -> %zu (header incl.)", src_len, zipped->written);
96 return zipped;
97} /* zip4_nstr */

106N_STR* unzip4_nstr(N_STR* src) {
107 __n_assert(src, return NULL);
108 __n_assert(src->data, return NULL);
109
110 if (src->written < 4) {
111 n_log(LOG_ERR, "compressed data too short (%zu bytes)", src->written);
112 return NULL;
113 }
114
115 uint32_t original_size = 0;
116 memcpy(&original_size, src->data, sizeof(uint32_t));
117 original_size = ntohl(original_size);
118 if (original_size == 0) {
119 n_log(LOG_ERR, "lz4 frame declares original_size == 0");
120 return NULL;
121 }
122 if (original_size > 256u * 1024u * 1024u) {
123 n_log(LOG_ERR, "lz4 frame declares absurd original_size %u", original_size);
124 return NULL;
125 }
126
127 N_STR* unzipped = new_nstr(original_size);
128 __n_assert(unzipped, return NULL);
129 __n_assert(unzipped->data, return NULL);
130
131 size_t compressed_bytes = src->written - 4;
132 if (compressed_bytes > (size_t)INT_MAX) {
133 n_log(LOG_ERR, "compressed payload > INT_MAX");
134 free_nstr(&unzipped);
135 return NULL;
136 }
137
138 int decoded = LZ4_decompress_safe(src->data + 4, unzipped->data,
139 (int)compressed_bytes, (int)original_size);
140 if (decoded < 0) {
141 n_log(LOG_ERR, "LZ4_decompress_safe failed (compressed=%zu, original=%u)",
142 compressed_bytes, original_size);
143 free_nstr(&unzipped);
144 return NULL;
145 }
146 unzipped->written = (size_t)decoded;
147 n_log(LOG_DEBUG, "lz4 unzip: %zu -> %zu (original=%u)",
148 src->written, unzipped->written, original_size);
149 return unzipped;
150} /* unzip4_nstr */
