92int n_kafka_get_status(N_KAFKA* kafka, size_t* nb_queued, size_t* nb_waiting, size_t* nb_error) {
93 __n_assert(kafka, return FALSE);
94 __n_assert(nb_waiting, return FALSE);
95 __n_assert(nb_error, return FALSE);
96 read_lock(kafka->rwlock);
97 int status = kafka->polling_thread_status;
98 *nb_queued = kafka->nb_queued;
99 *nb_waiting = kafka->nb_waiting;
100 *nb_error = kafka->nb_error;
101 unlock(kafka->rwlock);
102 return status;

114static void n_kafka_delivery_message_callback(rd_kafka_t* rk, const rd_kafka_message_t* rkmessage, void* opaque) {
115 (void)opaque;
116
117 __n_assert(rk, n_log(LOG_ERR, "rk=NULL is not a valid kafka handle"); return);
118 __n_assert(rkmessage, n_log(LOG_ERR, "rkmessage=NULL is not a valid kafka message"); return);
119
120 N_KAFKA_EVENT* event = (N_KAFKA_EVENT*)rkmessage->_private;
121
122 if (rkmessage->err) {
123 n_log(LOG_ERR, "message delivery failed: %s", rd_kafka_err2str(rkmessage->err));
124 if (!event) {
125 n_log(LOG_ERR, "fatal: event is NULL");
126 return;
127 }
128 if (event->parent_table)
129 write_lock(event->parent_table->rwlock);
130 event->status = N_KAFKA_EVENT_ERROR;
131 /* adjust counters: event was WAITING_ACK, now it's ERROR */
132 if (event->parent_table) {
133 event->parent_table->nb_waiting--;
134 event->parent_table->nb_error++;
135 }
136 if (event->parent_table)
137 unlock(event->parent_table->rwlock);
138 } else {
139 n_log(LOG_DEBUG, "message delivered (%ld bytes, partition %d)", rkmessage->len, rkmessage->partition);
140 if (event) {
141 // lock
142 if (event && event->parent_table)
143 write_lock(event->parent_table->rwlock);
144 // delete produce event linked files
145 if (event->event_files_to_delete) {
146 char** files_to_delete = split(_nstr(event->event_files_to_delete), ";", 0);
147 if (files_to_delete) {
148 if (split_count(files_to_delete) > 0) {
149 int index = 0;
150 while (files_to_delete[index]) {
151 int ret = unlink(files_to_delete[index]);
152 int error = errno;
153 if (ret == 0) {
154 n_log(LOG_DEBUG, "deleted on produce ack: %s", files_to_delete[index]);
155 } else {
156 n_log(LOG_ERR, "couldn't delete \"%s\": %s", files_to_delete[index], strerror(error));
157 }
158 index++;
159 }
160 } else {
161 n_log(LOG_ERR, "split result is empty !");
162 }
163 }
164 free_split_result(&files_to_delete);
165 }
166 // set status
167 event->status = N_KAFKA_EVENT_OK;
168 // unlock
169 if (event && event->parent_table)
170 unlock(event->parent_table->rwlock);
171 n_log(LOG_INFO, "kafka event %p received an ack !", event);
172 } else {
173 n_log(LOG_ERR, "fatal: event is NULL");
174 }
175 }
176 return;
177 /* The rkmessage is destroyed automatically by librdkafka */
178}

184void n_kafka_delete(N_KAFKA* kafka) {
185 __n_assert(kafka, return);
186
187 FreeNoLog(kafka->topic);
188 free_split_result(&kafka->topics);
189 FreeNoLog(kafka->groupid);
190
191 n_kafka_stop_polling_thread(kafka);
192
193 /* rd_kafka_topic_destroy must be called before rd_kafka_destroy */
194 if (kafka->rd_kafka_topic)
195 rd_kafka_topic_destroy(kafka->rd_kafka_topic);
196 kafka->rd_kafka_topic = NULL;
197
198 if (kafka->rd_kafka_handle) {
199 if (kafka->mode == RD_KAFKA_CONSUMER) {
200 /* close the consumer. May already be closed by stop_polling_thread,
201 * calling it again is safe (returns an error but does not crash) */
202 rd_kafka_consumer_close(kafka->rd_kafka_handle);
203 if (kafka->subscription)
204 rd_kafka_topic_partition_list_destroy(kafka->subscription);
205 }
206 if (kafka->mode == RD_KAFKA_PRODUCER) {
207 rd_kafka_flush(kafka->rd_kafka_handle, kafka->poll_timeout);
208 }
209 rd_kafka_destroy(kafka->rd_kafka_handle);
210 n_log(LOG_DEBUG, "kafka handle destroyed");
211 }
212
213 if (kafka->rd_kafka_conf) {
214 rd_kafka_conf_destroy(kafka->rd_kafka_conf);
215 }
216
217 if (kafka->configuration)
218 cJSON_Delete(kafka->configuration);
219
220 if (kafka->errstr)
221 free_nstr(&kafka->errstr);
222
223 if (kafka->events_to_send)
224 list_destroy(&kafka->events_to_send);
225
226 if (kafka->received_events)
227 list_destroy(&kafka->received_events);
228
229 rw_lock_destroy(kafka->rwlock);
230
231 Free(kafka->bootstrap_servers);
232
233 Free(kafka);
234 return;
235}

244N_KAFKA* n_kafka_new(int32_t poll_timeout, int32_t poll_interval, size_t errstr_len) {
245 N_KAFKA* kafka = NULL;
246 Malloc(kafka, N_KAFKA, 1);
247 __n_assert(kafka, return NULL);
248
249 kafka->errstr = new_nstr(errstr_len);
250 __n_assert(kafka->errstr, Free(kafka); return NULL);
251
252 kafka->events_to_send = NULL;
253 kafka->received_events = NULL;
254 kafka->rd_kafka_conf = NULL;
255 kafka->rd_kafka_handle = NULL;
256 kafka->configuration = NULL;
257 kafka->groupid = NULL;
258 kafka->topics = NULL;
259 kafka->subscription = NULL;
260 kafka->topic = NULL;
261 kafka->mode = -1;
262 kafka->schema_id = -1;
263 kafka->poll_timeout = poll_timeout;
264 kafka->poll_interval = poll_interval;
265 kafka->monitored_directory_interval = 3000; // 3000 msecs as a default
266 kafka->nb_queued = 0;
267 kafka->nb_waiting = 0;
268 kafka->nb_error = 0;
269 kafka->event_consumption_enabled = TRUE;
270 kafka->event_production_enabled = TRUE;
271 kafka->polling_thread_status = 0;
272 kafka->bootstrap_servers = NULL;
273
274 kafka->events_to_send = new_generic_list(MAX_LIST_ITEMS);

276

278 __n_assert(kafka->received_events, n_kafka_delete(kafka); return NULL);
279
280 if (init_lock(kafka->rwlock) != 0) {
281 n_log(LOG_ERR, "could not init kafka rwlock in kafka structure at address %p", kafka);
282 n_kafka_delete(kafka);
283 return NULL;
284 }
285
286 return kafka;
287}

295N_KAFKA* n_kafka_load_config(char* config_file, int mode) {
296 __n_assert(config_file, return NULL);
297
298 N_KAFKA* kafka = NULL;
299
300 kafka = n_kafka_new(-1, 100, 1024);
301 __n_assert(kafka, return NULL);
302
303 // initialize kafka object
304 kafka->rd_kafka_conf = rd_kafka_conf_new();
305
306 // load config file
307 N_STR* config_string = NULL;
308 config_string = file_to_nstr(config_file);
309 if (!config_string) {
310 n_log(LOG_ERR, "unable to read config from file %s !", config_file);
311 n_kafka_delete(kafka);
312 return NULL;
313 }
314 cJSON* json = NULL;
315 json = cJSON_Parse(_nstrp(config_string));
316 /* config_string is no longer needed after parsing */
317 free_nstr(&config_string);
318 if (!json) {
319 n_log(LOG_ERR, "unable to parse json from file %s", config_file);
320 n_kafka_delete(kafka);
321 return NULL;
322 }
323
324 int jsonIndex;
325 for (jsonIndex = 0; jsonIndex < cJSON_GetArraySize(json); jsonIndex++) {
326 cJSON* entry = cJSON_GetArrayItem(json, jsonIndex);
327
328 if (!entry) continue;
329 __n_assert(entry->string, continue);
330 if (!entry->valuestring) {
331 n_log(LOG_DEBUG, "no valuestring for entry %s", _str(entry->string));
332 continue;
333 }
334
335 if (entry->string[0] != '-') {
336 // if it's not one of the optionnal parameters not managed by kafka, then we can use rd_kafka_conf_set on them
337 if (strcmp("topic", entry->string) != 0 &&
338 strcmp("topics", entry->string) != 0 &&
339 strcmp("value.schema.id", entry->string) != 0 &&
340 strcmp("value.schema.type", entry->string) != 0 &&
341 strcmp("poll.interval", entry->string) != 0 &&
342 strcmp("poll.timeout", entry->string) != 0 &&
343 strcmp("group.id.autogen", entry->string) != 0 &&
344 strcmp("monitored.directory.interval", entry->string) != 0) {
345 if (!strcmp("group.id", entry->string)) {
346 // exclude group id for producer
347 if (mode == RD_KAFKA_PRODUCER)
348 continue;
349 kafka->groupid = strdup(entry->valuestring);
350 }
351
352 if (rd_kafka_conf_set(kafka->rd_kafka_conf, entry->string, entry->valuestring, _nstr(kafka->errstr), kafka->errstr->length) != RD_KAFKA_CONF_OK) {
353 n_log(LOG_ERR, "kafka config: %s", _nstr(kafka->errstr));
354 } else {
355 n_log(LOG_DEBUG, "kafka config enabled: %s => %s", entry->string, entry->valuestring);
356 }
357 }
358 } else {
359 n_log(LOG_DEBUG, "kafka disabled config: %s => %s", entry->string, entry->valuestring);
360 }
361 }
362
363 // other parameters, not directly managed by kafka API (will cause an error if used along rd_kafka_conf_set )
364 // producer topic
365 cJSON* jstr = NULL;
366 jstr = cJSON_GetObjectItem(json, "topic");
367 if (jstr && jstr->valuestring) {
368 kafka->topic = strdup(jstr->valuestring);
369 n_log(LOG_DEBUG, "kafka producer topic: %s", kafka->topic);
370 } else {
371 if (mode == RD_KAFKA_PRODUCER) {
372 n_log(LOG_ERR, "no topic configured !");
373 cJSON_Delete(json);
374 n_kafka_delete(kafka);
375 return NULL;
376 }
377 }
378 // consumer topics
379 jstr = cJSON_GetObjectItem(json, "topics");
380 if (jstr && jstr->valuestring) {
381 kafka->topics = split(jstr->valuestring, ",", 0);
382 n_log(LOG_DEBUG, "kafka consumer topics: %s", jstr->valuestring);
383 } else {
384 if (mode == RD_KAFKA_CONSUMER) {
385 n_log(LOG_ERR, "no topics configured !");
386 cJSON_Delete(json);
387 n_kafka_delete(kafka);
388 return NULL;
389 }
390 }
391 jstr = cJSON_GetObjectItem(json, "value.schema.id");
392 if (jstr && jstr->valuestring) {
393 int schem_v = atoi(jstr->valuestring);
394 if (schem_v < -1 || schem_v > 9999) {
395 n_log(LOG_ERR, "invalid schema id %d", schem_v);
396 cJSON_Delete(json);
397 n_kafka_delete(kafka);
398 return NULL;
399 }
400 n_log(LOG_DEBUG, "kafka schema id: %d", schem_v);
401 kafka->schema_id = schem_v;
402 }
403
404 jstr = cJSON_GetObjectItem(json, "poll.interval");
405 if (jstr && jstr->valuestring) {
406 kafka->poll_interval = atoi(jstr->valuestring);
407 n_log(LOG_DEBUG, "kafka poll interval: %d", kafka->poll_interval);
408 }
409
410 jstr = cJSON_GetObjectItem(json, "poll.timeout");
411 if (jstr && jstr->valuestring) {
412 kafka->poll_timeout = atoi(jstr->valuestring);
413 n_log(LOG_DEBUG, "kafka poll timeout: %d", kafka->poll_timeout);
414 }
415
416 jstr = cJSON_GetObjectItem(json, "monitored.directory.interval");
417 if (jstr && jstr->valuestring) {
418 kafka->monitored_directory_interval = atoi(jstr->valuestring);
419 n_log(LOG_DEBUG, "kafka monitored directory interval: %d", kafka->monitored_directory_interval);
420 }
421
422 // saving bootstrap servers in struct
423 jstr = cJSON_GetObjectItem(json, "bootstrap.servers");
424 if (jstr && jstr->valuestring) {
425 kafka->bootstrap_servers = strdup(jstr->valuestring);
426 n_log(LOG_DEBUG, "kafka bootstrap server: %s", kafka->bootstrap_servers);
427 }
428
429 if (mode == RD_KAFKA_PRODUCER) {
430 // set delivery callback
431 rd_kafka_conf_set_dr_msg_cb(kafka->rd_kafka_conf, n_kafka_delivery_message_callback);
432
433 kafka->rd_kafka_handle = rd_kafka_new(RD_KAFKA_PRODUCER, kafka->rd_kafka_conf, _nstr(kafka->errstr), kafka->errstr->length);
434 if (!kafka->rd_kafka_handle) {
435 n_log(LOG_ERR, "failed to create new producer: %s", _nstr(kafka->errstr));
436 cJSON_Delete(json);
437 n_kafka_delete(kafka);
438 return NULL;
439 }
440 // conf is now owned by kafka handle
441 kafka->rd_kafka_conf = NULL;
442 // Create topic object
443 kafka->rd_kafka_topic = rd_kafka_topic_new(kafka->rd_kafka_handle, kafka->topic, NULL);
444 if (!kafka->rd_kafka_topic) {
445 cJSON_Delete(json);
446 n_kafka_delete(kafka);
447 return NULL;
448 }
449 kafka->mode = RD_KAFKA_PRODUCER;
450 } else if (mode == RD_KAFKA_CONSUMER) {
451 /* If there is no previously committed offset for a partition
452 * the auto.offset.reset strategy will be used to decide where
453 * in the partition to start fetching messages.
454 * By setting this to earliest the consumer will read all messages
455 * in the partition if there was no previously committed offset. */
456 /*if (rd_kafka_conf_set(kafka -> rd_kafka_conf, "auto.offset.reset", "earliest", _nstr( kafka -> errstr ) , kafka -> errstr -> length) != RD_KAFKA_CONF_OK) {
457 n_log( LOG_ERR , "kafka conf set: %s", kafka -> errstr);
458 n_kafka_delete( kafka );
459 return NULL ;
460 }*/
461
462 // if groupid is not set, generate a unique one
463 if (!kafka->groupid) {
464 char computer_name[1024] = "";
465 get_computer_name(computer_name, 1024);
466 N_STR* groupid = new_nstr(1024);
467 char* topics = join(kafka->topics, "_");
468 // generating group id
469 jstr = cJSON_GetObjectItem(json, "group.id.autogen");
470 if (jstr && jstr->valuestring) {
471 if (strcmp(jstr->valuestring, "host-topic-group") == 0) {
472 // nstrprintf( groupid , "%s_%s_%s" , computer_name , topics ,kafka -> bootstrap_servers );
473 nstrprintf(groupid, "%s_%s", computer_name, topics);
474 } else if (strcmp(jstr->valuestring, "unique-group") == 0) {
475 // nstrprintf( groupid , "%s_%s_%s_%d" , computer_name , topics , kafka -> bootstrap_servers , getpid() );
476 nstrprintf(groupid, "%s_%s_%d", computer_name, topics, getpid());
477 }
478 } else // default unique group
479 {
480 // nstrprintf( groupid , "%s_%s_%s_%d" , computer_name , topics , kafka -> bootstrap_servers , getpid() );
481 nstrprintf(groupid, "%s_%s_%d", computer_name, topics, getpid());
482 n_log(LOG_DEBUG, "group.id is not set and group.id.autogen is not set, generated unique group id: %s", _nstr(groupid));
483 }
484 free(topics);
485 kafka->groupid = groupid->data;
486 groupid->data = NULL;
487 free_nstr(&groupid);
488 }
489 if (rd_kafka_conf_set(kafka->rd_kafka_conf, "group.id", kafka->groupid, _nstr(kafka->errstr), kafka->errstr->length) != RD_KAFKA_CONF_OK) {
490 n_log(LOG_ERR, "kafka consumer group.id error: %s", _nstr(kafka->errstr));
491 cJSON_Delete(json);
492 n_kafka_delete(kafka);
493 return NULL;
494 } else {
495 n_log(LOG_DEBUG, "kafka consumer group.id => %s", kafka->groupid);
496 }
497
498 /* Create consumer instance.
499 * NOTE: rd_kafka_new() takes ownership of the conf object
500 * and the application must not reference it again after
501 * this call.
502 */
503 kafka->rd_kafka_handle = rd_kafka_new(RD_KAFKA_CONSUMER, kafka->rd_kafka_conf, _nstr(kafka->errstr), kafka->errstr->length);
504 if (!kafka->rd_kafka_handle) {
505 n_log(LOG_ERR, "%% Failed to create new consumer: %s", kafka->errstr);
506 cJSON_Delete(json);
507 n_kafka_delete(kafka);
508 return NULL;
509 }
510 // conf is now owned by kafka handle
511 kafka->rd_kafka_conf = NULL;
512
513 /* Redirect all messages from per-partition queues to
514 * the main queue so that messages can be consumed with one
515 * call from all assigned partitions.
516 *
517 * The alternative is to poll the main queue (for events)
518 * and each partition queue separately, which requires setting
519 * up a rebalance callback and keeping track of the assignment:
520 * but that is more complex and typically not recommended. */
521 rd_kafka_poll_set_consumer(kafka->rd_kafka_handle);
522
523 /* Convert the list of topics to a format suitable for librdkafka */
524 int topic_cnt = split_count(kafka->topics);
525 kafka->subscription = rd_kafka_topic_partition_list_new(topic_cnt);
526 for (int i = 0; i < topic_cnt; i++)
527 rd_kafka_topic_partition_list_add(kafka->subscription, kafka->topics[i],
528 /* the partition is ignored
529 * by subscribe() */
530 RD_KAFKA_PARTITION_UA);
531 /* Assign the topic. This method is disabled as it does noat allow dynamic partition assignement
532 int err = rd_kafka_assign(kafka -> rd_kafka_handle, kafka -> subscription);
533 if( err )
534 {
535 n_log( LOG_ERR , "kafka consumer: failed to assign %d topics: %s", kafka -> subscription->cnt, rd_kafka_err2str(err));
536 n_kafka_delete( kafka );
537 return NULL;
538 } */
539
540 /* Subscribe to the list of topics */
541 int err = rd_kafka_subscribe(kafka->rd_kafka_handle, kafka->subscription);
542 if (err) {
543 n_log(LOG_ERR, "kafka consumer: failed to subscribe to %d topics: %s", kafka->subscription->cnt, rd_kafka_err2str(err));
544 cJSON_Delete(json);
545 n_kafka_delete(kafka);
546 return NULL;
547 }
548
549 n_log(LOG_DEBUG, "kafka consumer created and subscribed to %d topic(s), waiting for rebalance and messages...", kafka->subscription->cnt);
550
551 kafka->mode = RD_KAFKA_CONSUMER;
552 } else {
553 n_log(LOG_ERR, "invalid mode %d", mode);
554 cJSON_Delete(json);
555 n_kafka_delete(kafka);
556 return NULL;
557 }
558
559 cJSON_Delete(json);
560 return kafka;
561} /* n_kafka_load_config */

637 return FALSE;
638 }
639
640 return TRUE;
641}

666int n_kafka_produce(N_KAFKA* kafka, N_KAFKA_EVENT* event) {
667 __n_assert(kafka, return FALSE);
668 __n_assert(event, return FALSE);
669
670 event->parent_table = kafka;
671
672 write_lock(kafka->rwlock);
673 kafka->nb_queued++;
674 list_push(kafka->events_to_send, event, &n_kafka_event_destroy_ptr);
675 unlock(kafka->rwlock);
676
677 // Success to *enqueue* event

850int n_kafka_poll(N_KAFKA* kafka) {
851 __n_assert(kafka, return FALSE);
852
853 int event_consumption_enabled = TRUE;
854 int event_production_enabled = TRUE;
855
856 read_lock(kafka->rwlock);
857 event_consumption_enabled = kafka->event_consumption_enabled;
858 event_production_enabled = kafka->event_production_enabled;
859 unlock(kafka->rwlock);
860
861 // wait poll interval msecs for kafka response
862 if (kafka->mode == RD_KAFKA_PRODUCER) {
863 if (event_production_enabled) {
864 int nb_events = rd_kafka_poll(kafka->rd_kafka_handle, kafka->poll_interval);
865 (void)nb_events;
866 write_lock(kafka->rwlock);
867 // check events status in event table
868 LIST_NODE* node = kafka->events_to_send->start;
869 while (node) {
870 N_KAFKA_EVENT* event = (N_KAFKA_EVENT*)node->ptr;
871 if (event->status == N_KAFKA_EVENT_OK) {
872 kafka->nb_waiting--;
873 n_log(LOG_DEBUG, "removing event OK %p", event);
874 LIST_NODE* node_to_kill = node;
875 node = node->next;
876 N_KAFKA_EVENT* event_to_kill = remove_list_node(kafka->events_to_send, node_to_kill, N_KAFKA_EVENT);
877 n_kafka_event_destroy(&event_to_kill);
878 continue;
879 } else if (event->status == N_KAFKA_EVENT_QUEUED) {
880 if (n_kafka_produce_ex(kafka, event) == FALSE) {
881 /* produce_ex already set status to ERROR,
882 * adjust counters: QUEUED -> ERROR */
883 kafka->nb_queued--;
884 kafka->nb_error++;
885 } else {
886 kafka->nb_waiting++;
887 kafka->nb_queued--;
888 }
889 } else if (event->status == N_KAFKA_EVENT_ERROR) {
890 /* counters already adjusted at transition time
891 * (in produce_ex failure above or in delivery callback) */
892 /* TODO: implement retry of errored events after error_timeout has elapsed */
893 }
894 node = node->next;
895 }
896 unlock(kafka->rwlock);
897 } else {
898 uint64_t sleep_us = (kafka->poll_interval > 0) ? ((uint64_t)(uint32_t)kafka->poll_interval * 1000ULL) : 0ULL;
899 usleep((unsigned int)(sleep_us > (uint64_t)UINT_MAX ? (uint64_t)UINT_MAX : sleep_us)); // production is disabled, sleep instead of producing
900 }
901 } else if (kafka->mode == RD_KAFKA_CONSUMER) {
902 if (event_consumption_enabled) {
903 rd_kafka_message_t* rkm = NULL;
904 while (event_consumption_enabled && (rkm = rd_kafka_consumer_poll(kafka->rd_kafka_handle, kafka->poll_interval))) {
905 read_lock(kafka->rwlock);
906 event_consumption_enabled = kafka->event_consumption_enabled;
907 unlock(kafka->rwlock);
908 if (rkm->err) {
909 /* Consumer errors are generally to be considered
910 * informational as the consumer will automatically
911 * try to recover from all types of errors. */
912 n_log(LOG_ERR, "consumer: %s", rd_kafka_message_errstr(rkm));
913 rd_kafka_message_destroy(rkm);
914 uint64_t sleep_us = (kafka->poll_interval > 0) ? ((uint64_t)(uint32_t)kafka->poll_interval * 1000ULL) : 0ULL;
915 usleep((unsigned int)(sleep_us > (uint64_t)UINT_MAX ? (uint64_t)UINT_MAX : sleep_us));
916 return FALSE;
917 }
918 // Reminder of rkm contents
919 n_log(LOG_DEBUG, "Message on %s [%" PRId32 "] at offset %" PRId64 " (leader epoch %" PRId32 ")", rd_kafka_topic_name(rkm->rkt), rkm->partition, rkm->offset, rd_kafka_message_leader_epoch(rkm));
920
921 // Print the message key
922 if (rkm->key && rkm->key_len > 0)
923 n_log(LOG_DEBUG, "Key: %.*s", (int)rkm->key_len, (const char*)rkm->key);
924 else if (rkm->key)
925 n_log(LOG_DEBUG, "Key: (%d bytes)", (int)rkm->key_len);
926
927 if (rkm->payload && rkm->len > 0) {
928 write_lock(kafka->rwlock);
929 // make a copy of the event for further processing
930 N_KAFKA_EVENT* event = NULL;
931 event = n_kafka_new_event_from_char(rkm->payload, rkm->len, -1); // no schema id because we want a full raw copy here
932 event->parent_table = kafka;
933 // test if there are headers, save them
934 rd_kafka_headers_t* hdrs = NULL;
935 if (!rd_kafka_message_headers(rkm, &hdrs)) {
936 size_t idx = 0;
937 const char* name = NULL;
938 const void* val = NULL;
939 size_t size = 0;
940 event->received_headers = new_generic_list(MAX_LIST_ITEMS);
941 while (!rd_kafka_header_get_all(hdrs, idx, &name, &val, &size)) {
942 N_STR* header_entry = NULL;
943 nstrprintf(header_entry, "%s=%s", _str(name), _str((char*)val));
944 list_push(event->received_headers, header_entry, &free_nstr_ptr);
945 idx++;
946 }
947 }
948 // save originating topic (can help sorting if there are multiples one)
949 event->from_topic = strdup(rd_kafka_topic_name(rkm->rkt));
950 if (kafka->schema_id != -1)
951 event->schema_id = n_kafka_get_schema_from_nstr(event->event_string);
952 list_push(kafka->received_events, event, &n_kafka_event_destroy_ptr);
953 n_log(LOG_DEBUG, "Consumer received event of (%d bytes) from topic %s", (int)rkm->len, event->from_topic);
954 unlock(kafka->rwlock);
955 }
956 rd_kafka_message_destroy(rkm);
957 }
958 } else {
959 uint64_t sleep_us = (kafka->poll_interval > 0) ? ((uint64_t)(uint32_t)kafka->poll_interval * 1000ULL) : 0ULL;
960 usleep((unsigned int)(sleep_us > (uint64_t)UINT_MAX ? (uint64_t)UINT_MAX : sleep_us)); // consumption is disabled, sleep instead of consuming
961 }
962 }
963 // n_log( LOG_DEBUG , "kafka poll for handle %p returned %d elements" , kafka -> rd_kafka_handle , nb_events );
964 return TRUE;
965} /* n_kafka_poll */

972void* n_kafka_polling_thread(void* ptr) {
973 N_KAFKA* kafka = (N_KAFKA*)ptr;
974
975 int status = 1;
976
977 N_TIME chrono;
978
979 if (kafka->mode == RD_KAFKA_PRODUCER)
980 n_log(LOG_DEBUG, "starting polling thread for kafka handler %p mode PRODUCER (%d) topic %s", kafka->rd_kafka_handle, RD_KAFKA_PRODUCER, kafka->topic);
981 if (kafka->mode == RD_KAFKA_CONSUMER) {
982 char* topiclist = join(kafka->topics, ",");
983 n_log(LOG_DEBUG, "starting polling thread for kafka handler %p mode CONSUMER (%d) topic %s", kafka->rd_kafka_handle, RD_KAFKA_CONSUMER, _str(topiclist));
984 FreeNoLog(topiclist);
985 }
986
987 start_HiTimer(&chrono);
988
989 int64_t remaining_time = kafka->poll_timeout * 1000;
990 while (status == 1) {
991 if (n_kafka_poll(kafka) == FALSE) {
992 if (kafka->mode == RD_KAFKA_PRODUCER) {
993 n_log(LOG_ERR, "failed to poll kafka producer handle %p with topic %s", kafka->rd_kafka_handle, rd_kafka_topic_name(kafka->rd_kafka_topic));
994 } else if (kafka->topics) {
995 char* topiclist = join(kafka->topics, ",");
996 n_log(LOG_ERR, "failed to poll kafka consumer handle %p with topic %s", kafka->rd_kafka_handle, _str(topiclist));
997 FreeNoLog(topiclist);
998 }
999 }
1000
1001 read_lock(kafka->rwlock);
1002 status = kafka->polling_thread_status;
1003 unlock(kafka->rwlock);
1004
1005 if (status == 2)
1006 break;
1007
1008 int64_t elapsed_time = get_usec(&chrono);
1009 if (kafka->poll_timeout != -1) {
1010 remaining_time -= elapsed_time;
1011 if (remaining_time < 0) {
1012 if (kafka->mode == RD_KAFKA_PRODUCER) {
1013 n_log(LOG_DEBUG, "timeouted on kafka handle %p", kafka->rd_kafka_handle);
1014 } else if (kafka->mode == RD_KAFKA_CONSUMER) {
1015 n_log(LOG_DEBUG, "timeouted on kafka handle %p", kafka->rd_kafka_handle);
1016 }
1017 break;
1018 }
1019 }
1020 // n_log( LOG_DEBUG , "remaining time: %d on kafka handle %p" , remaining_time , kafka -> rd_kafka_handle );
1021 }
1022
1023 write_lock(kafka->rwlock);
1024 kafka->polling_thread_status = 0;
1025 unlock(kafka->rwlock);
1026
1027 n_log(LOG_DEBUG, "exiting polling thread for kafka handler %p mode %s", kafka->rd_kafka_handle, (kafka->mode == RD_KAFKA_PRODUCER) ? "PRODUCER" : "CONSUMER");
1028 pthread_exit(NULL);
1029 return NULL;
1030} /* n_kafka_polling_thread */
