93static void n_kafka_delivery_message_callback(rd_kafka_t* rk, const rd_kafka_message_t* rkmessage, void* opaque) {
94 (void)opaque;
95
96 __n_assert(rk, n_log(LOG_ERR, "rk=NULL is not a valid kafka handle"); return);
97 __n_assert(rkmessage, n_log(LOG_ERR, "rkmessage=NULL is not a valid kafka message"); return);
98
99 N_KAFKA_EVENT* event = (N_KAFKA_EVENT*)rkmessage->_private;
100
101 if (rkmessage->err) {
102 n_log(LOG_ERR, "message delivery failed: %s", rd_kafka_err2str(rkmessage->err));
103 if (!event) {
104 n_log(LOG_ERR, "fatal: event is NULL");
105 return;
106 }
107 if (event && event->parent_table)
108 write_lock(event->parent_table->rwlock);
109 event->status = N_KAFKA_EVENT_ERROR;
110 if (event && event->parent_table)
111 unlock(event->parent_table->rwlock);
112 } else {
113 n_log(LOG_DEBUG, "message delivered (%ld bytes, partition %d)", rkmessage->len, rkmessage->partition);
114 if (event) {
115 // lock
116 if (event && event->parent_table)
117 write_lock(event->parent_table->rwlock);
118 // delete produce event linked files
119 if (event->event_files_to_delete) {
120 char** files_to_delete = split(_nstr(event->event_files_to_delete), ";", 0);
121 if (files_to_delete) {
122 if (split_count(files_to_delete) > 0) {
123 int index = 0;
124 while (files_to_delete[index]) {
125 int ret = unlink(files_to_delete[index]);
126 int error = errno;
127 if (ret == 0) {
128 n_log(LOG_DEBUG, "deleted on produce ack: %s", files_to_delete[index]);
129 } else {
130 n_log(LOG_ERR, "couldn't delete \"%s\": %s", files_to_delete[index], strerror(error));
131 }
132 index++;
133 }
134 } else {
135 n_log(LOG_ERR, "split result is empty !");
136 }
137 }
138 free_split_result(&files_to_delete);
139 }
140 // set status
141 event->status = N_KAFKA_EVENT_OK;
142 // unlock
143 if (event && event->parent_table)
144 unlock(event->parent_table->rwlock);
145 n_log(LOG_INFO, "kafka event %p received an ack !", event);
146 } else {
147 n_log(LOG_ERR, "fatal: event is NULL");
148 }
149 }
150 return;
151 /* The rkmessage is destroyed automatically by librdkafka */
152}

190 list_destroy(&kafka->events_to_send);

786int n_kafka_poll(N_KAFKA* kafka) {
787 __n_assert(kafka, return FALSE);
788 int ret = TRUE;
789
790 // wait poll interval msecs for kafka response
791 if (kafka->mode == RD_KAFKA_PRODUCER) {
792 int nb_events = rd_kafka_poll(kafka->rd_kafka_handle, kafka->poll_interval);
793 (void)nb_events;
794 write_lock(kafka->rwlock);
795 // check events status in event table
796 LIST_NODE* node = kafka->events_to_send->start;
797 while (node) {
798 N_KAFKA_EVENT* event = (N_KAFKA_EVENT*)node->ptr;
799 if (event->status == N_KAFKA_EVENT_OK) {
800 kafka->nb_waiting--;
801 n_log(LOG_DEBUG, "removing event OK %p", event);
802 LIST_NODE* node_to_kill = node;
803 node = node->next;
804 N_KAFKA_EVENT* event_to_kill = remove_list_node(kafka->events_to_send, node_to_kill, N_KAFKA_EVENT);
805 n_kafka_event_destroy(&event_to_kill);
806 continue;
807 } else if (event->status == N_KAFKA_EVENT_QUEUED) {
808 if (n_kafka_produce_ex(kafka, event) == FALSE) {
809 kafka->nb_error++;
810 } else {
811 kafka->nb_waiting++;
812 kafka->nb_queued--;
813 }
814 } else if (event->status == N_KAFKA_EVENT_ERROR) {
815 kafka->nb_error++;
816 // TODO :
817 // Re try sending errored events after error_timeout is done
818 //
819 }
820 node = node->next;
821 }
822 unlock(kafka->rwlock);
823 } else if (kafka->mode == RD_KAFKA_CONSUMER) {
824 rd_kafka_message_t* rkm = NULL;
825 while ((rkm = rd_kafka_consumer_poll(kafka->rd_kafka_handle, kafka->poll_interval))) {
826 if (rkm->err) {
827 /* Consumer errors are generally to be considered
828 * informational as the consumer will automatically
829 * try to recover from all types of errors. */
830 n_log(LOG_ERR, "consumer: %s", rd_kafka_message_errstr(rkm));
831 continue;
832 }
833
834 /* Reminder of rkm contents */
835 // Proper message.
836 n_log(LOG_DEBUG, "Message on %s [%" PRId32 "] at offset %" PRId64 " (leader epoch %" PRId32 ")", rd_kafka_topic_name(rkm->rkt), rkm->partition, rkm->offset, rd_kafka_message_leader_epoch(rkm));
837
838 // Print the message key
839 if (rkm->key && rkm->key_len > 0)
840 n_log(LOG_DEBUG, "Key: %.*s", (int)rkm->key_len, (const char*)rkm->key);
841 else if (rkm->key)
842 n_log(LOG_DEBUG, "Key: (%d bytes)", (int)rkm->key_len);
843
844 if (rkm->payload && rkm->len > 0) {
845 write_lock(kafka->rwlock);
846 // make a copy of the event for further processing
847 N_KAFKA_EVENT* event = NULL;
848 event = n_kafka_new_event_from_char(rkm->payload, rkm->len, -1); // no schema id because we want a full raw copy here
849 event->parent_table = kafka;
850 // test if there are headers, save them
851 rd_kafka_headers_t* hdrs = NULL;
852 if (!rd_kafka_message_headers(rkm, &hdrs)) {
853 size_t idx = 0;
854 const char* name = NULL;
855 const void* val = NULL;
856 size_t size = 0;
857
858 event->received_headers = new_generic_list(MAX_LIST_ITEMS);
859 while (!rd_kafka_header_get_all(hdrs, idx, &name, &val, &size)) {
860 N_STR* header_entry = NULL;
861 nstrprintf(header_entry, "%s=%s", _str(name), _str((char*)val));
862 list_push(event->received_headers, header_entry, &free_nstr_ptr);
863 idx++;
864 }
865 }
866 // save originating topic (can help sorting if there are multiples one)
867 event->from_topic = strdup(rd_kafka_topic_name(rkm->rkt));
868 if (kafka->schema_id != -1)
869 event->schema_id = n_kafka_get_schema_from_nstr(event->event_string);
870 list_push(kafka->received_events, event, &n_kafka_event_destroy_ptr);
871 n_log(LOG_DEBUG, "Consumer received event of (%d bytes) from topic %s", (int)rkm->len, event->from_topic);
872 unlock(kafka->rwlock);
873 }
874 rd_kafka_message_destroy(rkm);
875 }
876 }
877 if (kafka->nb_error > 0) {
878 ret = FALSE;
879 }
880 // n_log( LOG_DEBUG , "kafka poll for handle %p returned %d elements" , kafka -> rd_kafka_handle , nb_events );
881 return ret;
882} /* n_kafka_poll */

889void* n_kafka_pooling_thread(void* ptr) {
890 N_KAFKA* kafka = (N_KAFKA*)ptr;
891
892 int status = 1;
893
894 N_TIME chrono;
895
896 if (kafka->mode == RD_KAFKA_PRODUCER)
897 n_log(LOG_DEBUG, "starting pooling thread for kafka handler %p mode PRODUCER (%d) topic %s", kafka->rd_kafka_handle, RD_KAFKA_PRODUCER, kafka->topic);
898 if (kafka->mode == RD_KAFKA_CONSUMER) {
899 char* topiclist = join(kafka->topics, ",");
900 n_log(LOG_DEBUG, "starting pooling thread for kafka handler %p mode CONSUMER (%d) topic %s", kafka->rd_kafka_handle, RD_KAFKA_CONSUMER, _str(topiclist));
901 FreeNoLog(topiclist);
902 }
903
904 start_HiTimer(&chrono);
905
906 int64_t remaining_time = kafka->poll_timeout * 1000;
907 while (status == 1) {
908 if (n_kafka_poll(kafka) == FALSE) {
909 if (kafka->mode == RD_KAFKA_PRODUCER) {
910 n_log(LOG_ERR, "failed to poll kafka producer handle %p with topic %s", kafka->rd_kafka_handle, rd_kafka_topic_name(kafka->rd_kafka_topic));
911 } else if (kafka->topics) {
912 char* topiclist = join(kafka->topics, ",");
913 n_log(LOG_ERR, "failed to poll kafka consumer handle %p with topic %s", kafka->rd_kafka_handle, _str(topiclist));
914 FreeNoLog(topiclist);
915 }
916 }
917
918 read_lock(kafka->rwlock);
919 status = kafka->pooling_thread_status;
920 unlock(kafka->rwlock);
921
922 if (status == 2)
923 break;
924
925 int64_t elapsed_time = get_usec(&chrono);
926 if (kafka->poll_timeout != -1) {
927 remaining_time -= elapsed_time;
928 if (remaining_time < 0) {
929 if (kafka->mode == RD_KAFKA_PRODUCER) {
930 n_log(LOG_DEBUG, "timeouted on kafka handle %p", kafka->rd_kafka_handle);
931 } else if (kafka->mode == RD_KAFKA_CONSUMER) {
932 n_log(LOG_DEBUG, "timeouted on kafka handle %p", kafka->rd_kafka_handle);
933 }
934 remaining_time = 0;
935 status = 2;
936 break;
937 }
938 }
939 // n_log( LOG_DEBUG , "remaining time: %d on kafka handle %p" , remaining_time , kafka -> rd_kafka_handle );
940 }
941
942 write_lock(kafka->rwlock);
943 kafka->pooling_thread_status = 0;
944 unlock(kafka->rwlock);
945
946 n_log(LOG_DEBUG, "exiting pooling thread for kafka handler %p mode %s", kafka->rd_kafka_handle, (kafka->mode == RD_KAFKA_PRODUCER) ? "PRODUCER" : "CONSUMER");
947 pthread_exit(NULL);
948 return NULL;
949} /* n_kafka_pooling_thread */
