62int tree_insert_child(TREE_NODE* parent, TREE_NODE* child) {

66

93int tree_delete_node(TREE* tree, TREE_NODE* node) {
94 if (tree == NULL || node == NULL) {
95 return FALSE;
96 }
97
98 // Recursively delete child nodes
99 LIST_NODE* child_node = node->children->start;
100 while (child_node) {
101 TREE_NODE* child = (TREE_NODE*)child_node->ptr;
102 tree_delete_node(tree, child);
103 child_node = child_node->next;
104 }
105
106 // Remove the node from its parent's child list, if applicable
107 if (node->parent && node->parent->children) {
108 remove_list_node_f(node->parent->children, node->parent_list_node);
109 }
110
111 // Free node data if necessary
112 if (node->destroy_func) {
113 node->destroy_func(node->data.value.ptr);
114 } else {
115 free(node->data.value.ptr);
116 }
117
118 // Free the node itself
119 free(node);
120 tree->nb_nodes--;
121
122 return TRUE;
123}

254void insert(QUADTREE* qt, QUADTREE_NODE** root, COORD_VALUE x, COORD_VALUE y, void* data_ptr) {
255 if (*root == NULL) {
256 *root = create_node(x, y, data_ptr);
257 return;
258 }
259
260 if (qt->compare(x, (*root)->x) < 0 && qt->compare(y, (*root)->y) < 0) {
261 insert(qt, &((*root)->sw), x, y, data_ptr);
262 } else if (qt->compare(x, (*root)->x) < 0 && qt->compare(y, (*root)->y) >= 0) {
263 insert(qt, &((*root)->nw), x, y, data_ptr);
264 } else if (qt->compare(x, (*root)->x) >= 0 && qt->compare(y, (*root)->y) < 0) {
265 insert(qt, &((*root)->se), x, y, data_ptr);
266 } else if (qt->compare(x, (*root)->x) >= 0 && qt->compare(y, (*root)->y) >= 0) {
267 insert(qt, &((*root)->ne), x, y, data_ptr);

351int determine_octant(POINT3D point, POINT3D center, int type) {
352 int octant = 0;
353 if (type == COORD_INT) {
354 if (point.x.i >= center.x.i) octant |= 4;
355 if (point.y.i >= center.y.i) octant |= 2;
356 if (point.z.i >= center.z.i) octant |= 1;
357 } else if (type == COORD_FLOAT) {
358 if (point.x.f >= center.x.f) octant |= 4;
359 if (point.y.f >= center.y.f) octant |= 2;
360 if (point.z.f >= center.z.f) octant |= 1;
361 } else if (type == COORD_DOUBLE) {
362 if (point.x.d >= center.x.d) octant |= 4;
363 if (point.y.d >= center.y.d) octant |= 2;
364 if (point.z.d >= center.z.d) octant |= 1;
365 }
366 return octant;
367}
