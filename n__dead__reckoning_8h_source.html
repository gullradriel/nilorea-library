172/* Vector operations (inline helpers) */
173
175static inline DR_VEC3 dr_vec3(double x, double y, double z) {
176 DR_VEC3 v = {x, y, z};
177 return v;
178}
179
181static inline DR_VEC3 dr_vec3_zero(void) {
182 DR_VEC3 v = {0.0, 0.0, 0.0};
183 return v;
184}
185
187static inline DR_VEC3 dr_vec3_add(DR_VEC3 a, DR_VEC3 b) {
188 DR_VEC3 v = {a.x + b.x, a.y + b.y, a.z + b.z};
189 return v;
190}
191
193static inline DR_VEC3 dr_vec3_sub(DR_VEC3 a, DR_VEC3 b) {
194 DR_VEC3 v = {a.x - b.x, a.y - b.y, a.z - b.z};
195 return v;
196}
197
199static inline DR_VEC3 dr_vec3_scale(DR_VEC3 v, double s) {
200 DR_VEC3 r = {v.x * s, v.y * s, v.z * s};
201 return r;
202}
203
205static inline DR_VEC3 dr_vec3_lerp(DR_VEC3 a, DR_VEC3 b, double t) {
206 DR_VEC3 r = {
207 a.x + (b.x - a.x) * t,
208 a.y + (b.y - a.y) * t,
209 a.z + (b.z - a.z) * t};
210 return r;
211}
212
214static inline double dr_vec3_length_sq(DR_VEC3 v) {
215 return v.x * v.x + v.y * v.y + v.z * v.z;
216}
217
219static inline double dr_vec3_length(DR_VEC3 v) {
220 return sqrt(dr_vec3_length_sq(v));
221}
222
224static inline double dr_vec3_distance(DR_VEC3 a, DR_VEC3 b) {
225 return dr_vec3_length(dr_vec3_sub(a, b));
226}
