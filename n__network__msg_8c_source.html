18double double_swap(double value) {
19 double swaped = 0;
20 unsigned char* src = (unsigned char*)&value;
21 unsigned char* dst = (unsigned char*)&swaped;
22
23 /* swap bytes */
24 dst[0] = src[7];
25 dst[1] = src[6];
26 dst[2] = src[5];
27 dst[3] = src[4];
28 dst[4] = src[3];
29 dst[5] = src[2];
30 dst[6] = src[1];
31 dst[7] = src[0];
32
33 return swaped;
34} /* double_swap() */

301N_STR* make_str_from_msg(NETW_MSG* msg) {
302 size_t str_length = 0;
303 LIST_NODE* node = NULL;
304
305 N_STR *strptr = NULL,
306 *generated_str = NULL;
307
308 char* charptr = NULL;
309
310 __n_assert(msg, return NULL);
311
312 /* the first thing to do is to computer the resulting string size
313 the first 12 octets are for the number of ints , floats, strings in the message
314 */
315 str_length = 3 * sizeof(int32_t) + (msg->tabint->nb_items * sizeof(int32_t)) + (msg->tabflt->nb_items * sizeof(double));
316
317 /* let's count the eventual string blocks */
318 node = msg->tabstr->start;
319 while (node) {
320 strptr = (N_STR*)node->ptr;
321 str_length += sizeof(int32_t) + sizeof(int32_t) + strptr->written;
322 node = node->next;
323 }
324
325 /* preparing the new string */
326 Malloc(generated_str, N_STR, 1);
327 __n_assert(generated_str, return NULL);
328 Malloc(generated_str->data, char, str_length + 1);
329 __n_assert(generated_str->data, return NULL);
330
331 generated_str->length = str_length + 1;
332 generated_str->written = str_length;
333
334 /* copying header */
335 charptr = generated_str->data;
336
337 /* testing limits */
338 if (msg->tabint->nb_items >= UINT32_MAX) {
339 n_log(LOG_ERR, "too much items (>=UINT32_MAX) in int list of message %p", msg);
340 free_nstr(&generated_str);
341 return NULL;
342 }
343 if (msg->tabflt->nb_items >= UINT32_MAX) {
344 n_log(LOG_ERR, "too much items (>=UINT32_MAX) in float list of message %p", msg);
345 free_nstr(&generated_str);
346 return NULL;
347 }
348 if (msg->tabstr->nb_items >= UINT32_MAX) {
349 n_log(LOG_ERR, "too much items (>=UINT32_MAX) in string list of message %p", msg);
350 free_nstr(&generated_str);
351 return NULL;
352 }
353
354 /* byte order */
355 uint32_t nb_int_items = htonl((uint32_t)msg->tabint->nb_items);
356 uint32_t nb_float_items = htonl((uint32_t)msg->tabflt->nb_items);
357 uint32_t nb_str_items = htonl((uint32_t)msg->tabstr->nb_items);
358
359 /* number of ints inside the message */
360 memcpy(charptr, &nb_int_items, sizeof(uint32_t));
361 charptr += sizeof(uint32_t);
362 /* number of doubles inside the message */
363 memcpy(charptr, &nb_float_items, sizeof(uint32_t));
364 charptr += sizeof(uint32_t);
365 /* number of string inside the message */
366 memcpy(charptr, &nb_str_items, sizeof(uint32_t));
367 charptr += sizeof(uint32_t);
368
369 /* if there are numbers, copy all them ! */
370 node = msg->tabint->start;
371 while (node) {
372 uint32_t* nbptr = (uint32_t*)node->ptr;
373 uint32_t val = htonl(nbptr[0]);
374 memcpy(charptr, &val, sizeof(uint32_t));
375 charptr += sizeof(uint32_t);
376 node = node->next;
377 }
378
379 /* if there are numbers, copy all them ! */
380 node = msg->tabflt->start;
381 while (node) {
382 double* nbptr = (double*)node->ptr;
383 double val = htond(nbptr[0]);
384 memcpy(charptr, &val, sizeof(double));
385 charptr += sizeof(double);
386 node = node->next;
387 }
388
389 /* if there are strings, copy all them ! */
390 if (msg->tabstr->nb_items > 0) {
391 node = msg->tabstr->start;
392 while (node) {
393 strptr = (N_STR*)node->ptr;
394 if (strptr->length >= UINT32_MAX) {
395 n_log(LOG_ERR, "string too long (>=UINT32_MAX) in string list of message %p", msg);
396 free_nstr(&generated_str);
397 return NULL;
398 }
399 uint32_t var = htonl((uint32_t)strptr->length);
400 memcpy(charptr, &var, sizeof(uint32_t));
401 charptr += sizeof(uint32_t);
402
403 if (strptr->written >= UINT32_MAX) {
404 n_log(LOG_ERR, "string too long (>=UINT32_MAX) in string list of message %p", msg);
405 free_nstr(&generated_str);
406 return NULL;
407 }
408 var = htonl((uint32_t)strptr->written);
409 memcpy(charptr, &var, sizeof(uint32_t));
410 charptr += sizeof(uint32_t);
411
412 memcpy(charptr, strptr->data, strptr->written);
413 charptr += strptr->written;
414 node = node->next;
415 }
416 }
417
418 return generated_str;
419} /* make_str_from_msg(...) */

426NETW_MSG* make_msg_from_str(N_STR* str) {
427 NETW_MSG* generated_msg = NULL;
428 N_STR* tmpstr = NULL;
429
430 char* charptr = NULL;
431
432 uint32_t it,
433 nb_int = 0,
434 nb_flt = 0,
435 nb_str = 0;
436 int32_t tmpnb = 0;
437
438 double tmpflt = 0;
439
440 __n_assert(str, return NULL);
441
442 charptr = str->data;
443
444 create_msg(&generated_msg);
445
446 __n_assert(generated_msg, return NULL);
447
448 memcpy(&nb_int, charptr, sizeof(uint32_t));
449 charptr += sizeof(uint32_t);
450 nb_int = ntohl(nb_int);
451
452 memcpy(&nb_flt, charptr, sizeof(uint32_t));
453 charptr += sizeof(uint32_t);
454 nb_flt = ntohl(nb_flt);
455
456 memcpy(&nb_str, charptr, sizeof(uint32_t));
457 charptr += sizeof(uint32_t);
458 nb_str = ntohl(nb_str);
459
460 if (nb_int > 0) {
461 for (it = 0; it < nb_int; it++) {
462 memcpy(&tmpnb, charptr, sizeof(uint32_t));
463 tmpnb = (int32_t)ntohl((uint32_t)tmpnb);
464 charptr += sizeof(uint32_t);
465 add_int_to_msg(generated_msg, tmpnb);
466 }
467 }
468
469 if (nb_flt > 0) {
470 for (it = 0; it < nb_flt; it++) {
471 memcpy(&tmpflt, charptr, sizeof(double));
472 tmpflt = ntohd(tmpflt);
473 charptr += sizeof(double);
474 add_nb_to_msg(generated_msg, tmpflt);
475 }
476 }
477
478 if (nb_str > 0) {
479 for (it = 0; it < nb_str; it++) {
480 Malloc(tmpstr, N_STR, 1);
481 __n_assert(tmpstr, return NULL);
482 uint32_t var = 0;
483 memcpy(&var, charptr, sizeof(uint32_t));
484 charptr += sizeof(uint32_t);
485 tmpstr->length = ntohl(var);
486
487 memcpy(&var, charptr, sizeof(uint32_t));
488 charptr += sizeof(uint32_t);
489 tmpstr->written = ntohl(var);
490
491 Malloc(tmpstr->data, char, tmpstr->length);
492 __n_assert(tmpstr->data, return NULL);
493
494 memcpy(tmpstr->data, charptr, tmpstr->written);
495 charptr += tmpstr->written;
496
497 add_nstrptr_to_msg(generated_msg, tmpstr);
498 }
499 }
500 return generated_msg;
501} /* make_msg_from_str(...) */
