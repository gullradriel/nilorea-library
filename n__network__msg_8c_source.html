36double double_swap(double value) {
37 double swaped = 0;
38 unsigned char* src = (unsigned char*)&value;
39 unsigned char* dst = (unsigned char*)&swaped;
40
41 /* swap bytes */
42 dst[0] = src[7];
43 dst[1] = src[6];
44 dst[2] = src[5];
45 dst[3] = src[4];
46 dst[4] = src[3];
47 dst[5] = src[2];
48 dst[6] = src[1];
49 dst[7] = src[0];
50
51 return swaped;
52} /* double_swap() */

59double htond(double value) {
60 double ret = value;
61 /* if byte order is little like x86, PC, ... we swap to network notation */
62#if (BYTEORDER_ENDIAN == BYTEORDER_LITTLE_ENDIAN)
63 ret = double_swap(value);
64#endif
65 return ret;

329N_STR* make_str_from_msg(NETW_MSG* msg) {
330 size_t str_length = 0;
331 LIST_NODE* node = NULL;
332
333 N_STR *strptr = NULL,
334 *generated_str = NULL;
335
336 char* charptr = NULL;
337
338 __n_assert(msg, return NULL);
339
340 /* the first thing to do is to computer the resulting string size
341 the first 12 octets are for the number of ints , floats, strings in the message
342 */
343 str_length = 3 * sizeof(int32_t) + (msg->tabint->nb_items * sizeof(int32_t)) + (msg->tabflt->nb_items * sizeof(double));
344
345 /* let's count the eventual string blocks */
346 node = msg->tabstr->start;
347 while (node) {
348 strptr = (N_STR*)node->ptr;
349 str_length += sizeof(int32_t) + sizeof(int32_t) + strptr->written;
350 node = node->next;
351 }
352
353 /* preparing the new string */
354 Malloc(generated_str, N_STR, 1);
355 __n_assert(generated_str, return NULL);
356 Malloc(generated_str->data, char, str_length + 1);
357 __n_assert(generated_str->data, Free(generated_str); return NULL);
358
359 generated_str->length = str_length + 1;
360 generated_str->written = str_length;
361
362 /* copying header */
363 charptr = generated_str->data;
364
365 /* testing limits */
366 if (msg->tabint->nb_items >= UINT32_MAX) {
367 n_log(LOG_ERR, "too much items (>=UINT32_MAX) in int list of message %p", msg);
368 free_nstr(&generated_str);
369 return NULL;
370 }
371 if (msg->tabflt->nb_items >= UINT32_MAX) {
372 n_log(LOG_ERR, "too much items (>=UINT32_MAX) in float list of message %p", msg);
373 free_nstr(&generated_str);
374 return NULL;
375 }
376 if (msg->tabstr->nb_items >= UINT32_MAX) {
377 n_log(LOG_ERR, "too much items (>=UINT32_MAX) in string list of message %p", msg);
378 free_nstr(&generated_str);
379 return NULL;
380 }
381
382 /* byte order */
383 uint32_t nb_int_items = htonl((uint32_t)msg->tabint->nb_items);
384 uint32_t nb_float_items = htonl((uint32_t)msg->tabflt->nb_items);
385 uint32_t nb_str_items = htonl((uint32_t)msg->tabstr->nb_items);
386
387 /* number of ints inside the message */
388 memcpy(charptr, &nb_int_items, sizeof(uint32_t));
389 charptr += sizeof(uint32_t);
390 /* number of doubles inside the message */
391 memcpy(charptr, &nb_float_items, sizeof(uint32_t));
392 charptr += sizeof(uint32_t);
393 /* number of string inside the message */
394 memcpy(charptr, &nb_str_items, sizeof(uint32_t));
395 charptr += sizeof(uint32_t);
396
397 /* if there are numbers, copy all them ! */
398 node = msg->tabint->start;
399 while (node) {
400 uint32_t* nbptr = (uint32_t*)node->ptr;
401 uint32_t val = htonl(nbptr[0]);
402 memcpy(charptr, &val, sizeof(uint32_t));
403 charptr += sizeof(uint32_t);
404 node = node->next;
405 }
406
407 /* if there are numbers, copy all them ! */
408 node = msg->tabflt->start;
409 while (node) {
410 double* nbptr = (double*)node->ptr;
411 double val = htond(nbptr[0]);
412 memcpy(charptr, &val, sizeof(double));
413 charptr += sizeof(double);
414 node = node->next;
415 }
416
417 /* if there are strings, copy all them ! */
418 if (msg->tabstr->nb_items > 0) {
419 node = msg->tabstr->start;
420 while (node) {
421 strptr = (N_STR*)node->ptr;
422 if (strptr->length >= UINT32_MAX) {
423 n_log(LOG_ERR, "string too long (>=UINT32_MAX) in string list of message %p", msg);
424 free_nstr(&generated_str);
425 return NULL;
426 }
427 uint32_t var = htonl((uint32_t)strptr->length);
428 memcpy(charptr, &var, sizeof(uint32_t));
429 charptr += sizeof(uint32_t);
430
431 if (strptr->written >= UINT32_MAX) {
432 n_log(LOG_ERR, "string too long (>=UINT32_MAX) in string list of message %p", msg);
433 free_nstr(&generated_str);
434 return NULL;
435 }
436 var = htonl((uint32_t)strptr->written);
437 memcpy(charptr, &var, sizeof(uint32_t));
438 charptr += sizeof(uint32_t);
439
440 memcpy(charptr, strptr->data, strptr->written);
441 charptr += strptr->written;
442 node = node->next;
443 }
444 }
445
446 return generated_str;
447} /* make_str_from_msg(...) */

454NETW_MSG* make_msg_from_str(N_STR* str) {
455 NETW_MSG* generated_msg = NULL;
456 N_STR* tmpstr = NULL;
457
458 char* charptr = NULL;
459
460 uint32_t it,
461 nb_int = 0,
462 nb_flt = 0,
463 nb_str = 0;
464 int32_t tmpnb = 0;
465
466 double tmpflt = 0;
467
468 __n_assert(str, return NULL);
469
470 charptr = str->data;
471
472 /* need at least 3 uint32_t for the header */
473 if (str->written < 3 * sizeof(uint32_t)) {
474 n_log(LOG_ERR, "message too short: %zu bytes, need at least %zu", str->written, 3 * sizeof(uint32_t));
475 return NULL;
476 }
477
478 create_msg(&generated_msg);
479
480 __n_assert(generated_msg, return NULL);
481
482 memcpy(&nb_int, charptr, sizeof(uint32_t));
483 charptr += sizeof(uint32_t);
484 nb_int = ntohl(nb_int);
485
486 memcpy(&nb_flt, charptr, sizeof(uint32_t));
487 charptr += sizeof(uint32_t);
488 nb_flt = ntohl(nb_flt);
489
490 memcpy(&nb_str, charptr, sizeof(uint32_t));
491 charptr += sizeof(uint32_t);
492 nb_str = ntohl(nb_str);
493
494 /* validate that the message is large enough for the claimed int and float counts.
495 * Use division-based checks to avoid size_t overflow on 32-bit platforms. */
496 size_t remaining = str->written - 3 * sizeof(uint32_t);
497
498 if (nb_int > remaining / sizeof(uint32_t)) {
499 n_log(LOG_ERR, "message claims %u ints but only %zu bytes remain", nb_int, remaining);
500 delete_msg(&generated_msg);
501 return NULL;
502 }
503 remaining -= (size_t)nb_int * sizeof(uint32_t);
504
505 if (nb_flt > remaining / sizeof(double)) {
506 n_log(LOG_ERR, "message claims %u floats but only %zu bytes remain", nb_flt, remaining);
507 delete_msg(&generated_msg);
508 return NULL;
509 }
510 remaining -= (size_t)nb_flt * sizeof(double);
511
512 if (nb_int > 0) {
513 for (it = 0; it < nb_int; it++) {
514 memcpy(&tmpnb, charptr, sizeof(uint32_t));
515 tmpnb = (int32_t)ntohl((uint32_t)tmpnb);
516 charptr += sizeof(uint32_t);
517 if (add_int_to_msg(generated_msg, tmpnb) == FALSE) {
518 n_log(LOG_ERR, "failed to add int %u/%u to message", it, nb_int);
519 delete_msg(&generated_msg);
520 return NULL;
521 }
522 }
523 }
524
525 if (nb_flt > 0) {
526 for (it = 0; it < nb_flt; it++) {
527 memcpy(&tmpflt, charptr, sizeof(double));
528 tmpflt = ntohd(tmpflt);
529 charptr += sizeof(double);
530 if (add_nb_to_msg(generated_msg, tmpflt) == FALSE) {
531 n_log(LOG_ERR, "failed to add float %u/%u to message", it, nb_flt);
532 delete_msg(&generated_msg);
533 return NULL;
534 }
535 }
536 }
537
538 if (nb_str > 0) {
539 /* sanity check: each string needs at least 2 uint32_t for headers */
540 if ((size_t)nb_str > remaining / (2 * sizeof(uint32_t))) {
541 n_log(LOG_ERR, "message claims %u strings but only %zu bytes remain (need at least %zu for headers)",
542 nb_str, remaining, (size_t)nb_str * 2 * sizeof(uint32_t));
543 delete_msg(&generated_msg);
544 return NULL;
545 }
546
547 for (it = 0; it < nb_str; it++) {
548 /* check space for length + written headers */
549 if (remaining < 2 * sizeof(uint32_t)) {
550 n_log(LOG_ERR, "string %u/%u: not enough data for string headers (%zu remaining)", it, nb_str, remaining);
551 delete_msg(&generated_msg);
552 return NULL;
553 }
554
555 Malloc(tmpstr, N_STR, 1);
556 __n_assert(tmpstr, { delete_msg(&generated_msg); return NULL; });
557
558 uint32_t var = 0;
559 memcpy(&var, charptr, sizeof(uint32_t));
560 charptr += sizeof(uint32_t);
561 tmpstr->length = ntohl(var);
562
563 memcpy(&var, charptr, sizeof(uint32_t));
564 charptr += sizeof(uint32_t);
565 tmpstr->written = ntohl(var);
566 remaining -= 2 * sizeof(uint32_t);
567
568 /* validate written < length (N_STR requires room for trailing '\0') */
569 if (tmpstr->length == 0 || tmpstr->written >= tmpstr->length) {
570 n_log(LOG_ERR, "string %u/%u: written (%zu) >= length (%zu), violates N_STR invariant", it, nb_str, tmpstr->written, tmpstr->length);
571 Free(tmpstr);
572 delete_msg(&generated_msg);
573 return NULL;
574 }
575
576 /* validate remaining buffer has enough for written bytes */
577 if (tmpstr->written > remaining) {
578 n_log(LOG_ERR, "string %u/%u: written size %zu exceeds remaining %zu bytes", it, nb_str, tmpstr->written, remaining);
579 Free(tmpstr);
580 delete_msg(&generated_msg);
581 return NULL;
582 }
583
584 /* sanity cap on allocation size to prevent DoS via huge allocation */
585 if (tmpstr->length > NETW_MSG_MAX_STRING_LENGTH) {
586 n_log(LOG_ERR, "string %u/%u: length %zu exceeds NETW_MSG_MAX_STRING_LENGTH (%zu)", it, nb_str, tmpstr->length, (size_t)NETW_MSG_MAX_STRING_LENGTH);
587 Free(tmpstr);
588 delete_msg(&generated_msg);
589 return NULL;
590 }
591
592 /* allocate based on written+1 (not untrusted length) to prevent memory-DoS
593 * where attacker sets length near cap but written tiny */
594 Malloc(tmpstr->data, char, tmpstr->written + 1);
595 __n_assert(tmpstr->data, { Free(tmpstr); delete_msg(&generated_msg); return NULL; });
596 tmpstr->length = tmpstr->written + 1;
597
598 memcpy(tmpstr->data, charptr, tmpstr->written);
599 tmpstr->data[tmpstr->written] = '\0';
600 charptr += tmpstr->written;
601 remaining -= tmpstr->written;
602
603 if (add_nstrptr_to_msg(generated_msg, tmpstr) == FALSE) {
604 Free(tmpstr->data);
605 Free(tmpstr);
606 delete_msg(&generated_msg);
607 return NULL;
608 }
609 }
610 }
611 return generated_msg;
612} /* make_msg_from_str(...) */

841int netw_msg_get_type(N_STR* msg) {

843 int32_t tmpnb = 0;
844 char* charptr = NULL;
845
846 __n_assert(msg, return -1);
847 __n_assert(msg->data, return -1);
848
849 /* need at least 3*uint32 header + 1*uint32 for the type int */
850 if (msg->written < 4 * sizeof(uint32_t)) return -1;
851
852 charptr = msg->data;
853
854 /* read nb_int from the header to verify the int section is present */
855 memcpy(&nb_int, charptr, sizeof(uint32_t));
856 nb_int = ntohl(nb_int);
857
858 if (nb_int < 1) {
859 n_log(LOG_ERR, "netw_msg_get_type: message has no int fields (nb_int=%u), cannot read type", nb_int);
860 return -1;
861 }
862
863 /* bypass nb_int, nb_flt, nb_str header fields to reach the first int (type) */
864 charptr += 3 * sizeof(uint32_t);
865 memcpy(&tmpnb, charptr, sizeof(uint32_t));
866 tmpnb = (int32_t)ntohl((uint32_t)tmpnb);
867
868 return tmpnb;
869} /* netw_msg_get_type(...) */
