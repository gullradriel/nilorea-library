84int _ht_is_leaf_node_trie(HASH_TABLE* table, const char* key) {
85 __n_assert(table, return -1);
86
87 /* checks if the prefix match of key and root is a leaf node */
88 HASH_NODE* node = table->root;
89 for (size_t it = 0; key[it]; it++) {
90 size_t index = (size_t)key[it] - table->alphabet_offset;
91 if (index >= table->alphabet_length) {
92 n_log(LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key);
93 index = 0;
94 }
95 if (node->children[index]) {
96 node = node->children[index];
97 } else {
98 return 0;
99 }
100 }
101 return node->is_leaf;
102} /* _ht_is_leaf_node_trie(...) */

108void _ht_node_destroy(void* node) {
109 HASH_NODE* node_ptr = (HASH_NODE*)node;
110 __n_assert(node_ptr, return);
111 if (node_ptr->type == HASH_STRING) {
112 Free(node_ptr->data.string);
113 }
114 if (node_ptr->type == HASH_PTR) {
115 if (node_ptr->destroy_func && node_ptr->data.ptr) {
116 node_ptr->destroy_func(node_ptr->data.ptr);
117 }
118 /* No free by default. must be passed as a destroy_func
119 else
120 {
121 Free( node_ptr -> data . ptr );
122 }
123 */
124 }
125 FreeNoLog(node_ptr->key);
126 if (node_ptr->alphabet_length > 0) {
127 for (size_t it = 0; it < node_ptr->alphabet_length; it++) {
128 if (node_ptr->children[it]) {
129 _ht_node_destroy(node_ptr->children[it]);
130 }
131 }
132 Free(node_ptr->children);
133 }
134 Free(node_ptr);
135} /* _ht_node_destroy */

143size_t _ht_check_trie_divergence(HASH_TABLE* table, const char* key) {
144 __n_assert(table, return SIZE_MAX);
145 __n_assert(key, return SIZE_MAX);
146 __n_assert(table->root, return SIZE_MAX);
147
148 HASH_NODE* node = table->root;
149
150 size_t last_index = 0;
151 /* `node` is non-null on entry (table->root checked above) and only ever
152 * gets reassigned from `node->children[index]` after that slot was
153 * verified non-null. We still gate the loop on `node` so the static
154 * analyzer narrows `node` to non-null at every dereference inside the
155 * body, older clang static analyzers (the one in Ubuntu's clang-tools
156 * package, used by CI) don't track that constraint through the
157 * children[] reassignment, even though it holds. */
158 for (size_t it = 0; node && key[it] != '\0'; it++) {
159 size_t index = (size_t)key[it] - table->alphabet_offset;
160 if (index >= table->alphabet_length) {
161 n_log(LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key);
162 index = 0;
163 }
164 HASH_NODE* child = node->children[index];
165 if (child) {
166 /* child check */
167 for (size_t it2 = 0; it2 < table->alphabet_length; it2++) {
168 if (it2 != (unsigned)index && node->children[it2]) {
169 /* child found, update branch index */
170 last_index = it + 1;
171 break;
172 }
173 }
174 /* Go to the next child in the sequence */
175 node = child;
176 }
177 }
178 return last_index;
179} /* _ht_check_trie_divergence(...) */

187char* _ht_find_longest_prefix_trie(HASH_TABLE* table, const char* key) {
188 __n_assert(table, return NULL);
189 __n_assert(key, return NULL);
190
191 size_t len = strlen(key);
192
193 /* start with full key and backtrack to search for divergences */
194 char* longest_prefix = NULL;
195 Malloc(longest_prefix, char, len + 1);
196 __n_assert(longest_prefix, return NULL);
197 memcpy(longest_prefix, key, len);
198 /* memcpy does not add a null terminator; set it explicitly */
199 longest_prefix[len] = '\0';
200
201 /* check branching from the root */
202 size_t branch_index = _ht_check_trie_divergence(table, longest_prefix);
203 if (branch_index > 0 && branch_index != SIZE_MAX) {
204 /* truncate to the branch point */
205 longest_prefix[branch_index - 1] = '\0';
206 /* shrink the allocation; Realloc preserves the original pointer on
207 failure so the caller still gets a valid terminated string */
208 Realloc(longest_prefix, char, branch_index + 1);
209 }
210 return longest_prefix;
211} /* _ht_find_longest_prefix_trie(...) */

219int _ht_remove_trie(HASH_TABLE* table, const char* key) {
220 __n_assert(table, return FALSE);
221 __n_assert(table->root, return FALSE);
222 __n_assert(key, return FALSE);
223
224 /* stop if matching node not a leaf node */
225 if (!_ht_is_leaf_node_trie(table, key)) {
226 return FALSE;
227 }
228
229 HASH_NODE* node = table->root;
230 /* find the longest prefix string that is not the current key */
231 char* longest_prefix = _ht_find_longest_prefix_trie(table, key);
232 if (!longest_prefix) {
233 n_log(LOG_ERR, "couldn't find longest prefix for key %s", key);
234 return FALSE;
235 }
236 if (longest_prefix[0] == '\0') {
237 Free(longest_prefix);
238 return FALSE;
239 }
240 /* keep track of position in the tree */
241 size_t it = 0;
242 for (it = 0; longest_prefix[it] != '\0'; it++) {
243 size_t index = (size_t)longest_prefix[it] - table->alphabet_offset;
244 if (index >= table->alphabet_length) {
245 n_log(LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key);
246 index = 0;
247 }
248 if (node->children[index] != NULL) {
249 /* common prefix, keep moving */
250 node = node->children[index];
251 } else {
252 /* not found */
253 Free(longest_prefix);
254 return FALSE;
255 }
256 }
257 /* deepest common node between the two strings */
258 /* deleting the sequence corresponding to key */
259 for (; key[it] != '\0'; it++) {
260 size_t index = (size_t)key[it] - table->alphabet_offset;
261 if (index >= table->alphabet_length) {
262 n_log(LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key);
263 index = 0;
264 }
265 if (node->children[index]) {
266 /* delete the remaining sequence */
267 HASH_NODE* node_to_kill = node->children[index];
268 node->children[index] = NULL;
269 _ht_node_destroy(node_to_kill);
270 }
271 }
272 Free(longest_prefix);
273
274 table->nb_keys--;
275
276 return TRUE;
277} /* _ht_remove_trie(...) */

302 __n_assert(node->children[index], return FALSE);
303 } else {
304 /* nothing to do since node is existing */
305 }

320 _ht_remove_trie(table, key);

322 }
323 /* Put the value */
324 node->data.ival = value;
325 node->type = HASH_INT;

862/* Classic hash table */
864#define ROTL64(x, r) (((x) << (r)) | ((x) >> (64 - (r))))
866#define ROTL32(x, r) (((x) << (r)) | ((x) >> (32 - (r))))
868#define BIG_CONSTANT(x) (x##LLU)
869
870/* NO-OP for little-endian platforms */
871#if defined(__BYTE_ORDER__) && defined(__ORDER_LITTLE_ENDIAN__)
872#if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
873#define BYTESWAP32(x) (x)
874#define BYTESWAP64(x) (x)
875#endif
876/* if __BYTE_ORDER__ is not predefined (like FreeBSD), use arch */
877#elif defined(__i386) || defined(__x86_64) || defined(__alpha) || defined(__vax)
878
879#define BYTESWAP32(x) (x)
880#define BYTESWAP64(x) (x)
881/* use __builtin_bswap32 if available */
882#elif defined(__GNUC__) || defined(__clang__)
883#ifdef __has_builtin
884#if __has_builtin(__builtin_bswap32)
885#define BYTESWAP32(x) __builtin_bswap32(x)
886#endif // __has_builtin(__builtin_bswap32)
887#if __has_builtin(__builtin_bswap64)
888#define BYTESWAP64(x) __builtin_bswap64(x)
889#endif // __has_builtin(__builtin_bswap64)
890#endif // __has_builtin
891#endif // defined(__GNUC__) || defined(__clang__)
892/* last resort (big-endian w/o __builtin_bswap) */
893#ifndef BYTESWAP32
895#define BYTESWAP32(x) ((((x) & 0xFF) << 24) | (((x) >> 24) & 0xFF) | (((x) & 0x0000FF00) << 8) | (((x) & 0x00FF0000) >> 8))
896#endif
897#ifndef BYTESWAP64
899#define BYTESWAP64(x) \
900 (((uint64_t)(x) << 56) | \
901 (((uint64_t)(x) << 40) & 0X00FF000000000000ULL) | \
902 (((uint64_t)(x) << 24) & 0X0000FF0000000000ULL) | \
903 (((uint64_t)(x) << 8) & 0X000000FF00000000ULL) | \
904 (((uint64_t)(x) >> 8) & 0X00000000FF000000ULL) | \
905 (((uint64_t)(x) >> 24) & 0X0000000000FF0000ULL) | \
906 (((uint64_t)(x) >> 40) & 0X000000000000FF00ULL) | \
907 ((uint64_t)(x) >> 56))
908#endif

977void MurmurHash3_x86_32(const void* key, const size_t len, const uint32_t seed, void* out) {
978 const uint8_t* data = (const uint8_t*)key;
979 const size_t nblocks = len / 4;
980
981 uint32_t h1 = seed;
982 const uint32_t c1 = 0xcc9e2d51;
983 const uint32_t c2 = 0x1b873593;
984
985 const uint32_t* blocks = (const uint32_t*)data;
986
987 for (size_t i = 0; i < nblocks; i++) {
988 uint32_t k1 = getblock32(blocks, i);
989 k1 *= c1;
990 k1 = ROTL32(k1, 15);
991 k1 *= c2;
992
993 h1 ^= k1;
994 h1 = ROTL32(h1, 13);
995 h1 = h1 * 5 + 0xe6546b64;
996 }
997
998 const uint8_t* tail = data + nblocks * 4;
999 uint32_t k1 = 0;
1000 switch (len & 3) {
1001 case 3:
1002 k1 ^= (uint32_t)tail[2] << 16; /* fall through */
1003 FALL_THROUGH;
1004 case 2:
1005 k1 ^= (uint32_t)tail[1] << 8; /* fall through */
1006 FALL_THROUGH;
1007 case 1:
1008 k1 ^= (uint32_t)tail[0];
1009 k1 *= c1;
1010 k1 = ROTL32(k1, 15);
1011 k1 *= c2;
1012 h1 ^= k1;
1013 break;
1014 default:
1015 break;
1016 }
1017
1018 h1 ^= (uint32_t)len;
1019 h1 = fmix32(h1);
1020 *(uint32_t*)out = h1;
1021} /* MurmurHash3_x86_32 */

1126 FALL_THROUGH;

2542 * GCC nested-function expansion is rejected by clang's parser. */

2578 }
2579 }
2580 }
2581 } else {

2583 for (size_t it = 0; it < (*table)->size; it++) {
2584 if ((*table)->hash_table[it]) {
