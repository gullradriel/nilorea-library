14int init_particle_system( PARTICLE_SYSTEM **psys, int max, double x, double y, double z, int max_sprites )
15{

33int add_particle( PARTICLE_SYSTEM *psys, int spr, int mode, int lifetime, int size, ALLEGRO_COLOR color, PHYSICS object )
34{
35 int it = 0 ;
36
37 PARTICLE *new_p = NULL ;
38
39
40 if( psys -> list -> nb_items == psys -> list -> nb_max_items )
41 return FALSE ;
42
43 for( it = 0 ; it < 3 ; it ++ )
44 {
45 object . position[ it ] += psys -> source[ it ] ;
46 }
47
48 Malloc( new_p, PARTICLE, 1 );
49 __n_assert( new_p, return FALSE );
50
51 new_p -> spr_id = spr ;
52 new_p -> mode = mode ;
53 new_p -> lifetime = lifetime ;
54 new_p -> color = color ;
55 new_p -> size = size ;
56
57 memcpy( new_p -> object . position, object . position, sizeof( VECTOR3D ) );
58 memcpy( new_p -> object . speed, object . speed, sizeof( VECTOR3D ) );
59 memcpy( new_p -> object . acceleration, object . acceleration, sizeof( VECTOR3D ) );
60 memcpy( new_p -> object . angular_speed, object . angular_speed, sizeof( VECTOR3D ) );
61 memcpy( new_p -> object . angular_acceleration, object . angular_acceleration, sizeof( VECTOR3D ) );
62 memcpy( new_p -> object . orientation, object . orientation, sizeof( VECTOR3D ) );
63 new_p -> object . sz = object . sz ;
64 new_p -> object . type = object . type ;
65
66 return list_push( psys -> list, new_p, &free );
67}

88int manage_particle( PARTICLE_SYSTEM *psys)
89{
90 LIST_NODE *node = NULL ;
91 PARTICLE *ptr = NULL ;
92
93 node = psys -> list -> start ;
94
95 double delta = 0 ;
96
97 delta = get_usec( &psys -> timer ) ;
98
99 while( node )
100 {
101 ptr = (PARTICLE *)node -> ptr ;
102 if( ptr -> lifetime != -1 )
103 {
104 ptr -> lifetime -= delta/1000.0 ;
105 if( ptr -> lifetime == -1 )
106 ptr -> lifetime = 0 ;
107 }
108
109 if( ptr -> lifetime > 0 || ptr -> lifetime == -1 )
110 {
111 update_physics_position( &ptr -> object, delta );
112 node = node -> next ;
113 }
114 else
115 {
116 LIST_NODE *node_to_kill = node ;
117 node = node -> next ;
118 ptr = remove_list_node( psys -> list, node_to_kill, PARTICLE );
119 Free( ptr );
120 }
121 }
122
123 return TRUE;
124}

128int draw_particle( PARTICLE_SYSTEM *psys, double xpos, double ypos, int w, int h, double range )
129{
130 LIST_NODE *node = NULL ;
131 PARTICLE *ptr = NULL ;
132
133
134 node = psys -> list -> start ;
135
136 while( node )
137 {
138 double x = 0, y = 0 ;
139
140 ptr = (PARTICLE *)node -> ptr ;
141 x = ptr -> object . position[ 0 ] - xpos ;
142 y = ptr -> object . position[ 1 ] - ypos ;
143
144 if( ( x < -range ) || ( x > ( w + range ) ) || ( y< -range ) || ( y > ( h + range ) ) )
145 {
146 node = node -> next ;
147 continue ;
148 }
149
150 while( ptr -> object . orientation[ 2 ] > 255 )
151 ptr -> object . orientation[ 2 ] -= 256 ;
152
153 while( ptr -> object . orientation[ 2 ] < 0 )
154 ptr -> object . orientation[ 2 ] += 256 ;
155
156 if( ptr -> mode == SINUS_PART )
157 {
158 if( ptr -> object . speed[ 0 ] != 0 )
159 x = x + ptr -> object . speed[ 0 ] * sin( (ptr -> object . position[ 0 ]/ptr -> object . speed[ 0 ]) ) ;
160 else
161 x = x + ptr -> object . speed[ 0 ] * sin( ptr -> object . position[ 0 ] );
162
163 if( ptr -> object . speed[ 1 ] != 0 )
164 y = y + ptr -> object . speed[ 1 ] * cos( (ptr -> object . speed[ 1 ]/ptr -> object . speed[ 1 ]) ) ;
165 else
166 y = y + ptr -> object . speed[ 1 ] * sin( ptr -> object . position[ 1 ] ) ;
167
168 if( ptr -> spr_id >= 0 && ptr -> spr_id < psys -> max_sprites && psys -> sprites[ ptr -> spr_id ] )
169 {
170 int spr_w = al_get_bitmap_width( psys -> sprites[ ptr -> spr_id ] );
171 int spr_h = al_get_bitmap_height( psys -> sprites[ ptr -> spr_id ] );
172
173 al_draw_rotated_bitmap( psys -> sprites[ ptr -> spr_id ], spr_w/2, spr_h/2, x - spr_w / 2, y - spr_h /2, al_ftofix( ptr -> object . orientation[ 2 ]), 0 );
174 }
175 else
176 al_draw_circle( x, y, 2, ptr -> color, 1 );
177 }
178
179 if( ptr -> mode & NORMAL_PART )
180 {
181 if( ptr -> spr_id >= 0 && ptr -> spr_id < psys -> max_sprites && psys -> sprites[ ptr -> spr_id ] )
182 {
183 int w = al_get_bitmap_width( psys -> sprites[ ptr -> spr_id ] );
184 int h = al_get_bitmap_height( psys -> sprites[ ptr -> spr_id ] );
185
186 al_draw_rotated_bitmap( psys -> sprites[ ptr -> spr_id ], w/2, h/2, x - w / 2, y - h /2, al_ftofix( ptr -> object . orientation[ 2 ]), 0 );
187 }
188 }
189 else if( ptr -> mode & PIXEL_PART )
190 {
191 al_draw_filled_rectangle( x - ptr -> size, y - ptr -> size, x + ptr -> size, y + ptr -> size, ptr -> color );
192 }
193 else
194 al_draw_circle( x, y, ptr -> size, ptr -> color, 1 );
195 node = node -> next ;
196 }
197
198 return TRUE;
199}
