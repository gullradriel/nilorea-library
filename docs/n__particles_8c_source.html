200int draw_particle( PARTICLE_SYSTEM *psys, double xpos, double ypos, int w, int h, double range )
201{
202 __n_assert( psys, return FALSE );
203
204 LIST_NODE *node = NULL ;
205 PARTICLE *ptr = NULL ;
206
207 node = psys -> list -> start ;
208
209 while( node )
210 {
211 double x = 0, y = 0 ;
212
213 ptr = (PARTICLE *)node -> ptr ;
214 x = ptr -> object . position[ 0 ] - xpos ;
215 y = ptr -> object . position[ 1 ] - ypos ;
216
217 if( ( x < -range ) || ( x > ( w + range ) ) || ( y< -range ) || ( y > ( h + range ) ) )
218 {
219 node = node -> next ;
220 continue ;
221 }
222
223 for( int it = 0 ; it < 3 ; it ++ )
224 {
225 while( ptr -> object . orientation[ it ] < 0.0 )
226 ptr -> object . orientation[ it ] += 256.0 ;
227
228 if( ptr -> object . orientation[ it ] >= 256.0 )
229 ptr -> object . orientation[ it ] = fmod( ptr -> object . orientation[ it ], 256.0 );
230 }
231
232 if( ptr -> mode == SINUS_PART )
233 {
234 if( ptr -> object . speed[ 0 ] != 0 )
235 x = x + ptr -> object . speed[ 0 ] * sin( (ptr -> object . position[ 0 ]/ptr -> object . speed[ 0 ]) ) ;
236 else
237 x = x + ptr -> object . speed[ 0 ] * sin( ptr -> object . position[ 0 ] );
238
239 if( ptr -> object . speed[ 1 ] != 0 )
240 y = y + ptr -> object . speed[ 1 ] * cos( (ptr -> object . speed[ 1 ]/ptr -> object . speed[ 1 ]) ) ;
241 else
242 y = y + ptr -> object . speed[ 1 ] * sin( ptr -> object . position[ 1 ] ) ;
243
244 if( ptr -> spr_id >= 0 && ptr -> spr_id < psys -> max_sprites && psys -> sprites[ ptr -> spr_id ] )
245 {
246 int spr_w = al_get_bitmap_width( psys -> sprites[ ptr -> spr_id ] );
247 int spr_h = al_get_bitmap_height( psys -> sprites[ ptr -> spr_id ] );
248
249 al_draw_rotated_bitmap( psys -> sprites[ ptr -> spr_id ], spr_w/2, spr_h/2, x - spr_w / 2, y - spr_h /2, al_ftofix( ptr -> object . orientation[ 2 ]), 0 );
250 }
251 else
252 al_draw_circle( x, y, ptr -> size, ptr -> color, 1 );
253 }
254
255 if( ptr -> mode & NORMAL_PART )
256 {
257 if( ptr -> spr_id >= 0 && ptr -> spr_id < psys -> max_sprites && psys -> sprites[ ptr -> spr_id ] )
258 {
259 int w = al_get_bitmap_width( psys -> sprites[ ptr -> spr_id ] );
260 int h = al_get_bitmap_height( psys -> sprites[ ptr -> spr_id ] );
261
262 al_draw_rotated_bitmap( psys -> sprites[ ptr -> spr_id ], w/2, h/2, x - w / 2, y - h /2, al_ftofix( ptr -> object . orientation[ 2 ]), 0 );
263 }
264 else
265 al_draw_circle( x, y, ptr -> size, ptr -> color, 1 );
266 }
267 else if( ptr -> mode & PIXEL_PART )
268 {
269 al_draw_filled_rectangle( x - ptr -> size, y - ptr -> size, x + ptr -> size, y + ptr -> size, ptr -> color );
270 }
271 else
272 al_draw_circle( x, y, ptr -> size, ptr -> color, 1 );
273 node = node -> next ;
274 }
275
276 return TRUE;
277} /* draw_particle() */
