154/*!\fn int db_mem_dump_leaked( char *file )
155 *\brief Dump the current debug memory table
156 *\param file The name of the logfile
157 *\return TRUE or FALSE
158 */
159int db_mem_dump_leaked( char *file )
160{
161 FILE *out = NULL ;
162 LIST_NODE *node = NULL ;
163 HASH_NODE *hash_entry = NULL ;
164 DEBUG_MEM_ITEM *item = NULL ;
165 size_t len = 0 ;
166
167 out = fopen( file, "w" );
168 if( !out )
169 {
170 n_log( LOG_ERR,"Unable to open %s", file );
171 return FALSE ;
172 }
173
174 fprintf( out, "%d malloc / %d free\n", __debug_mem_table . nb_malloc, __debug_mem_table . nb_free );
175
176 for( unsigned long int it = 0 ; it < __debug_mem_table . table -> size ; it ++ )
177 {
178 node = __debug_mem_table . table -> hash_table[ it ] -> start ;
179 while( node )
180 {
181 hash_entry= (HASH_NODE *)node -> ptr ;
182 item = hash_entry -> data . ptr ;
183
184 if( item )
185 {
186 fprintf( out, "%p size %zu file %s func %s line %d dump:", item -> ptr, item -> size, item -> file, item -> func, item -> line );
187 if( item -> size > DEBUG_MEM_DUMP_MAX_LEN )
188 len = DEBUG_MEM_DUMP_MAX_LEN ;
189 else
190 len = item -> size ;
191
192 fwrite( item -> ptr, len, 1, out );
193 fprintf( out, "\n" );
194 delete_db_mem_item( item );
195 item = NULL ;
196 }
197 else
198 {
199 n_log( LOG_ERR,"Unable to load item from node -> ptr" );
200 }
201 node = node -> next ;
202 }
203 }
204
205 fclose( out );
206 n_log( LOG_DEBUG, "Memory dump finished" );
207 return TRUE ;
208} /*db_mem_dump_leaked() */
