25/*!\fn init_nodup_log( int max )
26 *\brief initialize the no duplicate logging system
27 *\param max the max size of the internal hash table. Leave it to zero or
28 negative to keep the internal value in use ( 1024 )
29 *\return TRUE or FALSE */
30int init_nodup_log( int max )
31{
32 if( _n_nodup_table != NULL )
33 {
34 n_log( LOG_ERR, "Could not allocate the internal hash table because it's already done" );
35 return FALSE ;
36 }
37 if( max <= 0 )
38 max = 1024 ;
39
40 _n_nodup_table = new_ht( max );
41
42 if( _n_nodup_table == NULL )
43 {
44 n_log( LOG_ERR, "Could not allocate the internal hash with size %d", max );
45 return FALSE ;
46 }
47 else
48 {
49 n_log( LOG_DEBUG, "LOGGING: nodup system activated with a hash table of %d cells", max );
50 }
51
52 return TRUE ;
53} /* init_nodup_log() */

122/*!\fn check_n_log_dup( const char *log , const char *file , const char *func , int line )
123 * \brief check if a log was already done or not at the given line, func, file
124 * \param log the line of log
125 * \param file the file containing the log
126 * \param func the name of calling func
127 * \param line the line of the log
128 * \return 0 if not existing 1 if existing with the same value , 2 if key exist
129 and value is different
130 */
131int check_n_log_dup( const char *log, const char *file, const char *func, int line )
132{
133 HASH_NODE *node = NULL ;
134 char *key = NULL ;
135
136 /* check if the nopdup session is on, else do a normal n_log */
137 if( !_n_nodup_table )
138 {
139 return 3 ;
140 }
141
142
143 key = get_nodup_key( file, func, line );
144
145 __n_assert( key, return FALSE );
146
147 node = ht_get_node( _n_nodup_table, key );
148
149 Free( key );
150
151 if( node )
152 {
153 if( strcmp( log, node -> data . string ) == 0 )
154 {
155 return 1 ;
156 }
157 return 2 ;
158 }
159 return 0 ;
160
161} /* check_n_log_dup(...) */

206/*!\fn void _n_nodup_log( int LEVEL , const char *file , const char *func , int line , const char *format , ... )
207 *\brief Logging function. log( level , const char *format , ... ) is a macro around _log
208 *\param LEVEL Logging level
209 *\param file File containing the emmited log
210 *\param func Function emmiting the log
211 *\param line Line of the log
212 *\param format Format and string of the log, printf style
213 */
214void _n_nodup_log( int LEVEL, const char *file, const char *func, int line, const char *format, ... )
215{
216 HASH_NODE *node = NULL ;
217 va_list args ;
218
219 char *syslogbuffer = NULL ;
220
221 va_start (args, format);
222 vasprintf( &syslogbuffer, format, args );
223 va_end( args );
224
225 char *key = get_nodup_key( file, func, line );
226 int is_dup = check_n_log_dup( syslogbuffer, file, func, line );
227
228 switch( is_dup )
229 {
230 /* new log entry for file,func,line */
231 case 0 :
232 if( _n_nodup_table )
233 {
234 ht_put_string( _n_nodup_table, key, syslogbuffer );
235 }
236 _n_log( LEVEL, file, func, line, "%s", syslogbuffer );
237 break ;
238
239 /* exising and same log entry, do nothing (maybe latter we will add a timeout to repeat logging) */
240 case 1 :
241 break ;
242 /* existing but different entry. We have to refresh and log it one time*/
243 case 2 :
244 node = ht_get_node( _n_nodup_table, key );
245 if( node && node -> data . string )
246 {
247 Free( node -> data . string );
248 node -> data . string = syslogbuffer ;
249 }
250 _n_log( LEVEL, file, func, line, "%s", syslogbuffer );
251 break ;
252 /* no nodup session started, normal loggin */
253 case 3 :
254 default:
255 _n_log( LEVEL, file, func, line, "%s", syslogbuffer );
256 break ;
257 }
258
259 Free( syslogbuffer );
260 Free( key );
261
262} /* _n_nodup_log() */

327/*!\fn void dump_nodup_log( char *file )
328 *\brief Dump the duplicate error log hash table in a file
329 *\param file The path and filename to the dump file
330 *\return TRUE or FALSE
331 */
332int dump_nodup_log( char *file )
333{
334 FILE *out = NULL ;
335 HASH_NODE *hash_node = NULL ;
336 __n_assert( file, return FALSE );
337 __n_assert( _n_nodup_table, return FALSE );
338
339 char *tmpfile = NULL ;
340 n_log( LOG_DEBUG, "outfile:%s", file );
341 strprintf( tmpfile, "%s.tmp", file );
342
343 out = fopen( tmpfile, "wb" );
344 __n_assert( out, return FALSE );
345
346 if( _n_nodup_table )
347 {
348 for( unsigned long int it = 0 ; it < _n_nodup_table -> size ; it ++ )
349 {
350 list_foreach( list_node, _n_nodup_table -> hash_table[ it ] )
351 {
352 hash_node = (HASH_NODE *)list_node -> ptr ;
353 fprintf( out, "%s\n", hash_node -> data . string );
354 }
355 }
356 }
357 fclose( out );
358
359 char *cmd = NULL ;
360 strprintf( cmd, "mv -f %s %s ; chmod 644 %s", tmpfile, file, file );
361 n_log( LOG_DEBUG, "%s", cmd );
362 system( cmd );
363
364 Free( cmd );
365 Free( tmpfile );
366
367 return TRUE ;
368
369} /* dump_nodup_log() */
