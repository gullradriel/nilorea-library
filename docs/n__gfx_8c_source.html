115/*!\fn Force32BitBmpToAlpha( BITMAP *bmp )
116 *\brief Force if already not a 32bpp bmp to be the same with an alpha channel
117 *\param bmp the bitmap to check
118 *\return TRUE or FALSE
119 */
120int Force32BitBmpToAlpha( BITMAP *bmp )
121{
122 int x, y;
123 int r, g, b, alpha;
124 int col, mask;
125
126 __n_assert( bmp, return FALSE );
127
128 acquire_bitmap( bmp );
129 mask = bitmap_mask_color( bmp );
130
131 /* First check if an alpha channel already exists */
132 for ( y = 0; y < bmp->h; ++y )
133 {
134 for ( x = 0; x < bmp->w; ++x )
135 {
136 col = getpixel( bmp, x, y );
137 alpha = geta( col );
138
139 if ( alpha != 0 && col != mask )
140 {
141 release_bitmap( bmp );
142 return FALSE;
143 }
144 }
145 }
146
147 for ( y = 0; y < bmp->h; ++y )
148 {
149 for ( x = 0; x < bmp->w; ++x )
150 {
151 col = getpixel( bmp, x, y );
152 r = getr( col );
153 g = getg( col );
154 b = getb( col );
155 putpixel( bmp, x, y, makeacol( r, g, b, 255 ) );
156 }
157 }
158 release_bitmap( bmp );
159 return TRUE ;
160} /* Force32BitBmpToAlpha */
