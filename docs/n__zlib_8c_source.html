21/*!\fn int GetMaxCompressedLen( unsigned int nLenSrc )
22 *\brief Return the maximum compressed size
23 *\param nLenSrc
24 *\return The size in bytes
25 */
26int GetMaxCompressedLen( unsigned int nLenSrc )
27{
28 unsigned int n16kBlocks = (nLenSrc+16383) / 16384; // round up any fraction of a block
29 return ( nLenSrc + 6 + (n16kBlocks*5) );

34/*!\fn int CompressData( unsigned char *abSrc, unsigned int nLenSrc, unsigned char *abDst, unsigned int nLenDst )
35 *\brief Compress a string to another
36 *\param abSrc source string
37 *\param nLenSrc size of source string
38 *\param abDst destination string
39 *\param nLenDst destination length
40 *\return -1 or lenght of output
41 */
42int CompressData( unsigned char *abSrc, unsigned int nLenSrc, unsigned char *abDst, unsigned int nLenDst )
43{
44 __n_assert( abSrc, return -1 );
45 __n_assert( abDst, return -1 );
46
47 if( nLenSrc == 0 )
48 {
49 n_log( LOG_ERR, "nLenSrc (%d) <= 0", nLenSrc );
50 return -1 ;
51 }
52 if( nLenDst == 0 )
53 {
54 n_log( LOG_ERR, "nLenDst (%d) <= 0", nLenDst );
55 return -1 ;
56 }
57
58 z_stream zInfo ;
59 memset( &zInfo, 0, sizeof( zInfo ) );
60 zInfo.total_in= zInfo.avail_in = nLenSrc;
61 zInfo.total_out= zInfo.avail_out= nLenDst;
62 zInfo.next_in= (unsigned char *)abSrc;
63 zInfo.next_out= abDst;
64
65 int nErr, nRet= -1;
66 nErr= deflateInit( &zInfo, Z_DEFAULT_COMPRESSION ); // zlib function
67 switch( nErr )
68 {
69 case Z_OK:
70 // all is fine
71 break ;
72 default:
73 n_log( LOG_ERR, "%s on string %p size %d", zError( nErr ), abSrc, nLenSrc );
74 return -1 ;
75 }
76 nErr= deflate( &zInfo, Z_FINISH ); // zlib function
77 if ( nErr == Z_STREAM_END )
78 {
79 nRet= zInfo.total_out;
80 }
81 else
82 {
83 n_log( LOG_ERR, "%s on string %p size %d", zError( nErr ), abSrc, nLenSrc );
84 }
85 deflateEnd( &zInfo ); // zlib function
86 return( nRet );
87} /* CompressData */

148/*!\fn N_STR *zip_nstr( N_STR *src )
149 *\param src The source string
150 *\return The compressed string or NULL
151 */
152N_STR *zip_nstr( N_STR *src )
153{
154 __n_assert( src, return NULL );
155 __n_assert( src -> data, return NULL );
156
157 if( src -> length <= 0 )
158 {
159 n_log( LOG_ERR, "length of src (%d) <= 0", src -> length );
160 return NULL ;
161 }
162 if( src -> written <= 0 )
163 {
164 n_log( LOG_ERR, "written of src (%d) <= 0", src -> written );
165 return NULL ;
166 }
167
168 /* storage for original string size + zipped string + padding */
169 unsigned int zip_max_size = GetMaxCompressedLen( src -> length );
170
171 N_STR *zipped = new_nstr( 4 + zip_max_size );
172
173 __n_assert( zipped, return NULL );
174 __n_assert( zipped -> data, return NULL );
175
176 /* copying size */
177 int32_t src_length = htonl( src -> length );
178 memcpy( zipped -> data, &src_length, sizeof( int32_t ) );
179 char *dataptr = zipped -> data + 4 ;
180
181 int compressed_size = 0 ;
182 compressed_size = CompressData( (unsigned char *)src -> data, src -> written, (unsigned char *)dataptr, zip_max_size );
183 if( compressed_size == -1 )
184 {
185 free_nstr( &zipped );
186 n_log( LOG_ERR, "unable to zip string %p %d/%d bytes", src -> data, src -> written, src -> length );
187 return NULL ;
188 }
189 zipped -> written = 4 + compressed_size ;
190 n_log( LOG_DEBUG, "zip :%d original: %d", zipped -> written, src -> length );
191
192 return zipped ;
193} /* zip_nstr */

201N_STR *unzip_nstr( N_STR *src )
202{
203 __n_assert( src, return NULL );

205
206 if( src -> length <= 0 )

208 n_log( LOG_ERR, "length of src (%d) <= 0", src -> length );
209 return NULL ;
210 }
211 if( src -> written <= 0 )
212 {
213 n_log( LOG_ERR, "written of src (%d) <= 0", src -> written );
214 return NULL ;
215 }
216
217 int32_t original_size = 0 ;
218 memcpy( &original_size, src -> data, sizeof( int32_t ) );
219 original_size = ntohl( original_size );
220 if( original_size <= 0 )
221 {
222 n_log( LOG_ERR, "original size (%d) <= 0", original_size );
223 return NULL ;
224 }
225 /* storage for original string size + zipped string + padding */
226 N_STR *unzipped = new_nstr( original_size );
227 __n_assert( unzipped, return NULL );
228 __n_assert( unzipped -> data, return NULL );
229
230 /* copying size */
231 unzipped -> written = UncompressData( ((unsigned char *)src -> data) + 4, src -> written, (unsigned char *)unzipped -> data, original_size );
232 if( unzipped -> written == -1 )
233 {
234 n_log( LOG_ERR, "unable to unzip string %p %d/%d bytes", unzipped -> data, unzipped -> written, unzipped -> length );
235 free_nstr( &unzipped );
236 return NULL ;
237 }
238 n_log( LOG_DEBUG, "Size: zip: %d => unzip :%d original:%d", src -> written, unzipped -> written, original_size );
239
240 return unzipped ;
241} /* unzip_nstr */
