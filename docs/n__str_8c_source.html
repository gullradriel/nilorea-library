124char *trim_nocopy(char *s)
125{
126 __n_assert( s, return NULL );
127
128 if( strlen( s ) == 0 )
129 return s ;
130
131 char *start = s;
132
133 /* skip spaces at start */
134 while(*start && isspace(*start))
135 start++;
136
137 char *end = s + strlen( s ) - 1 ;
138 /* iterate over the rest remebering last non-whitespace */
139 while( *end && isspace( *end) && end > s )
140 {
141 end --;
142 }
143 end ++ ;
144 /* write the terminating zero after last non-whitespace */
145 *end = 0;
146
147 return start;
148} /* trim_nocopy */

289N_STR *file_to_nstr( char *filename )
290{
291 N_STR *tmpstr = NULL ;
292 struct stat filestat ;
293 FILE *in = NULL ;
294
295 __n_assert( filename, n_log( LOG_ERR, "Unable to make a string from NULL filename") ; return NULL );
296
297 if( stat( filename, &filestat ) != 0 )
298 {
299#ifdef __linux__
300 if( errno == EOVERFLOW )
301 {
302 n_log( LOG_ERR, "%s size is too big ,EOVERFLOW)", filename );
303 }
304 else
305 {
306#endif
307 n_log( LOG_ERR, "Couldn't stat %s. Errno: %s", filename, strerror( errno ) );
308#ifdef __linux__
309 }
310#endif
311 return NULL ;
312 }
313 if( ( filestat . st_size + 1 ) >= ( pow( 2, 32 ) - 1 ) )
314 {
315 n_log( LOG_ERR, "file size >= 2GB is not possible yet. %s is %lld oct", filename, filestat . st_size );
316 return NULL ;
317 }
318
319 n_log( LOG_DEBUG, "%s file size is: %lld", filename, filestat . st_size );
320
321 in = fopen( filename, "rb" );
322 __n_assert( in, n_log( LOG_ERR, "Unable to open %s", filename ); return NULL );
323
324 tmpstr = new_nstr( filestat . st_size + 1 );
325 __n_assert( tmpstr, fclose( in ); n_log( LOG_ERR, "Unable to get a new nstr of %ld octets", filestat . st_size + 1 ); return NULL );
326
327 tmpstr -> written = filestat . st_size ;
328
329 if( fread( tmpstr -> data, sizeof( char ), tmpstr -> written, in ) == 0 )
330 {
331 n_log( LOG_ERR, "Couldn't read %s, fread return 0", filename );
332 free_nstr( &tmpstr );
333 fclose( in );
334 return NULL ;
335 }
336 if( ferror( in ) )
337 {
338 n_log( LOG_ERR, "There were some errors when reading %s", filename );
339 free_nstr( &tmpstr );
340 fclose( in );
341 return NULL ;
342 }
343
344 fclose( in );
345
346 return tmpstr ;
347} /*file_to_nstr */

987char** split( const char* str, const char* delim, int empty )
988{
989 char** tab=NULL; /* result array */
990 char *ptr = NULL ; /* tmp pointer */
991 int sizeStr=0; /* string token size */
992 int sizeTab=0; /* array size */
993 const char* largestring = NULL; /* pointer to start of string */
994
995 __n_assert( str, return NULL );
996 __n_assert( delim, return NULL );
997
998 int sizeDelim=strlen(delim);
999 largestring = str;
1000
1001 while( ( ptr = strstr( largestring, delim ) ) != NULL )
1002 {
1003 sizeStr=ptr-largestring;
1004 if( empty == 1 || sizeStr != 0 )
1005 {
1006 sizeTab++;
1007 tab= (char**) realloc(tab,sizeof(char*)*sizeTab);
1008
1009 Malloc( tab[sizeTab -1], char, (int)(sizeof(char)*(sizeStr+1)) );
1010 __n_assert( tab[ sizeTab - 1 ], goto error );
1011
1012 memcpy(tab[sizeTab-1],largestring,sizeStr);
1013 tab[sizeTab-1][sizeStr]='\0';
1014 }
1015 ptr=ptr+sizeDelim;
1016 largestring=ptr;
1017 }
1018
1019 /* adding last part if any */
1020 if( strlen(largestring) !=0 )
1021 {
1022 sizeStr=strlen(largestring);
1023 sizeTab++;
1024

1380int wildmat(register const char *text, register const char *p)
1381{
1382 register int last = 0 ;
1383 register int matched = 0;
1384 register int reverse = 0;
1385
1386 for ( ; *p; text++, p++)
1387 {
1388 if (*text == '\0' && *p != '*')
1389 return WILDMAT_ABORT;
1390 switch (*p)
1391 {
1392 case '\\':
1393 /* Literal match with following character. */
1394 p++;
1395 /* FALLTHROUGH */
1396 default:
1397 if (*text != *p)
1398 return FALSE;
1399 continue;
1400 case '?':
1401 /* Match anything. */
1402 continue;
1403 case '*':
1404 while (*++p == '*')
1405 /* Consecutive stars act just like one. */
1406 continue;
1407 if (*p == '\0')
1408 /* Trailing star matches everything. */
1409 return TRUE;
1410 while (*text)
1411 if ((matched = wildmat(text++, p)) != FALSE)
1412 return matched;
1413 return WILDMAT_ABORT;
1414 case '[':
1415 reverse = p[1] == WILDMAT_NEGATE_CLASS ? TRUE : FALSE;
1416 if (reverse)
1417 /* Inverted character class. */
1418 p++;
1419 matched = FALSE;
1420 if (p[1] == ']' || p[1] == '-')
1421 if (*++p == *text)
1422 matched = TRUE;
1423 for (last = *p; *++p && *p != ']'; last = *p)
1424 /* This next line requires a good C compiler. */
1425 if (*p == '-' && p[1] != ']'
1426 ? *text <= *++p && *text >= last : *text == *p)
1427 matched = TRUE;
1428 if (matched == reverse)
1429 return FALSE;
1430 continue;
1431 }
1432 }
1433#ifdef WILDMAT_MATCH_TAR_PATTERN
1434 if (*text == '/')
1435 return TRUE;
1436#endif /* MATCH_TAR_ATTERN */
1437 return *text == '\0';
1438} /* wildmatch(...) */
