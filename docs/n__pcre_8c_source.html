15/*!\fn N_PCRE *npcre_new( char *str , int max_cap , int flags )
16 *\brief create a compiled regex
17 * From pcre doc, the flag bits are:
18 * PCRE_ANCHORED Force pattern anchoring
19 * PCRE_AUTO_CALLOUT Compile automatic callouts
20 * PCRE_BSR_ANYCRLF \\R matches only CR, LF, or CRLF
21 * PCRE_BSR_UNICODE \\R matches all Unicode line endings
22 * PCRE_CASELESS Do caseless matching
23 * PCRE_DOLLAR_ENDONLY $ not to match newline at end
24 * PCRE_DOTALL . matches anything including NL
25 * PCRE_DUPNAMES Allow duplicate names for subpatterns
26 * PCRE_EXTENDED Ignore white space and # comments
27 * PCRE_EXTRA PCRE extra features
28 * (not much use currently)
29 * PCRE_FIRSTLINE Force matching to be before newline
30 * PCRE_JAVASCRIPT_COMPAT JavaScript compatibility
31 * PCRE_MULTILINE ^ and $ match newlines within data
32 * PCRE_NEWLINE_ANY Recognize any Unicode newline sequence
33 * PCRE_NEWLINE_ANYCRLF Recognize CR, LF, and CRLF as newline
34 * sequences
35 * PCRE_NEWLINE_CR Set CR as the newline sequence
36 * PCRE_NEWLINE_CRLF Set CRLF as the newline sequence
37 * PCRE_NEWLINE_LF Set LF as the newline sequence
38 * PCRE_NO_AUTO_CAPTURE Disable numbered capturing paren-
39 * theses (named ones available)
40 * PCRE_NO_UTF16_CHECK Do not check the pattern for UTF-16
41 * validity (only relevant if
42 * PCRE_UTF16 is set)
43 * PCRE_NO_UTF32_CHECK Do not check the pattern for UTF-32
44 * validity (only relevant if
45 * PCRE_UTF32 is set)
46 * PCRE_NO_UTF8_CHECK Do not check the pattern for UTF-8
47 * validity (only relevant if
48 * PCRE_UTF8 is set)
49 * PCRE_UCP Use Unicode properties for \d, \w, etc.
50 * PCRE_UNGREEDY Invert greediness of quantifiers
51 * PCRE_UTF16 Run in pcre16_compile() UTF-16 mode
52 * PCRE_UTF32 Run in pcre32_compile() UTF-32 mode
53 * PCRE_UTF8 Run in pcre_compile() UTF-8 mode
54 *\param str The string containing the regexp
55 *\param max_cap Maximum number of captures. str The string containing the regexp
56 *\param flags pcre_compile flags
57 *\return a filled N_PCRE struct or NULL
58 */
59N_PCRE *npcre_new( char *str, int max_cap, int flags )
60{
61 N_PCRE *pcre = NULL ;
62 __n_assert( str, return NULL );
63
64 Malloc( pcre, N_PCRE, 1 );
65 __n_assert( pcre, return NULL );
66
67 pcre -> captured = 0 ;
68
69 if( max_cap <= 0 )
70 {
71 max_cap = 1 ;
72 }
73 pcre -> match_list = NULL ;
74
75 pcre -> regexp_str = strdup( str );
76 __n_assert( str, npcre_delete( &pcre ); return NULL );
77
78 pcre -> ovecount = max_cap ;
79 Malloc( pcre -> ovector, int, 3 * max_cap );
80 __n_assert( pcre -> ovector, npcre_delete( &pcre ); return NULL );
81
82 const char *error = NULL ;
83 int erroroffset = 0 ;
84
85 pcre -> regexp = pcre_compile( str, flags, &error, &erroroffset, NULL ) ;
86 if( !pcre -> regexp )
87 {
88 n_log( LOG_ERR, " pcre compilation of %s failed at offset %d : %s", str, erroroffset, error );
89 npcre_delete( &pcre );
90 return FALSE ;
91 }
92 /* no flags for study = no JIT compilation */
93 pcre -> extra = pcre_study( pcre -> regexp, 0, &error );
94
95 return pcre ;
96}/* npcre_new(...) */

102 *\brief Free a N_PCRE pointer

104 *
105 *\return TRUE or FALSE

108{

122 (*pcre) -> extra = NULL ;
123#else
124 pcre_free( (*pcre) -> extra );
125 (*pcre) -> extra = NULL ;
126#endif
127 if( (*pcre) -> captured > 0 )
128 {
129 pcre_free_substring_list( (*pcre) -> match_list );

131 FreeNoLog( (*pcre) -> regexp_str );
132 FreeNoLog( (*pcre) );
133 return TRUE ;
