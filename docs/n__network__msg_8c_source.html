13/*!\fn static double double_swap( double value )
14 *\brief Swap bytes endiannes for a double
15 *\param value The double to swap
16 *\return A swapped double
17 */
18double double_swap( double value )
19{
20 double swaped = 0 ;
21 unsigned char *src = (unsigned char *)&value ;
22 unsigned char *dst = (unsigned char *)&swaped ;
23
24 /* swap bytes */
25 dst[0] = src[7];
26 dst[1] = src[6];
27 dst[2] = src[5];
28 dst[3] = src[4];
29 dst[4] = src[3];
30 dst[5] = src[2];
31 dst[6] = src[1];
32 dst[7] = src[0];
33
34 return swaped ;
35} /* double_swap() */

40 *\brief If needed swap bytes for a double
41 *\param value The value to eventually swap
42 *\return The modified value

46 double ret = value ;
47 /* if byte order is little like x86, PC, ... we swap to network notation */

356/*!\fn make_str_from_msg( NETW_MSG *msg )
357 *\brief Make a single string of the message
358 *\param msg A NETW_MSG structure to convert into a N_STR one
359 *\return A pointer to the generated N_STR, do not forget to free it after use !
360 */
361N_STR *make_str_from_msg( NETW_MSG *msg )
362{
363 int str_length = 0 ;
364 LIST_NODE *node = NULL ;
365
366 N_STR *strptr = NULL,
367 *generated_str = NULL ;
368
369 char *charptr = NULL ;
370
371 __n_assert( msg, return NULL );
372
373 /* the first thing to do is to computer the resulting string size
374 the first 12 octets are for the number of ints , floats, strings in the message
375 */
376 str_length = 3 * sizeof( NSTRBYTE ) + ( msg -> tabint -> nb_items * sizeof( NSTRBYTE ) ) + ( msg -> tabflt -> nb_items * sizeof( double ) ) ;
377
378 /* let's count the eventual string blocks */
379 node = msg -> tabstr -> start ;
380 while( node )
381 {
382 strptr = (N_STR *)node -> ptr ;
383 str_length += sizeof( NSTRBYTE ) + sizeof( NSTRBYTE ) + strptr -> written ;
384 node = node -> next ;
385 }
386
387 /* preparing the new string */
388 Malloc( generated_str, N_STR, 1 );
389 __n_assert( generated_str, return NULL );
390 Malloc( generated_str -> data, char, str_length + 1 );
391 __n_assert( generated_str -> data, return NULL );
392
393 generated_str -> length = str_length + 1 ;
394 generated_str -> written = str_length ;
395
396
397 /* copying header */
398 charptr = generated_str -> data ;
399
400 /* byte order */
401 msg -> tabint -> nb_items = htonl( msg -> tabint -> nb_items );
402 msg -> tabflt -> nb_items = htonl( msg -> tabflt -> nb_items );
403 msg -> tabstr -> nb_items = htonl( msg -> tabstr -> nb_items );
404
405 /* number of ints inside the message */
406 memcpy ( charptr, &msg -> tabint -> nb_items, sizeof( NSTRBYTE ) );
407 charptr += sizeof( NSTRBYTE ) ;
408 /* number of doubles inside the message */
409 memcpy ( charptr, &msg -> tabflt -> nb_items, sizeof( NSTRBYTE ) );
410 charptr += sizeof( NSTRBYTE ) ;
411 /* number of string inside the message */
412 memcpy ( charptr, &msg -> tabstr -> nb_items, sizeof( NSTRBYTE ) );
413 charptr += sizeof( NSTRBYTE ) ;
414
415 /*reversing byte order to default host order */
416 msg -> tabint -> nb_items = ntohl( msg -> tabint -> nb_items );
417 msg -> tabflt -> nb_items = ntohl( msg -> tabflt -> nb_items );
418 msg -> tabstr -> nb_items = ntohl( msg -> tabstr -> nb_items );
419
420 /* if there are numbers, copy all them ! */
421 node = msg -> tabint -> start ;
422 while( node )
423 {
424 NSTRBYTE *nbptr = (NSTRBYTE *)node -> ptr ;
425 int val = htonl( nbptr[ 0 ] );
426 memcpy( charptr, &val, sizeof( NSTRBYTE ) );
427 charptr += sizeof( NSTRBYTE ) ;
428 node = node -> next ;
429 }
430
431 /* if there are numbers, copy all them ! */
432 node = msg -> tabflt -> start ;
433 while( node )
434 {
435 double *nbptr = (double *)node -> ptr ;
436 double val = htond( nbptr[ 0 ] );
437 memcpy( charptr, &val, sizeof( double ) );
438 charptr += sizeof( double ) ;
439 node = node -> next ;
440 }
441
442 /* if there are strings, copy all them ! */
443 if( msg -> tabstr -> nb_items > 0 )
444 {
445 node = msg -> tabstr -> start ;
446 while( node )
447 {
448 strptr = (N_STR *)node -> ptr ;
449
450 strptr -> length = htonl( strptr -> length );
451 memcpy( charptr, &strptr -> length, sizeof( NSTRBYTE ) );
452 charptr += sizeof( NSTRBYTE );
453 strptr -> length = ntohl( strptr -> length );
454
455 strptr -> written = htonl( strptr -> written );
456 memcpy( charptr, &strptr -> written, sizeof( NSTRBYTE ) );
457 charptr += sizeof( NSTRBYTE );
458 strptr -> written = ntohl( strptr -> written );
459
460 memcpy( charptr, strptr -> data, strptr -> written );
461 charptr += strptr -> written ;
462 node = node -> next ;
463 }
464
465 }
466
467 return generated_str;
468} /* make_str_from_msg(...) */

472/*!\fn make_msg_from_str( N_STR *str )
473 *\brief Make a single message of the string
474 *\param str A N_STR structure to convert into a NETW_MSG one
475 *\return A pointer to the generated NETW_MSG, do not forget to free it after use !
476 */
477NETW_MSG *make_msg_from_str( N_STR *str )
478{
479 NETW_MSG *generated_msg = NULL ;
480 N_STR *tmpstr = NULL;
481
482 char *charptr = NULL;
483
484 NSTRBYTE it,
485 nb_int = 0,
486 nb_flt = 0,
487 nb_str = 0,
488 tmpnb = 0 ;
489
490 double tmpflt = 0 ;
491
492 __n_assert( str, return NULL );
493
494 charptr = str -> data;
495
496 create_msg( &generated_msg );
497
498 __n_assert( generated_msg, return NULL );
499
500 memcpy( &nb_int, charptr, sizeof( NSTRBYTE ) );
501 charptr += sizeof( NSTRBYTE ) ;
502 nb_int = ntohl( nb_int );
503
504 memcpy( &nb_flt, charptr, sizeof( NSTRBYTE ) );
505 charptr += sizeof( NSTRBYTE ) ;
506 nb_flt = ntohl( nb_flt );
507
508 memcpy( &nb_str, charptr, sizeof( NSTRBYTE ) );
509 charptr += sizeof( NSTRBYTE ) ;
510 nb_str = ntohl( nb_str );
511
512 if( nb_int > 0 )
513 {
514 for( it = 0 ; it < nb_int ; it ++ )
515 {
516 memcpy( &tmpnb, charptr, sizeof( NSTRBYTE ) ) ;
517 tmpnb = ntohl( tmpnb );
518 charptr += sizeof( NSTRBYTE ) ;
519 add_int_to_msg( generated_msg, tmpnb );
520 }
521 }
522
523 if( nb_flt > 0 )
524 {
525 for( it = 0 ; it < nb_flt ; it ++ )
526 {
527 memcpy( &tmpflt, charptr, sizeof( double ) ) ;
528 tmpflt = ntohd( tmpflt );
529 charptr += sizeof( double ) ;
530 add_nb_to_msg( generated_msg, tmpflt );
531 }
532 }
533
534
535 if( nb_str > 0 )
536 {
537 for( it = 0 ; it < nb_str ; it ++ )
538 {
539
540 Malloc( tmpstr, N_STR, 1 );
541 __n_assert( tmpstr, return NULL );
542
543 memcpy( &tmpstr -> length, charptr, sizeof( NSTRBYTE ) );
544 charptr += sizeof( NSTRBYTE ) ;
545 memcpy( &tmpstr -> written, charptr, sizeof( NSTRBYTE ) );
546 charptr += sizeof( NSTRBYTE ) ;
547
548 tmpstr -> length = ntohl( tmpstr -> length );
549 tmpstr -> written = ntohl( tmpstr -> written );
550
551 Malloc( tmpstr -> data, char, tmpstr -> length );
552 __n_assert( tmpstr -> data, return NULL );
553
554 memcpy( tmpstr -> data, charptr, tmpstr -> written );
555 charptr += tmpstr -> written ;
556
557 add_nstrptr_to_msg( generated_msg, tmpstr );
558 }
559 }
560 return generated_msg ;
561} /* make_msg_from_str(...) */
