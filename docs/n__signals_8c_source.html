77/* Resolve symbol name and source location given the path to the executable
78 and an address */
79int addr2line(char const * const program_name, void const * const addr)
80{
81 char addr2line_cmd[4093] = "" ;
82
83 /* have addr2line map the address to the relent line in the code */
84#ifdef __APPLE__
85 /* apple does things differently... */
86 sprintf(addr2line_cmd,"atos -o %.256s ./%p", program_name, addr);
87#else
88#ifdef __windows__
89 sprintf(addr2line_cmd,"addr2line -f -p -e ./%s %p", program_name, addr);
90 LOGSIG( "%s", addr2line_cmd );
91 return 0 ;
92#else
93 sprintf(addr2line_cmd,"addr2line -f -p -e %.256s %p", program_name, addr);
94#endif
95#endif
96 N_STR *output = NULL ;
97 int ret = -1 ;
98 n_popen( addr2line_cmd, 1024, (void **)&output, &ret );
99 LOGSIG( "%s", output -> data );
100 free_nstr( &output );
101 return ret ;
102}

106void windows_print_stacktrace(CONTEXT* context)
107{
108 SymInitialize(GetCurrentProcess(), 0, true);
109
110 STACKFRAME frame = { 0 };
111
112 /* setup initial stack frame */
113#if (_WIN32_WINNT >= _WIN32_WINNT_WIN8)
114 /* Wrong values for W10 ? */
115 frame.AddrPC.Offset = context->Eip;
116 frame.AddrPC.Mode = AddrModeFlat;
117 frame.AddrStack.Offset = context->Esp;
118 frame.AddrStack.Mode = AddrModeFlat;
119 frame.AddrFrame.Offset = context->Ebp;
120 frame.AddrFrame.Mode = AddrModeFlat;
121#else
122 frame.AddrPC.Offset = context->Rip;
123 frame.AddrPC.Mode = AddrModeFlat;
124 frame.AddrStack.Offset = context->Rsp;
125 frame.AddrStack.Mode = AddrModeFlat;
126 frame.AddrFrame.Offset = context->Rbp;
127 frame.AddrFrame.Mode = AddrModeFlat;
128#endif
129
130
131 while (StackWalk(IMAGE_FILE_MACHINE_I386,
132 GetCurrentProcess(),
133 GetCurrentThread(),
134 &frame,
135 context,
136 0,
137 SymFunctionTableAccess,
138 SymGetModuleBase,
139 0 ) )
140 {
141 addr2line( icky_global_program_name, (void*)frame.AddrPC.Offset);
142 }
143
144 SymCleanup( GetCurrentProcess() );
145}

147LONG WINAPI windows_exception_handler(EXCEPTION_POINTERS * ExceptionInfo)
148{
149 switch(ExceptionInfo->ExceptionRecord->ExceptionCode)

151 case EXCEPTION_ACCESS_VIOLATION:

154 case EXCEPTION_ARRAY_BOUNDS_EXCEEDED:

156 break;
157 case EXCEPTION_BREAKPOINT:

159 break;
160 case EXCEPTION_DATATYPE_MISALIGNMENT:

162 break;
163 case EXCEPTION_FLT_DENORMAL_OPERAND:

165 break;
166 case EXCEPTION_FLT_DIVIDE_BY_ZERO:

168 break;
169 case EXCEPTION_FLT_INEXACT_RESULT:

172 case EXCEPTION_FLT_INVALID_OPERATION:

177 break;

179 LOGSIG("Error: EXCEPTION_FLT_STACK_CHECK" );

182 LOGSIG("Error: EXCEPTION_FLT_UNDERFLOW" );

192 break;

194 LOGSIG( "Error: EXCEPTION_INT_OVERFLOW" );
195 break;
196 case EXCEPTION_INVALID_DISPOSITION:
197 LOGSIG( "Error: EXCEPTION_INVALID_DISPOSITION" );

210 break;
211 default:

213 break;
214 }
215

217 where the error happened */
218 if (EXCEPTION_STACK_OVERFLOW != ExceptionInfo->ExceptionRecord->ExceptionCode)
219 {
220 windows_print_stacktrace(ExceptionInfo->ContextRecord);
221 }
222 else
223 {
224 addr2line(icky_global_program_name, (void*)ExceptionInfo->ContextRecord->Rip);
225 }
226 fflush(stdout);
227 fflush(stderr);
228 return EXCEPTION_EXECUTE_HANDLER;
229}

263void posix_signal_handler(int sig, siginfo_t *siginfo, void *context)
264{
265 (void)context;
266 switch(sig)
267 {
268 case SIGSEGV:
269 LOGSIG("Caught SIGSEGV: Segmentation Fault" );
270 break;
271 case SIGINT:
272 LOGSIG("Caught SIGINT: Interactive attention signal, (usually ctrl+c)" );
273 break;
274 case SIGFPE:
275 switch(siginfo->si_code)
276 {
277 case FPE_INTDIV:
278 LOGSIG("Caught SIGFPE: (integer divide by zero)" );
279 break;
280 case FPE_INTOVF:
281 LOGSIG("Caught SIGFPE: (integer overflow)" );
282 break;
283 case FPE_FLTDIV:
284 LOGSIG("Caught SIGFPE: (floating-point divide by zero)" );
285 break;
286 case FPE_FLTOVF:
287 LOGSIG("Caught SIGFPE: (floating-point overflow)" );
288 break;
289 case FPE_FLTUND:
290 LOGSIG("Caught SIGFPE: (floating-point underflow)" );
291 break;
292 case FPE_FLTRES:
293 LOGSIG("Caught SIGFPE: (floating-point inexact result)" );
294 break;
295 case FPE_FLTINV:
296 LOGSIG("Caught SIGFPE: (floating-point invalid operation)" );
297 break;
298 case FPE_FLTSUB:
299 LOGSIG("Caught SIGFPE: (subscript out of range)" );
300 break;
301 default:
302 LOGSIG("Caught SIGFPE: Arithmetic Exception" );
303 break;
304 }
305 break;
306 case SIGILL:
307 switch(siginfo->si_code)
308 {
309 case ILL_ILLOPC:
310 LOGSIG("Caught SIGILL: (illegal opcode)" );
311 break;
312 case ILL_ILLOPN:
313 LOGSIG("Caught SIGILL: (illegal operand)" );
314 break;
315 case ILL_ILLADR:
316 LOGSIG("Caught SIGILL: (illegal addressing mode)" );
317 break;
318 case ILL_ILLTRP:
319 LOGSIG("Caught SIGILL: (illegal trap)" );
320 break;
321 case ILL_PRVOPC:
322 LOGSIG("Caught SIGILL: (privileged opcode)" );
323 break;
324 case ILL_PRVREG:
325 LOGSIG("Caught SIGILL: (privileged register)" );
326 break;
327 case ILL_COPROC:
328 LOGSIG("Caught SIGILL: (coprocessor error)" );
329 break;
330 case ILL_BADSTK:
331 LOGSIG("Caught SIGILL: (internal stack error)" );
332 break;
333 default:
334 LOGSIG("Caught SIGILL: Illegal Instruction" );
335 break;
336 }
337 break;
338 case SIGTERM:
339 LOGSIG("Caught SIGTERM: a termination request was sent to the program" );
340 break;
341 case SIGABRT:
342 LOGSIG("Caught SIGABRT: usually caused by an abort() or assert()" );
343 break;
344 default:
345 break;
346 }
347 posix_print_stack_trace();
348 _Exit(1);
349}

351static uint8_t alternate_stack[SIGALTSTACK_SIZE];
352
353void set_signal_handler( const char *progname )
354{
355 icky_global_program_name = progname ;
356
357#ifdef RLIMIT_STACK
358 /* Before starting the endless recursion, try to be friendly to the user's
359 machine. On some Linux 2.2.x systems, there is no stack limit for user
360 processes at all. We don't want to kill such systems. */
361 struct rlimit rl;
362 rl.rlim_cur = rl.rlim_max = 0x100000; /* 1 MB */
363 setrlimit (RLIMIT_STACK, &rl);
364#endif
365
366 /* setup alternate stack */
367 {
368 stack_t ss ;
369
370 /* malloc is usually used here, I'm not 100% sure my static allocation
371 is valid but it seems to work just fine. */
372 ss.ss_sp = alternate_stack;
373 ss.ss_size = sizeof( alternate_stack );
374 ss.ss_flags = SS_ONSTACK ;
375
376 if (sigaltstack(&ss, NULL) != 0)
377 {
378 err(1, "sigaltstack");
379 }
380 }
381
382 /* register our signal handlers */
383 {
384 struct sigaction sig_action ;
385 sigemptyset( &sig_action.sa_mask );
386 sig_action.sa_sigaction = posix_signal_handler;
387
388#ifdef __APPLE__
389 /* for some reason we backtrace() doesn't work on osx
390 when we use an alternate stack */
391 sig_action.sa_flags = SA_SIGINFO;
392#else
393 sig_action.sa_flags = SA_SIGINFO | SA_ONSTACK;
394#endif
395
396 if (sigaction(SIGSEGV, &sig_action, NULL) != 0)
397 {
398 err(1, "sigaction");
399 }
400 if (sigaction(SIGFPE, &sig_action, NULL) != 0)
401 {
402 err(1, "sigaction");
403 }
404 if (sigaction(SIGINT, &sig_action, NULL) != 0)
405 {
406 err(1, "sigaction");
407 }
408 if (sigaction(SIGILL, &sig_action, NULL) != 0)
409 {
410 err(1, "sigaction");
411 }
412 /* No TERM catch
413 if (sigaction(SIGTERM, &sig_action, NULL) != 0) { err(1, "sigaction"); } */
414 if (sigaction(SIGABRT, &sig_action, NULL) != 0)
415 {
416 err(1, "sigaction");
417 }
418 }
419}
