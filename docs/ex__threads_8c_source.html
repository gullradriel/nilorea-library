94void *occupy_thread( void *rest )
95{
96 __n_assert( rest, return NULL );
97
98 intptr_t sleep_value = (intptr_t)(rest) ;
99
100 n_log( LOG_INFO, "Starting to sleep %d secs on thread %lld", sleep_value/1000000, pthread_self() );
101
102 if( sleep_value < 1000000 )
103 usleep( sleep_value );
104 else
105 {
106 usleep( (sleep_value)%1000000 );
107 sleep( (sleep_value/1000000) );
108 }
109
110 return NULL ;
111}
