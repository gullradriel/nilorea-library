110void lp_lock_precheck_diagnose(const pthread_mutex_t *mutex)
111{
112 int rc = -1;
113 /*Highest ranking mutex currently acquired by 'me'*/
114 int maxmidx = LP_INVALID_IDX;
115 int midx = LP_INVALID_IDX;
116 pthread_t me = pthread_self();
117
118 pthread_t thr = LP_INVALID_THREAD;
119 const pthread_mutex_t *mtx = NULL;
120
121 /* Thread tries to lock a mutex it has already locked? */
122 if ((rc=lp_is_mutex_locked(mutex, me)) != 0)
123 {
124 lp_report("Mutex M%lx is already locked by thread.\n", mutex);
125 lp_dump();
126 }
127
128 /* Is mutex order respected? */
129 maxmidx = lp_max_idx_mutex_locked(me);
130 midx = lp_mutex_idx(mutex);
131 if (midx < maxmidx)
132 {
133 lp_report("Mutex M%lx will be locked out of order (idx=%d, crt max=%d).\n",
134 mutex, midx, maxmidx);
135 lp_dump();
136 }
137
138 /* Will deadlock? Check for a circular chain. */
139 lp_report("Checking if it will deadlock...\n");
140 thr = me;
141 mtx = mutex;
142 while ((thr=lp_thread_owning(mtx)) != LP_INVALID_THREAD)
143 {
144 lp_report(" Mutex M%lx is owned by T%lx.\n", mtx, thr);
145 mtx = lp_mutex_wanted(thr);
146 lp_report(" Thread T%lx waits for M%lx.\n", thr, mtx);
147
148 if (mtx == LP_INVALID_MUTEX)
149 break; /*no circular dead; */
150
151 if (0 != pthread_equal(thr, me))
152 {
153 lp_report(" Deadlock condition detected.\n");
154 lp_dump();
155 break;
156 }
157 }
158}

160void lp_unlock_precheck_diagnose(const pthread_mutex_t *mutex)
161{
162 int rc = -1;
163 int maxmidx = LP_INVALID_IDX, midx = LP_INVALID_IDX;
164
165 /* Thread tries to unlock a mutex it does not have? */
166 if ((rc=lp_is_mutex_locked(mutex, pthread_self())) == 0)
167 {
168 lp_report("Attempt to release M%lx NOT locked by thread.\n", mutex);
169 lp_dump();
170 }
171
172 /* Are mutexes released in reverse order? */
173 maxmidx = lp_max_idx_mutex_locked(pthread_self());
174 midx = lp_mutex_idx(mutex);
175 if (midx < maxmidx)
176 {
177 lp_report("Mutex M%lx will be released out of order (idx=%d, crt max=%d).\n",
178 mutex, midx, maxmidx);
179 lp_dump();
180 }
181}

185int pthread_mutex_lock(pthread_mutex_t *mutex)
186{
187 int rc = EINVAL;
188
189 if (mutex != NULL)
190 {
191 lp_lock_precheck(mutex);
192
193 /* Call the real pthread_mutex_lock() */
194 rc = next_pthread_mutex_lock(mutex);
195
196 lp_lock_postcheck(mutex, rc);
197 }
198 else
199 {
200 lp_report("%s(): mutex* is NULL.\n", __FUNCTION__ );
201 }
202
203 return rc;
204}
