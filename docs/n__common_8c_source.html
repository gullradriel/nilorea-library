35/*! void n_abort( char const *format , ... )
36 *\brief abort program with a text
37 *\param format printf style format and args
38 */
39void n_abort( char const *format, ... )
40{
41 char str[1024] = "" ;
42 va_list args;
43 va_start(args, format);
44
45 vsnprintf(str, sizeof str, format, args);
46 va_end(args);

95 error = errno ;

99 return strdup( dirname( strbuf ) );
100 }

103 return NULL ;
104} /* get_prog_dir */

143/*!\fn int n_popen( char *cmd , int read_buf_size , void *nstr_output , int *ret )
144 * \brief launch a command abd return output and status
145 * \param cmd The command to launch
146 * \param read_buf_size popen read buf. Also serves as a minimum size for the dynamically allocated returned output if not already allocaded
147 * \param output Pointer to a valid N_STR or NULL (see read_buf_size)
148 * \param ret Command output if any. Should be initialized to -1 and tested against to check if it's valid.
149 * \return TRUE or FALSE
150 */
151int n_popen( char *cmd, int read_buf_size, void **nstr_output, int *ret )
152{
153 __n_assert( cmd, return FALSE );
154
155 N_STR *output_pointer = new_nstr( read_buf_size + 1 );
156
157 FILE *fp = NULL ;
158 int status = 0 ;
159 char read_buf[ read_buf_size ] ;
160
161 fp = popen( cmd, "w" );
162 int err = errno ;
163 if (fp == NULL)
164 {
165 n_log( LOG_ERR, "popen( %s ) returned NULL , %s (%d)", cmd, strerror( err ), err );
166 return FALSE ;
167 }
168
169 int length = 0 ;
170 while( fgets( read_buf, read_buf_size, fp ) != NULL )
171 {
172 length = strlen( read_buf );
173 if( read_buf[ length - 1 ] == '\n' )
174 {
175 read_buf[ length - 1 ] = '\0' ;
176 }
177 nstrprintf_cat( output_pointer, "%s", read_buf );
178 }
179 status = pclose( fp );
180 err = errno ;
181 if( status == -1 )
182 {
183 n_log( LOG_ERR, "pclose( %s ) returned -1 , %s (%d)", cmd, strerror( err ), err );
184 if( WIFEXITED( status ) )
185 {
186 n_log( LOG_ERR, "Child exited with RC=%d", WEXITSTATUS( status ) );
187 (*ret) = WEXITSTATUS( status );
188 }
189 return FALSE ;
190 }
191 (*ret) = WEXITSTATUS( status );
192 (*nstr_output) = output_pointer ;
193 return TRUE ;
194} /* n_popen( ... ) */

198/*!\fn n_daemonize( void )
199 *\brief Daemonize program
200 *\return TRUE or FALSE
201 */
202int n_daemonize( void )
203{
204 /* Fork off the parent process */
205 pid_t pid = fork();
206 if( pid < 0 )
207 {
208 n_log( LOG_ERR, "Error: unable to fork child: %s", strerror( errno ) );
209 exit( 1 );
210 }
211 if( pid > 0 )
212 {
213 n_log( LOG_NOTICE, "Child started successfuly !" );
214 exit( 0 );
215 }
216 /* Close all open file descriptors */
217 int x = 0 ;
218 for(x = sysconf( _SC_OPEN_MAX ) ; x >= 0 ; x-- )
219 {
220 close(x);
221 }
222 int fd = open("/dev/null",O_RDWR, 0);
223 if (fd != -1)
224 {
225 dup2 (fd, STDIN_FILENO);
226 dup2 (fd, STDOUT_FILENO);
227 dup2 (fd, STDERR_FILENO);
228 if (fd > 2)
229 {
230 close (fd);
231 }
232 }
233 else
234 {
235 n_log( LOG_ERR, "Failed to open /dev/null" );
236 }
237
238 /* On success: The child process becomes session leader */
239 if( setsid() < 0 )
240 {
241 n_log( LOG_ERR, "Error: unable to set session leader with setsid(): %s", strerror( errno ) );
242 exit( 1 );
243 }
244 else
245 {
246 n_log( LOG_NOTICE, "Session leader set !" );
247 }
248
249 return TRUE ;
250} /* n_daemonize(...) */

254/*!\fn pid_t system_nb(const char * command, int * infp, int * outfp)
255 *\brief Non blocking system call
256 *\param command to call
257 *\param infp stdin file descriptor holder or NULL
258 *\param outfp stdout file descriptor holder or NULL
259 *\return The system command pid or -1
260 */
261pid_t system_nb(const char * command, int * infp, int * outfp)
262{
263 __n_assert( command, return -1 );
264
265 int p_stdin[2] = { -1, -1 };
266 int p_stdout[2] = { -1, -1 };
267 pid_t pid = -1 ;
268
269 if( pipe( p_stdin ) == -1 )
270 {
271 n_log( LOG_ERR, "Couldn't pipe stdin for command %s", command );
272 return -1;
273 }
274 if (pipe(p_stdout) == -1)
275 {
276 close(p_stdin[0]);
277 close(p_stdin[1]);
278 n_log( LOG_ERR, "Couldn't pipe stdout for command %s", command );
279 return -1;
280 }
281
282 pid = fork();
283 if (pid < 0)
284 {
285 close(p_stdin[0]);
286 close(p_stdin[1]);
287 close(p_stdout[0]);
288 close(p_stdout[1]);
289 n_log( LOG_ERR, "Couldn't fork command %s", command );
290 return pid;
291 }
292 else if (pid == 0)
293 {
294 close(p_stdin[1]);
295 dup2(p_stdin[0], 0);
296 close(p_stdout[0]);
297 dup2(p_stdout[1], 1);
298 dup2( open("/dev/null", O_RDONLY), 2);
299
300 /// Close all other descriptors for the safety sake.
301 for (int i = 3; i < 4096; ++i)
302 {
303 close(i);
304 }
305 setsid();
306 execl("/bin/sh", "sh", "-c", command, (char *)NULL);
307 _exit(1);
308 }
309
310 close( p_stdin[0] );
311 close( p_stdout[1] );
312 if( infp == NULL)
313 {
314 close(p_stdin[1]);
315 }
316 else
317 {
318 *infp = p_stdin[1];
319 }
320 if (outfp == NULL)
321 {
322 close(p_stdout[0]);
323 }
324 else
325 {
326 *outfp = p_stdout[0];
327 }
328
329 return pid;
330} /* system_nb */
