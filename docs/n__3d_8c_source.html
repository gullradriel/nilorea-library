12/*!\fn double distance( VECTOR3D *p1 , VECTOR3D *p2 )
13 *\brief compute the distance between two VECTOR3D points
14 *\param p1 The first point
15 *\param p2 The second point
16 *\return The computed distance
17 */
18double distance( VECTOR3D *p1, VECTOR3D *p2 )
19{
20 return sqrt( ( (*p1)[ 0 ] - (*p2)[ 0 ] ) * ( (*p1)[ 0 ] - (*p2)[ 0 ] ) +
21 ( (*p1)[ 1 ] - (*p2)[ 1 ] ) * ( (*p1)[ 1 ] - (*p2)[ 1 ] ) +
22 ( (*p1)[ 2 ] - (*p2)[ 2 ] ) * ( (*p1)[ 2 ] - (*p2)[ 2 ] ) );
23} /* distance(...) */

28 *\brief Update object position componant.
29 *\param object The object physics to update
30 *\param it Componant of VECTOR3D to update. it values : 0 -> x, 1 -> y , 2 -> z
31 *\param delta_t Elapsed time since last call for componant update computing
32 *\return TRUE or FALSE
33 */
34int update_physics_position_nb( PHYSICS *object, int it, double delta_t )
35{
36 __n_assert( object, return FALSE );
37
38 object -> speed[ it ] = object -> speed[ it ] + ( object -> acceleration[ it ] * delta_t )/ 1000000.0 ;
39 object -> position[ it ] = object -> position[ it ] + ( object -> speed[ it ] * delta_t ) / 1000000.0 + ( object -> acceleration[ it ] * (delta_t / 1000000.0 ) * ( delta_t / 1000000.0 ) ) / 2.0 ;
40 object -> angular_speed[ it ] = object -> angular_speed[ it ] + ( object -> angular_acceleration[ it ] * delta_t ) / 1000000.0 ;
41 object -> speed[ it ] = object -> speed[ it ] + ( object -> gravity[ it ] * delta_t ) / 1000000.0 ;

123/*!\fn int vector_intersect( VECTOR3D *p1 , VECTOR3D *p2 , VECTOR3D *p3 , VECTOR3D *p4 , VECTOR3D *px )
124 *\brief Compute if two vectors are intersecting or not
125 *\param p1 First point of vector 1
126 *\param p2 Second poinf of vector 1
127 *\param p3 First point of vector 2
128 *\param p4 Second point of vector 2
129 *\param px Storage for the eventual point
130 *\return VECTOR3D_DONT_INTERSECT , VECTOR3D_COLLINEAR or VECTOR3D_DO_INTERSECT
131 */
132int vector_intersect( VECTOR3D *p1, VECTOR3D *p2, VECTOR3D *p3, VECTOR3D *p4, VECTOR3D *px )
133{
134 double a1 = 0, a2 = 0, b1 = 0, b2 = 0, c1 = 0, c2 = 0,
135 r1 = 0, r2 = 0, r3 = 0, r4 = 0,
136 denom = 0, offset = 0, num = 0 ;
137
138 /* Compute a1, b1, c1, where line joining points 1 and 2 is
139 "a1 x + b1 y + c1 = 0". */
140 a1 = (*p2)[ 1 ] - (*p1)[ 1 ];
141 b1 = (*p1)[ 0 ] - (*p2)[ 0 ];
142 c1 = ((*p2)[ 0 ] * (*p1)[ 1 ]) - ((*p1)[ 0 ] * (*p2)[ 1 ]);
143
144 /* Compute r3 and r4. */
145 r3 = ((a1 * (*p3)[ 0 ]) + (b1 * (*p3)[ 1 ]) + c1);
146 r4 = ((a1 * (*p4)[ 0 ]) + (b1 * (*p4)[ 1 ]) + c1);
147
148 /* Check signs of r3 and r4. If both point 3 and point 4 lie on
149 same side of line 1, the line segments do not intersect. */
150 if((r3 != 0) && (r4 != 0) && same_sign(r3, r4))
151 {
152 return VECTOR3D_DONT_INTERSECT;
153 }
154
155 /* Compute a2, b2, c2 */
156 a2 = (*p4)[ 1 ] - (*p3)[ 1 ];
157 b2 = (*p3)[ 0 ] - (*p4)[ 0 ];
158 c2 = ((*p4)[ 0 ] * (*p3)[ 1 ]) - ((*p3)[ 0 ] * (*p4)[ 1 ]);
159
160 /* Compute r1 and r2 */
161 r1 = (a2 * (*p1)[ 0 ]) + (b2 * (*p1)[ 1 ]) + c2;
162 r2 = (a2 * (*p2)[ 0 ]) + (b2 * (*p2)[ 1 ]) + c2;
163
164 /* Check signs of r1 and r2. If both point 1 and point 2 lie
165 on same side of second line segment, the line segments do
166 not intersect. */
167 if( (r1 != 0) && (r2 != 0) && same_sign(r1, r2) )
168 {
169 return VECTOR3D_DONT_INTERSECT;
170 }
171
172 /* Line segments intersect: compute intersection point. */
173 denom = (a1 * b2) - (a2 * b1);
174
175 if( denom == 0 )
176 {
177 return VECTOR3D_COLLINEAR;
178 }
179
180 if( denom < 0 )
181 {
182 offset = -denom / 2;
183 }
184 else
185 {
186 offset = denom / 2 ;
187 }
188
189 /* The denom/2 is to get rounding instead of truncating. It
190 is added or subtracted to the numerator, depending upon the
191 sign of the numerator. */
192 num = (b1 * c2) - (b2 * c1);
193 if(num < 0)
194 {
195 (*px)[ 0 ] = (num - offset) / denom;
196 }
197 else
198 {
199 (*px)[ 0 ] = (num + offset) / denom;
200 }
201
202 num = (a2 * c1) - (a1 * c2);
203 if(num < 0)
204 {
205 (*px)[ 1 ] = ( num - offset) / denom;
206 }
207 else
208 {
209 (*px)[ 1 ] = (num + offset) / denom;
210 }
211
212 /* lines_intersect */
213 return VECTOR3D_DO_INTERSECT;
214} /* vector_intersect(...) */

217double vector_dot_prod( VECTOR3D *vec1, VECTOR3D *vec2 )
218{
219 return (*vec1)[ 0 ] * (*vec2)[ 0 ] + (*vec1)[ 1 ] * (*vec2)[ 1 ] + (*vec1)[ 2 ] * (*vec2)[ 2 ];
220}

222double vector_normalize( VECTOR3D *vec )
223{
224 double res = 0.0 ;
225 for( int i=0 ; i < 3 ; i++ )
226 {
227 res += pow( (*vec)[ i ], 2 );
228 }
229 return sqrt( res );
230}

233double vector_angle_between( VECTOR3D *vec1, VECTOR3D *vec2 )
234{
235 return acos( vector_dot_prod( &(*vec1), &(*vec2) ) / ( vector_normalize( &(*vec1) ) * vector_normalize( &(*vec2) ) ) );
