121  *\brief Compute if two vectors are intersecting or not
122  *\param p1 First point of vector 1
123  *\param p2 Second poinf of vector 1
124  *\param p3 First point of vector 2
125  *\param p4 Second point of vector 2
126  *\param px Storage for the eventual point
127  *\return VECTOR3D_DONT_INTERSECT , VECTOR3D_COLLINEAR or VECTOR3D_DO_INTERSECT
128  */
129 int vector_intersect( VECTOR3D p1, VECTOR3D p2, VECTOR3D p3, VECTOR3D p4, VECTOR3D *px )
130 {
131  double a1 = 0, a2 = 0, b1 = 0, b2 = 0, c1 = 0, c2 = 0,
132  r1 = 0, r2 = 0, r3 = 0, r4 = 0,
133  denom = 0, offset = 0, num = 0 ;
134 
135  /* Compute a1, b1, c1, where line joining points 1 and 2 is
136  "a1 x + b1 y + c1 = 0". */
137  a1 = p2[ 1 ] - p1[ 1 ];
138  b1 = p1[ 0 ] - p2[ 0 ];
139  c1 = (p2[ 0 ] * p1[ 1 ]) - (p1[ 0 ] * p2[ 1 ]);
140 
141  /* Compute r3 and r4. */
142  r3 = ((a1 * p3[ 0 ]) + (b1 * p3[ 1 ]) + c1);
143  r4 = ((a1 * p4[ 0 ]) + (b1 * p4[ 1 ]) + c1);
144 
145  /* Check signs of r3 and r4. If both point 3 and point 4 lie on
146  same side of line 1, the line segments do not intersect. */
147  if((r3 != 0) && (r4 != 0) && same_sign(r3, r4))
148  {
149  return VECTOR3D_DONT_INTERSECT;
150  }
151 
152  /* Compute a2, b2, c2 */
153  a2 = p4[ 1 ] - p3[ 1 ];
154  b2 = p3[ 0 ] - p4[ 0 ];
155  c2 = (p4[ 0 ] * p3[ 1 ]) - (p3[ 0 ] * p4[ 1 ]);
156 
157  /* Compute r1 and r2 */
158  r1 = (a2 * p1[ 0 ]) + (b2 * p1[ 1 ]) + c2;
159  r2 = (a2 * p2[ 0 ]) + (b2 * p2[ 1 ]) + c2;
160 
161  /* Check signs of r1 and r2. If both point 1 and point 2 lie
162  on same side of second line segment, the line segments do
163  not intersect. */
164  if( (r1 != 0) && (r2 != 0) && same_sign(r1, r2) )
165  {
166  return VECTOR3D_DONT_INTERSECT;
167  }
168 
169  /* Line segments intersect: compute intersection point. */
170  denom = (a1 * b2) - (a2 * b1);
171

174  return VECTOR3D_COLLINEAR;
175  }
176

178  {
179  offset = -denom / 2;
180  }
181  else

183  offset = denom / 2 ;
184  }
185

187  is added or subtracted to the numerator, depending upon the
188  sign of the numerator. */
189  num = (b1 * c2) - (b2 * c1);
190  if(num < 0)
191  {
192  (*px)[ 0 ] = (num - offset) / denom;
193  }
