62/*!\fn void *remove_list_node_f( LIST *list , LIST_NODE *node )
63 *\brief Internal function called each time we need to get a node out of a list
64 *\param list The list to pick in
65 *\param node The node to remove in the list
66 *\return The node or NULL
67 */
68void *remove_list_node_f( LIST *list, LIST_NODE *node )
69{
70 void *ptr = NULL ;
71
72 __n_assert( list, n_log( LOG_ERR, "can't remove from NULL list" ); return NULL );
73 __n_assert( list->start, n_log( LOG_ERR, "can't remove from NULL list->start" ); return NULL );
74 __n_assert( list->end, n_log( LOG_ERR, "can't remove from NULL list->end" ); return NULL );
75 __n_assert( node, n_log( LOG_ERR, "can't remove from NULL node" ); return NULL );
76
77 ptr = node -> ptr ;
78 if( node -> prev && node -> next )
79 {
80 node -> prev -> next = node -> next ;
81 node -> next -> prev = node -> prev ;
82 }
83 else
84 {
85 if( node -> prev == NULL && node -> next )
86 {
87 node -> next -> prev = NULL ;
88 list -> start = node -> next ;
89 }
90 else
91 {
92 if( node -> prev && node -> next == NULL )
93 {
94 node -> prev -> next = NULL ;
95 list -> end = node -> prev ;
96 }
97 else
98 {
99 if( node -> prev == NULL && node -> next == NULL )
100 {
101 /* removing last item */
102 list -> start = list -> end = NULL ;
103 }
104 }
105 }
106 }
107 Free( node );
108 list -> nb_items -- ;
109 return ptr ;
110} /* remove_list_node_f(...) */

114/*!\fn int list_push( LIST *list , void *ptr , void (*destructor)( void *ptr ) )
115 *\brief Add a pointer to the end of the list.
116 *\param list An initilized list container. A null value will cause an error and a _log message.
117 *\param ptr The pointer you wan to put in the list. A null value will cause an error and a _log message.
118 *\param destructor Pointer to the ptr type destructor function. Leave to NULL if there isn't.
119 *\return TRUE or FALSE. Check error messages in DEBUG mode.
120 */
121int list_push( LIST *list, void *ptr, void (*destructor)( void *ptr ) )
122{
123 LIST_NODE *node = NULL ;
124
125 __n_assert( list, n_log( LOG_ERR, "invalid list: NULL" ); return FALSE );
126 __n_assert( ptr, n_log( LOG_ERR, "invalid ptr: NULL" ); return FALSE );
127
128 if( list -> nb_max_items > 0 && ( list -> nb_items >= list -> nb_max_items ) )
129 {
130 n_log( LOG_ERR, "list is full" );
131 return FALSE ;
132 }
133
134 node = new_list_node( ptr, destructor );
135 __n_assert( node, n_log( LOG_ERR, "Couldn't allocate new node" ); return FALSE );
136
137 list -> nb_items ++ ;
138
139 if( list -> end )
140 {
141 list -> end -> next = node ;
142 node -> prev = list -> end ;
143 list -> end = node ;
144 }
145 else
146 {
147 list -> start = list -> end = node ;
148 }
149 return TRUE ;
150} /* list_push( ... ) */

154/*!\fn int list_push_sorted( LIST *list , void *ptr , int (*comparator)( const void *a , const void *b ) , void (*destructor)( void *ptr ) )
155 *\brief Add a pointer sorted in the list , starting by the end of the list.
156 *\param list An initilized list container. A null value will cause an error and a _log message.
157 *\param ptr The pointer you wan to put in the list. A null value will cause an error and a _log message.
158 *\param comparator A pointer to a function which take two void * pointers and return an int.
159 *\param destructor Pointer to the ptr type destructor function. Leave to NULL if there isn't.
160 *\return TRUE or FALSE. Check error messages in DEBUG mode.
161 */
162int list_push_sorted( LIST *list, void *ptr, int (*comparator)( const void *a, const void *b ), void (*destructor)( void *ptr ) )
163{
164 LIST_NODE *nodeptr = NULL ;
165
166 __n_assert( list, n_log( LOG_ERR, "invalid list: NULL" ); return FALSE );
167 __n_assert( ptr, n_log( LOG_ERR, "invalid ptr: NULL" ); return FALSE );
168
169 if( list -> nb_max_items > 0 && ( list -> nb_items >= list -> nb_max_items ) )
170 {
171 n_log( LOG_ERR, "list is full" );
172 return FALSE ;
173 }
174
175 if( list -> end )
176 {
177 nodeptr = list -> end ;
178 while( nodeptr && ( comparator( ptr, nodeptr -> ptr ) < 0 ) )
179 nodeptr = nodeptr -> prev ;
180
181 if( !nodeptr )
182 {
183 /* It's the lower ranked element in the sort */
184 list_unshift( list, ptr, destructor );
185 }
186 else
187 {
188 /* we have a match inside the list. let's insert the datas */
189 LIST_NODE *node_next = nodeptr -> next ;
190 LIST_NODE *newnode = new_list_node( ptr, destructor );
191 __n_assert( newnode, n_log( LOG_ERR, "Couldn't allocate new node" ); return FALSE );
192
193 if( node_next )
194 {
195 link_node( newnode, node_next );
196 }
197 else
198 list -> end = newnode ;
199
200 link_node( nodeptr, newnode );
201 list -> nb_items ++ ;
202 }
203 }
204 else
205 list_push( list, ptr, destructor );
206
207 return TRUE ;
208} /* list_push_sorted( ... ) */

420 *\brief Empty a LIST list of pointers.
421 *\param list The list to empty. An already empty or null list will cause an error and a _log message.
422 *\return TRUE or FALSE. Check error messages in DEBUG mode.
