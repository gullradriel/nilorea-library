179#ifndef vasprintf
180int vasprintf(char **strp, const char *fmt, va_list ap)
181{
182 long long int len = _vscprintf_so(fmt, ap);
183 if (len == -1)
184 return -1;
185 char *str = NULL ;
186 Malloc( str, char, (size_t) len + 1 + sizeof( void *) ); // len + EndOfString + padding
187 if (!str)
188 return -1;
189 int r = vsnprintf(str, (size_t)(len+1), fmt, ap); /* "secure" version of vsprintf */
190 if (r == -1)
191 return free(str), -1;
192 *strp = str;
193 return r;
194}
195#endif

208/*!\fn void _n_log( int level , const char *file , const char *func , int line , const char *format , ... )
209 *\brief Logging function. log( level , const char *format , ... ) is a macro around _log
210 *\param level Logging level
211 *\param file File containing the emmited log
212 *\param func Function emmiting the log
213 *\param line Line of the log
214 *\param format Format and string of the log, printf style
215 */
216void _n_log( int level, const char *file, const char *func, int line, const char *format, ... )
217{
218 FILE *out = NULL ;
219
220 if( level == LOG_NULL )
221 return ;
222
223 if( !log_file )
224 out = stderr ;
225 else
226 out = log_file ;
227
228 int log_level = get_log_level();
229
230 if( level <= log_level )
231 {
232 va_list args ;
233 char *syslogbuffer = NULL ;
234 char *eventbuffer = NULL ;
235#ifdef __windows__
236 size_t needed = 0 ;
237 char *name = "NULL" ;
238 if( proc_name )
239 name = proc_name ;
240#endif
241
242 switch( LOG_TYPE )
243 {
244 case LOG_SYSJRNL :
245 va_start (args, format);
246 vasprintf( &syslogbuffer, format, args );
247 va_end( args );
248#ifdef __windows__
249 needed = (unsigned long long)snprintf( NULL, 0, "start /B EventCreate /t %s /id 666 /l APPLICATION /so %s /d \"%s\" > NUL 2>&1", prioritynames[ level ] . w_name, name, syslogbuffer );
250 Malloc( eventbuffer, char, needed + 4 );
251 sprintf( eventbuffer, "start /B EventCreate /t %s /id 666 /l APPLICATION /so %s /d \"%s\" > NUL 2>&1", prioritynames[ level ] . w_name, name, syslogbuffer );
252 system( eventbuffer );
253#else
254 syslog( level, "%s->%s:%d %s", file, func, line, syslogbuffer );
255#endif
256 FreeNoLog( syslogbuffer );
257 FreeNoLog( eventbuffer );
258 break ;
259 default:
260 fprintf( out, "%s:%jd:%s->%s:%d ", prioritynames[ level ] . c_name, (intmax_t)time( NULL ), file, func, line );
261 va_start (args, format);
262 vfprintf( out, format, args );
263 va_end( args );
264 fprintf( out, "\n" );
265 break ;
266 }
267 fflush( out );
268 }
269} /* _n_log( ... ) */

273/*!\fn open_safe_logging( TS_LOG **log , char *pathname , char *opt )
274 *\brief Open a thread-safe logging file
275 *\param log A TS_LOG handler
276 *\param pathname The file path (if any) and name
277 *\param opt Options for opening (please never forget to use "w")
278 *\return TRUE on success , FALSE on error , -1000 if already open
279 */
280int open_safe_logging( TS_LOG **log, char *pathname, char *opt )
281{
282
283 if( (*log) )
284 {
285
286 if( (*log) -> file )
287 return -1000; /* already open */
288
289 return FALSE;
290 }
291
292 if( !pathname )
293 return FALSE; /* no path/filename */
294
295 Malloc( (*log), TS_LOG, 1 );
296
297 if( !(*log) )
298 return FALSE;
299
300 pthread_mutex_init( &(*log) -> LOG_MUTEX, NULL );
301
302 (*log) -> file = fopen( pathname, opt );
303
304 if( !(*log) -> file )
305 return FALSE;
306
307 return TRUE;
308
309}/* open_safe_logging(...) */

313/*!\fn write_safe_log( TS_LOG *log , char *pat , ... )
314 *\brief write to a thread-safe logging file
315 *\param log A TS_LOG handler
316 *\param pat Pattern for writting (i.e "%d %d %s")
317 *\return TRUE on success, FALSE on error
318 */
319int write_safe_log( TS_LOG *log, char *pat, ... )
320{
321 /* argument list */
322 va_list arg;
323 char str[2048] = "" ;
324
325 if( !log )
326 return FALSE;
327
328 va_start( arg, pat );
329
330 vsnprintf( str, sizeof(str), pat, arg );
331
332 va_end( arg );
333
334 pthread_mutex_lock( &log -> LOG_MUTEX );
335 fprintf( log -> file, "%s", str );
336 fflush( log -> file );

341} /* write_safe_log( ... ) */
