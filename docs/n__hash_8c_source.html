481int _ht_check_trie_divergence( HASH_TABLE *table, char *key )
482{
483 __n_assert( table, return FALSE );
484 __n_assert( key, return FALSE );
485
486 /* check and return branching if any */
487 HASH_NODE* node = table -> root;
488
489 int last_index = 0;
490 for( uint32_t it = 0 ; key[ it ] != '\0' ; it ++ )
491 {
492 int index = key[ it ] - table -> alphabet_offset ;
493 if( index < 0 || (unsigned)index >= table -> alphabet_length )
494 {
495 n_log( LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key );
496 index = 0 ;
497 }
498 if( node -> children[ index ] )
499 {
500 /* child check */
501 for( uint32_t it2 = 0 ; it2 < table -> alphabet_length ; it2++ )
502 {
503 if( it2 != (unsigned)index && node -> children[ it2 ] )
504 {
505 /* child found, update branch index */
506 last_index = it + 1;
507 break;
508 }
509 }
510 /* Go to the next child in the sequence */
511 node = node->children[ index ];
512 }
513 }
514 return last_index;
515}

609int _ht_remove_trie( HASH_TABLE *table, char *key )
610{
611 __n_assert( table, return FALSE );
612 __n_assert( table -> root, return FALSE );
613 __n_assert( key, return FALSE );
614
615 /* stop if matching node not a leaf node */
616 if( !_ht_is_leaf_node_trie( table, key ) )
617 {
618 return FALSE ;
619 }
620
621 HASH_NODE *node = table -> root ;
622 /* find the longest prefix string that is not the current key */
623 char *longest_prefix = _ht_find_longest_prefix_trie( table, key );
624 if( longest_prefix && longest_prefix[ 0 ] == '\0' )
625 {
626 Free( longest_prefix );
627 return FALSE ;
628 }
629 /* keep track of position in the tree */
630 uint32_t it = 0 ;
631 for( it = 0 ; longest_prefix[ it ] != '\0'; it++ )
632 {
633 int index = (int)longest_prefix[ it ] - table -> alphabet_offset ;
634 if( index < 0 || (unsigned)index >= table -> alphabet_length )
635 {
636 n_log( LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key );
637 index = 0 ;
638 }
639 if( node -> children[ index ] != NULL )
640 {
641 /* common prefix, keep moving */
642 node = node -> children[ index ];
643 }
644 else
645 {
646 /* not found */
647 Free( longest_prefix );
648 return FALSE ;
649 }
650 }
651 /* deepest common node between the two strings */
652 /* deleting the sequence corresponding to key */
653 for( ; key[ it ] != '\0' ; it++ )
654 {
655 int index = (int)key[ it ] - table -> alphabet_offset ;
656 if( index < 0 || (unsigned)index >= table -> alphabet_length )
657 {
658 n_log( LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key );
659 index = 0 ;
660 }
661 if( node -> children[ index] )
662 {
663 /* delete the remaining sequence */
664 HASH_NODE* node_to_kill = node -> children[ index ];
665 node -> children[ index ] = NULL;
666 _ht_node_destroy( node_to_kill );
667 }
668 }
669 Free( longest_prefix );
670 return TRUE ;
671}

930{

933
934 uint32_t h1 = seed;
935
936 uint32_t c1 = 0xcc9e2d51;
937 uint32_t c2 = 0x1b873593;

941
942 int i = 0 ;

944 {
945 uint32_t k1 = getblock(blocks,i);
946
947 k1 *= c1;
948 k1 = ROTL32(k1,15);
949 k1 *= c2;

952 h1 = ROTL32(h1,13);
953 h1 = h1*5+0xe6546b64;
