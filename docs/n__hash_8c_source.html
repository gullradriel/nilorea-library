492int _ht_check_trie_divergence( HASH_TABLE *table, const char *key )
493{
494 __n_assert( table, return FALSE );
495 __n_assert( key, return FALSE );
496
497
498 HASH_NODE* node = table -> root;
499
500 int last_index = 0;
501 for( uint32_t it = 0 ; key[ it ] != '\0' ; it ++ )
502 {
503 int index = key[ it ] - table -> alphabet_offset ;
504 if( index < 0 || (unsigned)index >= table -> alphabet_length )
505 {
506 n_log( LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key );
507 index = 0 ;
508 }
509 if( node -> children[ index ] )
510 {
511 /* child check */
512 for( uint32_t it2 = 0 ; it2 < table -> alphabet_length ; it2++ )
513 {
514 if( it2 != (unsigned)index && node -> children[ it2 ] )
515 {
516 /* child found, update branch index */
517 last_index = it + 1;
518 break;
519 }
520 }
521 /* Go to the next child in the sequence */
522 node = node->children[ index ];
523 }
524 }
525 return last_index;

539{

648int _ht_remove_trie( HASH_TABLE *table, const char *key )
649{
650 __n_assert( table, return FALSE );
651 __n_assert( table -> root, return FALSE );
652 __n_assert( key, return FALSE );
653
654 /* stop if matching node not a leaf node */
655 if( !_ht_is_leaf_node_trie( table, key ) )
656 {
657 return FALSE ;
658 }
659
660 HASH_NODE *node = table -> root ;
661 /* find the longest prefix string that is not the current key */
662 char *longest_prefix = _ht_find_longest_prefix_trie( table, key );
663 if( longest_prefix && longest_prefix[ 0 ] == '\0' )
664 {
665 Free( longest_prefix );
666 return FALSE ;
667 }
668 /* keep track of position in the tree */
669 uint32_t it = 0 ;
670 for( it = 0 ; longest_prefix[ it ] != '\0'; it++ )
671 {
672 int index = (int)longest_prefix[ it ] - table -> alphabet_offset ;
673 if( index < 0 || (unsigned)index >= table -> alphabet_length )
674 {
675 n_log( LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key );
676 index = 0 ;
677 }
678 if( node -> children[ index ] != NULL )
679 {
680 /* common prefix, keep moving */
681 node = node -> children[ index ];
682 }
683 else
684 {
685 /* not found */
686 Free( longest_prefix );
687 return FALSE ;
688 }
689 }
690 /* deepest common node between the two strings */
691 /* deleting the sequence corresponding to key */
692 for( ; key[ it ] != '\0' ; it++ )
693 {
694 int index = (int)key[ it ] - table -> alphabet_offset ;
695 if( index < 0 || (unsigned)index >= table -> alphabet_length )
696 {
697 n_log( LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key );
698 index = 0 ;
699 }
700 if( node -> children[ index] )
701 {
702 /* delete the remaining sequence */
703 HASH_NODE* node_to_kill = node -> children[ index ];
704 node -> children[ index ] = NULL;
705 _ht_node_destroy( node_to_kill );
706 }
707 }
708 Free( longest_prefix );
709 return TRUE ;
710} /* _ht_remove_trie(...) */

1084 h3 = ROTL32(h3,15);
1085 h3 += h4;
1086 h3 = h3*5+0x96cd1c35;
1087

1089 k4 = ROTL32(k4,18);

1091 h4 ^= k4;
1092 h4 = ROTL32(h4,13);
1093 h4 += h1;
1094 h4 = h4*5+0x32ac3b17;
1095 }

1123 case 11:
1124 k3 ^= tail[10] << 16;
1125 FALL_THROUGH ;
1126 case 10:
1127 k3 ^= tail[ 9] << 8;
1128 FALL_THROUGH ;
1129 case 9:
1130 k3 ^= tail[ 8] << 0;
1131 k3 *= c3;

1138 FALL_THROUGH ;
1139 case 7:

1141 FALL_THROUGH ;
1142 case 6:
