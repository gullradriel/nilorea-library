491int _ht_check_trie_divergence( HASH_TABLE *table, char *key )
492{
493 __n_assert( table, return FALSE );
494 __n_assert( key, return FALSE );
495
496
497 HASH_NODE* node = table -> root;
498
499 int last_index = 0;
500 for( uint32_t it = 0 ; key[ it ] != '\0' ; it ++ )
501 {
502 int index = key[ it ] - table -> alphabet_offset ;
503 if( index < 0 || (unsigned)index >= table -> alphabet_length )
504 {
505 n_log( LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key );
506 index = 0 ;
507 }
508 if( node -> children[ index ] )
509 {
510 /* child check */
511 for( uint32_t it2 = 0 ; it2 < table -> alphabet_length ; it2++ )
512 {
513 if( it2 != (unsigned)index && node -> children[ it2 ] )
514 {
515 /* child found, update branch index */
516 last_index = it + 1;
517 break;
518 }
519 }
520 /* Go to the next child in the sequence */
521 node = node->children[ index ];
522 }
523 }
524 return last_index;

647int _ht_remove_trie( HASH_TABLE *table, char *key )
648{
649 __n_assert( table, return FALSE );
650 __n_assert( table -> root, return FALSE );
651 __n_assert( key, return FALSE );
652
653 /* stop if matching node not a leaf node */
654 if( !_ht_is_leaf_node_trie( table, key ) )
655 {
656 return FALSE ;
657 }
658
659 HASH_NODE *node = table -> root ;
660 /* find the longest prefix string that is not the current key */
661 char *longest_prefix = _ht_find_longest_prefix_trie( table, key );
662 if( longest_prefix && longest_prefix[ 0 ] == '\0' )
663 {
664 Free( longest_prefix );
665 return FALSE ;
666 }
667 /* keep track of position in the tree */
668 uint32_t it = 0 ;
669 for( it = 0 ; longest_prefix[ it ] != '\0'; it++ )
670 {
671 int index = (int)longest_prefix[ it ] - table -> alphabet_offset ;
672 if( index < 0 || (unsigned)index >= table -> alphabet_length )
673 {
674 n_log( LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key );
675 index = 0 ;
676 }
677 if( node -> children[ index ] != NULL )
678 {
679 /* common prefix, keep moving */
680 node = node -> children[ index ];
681 }
682 else
683 {
684 /* not found */
685 Free( longest_prefix );
686 return FALSE ;
687 }
688 }
689 /* deepest common node between the two strings */
690 /* deleting the sequence corresponding to key */
691 for( ; key[ it ] != '\0' ; it++ )
692 {
693 int index = (int)key[ it ] - table -> alphabet_offset ;
694 if( index < 0 || (unsigned)index >= table -> alphabet_length )
695 {
696 n_log( LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key );
697 index = 0 ;
698 }
699 if( node -> children[ index] )
700 {
701 /* delete the remaining sequence */
702 HASH_NODE* node_to_kill = node -> children[ index ];
703 node -> children[ index ] = NULL;
704 _ht_node_destroy( node_to_kill );
705 }
706 }
707 Free( longest_prefix );
708 return TRUE ;
709} /* _ht_remove_trie(...) */
