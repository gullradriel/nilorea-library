26/*!\fn HASH_NODE *_ht_new_node_trie( HASH_TABLE *table , char key )
27 *\internal
28 *\brief node creation, HASH_CLASSIC mode
29 *\param table Target table
30 *\param key key of new node
31 *\return NULL or a new HASH_NODE *
32 */
33HASH_NODE *_ht_new_node_trie( HASH_TABLE *table, char key )
34{
35 __n_assert( table, return NULL );
36
37 HASH_NODE *new_hash_node = NULL ;
38 Malloc( new_hash_node, HASH_NODE, 1 );
39 __n_assert( new_hash_node, n_log( LOG_ERR, "Could not allocate new_hash_node" ); return NULL );
40 new_hash_node -> key = NULL ;
41 new_hash_node -> hash_value = 0 ;
42 new_hash_node -> data. ptr = NULL ;
43 new_hash_node -> destroy_func = NULL ;
44 new_hash_node -> children = NULL ;
45 new_hash_node -> is_leaf = 0 ;
46 new_hash_node -> alphabet_length = table -> alphabet_length ;
47
48 Malloc( new_hash_node -> children, HASH_NODE *, table -> alphabet_length );
49 __n_assert( new_hash_node -> children, n_log( LOG_ERR, "Could not allocate new_hash_node" ); Free( new_hash_node ); return NULL );
50
51 /* n_log( LOG_DEBUG , "node: %d %c table: alpha: %d / offset %d" , key , key , table -> alphabet_length , table -> alphabet_offset ); */
52
53 for( uint32_t it = 0 ; it < table -> alphabet_length ; it++ )
54 {
55 new_hash_node -> children[ it ] = NULL;
56 }
57 new_hash_node -> is_leaf = 0 ;
58 new_hash_node -> key_id = key ;
59 return new_hash_node;
60}

64int _ht_put_int_trie( HASH_TABLE *table, char *key, int value )
65{
66 __n_assert( table, return FALSE );
67 __n_assert( key, return FALSE );
68
69 HASH_NODE *node = table -> root ;
70
71 for( uint32_t it = 0 ; key[ it ] != '\0' ; it++ )
72 {
73 int index = (int)key[ it ] - table -> alphabet_offset ;
74 if( index < 0 || (unsigned)index >= table -> alphabet_length )
75 {
76 n_log( LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key );
77 index = 0 ;
78 }
79 /* n_log( LOG_DEBUG , "index:%d" , index ); */
80 if( node -> children[ index ] == NULL )
81 {
82 /* create a node */
83 node -> children[ index ] = _ht_new_node_trie( table, key[ it ] );
84 __n_assert( node -> children[ index ], return FALSE );
85 }
86 else
87 {
88 /* nothing to do since node is existing */
89 }
90 /* go down a level, to the child referenced by index */
91 node = node -> children[ index ];
92 }
93 /* At the end of the key, mark this node as the leaf node */
94 node -> is_leaf = 1;
95 /* Put the key */
96 node -> key = strdup( key );
97 /* Put the value */
98 node -> data . ival = value ;
99 node -> type = HASH_INT ;
100
101 return TRUE;
102}

425int _ht_check_trie_divergence( HASH_TABLE *table, char *key )
426{
427 __n_assert( table, return FALSE );
428 __n_assert( key, return FALSE );
429
430 /* check and return branching if any */
431 HASH_NODE* node = table -> root;
432
433 int last_index = 0;
434 for( uint32_t it = 0 ; key[ it ] != '\0' ; it ++ )
435 {
436 int index = key[ it ] - table -> alphabet_offset ;
437 if( index < 0 || (unsigned)index >= table -> alphabet_length )
438 {
439 n_log( LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key );
440 index = 0 ;
441 }
442 if( node -> children[ index ] )
443 {
444 /* child check */
445 for( uint32_t it2 = 0 ; it2 < table -> alphabet_length ; it2++ )
446 {
447 if( it2 != (unsigned)index && node -> children[ it2 ] )
448 {
449 /* child found, update branch index */
450 last_index = it + 1;
451 break;
452 }
453 }
454 /* Go to the next child in the sequence */
455 node = node->children[ index ];
456 }
457 }
458 return last_index;
459}

463char* _ht_find_longest_prefix_trie( HASH_TABLE *table, char *key )
464{
465 __n_assert( table, return NULL );
466 __n_assert( key, return NULL );
467
468 int len = strlen( key );
469
470 /* start with full key and backtrack to search for divergences */
471 char *longest_prefix = NULL ;
472 Malloc( longest_prefix, char, len + 1 );
473 memcpy( longest_prefix, key, len );
474
475 /* check branching from the root */
476 int branch_index = _ht_check_trie_divergence( table, longest_prefix ) - 1 ;
477 if(branch_index >= 0 )
478 {
479 /* there is branching, update the position to the longest match and update the longest prefix by the branch index length */
480 longest_prefix[ branch_index ] = '\0';
481 Realloc( longest_prefix, char, (branch_index + 1) );
482 __n_assert( longest_prefix, return NULL );
483 }
484 return longest_prefix;
485}

512 *\brief destroy a HASH_NODE by first calling the HASH_NODE destructor
513 *\param node The node to kill
514 */
515void _ht_node_destroy( void *node )
516{

518 __n_assert( node_ptr, return );
519 if( node_ptr -> type == HASH_STRING )
520 {

532 Free( node_ptr -> data . ptr );

538 {

540 {

553int _ht_remove_trie( HASH_TABLE *table, char *key )
554{
555 __n_assert( table, return FALSE );
556 __n_assert( table -> root, return FALSE );
557 __n_assert( key, return FALSE );
558
559 /* stop if matching node not a leaf node */
560 if( !_ht_is_leaf_node_trie( table, key ) )
561 {
562 return FALSE ;
563 }
564
565 HASH_NODE *node = table -> root ;
566 /* find the longest prefix string that is not the current key */
567 char *longest_prefix = _ht_find_longest_prefix_trie( table, key );
568 if( longest_prefix && longest_prefix[ 0 ] == '\0' )
569 {
570 Free( longest_prefix );
571 return FALSE ;
572 }
573 /* keep track of position in the tree */
574 uint32_t it = 0 ;
575 for( it = 0 ; longest_prefix[ it ] != '\0'; it++ )
576 {
577 int index = (int)longest_prefix[ it ] - table -> alphabet_offset ;
578 if( index < 0 || (unsigned)index >= table -> alphabet_length )
579 {
580 n_log( LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key );
581 index = 0 ;
582 }
583 if( node -> children[ index ] != NULL )
584 {
585 /* common prefix, keep moving */
586 node = node -> children[ index ];
587 }
588 else
589 {
590 /* not found */
591 Free( longest_prefix );
592 return FALSE ;
593 }
594 }
595 /* deepest common node between the two strings */
596 /* deleting the sequence corresponding to key */
597 for( ; key[ it ] != '\0' ; it++ )
598 {
599 int index = (int)key[ it ] - table -> alphabet_offset ;
600 if( index < 0 || (unsigned)index >= table -> alphabet_length )
601 {
602 n_log( LOG_ERR, "Invalid value %d for charater at position %d of %s, set to 0", index, it, key );
603 index = 0 ;
604 }
605 if( node -> children[ index] )
606 {
607 /* delete the remaining sequence */
608 HASH_NODE* node_to_kill = node -> children[ index ];
609 node -> children[ index ] = NULL;
610 _ht_node_destroy( node_to_kill );
611 }
612 }
613 Free( longest_prefix );
614 return TRUE ;
615}

878/*!\fn void MurmurHash3_x86_32 ( const void * key, int len, uint32_t seed, void * out )
879 * brief MurmurHash3 was written by Austin Appleby, and is placed in the public
880 domain. The author hereby disclaims copyright to this source code.
881 Note - The x86 and x64 versions do _not_ produce the same results, as the
882 algorithms are optimized for their respective platforms. You can still
883 compile and run any of them on any platform, but your performance with the
884 non-native version will be less than optimal.
885 *\param key char *string as the key
886 *\param len size of the key
887 *\param seed seed value for murmur hash
888 *\param out generated hash
889 */
890void MurmurHash3_x86_32 ( const void * key, int len, uint32_t seed, void * out )
891{
892 const uint8_t * data = (const uint8_t*)key;
893 const int nblocks = len / 4;
894
895 uint32_t h1 = seed;
896
897 uint32_t c1 = 0xcc9e2d51;
898 uint32_t c2 = 0x1b873593;
899
900 /* body */
901 const uint32_t * blocks = (const void *)(data + nblocks*4);
902
903 int i = 0 ;
904 for(i = -nblocks; i; i++)
905 {
906 uint32_t k1 = getblock(blocks,i);
907
908 k1 *= c1;
909 k1 = ROTL32(k1,15);
910 k1 *= c2;
911
912 h1 ^= k1;
913 h1 = ROTL32(h1,13);
914 h1 = h1*5+0xe6546b64;
915 }
916
917 /* tail */
918 const uint8_t * tail = (const uint8_t*)(data + nblocks*4);
919
920 uint32_t k1 = 0;
921
922 switch(len & 3)
923 {
924 case 3:
925 k1 ^= tail[2] << 16;
926 FALL_THROUGH ;
927 case 2:
928 k1 ^= tail[1] << 8;
929 FALL_THROUGH ;
930 case 1:
931 k1 ^= tail[0];
932 k1 *= c1;
933 k1 = ROTL32(k1,15);
934 k1 *= c2;
935 h1 ^= k1;
936 default:
937 break ;
938 };
939
940 /* finalization */
941 h1 ^= len;
942 h1 = fmix32(h1);
943 *(uint32_t*)out = h1;
944} /* MurmurHash3_x86_32(...) */
