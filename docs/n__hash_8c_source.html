103  const int nblocks = len / 4;
104

106

108  uint32_t c2 = 0x1b873593;
109 
110  /* body */
111  const uint32_t * blocks = (const uint32_t *)(data + nblocks*4);
112 
113  int i = 0 ;
114  for(i = -nblocks; i; i++)
115  {
116  uint32_t k1 = getblock(blocks,i);
117 
118  k1 *= c1;
119  k1 = ROTL32(k1,15);
120  k1 *= c2;

123  h1 = ROTL32(h1,13);
124  h1 = h1*5+0xe6546b64;
125  }
