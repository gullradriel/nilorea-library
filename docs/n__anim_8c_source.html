13/*!\fn ANIM_LIB *create_anim_library( char *name, unsigned int size )
14 *\brief Allocate an animation stream
15 *\param name Name for the stream or NULL (in which case 'generic_name' will be assigned)
16 *\param size Number of frames
17 *\return An allocated ANIM_LIB *object or NULL
18 */
19ANIM_LIB *create_anim_library( char *name, unsigned int size )
20{
21 if( size < 1 )
22 {
23 n_log( LOG_ERR, "Invalid size (<1) for anim lib creation" );
24 return NULL ;
25 }
26
27 ANIM_LIB *lib = NULL ;
28 Malloc( lib, ANIM_LIB, 1 );
29 __n_assert( lib, return NULL );
30
31 lib -> gfxs = NULL ;
32 Malloc( lib -> gfxs, ANIM_GFX *, size );
33 if( !lib -> gfxs )
34 {
35 n_log( LOG_ERR, "Unable to allocate a gfx_lib of size %d", size );
36 return NULL ;
37 }
38
39 if( name )
40 lib -> name = strdup( name );
41 else
42 lib -> name = strdup( "generic-name" );
43
44 lib -> nb_max_gfxs = size ;
45
46 return lib ;
47} /* create_anim_lib */

68/*!\fn int add_bmp_to_lib( ANIM_LIB *lib, unsigned int pos, char *file, char *resfile )
69 *\brief
70 *\param
71 *\return
72 */
73int add_bmp_to_lib( ANIM_LIB *lib, unsigned int pos, char *file, char *resfile )
74{
75 __n_assert( lib, return FALSE );
76 __n_assert( file, return FALSE );
77 __n_assert( resfile, return FALSE );
78
79 if( pos >= lib -> nb_max_gfxs )
80 {
81 n_log( LOG_ERR, "invalid position %d, can only go from 0 to %d in anim lib %s", pos, lib -> nb_max_gfxs, lib -> name );
82 return FALSE ;
83 }
84 if( lib -> gfxs[ pos ] != NULL )
85 {
86 n_log( LOG_ERR, "there already is a gfx at pos %d in anim lib %s", pos, lib -> name );
87 return FALSE ;
88 }
89
90 FILE *data = fopen( resfile,"r");
91 if( !data )
92 {
93 n_log( LOG_ERR, "Unable to open %s !", resfile );
94 }
95 int check_id ;
96 fscanf( data, "%d", &check_id );
97 if( check_id !=211282 )
98 {
99 n_log( LOG_ERR, "file %s: invalid check_id of %d, should be 211282", resfile, check_id );
100 fclose( data );
101 return FALSE ;
102 }
103
104 ANIM_GFX *gfx = NULL ;
105 Malloc( gfx, ANIM_GFX, 1 );
106
107 fscanf( data, "%d %d %d", &gfx -> w, &gfx -> h, &gfx -> nb_frames );
108
109 Malloc( gfx -> frames, ANIM_FRAME, gfx -> nb_frames );
110
111 gfx -> bmp = al_load_bitmap( file );
112 if( !gfx -> bmp )
113 {
114 n_log( LOG_ERR, "file %s: unable to load image", file );
115 fclose( data );
116 return FALSE ;
117 }
118
119 for( unsigned int it = 0 ; it < gfx -> nb_frames ; it ++ )
120 {
121 fscanf( data, "%d %d %d", &gfx -> frames[ it ].x, &gfx -> frames[ it ].y, &gfx -> frames[ it ] . duration );
122 }
123 fclose( data );
124
125 lib -> gfxs[ pos ] = gfx ;
126
127 return TRUE ;
128}

136int update_anim( ANIM_DATA *data, unsigned int delta_t )
137{
138 __n_assert( data, return FALSE );
139
140 data -> elapsed += delta_t ;
141 if( data -> elapsed > data -> lib -> gfxs[ data -> id ] -> frames[ data -> frame ].duration )
142 {
143 data -> elapsed = 0 ;
144 data -> frame ++ ;
145 if( data -> frame >= data -> lib -> gfxs[ data -> id ] -> nb_frames )
146 data -> frame = 0 ;
147 }
148 return TRUE ;
149}
