89/*!\fn time_t get_usec( N_TIME *timer )
90 *\brief Poll any N_TIME HiTimer, returning usec, and moving currentTime to startTime
91 *\param timer Any N_TIMER *timer you wanna poll
92 *\return The elapsed number of usec for the given N_TIME *timer
93 */
94time_t get_usec( N_TIME *timer )
95{
96#ifdef __windows__
97 QueryPerformanceCounter( ( LARGE_INTEGER * ) & timer -> currentTime );
98 timer -> delta = 1000000 * ( timer -> currentTime . QuadPart - timer -> startTime . QuadPart ) / timer -> freq . QuadPart ;
99 timer -> startTime = timer -> currentTime ;
100#else
101 gettimeofday( &timer -> currentTime, 0 );
102 timer -> delta = ( timer -> currentTime . tv_sec - timer -> startTime . tv_sec ) * 1000000 + ( timer -> currentTime . tv_usec - timer -> startTime . tv_usec );
103 timer -> startTime . tv_sec = timer -> currentTime . tv_sec ;
104 timer -> startTime . tv_usec = timer -> currentTime . tv_usec ;
105#endif
106
107 return timer -> delta;
108} /* get_usec( ... ) */
