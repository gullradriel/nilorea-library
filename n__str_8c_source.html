102char* trim_nocopy(char* s) {
103 __n_assert(s, return NULL);
104
105 if (strlen(s) == 0)
106 return s;
107
108 char* start = s;
109
110 /* skip spaces at start */
111 while (*start && isspace(*start))
112 start++;
113
114 char* end = s + strlen(s) - 1;
115 /* iterate over the rest remebering last non-whitespace */
116 while (*end && isspace(*end) && end > s) {
117 end--;
118 }
119 end++;
120 /* write the terminating zero after last non-whitespace */
121 *end = 0;
122
123 return start;

869char** split(const char* str, const char* delim, int empty) {
870 char** tab = NULL; /* result array */
871 char* ptr = NULL; /* tmp pointer */
872 size_t sizeStr = 0; /* string token size */
873 size_t sizeTab = 0; /* array size */

875
876 __n_assert(str, return NULL);
877 __n_assert(delim, return NULL);
878
879 size_t sizeDelim = strlen(delim);
880 largestring = str;
881
882 while ((ptr = strstr(largestring, delim)) != NULL) {
883 sizeStr = (size_t)(ptr - largestring);
884 if (empty == 1 || sizeStr != 0) {
885 sizeTab++;
886 tab = (char**)realloc(tab, sizeof(char*) * sizeTab);
887
888 Malloc(tab[sizeTab - 1], char, sizeof(char) * (sizeStr + 1));

893 }

896 }

900 sizeStr = strlen(largestring);
901 sizeTab++;
902

1196int wildmat(register const char* text, register const char* p) {
1197 register int last = 0;
1198 register int matched = 0;
1199 register int reverse = 0;
1200
1201 for (; *p; text++, p++) {
1202 if (*text == '\0' && *p != '*')
1203 return WILDMAT_ABORT;
1204 switch (*p) {
1205 case '\\':
1206 /* Literal match with following character. */
1207 p++;
1208 /* FALLTHROUGH */
1209 default:
1210 if (*text != *p)
1211 return FALSE;
1212 continue;
1213 case '?':
1214 /* Match anything. */
1215 continue;
1216 case '*':
1217 while (*++p == '*')
1218 /* Consecutive stars act just like one. */
1219 continue;
1220 if (*p == '\0')
1221 /* Trailing star matches everything. */
1222 return TRUE;
1223 while (*text)
1224 if ((matched = wildmat(text++, p)) != FALSE)
1225 return matched;
1226 return WILDMAT_ABORT;
1227 case '[':
1228 reverse = p[1] == WILDMAT_NEGATE_CLASS ? TRUE : FALSE;
1229 if (reverse)
1230 /* Inverted character class. */
1231 p++;
1232 matched = FALSE;
1233 if (p[1] == ']' || p[1] == '-')
1234 if (*++p == *text)
1235 matched = TRUE;
1236 for (last = *p; *++p && *p != ']'; last = *p)
1237 /* This next line requires a good C compiler. */
1238 if (*p == '-' && p[1] != ']'
1239 ? *text <= *++p && *text >= last
1240 : *text == *p)
1241 matched = TRUE;
1242 if (matched == reverse)
1243 return FALSE;
1244 continue;
1245 }
1246 }
1247#ifdef WILDMAT_MATCH_TAR_PATTERN
1248 if (*text == '/')
1249 return TRUE;
1250#endif /* MATCH_TAR_ATTERN */
1251 return *text == '\0';
1252} /* wildmatch(...) */

1326char* str_replace(const char* string, const char* substr, const char* replacement) {
1327 char* tok = NULL;
1328 char* newstr = NULL;
1329 char* oldstr = NULL;
1330 char* head = NULL;
1331
1332 /* if either substr or replacement is NULL, duplicate string a let caller handle it */
1333 if (substr == NULL || replacement == NULL)
1334 return strdup(string);
1335 newstr = strdup(string);
1336 head = newstr;
1337 while ((tok = strstr(head, substr))) {
1338 oldstr = newstr;
1339 Malloc(newstr, char, strlen(oldstr) - strlen(substr) + strlen(replacement) + 8);
1340 /*failed to alloc mem, free old string and return NULL */
1341 if (newstr == NULL) {
1342 free(oldstr);
1343 return NULL;
1344 }
1345 memcpy(newstr, oldstr, (size_t)(tok - oldstr));
1346 memcpy(newstr + (tok - oldstr), replacement, strlen(replacement));
1347 memcpy(newstr + (tok - oldstr) + strlen(replacement), tok + strlen(substr), strlen(oldstr) - strlen(substr) - (size_t)(tok - oldstr));
1348 memset(newstr + strlen(oldstr) - strlen(substr) + strlen(replacement), 0, 1);
1349 /* move back head right after the last replacement */
1350 head = newstr + (tok - oldstr) + strlen(replacement);
1351 free(oldstr);
1352 }
