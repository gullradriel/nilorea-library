121char* trim_nocopy(char* s) {
122 __n_assert(s, return NULL);
123
124 if (strlen(s) == 0)
125 return s;
126
127 char* start = s;
128
129 /* skip spaces at start */
130 while (*start && isspace((unsigned char)*start))
131 start++;
132
133 char* end = s + strlen(s) - 1;
134 /* iterate over the rest remebering last non-whitespace */
135 while (*end && isspace((unsigned char)*end) && end > s) {
136 end--;
137 }
138 end++;
139 /* write the terminating zero after last non-whitespace */
140 *end = 0;
141
142 return start;
143} /* trim_nocopy */

285N_STR* file_to_nstr(char* filename) {
286 N_STR* tmpstr = NULL;
287 struct stat filestat;
288 FILE* in = NULL;
289 int error = 0;
290
291 __n_assert(filename, n_log(LOG_ERR, "Unable to make a string from NULL filename"); return NULL);
292
293 int fd = open(filename, O_RDONLY);
294 error = errno;
295 if (fd == -1) {
296 n_log(LOG_ERR, "Unable to open %s for reading. Errno: %s", filename, strerror(error));
297 return NULL;
298 }
299
300 if (fstat(fd, &filestat) != 0) {
301 error = errno;
302#ifdef __linux__
303 if (error == EOVERFLOW) {
304 n_log(LOG_ERR, "%s size is too big ,EOVERFLOW)", filename);
305 } else {
306#endif
307 n_log(LOG_ERR, "Couldn't stat %s. Errno: %s", filename, strerror(error));
308#ifdef __linux__
309 }
310#endif
311 close(fd);
312 return NULL;
313 }
314
315 if ((uint64_t)filestat.st_size >= (uint64_t)UINT32_MAX) {
316 n_log(LOG_ERR, "file size >= 4GB is not possible yet. %s is %lld oct", filename, (long long)filestat.st_size);
317 close(fd);
318 return NULL;
319 }
320
321 n_log(LOG_DEBUG, "%s file size is: %lld", filename, (long long)filestat.st_size);
322
323 in = fdopen(fd, "rb");
324 if (!in) {
325 n_log(LOG_ERR, "fdopen failed on %s (fd=%d)", filename, fd);
326 close(fd);
327 return NULL;
328 }
329
330 tmpstr = new_nstr((size_t)filestat.st_size + 1);
331 __n_assert(tmpstr, fclose(in); n_log(LOG_ERR, "Unable to get a new nstr of %ld octets", filestat.st_size + 1); return NULL);
332
333 tmpstr->written = (size_t)filestat.st_size;
334
335 if (fread(tmpstr->data, sizeof(char), tmpstr->written, in) == 0) {
336 n_log(LOG_ERR, "Couldn't read %s, fread return 0", filename);
337 free_nstr(&tmpstr);
338 fclose(in);
339 return NULL;
340 }
341 if (ferror(in)) {
342 n_log(LOG_ERR, "There were some errors when reading %s", filename);
343 free_nstr(&tmpstr);
344 fclose(in);
345 return NULL;
346 }
347
348 fclose(in);
349
350 return tmpstr;
351} /*file_to_nstr */

362int nstr_to_fd(N_STR* str, FILE* out, int lock) {
363#ifdef __linux__
364 struct flock out_lock;
365#endif
366 __n_assert(out, return FALSE);
367 __n_assert(str, return FALSE);
368
369 int ret = TRUE;
370
371 if (lock == 1) {
372#ifdef __linux__
373 memset(&out_lock, 0, sizeof(out_lock));
374 out_lock.l_type = F_WRLCK;
375 /* Place a write lock on the file. */
376 fcntl(fileno(out), F_SETLKW, &out_lock);
377#else
378 lock = 2; /* compiler warning suppressor */
379#endif
380 }
381
382 size_t written_to_file = 0;
383 if ((written_to_file = fwrite(str->data, sizeof(char), str->written, out)) != (size_t)str->written) {
384 n_log(LOG_ERR, "Couldn't write file, fwrite %d of %d octets", written_to_file, str->written);
385 ret = FALSE;
386 }
387 if (ferror(out)) {
388 n_log(LOG_ERR, "There were some errors when writing to %d", fileno(out));
389 ret = FALSE;
390 }
391
392 if (lock == 1) {
393#ifdef __linux__
394 memset(&out_lock, 0, sizeof(out_lock));
395 out_lock.l_type = F_WRLCK;
396 /* Place a write lock on the file. */
397 fcntl(fileno(out), F_SETLKW, &out_lock);
398#else
399 lock = 2; /* compiler warning suppressor */
400#endif
401 }
402
403 return ret;
404} /* nstr_to_fd( ... ) */

450int str_to_int_ex(const char* s, NSTRBYTE start, NSTRBYTE end, int* i, const int base) {
451 char* tmpstr = NULL;
452 char* endstr = NULL;
453 long int l = 0;
454
455 __n_assert(s, return FALSE);
456
457 if (start > end) return FALSE;
458
459 Malloc(tmpstr, char, sizeof(int) + end - start + 8);
460 __n_assert(tmpstr, n_log(LOG_ERR, "Unable to Malloc( tmpstr , char , sizeof( int ) + %d - %d )", end, start); return FALSE);
461
462 memcpy(tmpstr, s + start, end - start);
463
464 errno = 0;
465 l = strtol(tmpstr, &endstr, base);
466 if ((errno == ERANGE && l == LONG_MAX) || l > INT_MAX) {
467 n_log(LOG_ERR, "OVERFLOW reached when converting %s to int", tmpstr);
468 Free(tmpstr);
469 return FALSE;
470 }
471 if ((errno == ERANGE && l == LONG_MIN) || l < INT_MIN) {
472 n_log(LOG_ERR, "UNDERFLOW reached when converting %s to int", tmpstr);
473 Free(tmpstr);
474 return FALSE;
475 }
476 if (*endstr != '\0' && *endstr != '\n') {
477 n_log(LOG_ERR, "Impossible conversion for %s", tmpstr);
478 Free(tmpstr);
479 return FALSE;
480 }
481 Free(tmpstr);
482 *i = (int)l;
483 return TRUE;
484} /* str_to_int_ex( ... ) */

560int str_to_long_ex(const char* s, NSTRBYTE start, NSTRBYTE end, long int* i, const int base) {
561 char* tmpstr = NULL;
562 char* endstr = NULL;
563 long l = 0;
564
565 __n_assert(s, return FALSE);
566
567 if (start > end) return FALSE;
568
569 Malloc(tmpstr, char, sizeof(int) + end - start + 8);
570 __n_assert(tmpstr, n_log(LOG_ERR, "Unable to Malloc( tmpstr , char , sizeof( int ) + %d - %d )", end, start); return FALSE);
571
572 memcpy(tmpstr, s + start, end - start);
573
574 errno = 0;
575 l = strtol(tmpstr, &endstr, base);
576 int error = errno;
577
578 /* test return to number and errno values */
579 if (tmpstr == endstr) {
580 n_log(LOG_ERR, " number : %ld invalid (no digits found, 0 returned)", l);
581 Free(tmpstr);
582 return FALSE;
583 } else if (error == ERANGE && l == LONG_MIN) {
584 n_log(LOG_ERR, " number : %ld invalid (underflow occurred)", l);
585 Free(tmpstr);
586 return FALSE;
587 } else if (error == ERANGE && l == LONG_MAX) {
588 n_log(LOG_ERR, " number : %ld invalid (overflow occurred)", l);
589 Free(tmpstr);
590 return FALSE;
591 } else if (error == EINVAL) /* not in all c99 implementations - gcc OK */
592 {
593 n_log(LOG_ERR, " number : %ld invalid (base contains unsupported value)", l);
594 Free(tmpstr);
595 return FALSE;
596 } else if (error != 0 && l == 0) {
597 n_log(LOG_ERR, " number : %ld invalid (unspecified error occurred)", l);
598 Free(tmpstr);
599 return FALSE;
600 } else if (error == 0 && tmpstr && !*endstr) {
601 n_log(LOG_DEBUG, " number : %ld valid (and represents all characters read)", l);
602 } else if (error == 0 && tmpstr && *endstr != 0) {
603 n_log(LOG_DEBUG, " number : %ld valid (but additional characters remain", l);
604 }
605 Free(tmpstr);
606 *i = l;
607 return TRUE;

735int skipw(char* string, char toskip, NSTRBYTE* iterator, int inc) {
736 int error_flag = 0;
737
738 __n_assert(string, return FALSE);
739
740 NSTRBYTE slen = (NSTRBYTE)strlen(string);
741 if (toskip == ' ') {
742 while (*iterator <= slen && isspace((unsigned char)string[*iterator])) {
743 if (inc < 0 && *iterator == 0) {
744 error_flag = 1;
745 break;
746 } else {
747 if (inc > 0)
748 *iterator = *iterator + (NSTRBYTE)inc;
749 else
750 *iterator = (*iterator > (NSTRBYTE)(-inc)) ? (*iterator - (NSTRBYTE)(-inc)) : 0;
751 }
752 }
753 } else {
754 while (*iterator <= slen && string[*iterator] == toskip) {
755 if (inc < 0 && *iterator == 0) {
756 error_flag = 1;
757 break;
758 } else {
759 if (inc > 0)
760 *iterator = *iterator + (NSTRBYTE)inc;
761 else
762 *iterator = (*iterator > (NSTRBYTE)(-inc)) ? (*iterator - (NSTRBYTE)(-inc)) : 0;
763 }
764 }
765 }
766 if (error_flag == 1 || *iterator > slen) {
767 return FALSE;
768 }
769
770 return TRUE;

872int strcpy_u(char* from, char* to, NSTRBYTE to_size, char split, NSTRBYTE* it) {
873 NSTRBYTE _it = 0;
874
875 __n_assert(from, return FALSE);
876 __n_assert(to, return FALSE);
877 __n_assert(it, return FALSE);
878
879 while (_it < to_size && from[(*it)] != '\0' && from[(*it)] != split) {
880 to[_it] = from[(*it)];

893
894 if (from[(*it)] != split) {

908char** split(const char* str, const char* delim, int empty) {

1284int wildmat(register const char* text, register const char* p) {
1285 register int last = 0;
1286 register int matched = 0;
1287 register int reverse = 0;
1288
1289 for (; *p; text++, p++) {
1290 if (*text == '\0' && *p != '*')
1291 return WILDMAT_ABORT;
1292 switch (*p) {
1293 case '\\':
1294 /* Literal match with following character. */
1295 p++;
1296 /* FALLTHROUGH */
1297 default:
1298 if (*text != *p)
1299 return FALSE;
1300 continue;
1301 case '?':
1302 /* Match anything. */
1303 continue;
1304 case '*':
1305 while (*++p == '*')
1306 /* Consecutive stars act just like one. */
1307 continue;
1308 if (*p == '\0')
1309 /* Trailing star matches everything. */
1310 return TRUE;
1311 while (*text)
1312 if ((matched = wildmat(text++, p)) != FALSE)
1313 return matched;
1314 return WILDMAT_ABORT;
1315 case '[':
1316 reverse = p[1] == WILDMAT_NEGATE_CLASS ? TRUE : FALSE;
1317 if (reverse)
1318 /* Inverted character class. */
1319 p++;
1320 matched = FALSE;
1321 if (p[1] == ']' || p[1] == '-')
1322 if (*++p == *text)
1323 matched = TRUE;
1324 for (last = *p; *++p && *p != ']'; last = *p)
1325 /* This next line requires a good C compiler. */
1326 if (*p == '-' && p[1] != ']'
1327 ? *text <= *++p && *text >= last
1328 : *text == *p)
1329 matched = TRUE;
1330 if (matched == reverse)
1331 return FALSE;
1332 continue;
1333 }
1334 }
1335#ifdef WILDMAT_MATCH_TAR_PATTERN
1336 if (*text == '/')
1337 return TRUE;
1338#endif /* MATCH_TAR_ATTERN */
1339 return *text == '\0';
1340} /* wildmatch(...) */
