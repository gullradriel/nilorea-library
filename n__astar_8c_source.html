43static inline int grid_index(const ASTAR_GRID* grid, int x, int y, int z) {
44 return (z * grid->height + y) * grid->width + x;
45}

48static inline int in_bounds(const ASTAR_GRID* grid, int x, int y, int z) {
49 return x >= 0 && x < grid->width &&
50 y >= 0 && y < grid->height &&
51 z >= 0 && z < grid->depth;
52}

60typedef struct ASTAR_BOX {
61 int minx, miny, minz;
62 int bw, bh, bd;
63} ASTAR_BOX;
64
66static inline int box_index(const ASTAR_BOX* b, int x, int y, int z) {
67 return ((z - b->minz) * b->bh + (y - b->miny)) * b->bw + (x - b->minx);
68}

71static inline int in_box(const ASTAR_BOX* b, int x, int y, int z) {
72 return x >= b->minx && x < b->minx + b->bw &&
73 y >= b->miny && y < b->miny + b->bh &&
74 z >= b->minz && z < b->minz + b->bd;
75}

82/* Binary Min-Heap (priority queue) */
83
84static ASTAR_HEAP* heap_new(int capacity) {
85 ASTAR_HEAP* h = (ASTAR_HEAP*)calloc(1, sizeof(ASTAR_HEAP));
86 if (!h) return NULL;
87
88 h->data = (ASTAR_HEAP_NODE*)malloc((size_t)capacity * sizeof(ASTAR_HEAP_NODE));
89 if (!h->data) {
90 free(h);
91 return NULL;
92 }
93 h->size = 0;
94 h->capacity = capacity;
95 return h;
96}
97
98static void heap_free(ASTAR_HEAP* h) {
99 if (!h) return;
100 free(h->data);
101 free(h);
102}
103
104static int heap_grow(ASTAR_HEAP* h) {
105 int new_cap = h->capacity * 2;
106 ASTAR_HEAP_NODE* new_data = (ASTAR_HEAP_NODE*)realloc(
107 h->data, (size_t)new_cap * sizeof(ASTAR_HEAP_NODE));
108 if (!new_data) return 0;
109 h->data = new_data;
110 h->capacity = new_cap;
111 return 1;
112}
113
114static void heap_swap(ASTAR_HEAP_NODE* a, ASTAR_HEAP_NODE* b) {
115 ASTAR_HEAP_NODE tmp = *a;
116 *a = *b;
117 *b = tmp;
118}
119
120static void heap_push(ASTAR_HEAP* h, int x, int y, int z, int f) {
121 if (h->size >= h->capacity) {
122 if (!heap_grow(h)) return;
123 }
124
125 int i = h->size++;
126 h->data[i].x = x;
127 h->data[i].y = y;
128 h->data[i].z = z;
129 h->data[i].f = f;
130
131 /* Sift up */
132 while (i > 0) {
133 int parent = (i - 1) / 2;
134 if (h->data[i].f < h->data[parent].f) {
135 heap_swap(&h->data[i], &h->data[parent]);
136 i = parent;
137 } else {
138 break;
139 }
140 }
141}
142
143static ASTAR_HEAP_NODE heap_pop(ASTAR_HEAP* h) {
144 ASTAR_HEAP_NODE top = h->data[0];
145 h->data[0] = h->data[--h->size];
146
147 /* Sift down */
148 int i = 0;
149 for (;;) {
150 int left = 2 * i + 1;
151 int right = 2 * i + 2;
152 int smallest = i;
153
154 if (left < h->size && h->data[left].f < h->data[smallest].f)
155 smallest = left;
156 if (right < h->size && h->data[right].f < h->data[smallest].f)
157 smallest = right;
158
159 if (smallest != i) {
160 heap_swap(&h->data[i], &h->data[smallest]);
161 i = smallest;
162 } else {
163 break;
164 }
165 }
166
167 return top;
168}

183int n_astar_heuristic(int x1, int y1, int z1, int x2, int y2, int z2, ASTAR_HEURISTIC heuristic) {
184 int dx = iabs(x2 - x1);
185 int dy = iabs(y2 - y1);
186 int dz = iabs(z2 - z1);
187
188 switch (heuristic) {
189 case ASTAR_HEURISTIC_MANHATTAN:
190 return (dx + dy + dz) * ASTAR_COST_CARDINAL;
191
192 case ASTAR_HEURISTIC_EUCLIDEAN: {
193 double dist = sqrt((double)(dx * dx + dy * dy + dz * dz));
194 return (int)(dist * ASTAR_COST_CARDINAL);
195 }
196
197 case ASTAR_HEURISTIC_CHEBYSHEV:
198 default: {
199 int vals[3] = {dx, dy, dz};
200 /* Sort ascending */
201 if (vals[0] > vals[1]) {
202 int t = vals[0];
203 vals[0] = vals[1];
204 vals[1] = t;
205 }
206 if (vals[1] > vals[2]) {
207 int t = vals[1];
208 vals[1] = vals[2];
209 vals[2] = t;
210 }
211 if (vals[0] > vals[1]) {
212 int t = vals[0];
213 vals[0] = vals[1];
214 vals[1] = t;
215 }
216
217 int cost = vals[2] * ASTAR_COST_CARDINAL;
218 cost += vals[0] * (ASTAR_COST_DIAGONAL3D - ASTAR_COST_CARDINAL);
219 cost += (vals[1] - vals[0]) * (ASTAR_COST_DIAGONAL - ASTAR_COST_CARDINAL);
220 return cost;
221 }
222 }

234ASTAR_GRID* n_astar_grid_new(int width, int height, int depth) {
235 if (width <= 0 || height <= 0 || depth <= 0) return NULL;
236
237 ASTAR_GRID* grid = (ASTAR_GRID*)calloc(1, sizeof(ASTAR_GRID));
238 if (!grid) return NULL;
239
240 grid->width = width;
241 grid->height = height;
242 grid->depth = depth;
243
244 size_t total = (size_t)width * (size_t)height * (size_t)depth;
245
246 grid->walkable = (uint8_t*)malloc(total);
247 if (!grid->walkable) {
248 free(grid);
249 return NULL;
250 }
251 memset(grid->walkable, 1, total);
252
253 grid->cost = (int*)malloc(total * sizeof(int));
254 if (!grid->cost) {
255 free(grid->walkable);
256 free(grid);
257 return NULL;
258 }
259 for (size_t i = 0; i < total; i++) {
260 grid->cost[i] = ASTAR_COST_CARDINAL;
261 }
262
263 return grid;
264}

270void n_astar_grid_free(ASTAR_GRID* grid) {
271 if (!grid) return;
272 free(grid->walkable);
273 free(grid->cost);
274 free(grid);
275}

372static ASTAR_PATH* reconstruct_path(ASTAR_CELL* cells, const ASTAR_BOX* box, int sx, int sy, int sz, int gx, int gy, int gz) {
373 /* Count path length */
374 int length = 0;
375 int cx = gx, cy = gy, cz = gz;
376 while (cx != sx || cy != sy || cz != sz) {
377 length++;
378 int idx = box_index(box, cx, cy, cz);
379 int px = cells[idx].parent_x;
380 int py = cells[idx].parent_y;
381 int pz = cells[idx].parent_z;
382 cx = px;
383 cy = py;
384 cz = pz;
385 if (length > box->bw * box->bh * box->bd) {
386 return NULL;
387 }
388 }
389 length++;
390
391 ASTAR_PATH* path = (ASTAR_PATH*)calloc(1, sizeof(ASTAR_PATH));
392 if (!path) return NULL;
393
394 path->nodes = (ASTAR_NODE*)malloc((size_t)length * sizeof(ASTAR_NODE));
395 if (!path->nodes) {
396 free(path);
397 return NULL;
398 }
399 path->length = length;
400 path->cost = cells[box_index(box, gx, gy, gz)].g;
401
402 int i = length - 1;
403 cx = gx;
404 cy = gy;
405 cz = gz;
406 while (i >= 0) {
407 path->nodes[i].x = cx;
408 path->nodes[i].y = cy;
409 path->nodes[i].z = cz;
410 if (cx == sx && cy == sy && cz == sz) break;
411 int idx = box_index(box, cx, cy, cz);
412 int px = cells[idx].parent_x;
413 int py = cells[idx].parent_y;
414 int pz = cells[idx].parent_z;
415 cx = px;
416 cy = py;
417 cz = pz;
418 i--;
419 }
420
421 return path;
422}

468/* A* Core Algorithm */
469
484ASTAR_PATH* n_astar_find_path(const ASTAR_GRID* grid,
485 int sx,
486 int sy,
487 int sz,
488 int gx,
489 int gy,
490 int gz,
491 int diagonal,
492 ASTAR_HEURISTIC heuristic) {
493 if (!grid) return NULL;
494 if (!in_bounds(grid, sx, sy, sz) || !in_bounds(grid, gx, gy, gz)) return NULL;
495 if (!grid->walkable[grid_index(grid, sx, sy, sz)]) return NULL;
496 if (!grid->walkable[grid_index(grid, gx, gy, gz)]) return NULL;
497
498 /* Trivial case */
499 if (sx == gx && sy == gy && sz == gz) {
500 ASTAR_PATH* path = (ASTAR_PATH*)calloc(1, sizeof(ASTAR_PATH));
501 if (!path) return NULL;
502 path->nodes = (ASTAR_NODE*)malloc(sizeof(ASTAR_NODE));
503 if (!path->nodes) {
504 free(path);
505 return NULL;
506 }
507 path->nodes[0].x = sx;
508 path->nodes[0].y = sy;
509 path->nodes[0].z = sz;
510 path->length = 1;
511 path->cost = 0;
512 return path;
513 }
514
515 int is_3d = (grid->depth > 1);
516
517 /* Window the per-call working set to the bounding box of (start, goal)
518 * expanded by ASTAR_SEARCH_MARGIN and clamped to the grid, instead of
519 * allocating one ASTAR_CELL per map tile. A 2400x2400 grid is 5.76M
520 * cells (~161 MB) per calloc/free — fired on every chase repath, this
521 * was the cause of multi-hundred-ms tick spikes. The grid's persistent
522 * walkable[] / cost[] planes stay full-size and keep using the absolute
523 * grid_index(); only the transient `cells` scratch is windowed, indexed
524 * by box_index(). So per-call cost scales with the path length, not the
525 * map size: a few KB for a short path. Trade-off: an optimal route that
526 * detours more than ASTAR_SEARCH_MARGIN tiles outside the start-goal box
527 * is not found through this window (returns NULL). For bounded game
528 * paths that never happens; raise ASTAR_SEARCH_MARGIN if a caller needs
529 * wider detours.
530 *
531 * calloc keeps zero-initialised cells (ASTAR_NODE_NONE == 0 == the
532 * "unvisited" state); the search only touches the handful of cells it
533 * explores, and reconstruct_path walks parents it has already written. */
534 ASTAR_BOX box;
535 {
536 int lo = (sx < gx) ? sx : gx, hi = (sx > gx) ? sx : gx;
537 box.minx = lo - ASTAR_SEARCH_MARGIN; if (box.minx < 0) box.minx = 0;
538 hi += ASTAR_SEARCH_MARGIN; if (hi > grid->width - 1) hi = grid->width - 1;
539 box.bw = hi - box.minx + 1;
540
541 lo = (sy < gy) ? sy : gy; hi = (sy > gy) ? sy : gy;
542 box.miny = lo - ASTAR_SEARCH_MARGIN; if (box.miny < 0) box.miny = 0;
543 hi += ASTAR_SEARCH_MARGIN; if (hi > grid->height - 1) hi = grid->height - 1;
544 box.bh = hi - box.miny + 1;
545
546 if (is_3d) {
547 lo = (sz < gz) ? sz : gz; hi = (sz > gz) ? sz : gz;
548 box.minz = lo - ASTAR_SEARCH_MARGIN; if (box.minz < 0) box.minz = 0;
549 hi += ASTAR_SEARCH_MARGIN; if (hi > grid->depth - 1) hi = grid->depth - 1;
550 box.bd = hi - box.minz + 1;
551 } else {
552 box.minz = 0;
553 box.bd = 1;
554 }
555 }
556
557 size_t btotal = (size_t)box.bw * (size_t)box.bh * (size_t)box.bd;
558 ASTAR_CELL* cells = (ASTAR_CELL*)calloc(btotal, sizeof(ASTAR_CELL));
559 if (!cells) return NULL;
560
561 ASTAR_HEAP* open = heap_new(256);
562 if (!open) {
563 free(cells);
564 return NULL;
565 }
566
567 int si = box_index(&box, sx, sy, sz);
568 cells[si].g = 0;
569 cells[si].h = n_astar_heuristic(sx, sy, sz, gx, gy, gz, heuristic);
570 cells[si].f = cells[si].h;
571 cells[si].status = ASTAR_NODE_OPEN;
572 heap_push(open, sx, sy, sz, cells[si].f);
573
574 ASTAR_PATH* result = NULL;
575
576 while (open->size > 0) {
577 ASTAR_HEAP_NODE current = heap_pop(open);
578 int cx = current.x, cy = current.y, cz = current.z;
579 int ci = grid_index(grid, cx, cy, cz); /* absolute: cost[] */
580 int lci = box_index(&box, cx, cy, cz); /* window-local: cells[] */
581
582 if (cells[lci].status == ASTAR_NODE_CLOSED)
583 continue;
584
585 if (cx == gx && cy == gy && cz == gz) {
586 result = reconstruct_path(cells, &box, sx, sy, sz, gx, gy, gz);
587 break;
588 }
589
590 cells[lci].status = ASTAR_NODE_CLOSED;
591
592 if (is_3d) {
593 /* 3D: 6 cardinal directions */
594 for (int d = 0; d < 6; d++) {
595 int nx = cx + dir3d_cardinal[d][0];
596 int ny = cy + dir3d_cardinal[d][1];
597 int nz = cz + dir3d_cardinal[d][2];
598 if (!in_box(&box, nx, ny, nz)) continue;
599
600 int ni = grid_index(grid, nx, ny, nz);
601 int lni = box_index(&box, nx, ny, nz);
602 if (!grid->walkable[ni] || cells[lni].status == ASTAR_NODE_CLOSED)
603 continue;
604
605 int move_cost = (grid->cost[ci] + grid->cost[ni]) / 2;
606 int tentative_g = cells[lci].g + move_cost;
607
608 if (cells[lni].status == ASTAR_NODE_NONE || tentative_g < cells[lni].g) {
609 cells[lni].g = tentative_g;
610 cells[lni].h = n_astar_heuristic(nx, ny, nz, gx, gy, gz, heuristic);
611 cells[lni].f = tentative_g + cells[lni].h;
612 cells[lni].parent_x = cx;
613 cells[lni].parent_y = cy;
614 cells[lni].parent_z = cz;
615 cells[lni].status = ASTAR_NODE_OPEN;
616 heap_push(open, nx, ny, nz, cells[lni].f);
617 }
618 }
619
620 /* 3D: 20 diagonal directions */
621 if (diagonal) {
622 for (int d = 0; d < 20; d++) {
623 int ddx = dir3d_diagonal[d][0];
624 int ddy = dir3d_diagonal[d][1];
625 int ddz = dir3d_diagonal[d][2];
626 int nx = cx + ddx;
627 int ny = cy + ddy;
628 int nz = cz + ddz;
629 if (!in_box(&box, nx, ny, nz)) continue;
630
631 int ni = grid_index(grid, nx, ny, nz);
632 int lni = box_index(&box, nx, ny, nz);
633 if (!grid->walkable[ni] || cells[lni].status == ASTAR_NODE_CLOSED)
634 continue;
635
636 /* Corner-cutting check */
637 int blocked = 0;
638 if (ddx != 0 && ddy != 0 && !grid->walkable[grid_index(grid, cx + ddx, cy, cz)])
639 blocked = 1;
640 if (ddx != 0 && ddy != 0 && !grid->walkable[grid_index(grid, cx, cy + ddy, cz)])
641 blocked = 1;
642 if (ddx != 0 && ddz != 0 && !grid->walkable[grid_index(grid, cx + ddx, cy, cz)])
643 blocked = 1;
644 if (ddx != 0 && ddz != 0 && !grid->walkable[grid_index(grid, cx, cy, cz + ddz)])
645 blocked = 1;
646 if (ddy != 0 && ddz != 0 && !grid->walkable[grid_index(grid, cx, cy + ddy, cz)])
647 blocked = 1;
648 if (ddy != 0 && ddz != 0 && !grid->walkable[grid_index(grid, cx, cy, cz + ddz)])
649 blocked = 1;
650 if (blocked) continue;
651
652 int axes = (ddx != 0) + (ddy != 0) + (ddz != 0);
653 int base_cost;
654 if (axes == 3)
655 base_cost = ASTAR_COST_DIAGONAL3D;
656 else
657 base_cost = ASTAR_COST_DIAGONAL;
658
659 int cell_cost = (grid->cost[ci] + grid->cost[ni]) / 2;
660 int move_cost = (base_cost * cell_cost) / ASTAR_COST_CARDINAL;
661 int tentative_g = cells[lci].g + move_cost;
662
663 if (cells[lni].status == ASTAR_NODE_NONE || tentative_g < cells[lni].g) {
664 cells[lni].g = tentative_g;
665 cells[lni].h = n_astar_heuristic(nx, ny, nz, gx, gy, gz, heuristic);
666 cells[lni].f = tentative_g + cells[lni].h;
667 cells[lni].parent_x = cx;
668 cells[lni].parent_y = cy;
669 cells[lni].parent_z = cz;
670 cells[lni].status = ASTAR_NODE_OPEN;
671 heap_push(open, nx, ny, nz, cells[lni].f);
672 }
673 }
674 }
675 } else {
676 /* 2D: 4 cardinal directions */
677 for (int d = 0; d < 4; d++) {
678 int nx = cx + dir2d_cardinal[d][0];
679 int ny = cy + dir2d_cardinal[d][1];
680 if (!in_box(&box, nx, ny, 0)) continue;
681
682 int ni = grid_index(grid, nx, ny, 0);
683 int lni = box_index(&box, nx, ny, 0);
684 if (!grid->walkable[ni] || cells[lni].status == ASTAR_NODE_CLOSED)
685 continue;
686
687 int move_cost = (grid->cost[ci] + grid->cost[ni]) / 2;
688 int tentative_g = cells[lci].g + move_cost;
689
690 if (cells[lni].status == ASTAR_NODE_NONE || tentative_g < cells[lni].g) {
691 cells[lni].g = tentative_g;
692 cells[lni].h = n_astar_heuristic(nx, ny, 0, gx, gy, 0, heuristic);
693 cells[lni].f = tentative_g + cells[lni].h;
694 cells[lni].parent_x = cx;
695 cells[lni].parent_y = cy;
696 cells[lni].parent_z = 0;
697 cells[lni].status = ASTAR_NODE_OPEN;
698 heap_push(open, nx, ny, 0, cells[lni].f);
699 }
700 }
701
702 /* 2D: 4 diagonal directions */
703 if (diagonal) {
704 for (int d = 0; d < 4; d++) {
705 int ddx = dir2d_diagonal[d][0];
706 int ddy = dir2d_diagonal[d][1];
707 int nx = cx + ddx;
708 int ny = cy + ddy;
709 if (!in_box(&box, nx, ny, 0)) continue;
710
711 int ni = grid_index(grid, nx, ny, 0);
712 int lni = box_index(&box, nx, ny, 0);
713 if (!grid->walkable[ni] || cells[lni].status == ASTAR_NODE_CLOSED)
714 continue;
715
716 if (!grid->walkable[grid_index(grid, cx + ddx, cy, 0)] ||
717 !grid->walkable[grid_index(grid, cx, cy + ddy, 0)])
718 continue;
719
720 int cell_cost = (grid->cost[ci] + grid->cost[ni]) / 2;
721 int move_cost = (ASTAR_COST_DIAGONAL * cell_cost) / ASTAR_COST_CARDINAL;
722 int tentative_g = cells[lci].g + move_cost;
723
724 if (cells[lni].status == ASTAR_NODE_NONE || tentative_g < cells[lni].g) {
725 cells[lni].g = tentative_g;
726 cells[lni].h = n_astar_heuristic(nx, ny, 0, gx, gy, 0, heuristic);
727 cells[lni].f = tentative_g + cells[lni].h;
728 cells[lni].parent_x = cx;
729 cells[lni].parent_y = cy;
730 cells[lni].parent_z = 0;
731 cells[lni].status = ASTAR_NODE_OPEN;
732 heap_push(open, nx, ny, 0, cells[lni].f);
