61N_GUI_TOKEN n_gui_next_token(N_GUI_TOKENIZER* tokenizer, bool* inside_open_tag) {
62 N_GUI_TOKEN token = {.type = N_GUI_TOKEN_EOF, .value = {0}};
63 char ch;
64
65 // Skip whitespace
66 while (isspace(n_gui_peek_char(tokenizer))) {
67 n_gui_next_char(tokenizer);
68 }
69
70 ch = n_gui_peek_char(tokenizer);
71
72 // Handle end of file
73 if (ch == '\0') {
74 token.type = N_GUI_TOKEN_EOF;
75 return token;
76 }
77
78 // Handle opening tag '<'
79 if (ch == '<') {
80 n_gui_next_char(tokenizer);
81 ch = n_gui_peek_char(tokenizer);
82 if (ch == '/') {
83 n_gui_next_char(tokenizer);
84 *inside_open_tag = false;
85 token.type = N_GUI_TOKEN_TAG_CLOSE;
86 strncpy(token.value, "/", N_GUI_MAX_TOKEN_SIZE - 1);
87 } else {
88 *inside_open_tag = true;
89 token.type = N_GUI_TOKEN_TAG_OPEN;
90 strncpy(token.value, "<", N_GUI_MAX_TOKEN_SIZE - 1);
91 }
92 return token;
93 }
94
95 // Handle closing tag '>'
96 if (ch == '>') {
97 n_gui_next_char(tokenizer);
98 *inside_open_tag = false;
99 token.type = N_GUI_TOKEN_TAG_CLOSE;
100 strncpy(token.value, ">", N_GUI_MAX_TOKEN_SIZE - 1);
101 return token;
102 }
103
104 // Handle attribute values
105 if (ch == '=') {
106 n_gui_next_char(tokenizer); // consume '='
107 token.type = N_GUI_TOKEN_ATTR_VALUE;
108 ch = n_gui_peek_char(tokenizer);
109 if (ch == '"' || ch == '\'') {
110 char quote = n_gui_next_char(tokenizer);
111 int pos = 0;
112 while (n_gui_peek_char(tokenizer) != quote && !n_gui_is_eof(tokenizer)) {
113 if (pos < N_GUI_MAX_TOKEN_SIZE - 1) {
114 token.value[pos++] = n_gui_next_char(tokenizer);
115 }
116 }
117 n_gui_next_char(tokenizer); // consume closing quote
118 token.value[pos] = '\0';
119 }
120 return token;
121 }
122
123 // Handle tag names and attribute names
124 if (*inside_open_tag && (isalpha(ch) || ch == '-')) {
125 int pos = 0;
126 while (isalnum(n_gui_peek_char(tokenizer)) || n_gui_peek_char(tokenizer) == '-' || n_gui_peek_char(tokenizer) == '_') {
127 if (pos < N_GUI_MAX_TOKEN_SIZE - 1) {
128 token.value[pos++] = n_gui_next_char(tokenizer);
129 }
130 }
131 token.value[pos] = '\0';
132
133 // Check if the next character is '=' to identify as attribute name
134 if (n_gui_peek_char(tokenizer) == '=') {
135 token.type = N_GUI_TOKEN_ATTR_NAME;
136 } else {
137 token.type = N_GUI_TOKEN_TAG_NAME;
138 }
139 return token;
140 }
141
142 // Handle text content
143 if (ch != '<' && ch != '>') {
144 if (!*inside_open_tag) { // Only capture text when not inside a tag
145 int pos = 0;
146 while (ch != '<' && ch != '>' && !n_gui_is_eof(tokenizer)) {
147 if (pos < N_GUI_MAX_TOKEN_SIZE - 1) {
148 token.value[pos++] = n_gui_next_char(tokenizer);
149 }
150 ch = n_gui_peek_char(tokenizer);
151 }
152 token.value[pos] = '\0';
153 token.type = N_GUI_TOKEN_TEXT;
154
155 // Handle text content if there is a valid length
156 if (pos > 0) {
157 return token;
158 }
159 }
160 }
161 // Default case for unknown or unsupported characters
162 n_gui_next_char(tokenizer); // consume character
163 return n_gui_next_token(tokenizer, inside_open_tag);
164}
