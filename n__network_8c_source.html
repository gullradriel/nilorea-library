143/*
144 * Copy src to string dst of size siz. At most siz-1 characters
145 * will be copied. Always NUL terminates (unless siz == 0).
146 * Returns strlen(src); if retval >= siz, truncation occurred.
147 */
148size_t strlcpy(char* dst, const char* src, size_t siz) {
149 char* d = dst;
150 const char* s = src;
151 size_t n = siz;
152
153 /* Copy as many bytes as will fit */
154 if (n != 0) {
155 while (--n != 0) {
156 if ((*d++ = *s++) == '\0')
157 break;
158 }
159 }
160
161 /* Not enough room in dst, add NUL and traverse rest of src */
162 if (n == 0) {
163 if (siz != 0)
164 *d = '\0'; /* NUL-terminate dst */
165 while (*s++);
166 }
167
168 return (s - src - 1); /* count does not include NUL */
169}

240/* const char *
241 * inet_ntop6(src, dst, size)
242 * convert IPv6 binary address into presentation (printable) format
243 * author:
244 * Paul Vixie, 1996.
245 */
246static char* inet_ntop6(const unsigned char* src, char* dst, socklen_t size) {
247 /*
248 * Note that int32_t and int16_t need only be "at least" large enough
249 * to contain a value of the specified size. On some systems, like
250 * Crays, there is no such thing as an integer variable with 16 bits.
251 * Keep this in mind if you think this function should have been coded
252 * to use pointer overlays. All the world's not a VAX.
253 */
254 char tmp[sizeof "ffff:ffff:ffff:ffff:ffff:ffff:255.255.255.255"], *tp;
255 struct
256 {
257 int base, len;
258 } best, cur;
259#define NS_IN6ADDRSZ 16
260#define NS_INT16SZ 2
261 u_int words[NS_IN6ADDRSZ / NS_INT16SZ];
262 int i;
263
264 /*
265 * Preprocess:
266 * Copy the input (bytewise) array into a wordwise array.
267 * Find the longest run of 0x00's in src[] for :: shorthanding.
268 */
269 memset(words, '\0', sizeof words);
270 for (i = 0; i < NS_IN6ADDRSZ; i++)
271 words[i / 2] |= (src[i] << ((1 - (i % 2)) << 3));
272 best.base = -1;
273 best.len = 0;
274 cur.base = -1;
275 cur.len = 0;
276 for (i = 0; i < (NS_IN6ADDRSZ / NS_INT16SZ); i++) {
277 if (words[i] == 0) {
278 if (cur.base == -1)
279 cur.base = i, cur.len = 1;
280 else
281 cur.len++;
282 } else {
283 if (cur.base != -1) {
284 if (best.base == -1 || cur.len > best.len)
285 best = cur;
286 cur.base = -1;
287 }
288 }
289 }
290 if (cur.base != -1) {
291 if (best.base == -1 || cur.len > best.len)
292 best = cur;
293 }
294 if (best.base != -1 && best.len < 2)
295 best.base = -1;
296
297 /*
298 * Format the result.
299 */
300 tp = tmp;
301 for (i = 0; i < (NS_IN6ADDRSZ / NS_INT16SZ); i++) {
302 /* Are we inside the best run of 0x00's? */
303 if (best.base != -1 && i >= best.base &&
304 i < (best.base + best.len)) {
305 if (i == best.base)
306 *tp++ = ':';
307 continue;
308 }
309 /* Are we following an initial run of 0x00s or any real hex? */
310 if (i != 0)
311 *tp++ = ':';
312 /* Is this address an encapsulated IPv4? */
313 if (i == 6 && best.base == 0 && (best.len == 6 || (best.len == 7 && words[7] != 0x0001) || (best.len == 5 && words[5] == 0xffff))) {
314 if (!inet_ntop4(src + 12, tp, sizeof tmp - (tp - tmp)))
315 return (NULL);
316 tp += strlen(tp);
317 break;
318 }
319 tp += sprintf(tp, "%x", words[i]);
320 }
321 /* Was it a trailing run of 0x00's? */
322 if (best.base != -1 && (best.base + best.len) ==
323 (NS_IN6ADDRSZ / NS_INT16SZ))
324 *tp++ = ':';
325 *tp++ = '\0';
326
327 /*
328 * Check for overflow, copy, and we're done.
329 */
330 if ((socklen_t)(tp - tmp) > size) {
331 return (NULL);
332 }
333 strcpy(dst, tmp);
334 return (dst);
335}

385/* int
386 * inet_pton4(src, dst)
387 * like inet_aton() but without all the hexadecimal and shorthand.
388 * return:
389 * 1 if `src' is a valid dotted quad, else 0.
390 * notice:
391 * does not touch `dst' unless it's returning 1.
392 * author:
393 * Paul Vixie, 1996.
394 */
395static int inet_pton4(const char* src, u_char* dst) {
396 static const char digits[] = "0123456789";
397 int saw_digit, octets, ch;
398#define NS_INADDRSZ 4
399 u_char tmp[NS_INADDRSZ], *tp;
400
401 saw_digit = 0;
402 octets = 0;
403 *(tp = tmp) = 0;
404 while ((ch = *src++) != '\0') {
405 const char* pch;
406
407 if ((pch = strchr(digits, ch)) != NULL) {
408 u_int uiNew = *tp * 10 + (pch - digits);
409
410 if (saw_digit && *tp == 0)
411 return (0);
412 if (uiNew > 255)
413 return (0);
414 *tp = uiNew;
415 if (!saw_digit) {
416 if (++octets > 4)
417 return (0);
418 saw_digit = 1;
419 }
420 } else if (ch == '.' && saw_digit) {
421 if (octets == 4)
422 return (0);
423 *++tp = 0;
424 saw_digit = 0;
425 } else
426 return (0);
427 }
428 if (octets < 4)
429 return (0);
430 memcpy(dst, tmp, NS_INADDRSZ);
431 return (1);
432}

434/* int
435 * inet_pton6(src, dst)
436 * convert presentation level address to network order binary form.
437 * return:
438 * 1 if `src' is a valid [RFC1884 2.2] address, else 0.
439 * notice:
440 * (1) does not touch `dst' unless it's returning 1.
441 * (2) :: in a full address is silently ignored.
442 * credit:
443 * inspired by Mark Andrews.
444 * author:
445 * Paul Vixie, 1996.
446 */
447static int inet_pton6(const char* src, u_char* dst) {
448 static const char xdigits_l[] = "0123456789abcdef",
449 xdigits_u[] = "0123456789ABCDEF";
450#define NS_IN6ADDRSZ 16
451#define NS_INT16SZ 2
452 u_char tmp[NS_IN6ADDRSZ], *tp, *endp, *colonp;
453 const char* curtok;
454 int ch, seen_xdigits;
455 u_int val;
456
457 memset((tp = tmp), '\0', NS_IN6ADDRSZ);
458 endp = tp + NS_IN6ADDRSZ;
459 colonp = NULL;
460 /* Leading :: requires some special handling. */
461 if (*src == ':')
462 if (*++src != ':')
463 return (0);
464 curtok = src;
465 seen_xdigits = 0;
466 val = 0;
467 while ((ch = *src++) != '\0') {
468 const char* xdigits : const char* pch;
469
470 if ((pch = strchr((xdigits = xdigits_l), ch)) == NULL)
471 pch = strchr((xdigits = xdigits_u), ch);
472 if (pch != NULL) {
473 val <<= 4;
474 val |= (pch - xdigits);
475 if (++seen_xdigits > 4)
476 return (0);
477 continue;
478 }
479 if (ch == ':') {
480 curtok = src;
481 if (!seen_xdigits) {
482 if (colonp)
483 return (0);
484 colonp = tp;
485 continue;
486 } else if (*src == '\0') {
487 return (0);
488 }
489 if (tp + NS_INT16SZ > endp)
490 return (0);
491 *tp++ = (u_char)(val >> 8) & 0xff;
492 *tp++ = (u_char)val & 0xff;
493 seen_xdigits = 0;
494 val = 0;
495 continue;
496 }
497 if (ch == '.' && ((tp + NS_INADDRSZ) <= endp) &&
498 inet_pton4(curtok, tp) > 0) {
499 tp += NS_INADDRSZ;
500 seen_xdigits = 0;
501 break; /* '\\0' was seen by inet_pton4(). */
502 }
503 return (0);
504 }
505 if (seen_xdigits) {
506 if (tp + NS_INT16SZ > endp)
507 return (0);
508 *tp++ = (u_char)(val >> 8) & 0xff;
509 *tp++ = (u_char)val & 0xff;
510 }
511 if (colonp != NULL) {
512 /*
513 * Since some memmove()'s erroneously fail to handle
514 * overlapping regions, we'll do the shift by hand.
515 */
516 const int n = tp - colonp;
517 int i;
518
519 if (tp == endp)
520 return (0);
521 for (i = 1; i <= n; i++) {
522 endp[-i] = colonp[n - i];
523 colonp[n - i] = 0;
524 }
525 tp = endp;
526 }
527 if (tp != endp)
528 return (0);
529 memcpy(dst, tmp, NS_IN6ADDRSZ);
530 return (1);
531}

750 if ((flags & O_NONBLOCK) && !is_blocking) {
751 n_log(LOG_DEBUG, "socket %d was already in non-blocking mode", netw->link.sock);
752 /* in case we missed it, let's update the link mode */
753 netw->link.is_blocking = 0;
754 return TRUE;
755 }
756 } else if (netw->link.is_blocking != 1 && is_blocking) {
757 if (!(flags & O_NONBLOCK) && is_blocking) {
758 n_log(LOG_DEBUG, "socket %d was already in blocking mode", netw->link.sock);
759 /* in case we missed it, let's update the link mode */
760 netw->link.is_blocking = 1;
761 return TRUE;
762 }
763 }
764 if (fcntl(netw->link.sock, F_SETFL, is_blocking ? flags ^ O_NONBLOCK : flags | O_NONBLOCK) == -1) {
765 error = neterrno;
766 errmsg = netstrerror(error);
767 n_log(LOG_ERR, "couldn't set blocking mode %d on %d: %s", is_blocking, netw->link.sock, _str(errmsg));
768 FreeNoLog(errmsg);
769 return FALSE;
770 }
771#else
772 unsigned long int blocking = 1 - is_blocking;
773 int res = ioctlsocket(netw->link.sock, FIONBIO, &blocking);
774 error = neterrno;
775 if (res != NO_ERROR) {
776 errmsg = netstrerror(error);
777 n_log(LOG_ERR, "ioctlsocket failed with error: %ld , neterrno: %s", res, _str(errmsg));
778 FreeNoLog(errmsg);
779 netw_close(&netw);
780 return FALSE;
781 }
782#endif
783 netw->link.is_blocking = is_blocking;
784 return TRUE;
785} /* netw_set_blocking */

1072__attribute__((unused)) static void netw_ssl_lock_callback(int mode, int type, char* file, int line) {
1073 (void)file;
1074 (void)line;
1075 if (mode & CRYPTO_LOCK) {
1076 pthread_mutex_lock(&(netw_ssl_lockarray[type]));
1077 } else {
1078 pthread_mutex_unlock(&(netw_ssl_lockarray[type]));
1079 }
1080}

1083 unsigned long ret;
1084
1085 ret = (unsigned long)pthread_self();
1086 return ret;
1087}

1089static void netw_init_locks(void) {
1090 int i;
1091
1092 netw_ssl_lockarray = (pthread_mutex_t*)OPENSSL_malloc(CRYPTO_num_locks() *
1093 sizeof(pthread_mutex_t));
1094 for (i = 0; i < CRYPTO_num_locks(); i++) {
1095 pthread_mutex_init(&(netw_ssl_lockarray[i]), NULL);
1096 }
1097
1098 CRYPTO_set_id_callback((unsigned long (*)())thread_id);
1099 CRYPTO_set_locking_callback((void (*)())netw_ssl_lock_callback);
1100}

1104
1105 CRYPTO_set_locking_callback(NULL);
1106 for (i = 0; i < CRYPTO_num_locks(); i++)
1107 pthread_mutex_destroy(&(netw_ssl_lockarray[i]));

1222int netw_connect_ex(NETWORK** netw, char* host, char* port, size_t send_list_limit, size_t recv_list_limit, int ip_version, char* ssl_key_file, char* ssl_cert_file) {
1223 // kill compilation warning when there is no openssl
1224 (void)ssl_key_file;
1225 (void)ssl_cert_file;
1226 int error = 0, net_status = 0;
1227 char* errmsg = NULL;
1228
1229 /*do not work over an already used netw*/
1230 if ((*netw)) {
1231 n_log(LOG_ERR, "Unable to allocate (*netw), already existing. You must use empty NETWORK *structs.");
1232 return FALSE;
1233 }
1234
1235 /*creating array*/
1236 (*netw) = netw_new(send_list_limit, recv_list_limit);
1237 __n_assert(netw && (*netw), return FALSE);
1238
1239 /*checking WSA when under windows*/
1240 if (netw_init_wsa(1, 2, 2) == FALSE) {
1241 n_log(LOG_ERR, "Unable to load WSA dll's");
1242 Free((*netw));
1243 return FALSE;
1244 }
1245
1246 /* choose ip version */
1247 if (ip_version == NETWORK_IPV4) {
1248 (*netw)->link.hints.ai_family = AF_INET; /* Allow IPv4 */
1249 } else if (ip_version == NETWORK_IPV6) {
1250 (*netw)->link.hints.ai_family = AF_INET6; /* Allow IPv6 */
1251 } else {
1252 /* NETWORK_ALL or unknown value */
1253 (*netw)->link.hints.ai_family = AF_UNSPEC; /* Allow IPv4 or IPv6 */
1254 }
1255
1256 (*netw)->link.hints.ai_socktype = SOCK_STREAM;
1257 (*netw)->link.hints.ai_protocol = IPPROTO_TCP;
1258 (*netw)->link.hints.ai_flags = AI_PASSIVE;
1259 (*netw)->link.hints.ai_canonname = NULL;
1260 (*netw)->link.hints.ai_next = NULL;
1261
1262 /* Note: on some system, i.e Solaris, it WILL show leak in getaddrinfo.
1263 * Testing it inside a 1,100 loop showed not effect on the amount of leaked
1264 * memory */
1265 error = getaddrinfo(host, port, &(*netw)->link.hints, &(*netw)->link.rhost);
1266 if (error != 0) {
1267 n_log(LOG_ERR, "Error when resolving %s:%s getaddrinfo: %s", host, port, gai_strerror(error));
1268 netw_close(&(*netw));
1269 return FALSE;
1270 }
1271 (*netw)->addr_infos_loaded = 1;
1272 Malloc((*netw)->link.ip, char, 64);
1273 __n_assert((*netw)->link.ip, netw_close(&(*netw)); return FALSE);
1274
1275 /* getaddrinfo() returns a list of address structures. Try each address until we successfully connect. If socket or connect fails, we close the socket and try the next address. */
1276 struct addrinfo* rp = NULL;
1277 for (rp = (*netw)->link.rhost; rp != NULL; rp = rp->ai_next) {
1278 int sock = socket(rp->ai_family, rp->ai_socktype, rp->ai_protocol);
1279 if (sock == -1) {
1280 error = neterrno;
1281 errmsg = netstrerror(error);
1282 n_log(LOG_ERR, "Error while trying to make a socket: %s", _str(errmsg));
1283 FreeNoLog(errmsg);
1284 continue;
1285 }
1286
1287 (*netw)->link.sock = sock;
1288
1289 net_status = connect(sock, rp->ai_addr, rp->ai_addrlen);
1290 if (net_status == -1) {
1291 error = neterrno;
1292 errmsg = netstrerror(error);
1293 n_log(LOG_INFO, "connecting to %s:%s : %s", host, port, _str(errmsg));
1294 FreeNoLog(errmsg);
1295 closesocket(sock);
1296 (*netw)->link.sock = -1;
1297 continue;
1298 } else {
1299 /*storing connected port and ip adress*/
1300 if (!inet_ntop(rp->ai_family, get_in_addr(rp->ai_addr), (*netw)->link.ip, 64)) {
1301 error = neterrno;
1302 errmsg = netstrerror(error);
1303 n_log(LOG_ERR, "inet_ntop: %p , %s", rp, _str(errmsg));
1304 FreeNoLog(errmsg);
1305 }
1306 break; /* Success */
1307 }
1308 }
1309 if (rp == NULL) {
1310 /* No address succeeded */
1311 n_log(LOG_ERR, "Couldn't connect to %s:%s : no adress succeeded", host, port);
1312 netw_close(&(*netw));
1313 return FALSE;
1314 }
1315
1316 (*netw)->link.port = strdup(port);
1317 __n_assert((*netw)->link.port, netw_close(&(*netw)); return FALSE);
1318
1319 if (ssl_key_file && ssl_cert_file) {
1320#ifdef HAVE_OPENSSL
1321 if (netw_set_crypto((*netw), ssl_key_file, ssl_cert_file) == FALSE) {
1322 /* could not initialize SSL */
1323 n_log(LOG_ERR, "couldn't initialize SSL !");
1324 netw_close(&(*netw));
1325 return FALSE;
1326 }
1327
1328 (*netw)->ssl = SSL_new((*netw)->ctx);
1329 SSL_set_fd((*netw)->ssl, (*netw)->link.sock);
1330
1331 // Perform SSL Handshake
1332 if (SSL_connect((*netw)->ssl) <= 0) {
1333 /* could not connect with SSL */
1334 n_log(LOG_ERR, "SSL Handshake error !");
1335 netw_close(&(*netw));
1336 return FALSE;
1337 }
1338 n_log(LOG_DEBUG, "SSL-Connected to %s:%s", (*netw)->link.ip, (*netw)->link.port);
1339#else
1340 n_log(LOG_ERR, "%s:%s trying to configure SSL but application was compiled without SSL support !", (*netw)->link.ip, (*netw)->link.port);
1341#endif
1342 } else {
1343 n_log(LOG_DEBUG, "Connected to %s:%s", (*netw)->link.ip, (*netw)->link.port);
1344 }
1345
1346 netw_set((*netw), NETW_CLIENT | NETW_RUN | NETW_THR_ENGINE_STOPPED);
1347
1348 return TRUE;
1349} /* netw_connect_ex(...)*/

1644int netw_make_listening(NETWORK** netw, char* addr, char* port, int nbpending, int ip_version) {

1670 (*netw)->link.hints.ai_family = AF_UNSPEC; /* Allow IPv4 or IPv6 */
1671 }

1673 (*netw)->link.hints.ai_flags = AI_PASSIVE; /* For wildcard IP address */

1676 (*netw)->link.hints.ai_protocol = IPPROTO_TCP;
1677 (*netw)->link.hints.ai_canonname = NULL;

1681 if (error != 0) {

1683 netw_close(&(*netw));

1708 n_log(LOG_ERR, "Error allocating 64 bytes for ip");
1709 netw_close(&(*netw));
1710 return FALSE;

1713 error = neterrno;
1714 errmsg = netstrerror(error);
1715 n_log(LOG_ERR, "inet_ntop: %p , %s", (*netw)->link.raddr, _str(errmsg));
1716 FreeNoLog(errmsg);
1717 }
1718 /*n_log( LOG_DEBUG, "Socket %d successfully binded to %s %s", (*netw) -> link . sock, _str( ip ) , _str( port ) );*/

1720 break; /* Success */
1721 }
1722 error = neterrno;
1723 errmsg = netstrerror(error);
1724 n_log(LOG_ERR, "Error from bind() on port %s neterrno: %s", port, errmsg);
1725 FreeNoLog(errmsg);
1726 closesocket((*netw)->link.sock);
1727 }
1728 if (rp == NULL) {
1729 /* No address succeeded */
1730 n_log(LOG_ERR, "Couldn't get a socket for listening on port %s", port);
1731 netw_close(&(*netw));
1732 return FALSE;
1733 }

1737 listen((*netw)->link.sock, (*netw)->nb_pending);
1738
1739 netw_set((*netw), NETW_SERVER | NETW_RUN | NETW_THR_ENGINE_STOPPED);
