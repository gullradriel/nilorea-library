134size_t n_base64_decode_len(N_STR* string) {
135 __n_assert(string, return 0);
136 size_t nbytesdecoded = 0;
137 size_t nprbytes = 0;
138
139 const unsigned char* bufin = (const unsigned char*)string->data;
140 const unsigned char* bufend = bufin + string->length; // End of the buffer
141
142 while (pr2six[*(bufin++)] <= 63) {
143 if (bufin >= bufend) {
144 n_log(LOG_ERR, "could not detect end of encoded string for N_STR %p", string);
145 return 0;
146 }
147 }
148
149 // Safely calculate the difference
150 if (bufin > (const unsigned char*)string->data) {
151 nprbytes = (size_t)(bufin - (const unsigned char*)string->data - 1);
152 } else {
153 // Handle unexpected cases where subtraction could be invalid
154 n_log(LOG_ERR, "invalid size of 0 for N_STR %p", string);
155 return 0;
156 }
157
158 nbytesdecoded = ((nprbytes + 3) / 4) * 3;
159
160 return nbytesdecoded;
161}

252
253 size_t i = 0;

272 *p++ = basis_64[((string[i + 1] & 0xF) << 2) |
273 ((int)(string[i + 2] & 0xC0) >> 6)];

280 *p++ = '=';
