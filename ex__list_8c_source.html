39int nstrcmp(const void* a, const void* b) {
40 const N_STR* s1 = a;
41 const N_STR* s2 = b;
42
43 if (!s1 || !s1->data)
44 return 1;
45 if (!s2 || !s2->data)
46 return -1;
47
48 return strcmp(s1->data, s2->data);
49}
