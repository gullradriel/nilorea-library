132static int _n_git_diff_print_cb(const git_diff_delta* delta, const git_diff_hunk* hunk, const git_diff_line* line, void* payload) {
133 (void)delta;
134 (void)hunk;
135 N_STR** nstr = (N_STR**)payload;
136
137 /* line->content is not null-terminated; copy it */
138 if (line->content && line->content_len > 0) {
139 char* tmp = NULL;
140 Malloc(tmp, char, line->content_len + 2);
141 if (tmp) {
142 memcpy(tmp, line->content, line->content_len);
143 tmp[line->content_len] = '\0';
144 nstrprintf_cat(*nstr, "%c%s", line->origin, tmp);
145 FreeNoLog(tmp);
146 }
147 }
148 return 0;
149}

257LIST* n_git_status(N_GIT_REPO* repo) {
258 __n_assert(repo, return NULL);
259 __n_assert(repo->repo, return NULL);
260
261 git_status_options opts = GIT_STATUS_OPTIONS_INIT;
262 opts.show = GIT_STATUS_SHOW_INDEX_AND_WORKDIR;
263 opts.flags = GIT_STATUS_OPT_INCLUDE_UNTRACKED | GIT_STATUS_OPT_RECURSE_UNTRACKED_DIRS | GIT_STATUS_OPT_RENAMES_HEAD_TO_INDEX;
264
265 git_status_list* status_list = NULL;
266 int err = git_status_list_new(&status_list, repo->repo, &opts);
267 if (err < 0) {
268 _n_git_log_error("n_git_status");
269 return NULL;
270 }
271
272 LIST* list = new_generic_list(UNLIMITED_LIST_ITEMS);
273 if (!list) {
274 git_status_list_free(status_list);
275 return NULL;
276 }
277
278 size_t count = git_status_list_entrycount(status_list);
279 for (size_t i = 0; i < count; i++) {
280 const git_status_entry* entry = git_status_byindex(status_list, i);
281 if (!entry) continue;
282
283 N_GIT_STATUS_ENTRY* se = NULL;
284 Malloc(se, N_GIT_STATUS_ENTRY, 1);
285 if (!se) continue;
286
287 const char* fpath = NULL;
288 if (entry->head_to_index && entry->head_to_index->new_file.path) {
289 fpath = entry->head_to_index->new_file.path;
290 } else if (entry->index_to_workdir && entry->index_to_workdir->new_file.path) {
291 fpath = entry->index_to_workdir->new_file.path;
292 } else if (entry->index_to_workdir && entry->index_to_workdir->old_file.path) {
293 fpath = entry->index_to_workdir->old_file.path;
294 }
295
296 if (fpath) {
297 snprintf(se->path, sizeof(se->path), "%s", fpath);
298 }
299
300 se->flags = 0;
301
302 /* index (staged) flags */
303 if (entry->status & GIT_STATUS_INDEX_NEW) {
304 se->flags |= N_GIT_STATUS_NEW | N_GIT_STATUS_STAGED;
305 }
306 if (entry->status & GIT_STATUS_INDEX_MODIFIED) {
307 se->flags |= N_GIT_STATUS_MODIFIED | N_GIT_STATUS_STAGED;
308 }
309 if (entry->status & GIT_STATUS_INDEX_DELETED) {
310 se->flags |= N_GIT_STATUS_DELETED | N_GIT_STATUS_STAGED;
311 }
312
313 /* workdir (unstaged) flags */
314 if (entry->status & GIT_STATUS_WT_NEW) {
315 se->flags |= N_GIT_STATUS_NEW;
316 }
317 if (entry->status & GIT_STATUS_WT_MODIFIED) {
318 se->flags |= N_GIT_STATUS_MODIFIED;
319 }
320 if (entry->status & GIT_STATUS_WT_DELETED) {
321 se->flags |= N_GIT_STATUS_DELETED;
322 }
323
324 list_push(list, se, _n_git_status_entry_destroy);
325 }
326
327 git_status_list_free(status_list);
328 return list;
329} /* n_git_status */

345int n_git_stage(N_GIT_REPO* repo, const char* filepath) {
346 __n_assert(repo, return -1);
347 __n_assert(repo->repo, return -1);
348 __n_assert(filepath, return -1);
349
350 git_index* index = NULL;
351 int ret = -1;
352
353 int err = git_repository_index(&index, repo->repo);
354 if (err < 0) {
355 _n_git_log_error("n_git_stage");
356 _n_git_set_status(repo, "stage failed: %s", filepath);
357 return -1;
358 }
359
360 size_t len = strlen(filepath);
361
362 /* directory paths (ending with '/') cannot be staged with
363 * git_index_add_bypath; use git_index_add_all with a pathspec instead.
364 * Only '/' is checked because libgit2 uses forward slashes on all
365 * platforms; on POSIX '\\' is a valid filename character. */
366 if (len > 0 && filepath[len - 1] == '/') {
367 char* pathspec_str = (char*)filepath;
368 git_strarray pathspec;
369 pathspec.strings = &pathspec_str;
370 pathspec.count = 1;
371
372 err = git_index_add_all(index, &pathspec, GIT_INDEX_ADD_DEFAULT, NULL, NULL);
373 if (err < 0) {
374 _n_git_log_error("n_git_stage");
375 _n_git_set_status(repo, "stage failed: %s", filepath);
376 goto cleanup;
377 }
378 } else {
379 err = git_index_add_bypath(index, filepath);
380 if (err < 0) {
381 _n_git_log_error("n_git_stage");
382 _n_git_set_status(repo, "stage failed: %s", filepath);
383 goto cleanup;
384 }
385 }
386
387 err = git_index_write(index);
388 if (err < 0) {
389 _n_git_log_error("n_git_stage");
390 _n_git_set_status(repo, "stage failed: %s", filepath);
391 goto cleanup;
392 }
393
394 _n_git_set_status(repo, "staged: %s", filepath);
395 ret = 0;
396
397cleanup:
398 git_index_free(index);
399 return ret;
400} /* n_git_stage */

449int n_git_unstage(N_GIT_REPO* repo, const char* filepath) {
450 __n_assert(repo, return -1);
451 __n_assert(repo->repo, return -1);
452 __n_assert(filepath, return -1);
453
454 git_reference* head_ref = NULL;
455 git_object* head_obj = NULL;
456 int ret = -1;
457
458 /* try to get HEAD; for initial commit there may be no HEAD */
459 int err = git_repository_head(&head_ref, repo->repo);
460 if (err == 0) {
461 err = git_reference_peel(&head_obj, head_ref, GIT_OBJECT_COMMIT);
462 if (err < 0) {
463 _n_git_log_error("n_git_unstage");
464 _n_git_set_status(repo, "unstage failed: %s", filepath);
465 goto cleanup;
466 }
467 }
468
469 git_strarray paths;
470 char* path_str = (char*)filepath;
471 paths.strings = &path_str;
472 paths.count = 1;
473
474 err = git_reset_default(repo->repo, head_obj, &paths);
475 if (err < 0) {
476 _n_git_log_error("n_git_unstage");
477 _n_git_set_status(repo, "unstage failed: %s", filepath);
478 goto cleanup;
479 }
480
481 _n_git_set_status(repo, "unstaged: %s", filepath);
482 ret = 0;
483
484cleanup:
485 if (head_obj) git_object_free(head_obj);
486 if (head_ref) git_reference_free(head_ref);
487 return ret;
488} /* n_git_unstage */

498int n_git_commit(N_GIT_REPO* repo, const char* message, const char* author_name, const char* author_email) {
499 __n_assert(repo, return -1);
500 __n_assert(repo->repo, return -1);
501 __n_assert(message, return -1);
502 __n_assert(author_name, return -1);
503 __n_assert(author_email, return -1);
504
505 git_index* index = NULL;
506 git_oid tree_oid;
507 git_oid commit_oid;
508 git_tree* tree = NULL;
509 git_signature* sig = NULL;
510 git_commit* parent_commit = NULL;
511 git_reference* head_ref = NULL;
512 int ret = -1;
513
514 int err = git_repository_index(&index, repo->repo);
515 if (err < 0) {
516 _n_git_log_error("n_git_commit");
517 _n_git_set_status(repo, "commit failed: cannot open index");
518 return -1;
519 }
520
521 err = git_index_write_tree(&tree_oid, index);
522 if (err < 0) {
523 _n_git_log_error("n_git_commit");
524 _n_git_set_status(repo, "commit failed: cannot write tree");
525 goto cleanup;
526 }
527
528 err = git_tree_lookup(&tree, repo->repo, &tree_oid);
529 if (err < 0) {
530 _n_git_log_error("n_git_commit");
531 _n_git_set_status(repo, "commit failed: cannot lookup tree");
532 goto cleanup;
533 }
534
535 err = git_signature_now(&sig, author_name, author_email);
536 if (err < 0) {
537 _n_git_log_error("n_git_commit");
538 _n_git_set_status(repo, "commit failed: cannot create signature");
539 goto cleanup;
540 }
541
542 /* try to get HEAD as parent; if no HEAD this is the initial commit */
543 int have_parent = 0;
544 err = git_repository_head(&head_ref, repo->repo);
545 if (err == 0) {
546 err = git_reference_peel((git_object**)&parent_commit, head_ref, GIT_OBJECT_COMMIT);
547 if (err == 0) {
548 have_parent = 1;
549 }
550 }
551
552 if (have_parent) {
553 const git_commit* parents[] = {parent_commit};
554 err = git_commit_create(&commit_oid, repo->repo, "HEAD", sig, sig, "UTF-8", message, tree, 1, parents);
555 } else {
556 err = git_commit_create(&commit_oid, repo->repo, "HEAD", sig, sig, "UTF-8", message, tree, 0, NULL);
557 }
558
559 if (err < 0) {
560 _n_git_log_error("n_git_commit");
561 _n_git_set_status(repo, "commit failed");
562 goto cleanup;
563 }
564
565 {
566 char hex[GIT_OID_SHA1_HEXSIZE + 1];
567 git_oid_tostr(hex, sizeof(hex), &commit_oid);
568 _n_git_set_status(repo, "committed %.8s: %s", hex, message);
569 }
570 ret = 0;
571
572cleanup:
573 if (parent_commit) git_commit_free(parent_commit);
574 if (head_ref) git_reference_free(head_ref);
575 if (sig) git_signature_free(sig);
576 if (tree) git_tree_free(tree);
577 git_index_free(index);
578 return ret;
579} /* n_git_commit */

587LIST* n_git_log(N_GIT_REPO* repo, size_t max_entries) {
588 __n_assert(repo, return NULL);
589 __n_assert(repo->repo, return NULL);
590
591 git_revwalk* walker = NULL;
592 int err = git_revwalk_new(&walker, repo->repo);
593 if (err < 0) {
594 _n_git_log_error("n_git_log");
595 return NULL;
596 }
597
598 git_revwalk_sorting(walker, GIT_SORT_TIME);
599
600 err = git_revwalk_push_head(walker);
601 if (err < 0) {
602 /* Unborn HEAD (empty repo, no commits), return an empty list.
603 * git_repository_head_unborn() returns:
604 * 1 => unborn HEAD
605 * 0 => not unborn
606 * <0 => error
607 */
608 git_revwalk_free(walker);
609 int unborn = git_repository_head_unborn(repo->repo);
610 if (unborn == 1) {
611 return new_generic_list(UNLIMITED_LIST_ITEMS);
612 }
613 if (unborn < 0) {
614 _n_git_log_error("n_git_log");
615 }
616 return NULL;
617 }
618
619 LIST* list = new_generic_list(UNLIMITED_LIST_ITEMS);
620 if (!list) {
621 git_revwalk_free(walker);
622 return NULL;
623 }
624
625 git_oid oid;
626 size_t count = 0;
627 while (count < max_entries && git_revwalk_next(&oid, walker) == 0) {
628 git_commit* commit = NULL;
629 err = git_commit_lookup(&commit, repo->repo, &oid);
630 if (err < 0) {
631 _n_git_log_error("n_git_log");
632 continue;
633 }
634
635 N_GIT_COMMIT_INFO* info = NULL;
636 Malloc(info, N_GIT_COMMIT_INFO, 1);
637 if (!info) {
638 git_commit_free(commit);
639 continue;
640 }
641
642 /* hash */
643 char hex[GIT_OID_SHA1_HEXSIZE + 1];
644 git_oid_tostr(hex, sizeof(hex), &oid);
645 snprintf(info->hash, sizeof(info->hash), "%s", hex);
646 snprintf(info->short_hash, sizeof(info->short_hash), "%.8s", hex);
647
648 /* message */
649 const char* msg = git_commit_message(commit);
650 if (msg) {
651 snprintf(info->message, sizeof(info->message), "%s", msg);
652 }
653
654 /* author */
655 const git_signature* author = git_commit_author(commit);
656 if (author) {
657 snprintf(info->author, sizeof(info->author), "%s <%s>", author->name ? author->name : "", author->email ? author->email : "");
658 info->timestamp = (int64_t)author->when.time;
659 }
660
661 list_push(list, info, _n_git_commit_info_destroy);
662 git_commit_free(commit);
663 count++;
664 }
665
666 git_revwalk_free(walker);
667 return list;
668} /* n_git_log */

705N_STR* n_git_diff_commits(N_GIT_REPO* repo, const char* hash_a, const char* hash_b) {
706 __n_assert(repo, return NULL);
707 __n_assert(repo->repo, return NULL);
708 __n_assert(hash_a, return NULL);
709 __n_assert(hash_b, return NULL);
710
711 git_object* obj_a = NULL;
712 git_object* obj_b = NULL;
713 git_commit* commit_a = NULL;
714 git_commit* commit_b = NULL;
715 git_tree* tree_a = NULL;
716 git_tree* tree_b = NULL;
717 git_diff* diff = NULL;
718 N_STR* result = NULL;
719 int err = 0;
720
721 err = git_revparse_single(&obj_a, repo->repo, hash_a);
722 if (err < 0) {
723 _n_git_log_error("n_git_diff_commits");
724 goto cleanup;
725 }
726
727 err = git_revparse_single(&obj_b, repo->repo, hash_b);
728 if (err < 0) {
729 _n_git_log_error("n_git_diff_commits");
730 goto cleanup;
731 }
732
733 err = git_commit_lookup(&commit_a, repo->repo, git_object_id(obj_a));
734 if (err < 0) {
735 _n_git_log_error("n_git_diff_commits");
736 goto cleanup;
737 }
738
739 err = git_commit_lookup(&commit_b, repo->repo, git_object_id(obj_b));
740 if (err < 0) {
741 _n_git_log_error("n_git_diff_commits");
742 goto cleanup;
743 }
744
745 err = git_commit_tree(&tree_a, commit_a);
746 if (err < 0) {
747 _n_git_log_error("n_git_diff_commits");
748 goto cleanup;
749 }
750
751 err = git_commit_tree(&tree_b, commit_b);
752 if (err < 0) {
753 _n_git_log_error("n_git_diff_commits");
754 goto cleanup;
755 }
756
757 err = git_diff_tree_to_tree(&diff, repo->repo, tree_a, tree_b, NULL);
758 if (err < 0) {
759 _n_git_log_error("n_git_diff_commits");
760 goto cleanup;
761 }
762
763 result = new_nstr(256);
764 if (!result) {
765 goto cleanup;
766 }
767
768 git_diff_print(diff, GIT_DIFF_FORMAT_PATCH, _n_git_diff_print_cb, &result);
769
770cleanup:
771 if (diff) git_diff_free(diff);
772 if (tree_b) git_tree_free(tree_b);
773 if (tree_a) git_tree_free(tree_a);
774 if (commit_b) git_commit_free(commit_b);

902int n_git_create_branch(N_GIT_REPO* repo, const char* branch_name) {
903 __n_assert(repo, return -1);
904 __n_assert(repo->repo, return -1);
905 __n_assert(branch_name, return -1);
906
907 git_reference* head_ref = NULL;
908 git_commit* head_commit = NULL;
909 git_reference* new_branch = NULL;
910 int ret = -1;
911
912 int err = git_repository_head(&head_ref, repo->repo);
913 if (err == GIT_EUNBORNBRANCH) {
914 /* Empty repository (no commits yet). We cannot create a real branch
915 * ref because there is no commit to point to. Instead, update the
916 * symbolic HEAD to target the requested branch, this is equivalent
917 * to `git checkout -b <branch>` on an empty repo. */
918 char refname[512];
919 int n = snprintf(refname, sizeof(refname), "refs/heads/%s", branch_name);
920 if (n < 0 || (size_t)n >= sizeof(refname)) {
921 n_log(LOG_ERR, "n_git_create_branch: branch name too long");
922 return -1;
923 }
924 git_reference* new_ref = NULL;
925 int sym_err = git_reference_symbolic_create(&new_ref, repo->repo, "HEAD", refname, 1, "n_git_create_branch: set HEAD on empty repo");
926 if (sym_err < 0) {
927 _n_git_log_error("n_git_create_branch (unborn HEAD)");
928 return -1;
929 }
930 git_reference_free(new_ref);
931 n_log(LOG_INFO, "n_git_create_branch: set HEAD to '%s' on empty repository", branch_name);
932 _n_git_set_status(repo, "created branch '%s' (empty repo)", branch_name);
933 return 0;
934 }
935 if (err < 0) {
936 _n_git_log_error("n_git_create_branch");
937 _n_git_set_status(repo, "create branch '%s' failed", branch_name);
938 return -1;
939 }
940
941 err = git_reference_peel((git_object**)&head_commit, head_ref, GIT_OBJECT_COMMIT);
942 if (err < 0) {
943 _n_git_log_error("n_git_create_branch");
944 _n_git_set_status(repo, "create branch '%s' failed", branch_name);
945 goto cleanup;
946 }
947
948 err = git_branch_create(&new_branch, repo->repo, branch_name, head_commit, 0);
949 if (err < 0) {
950 _n_git_log_error("n_git_create_branch");
951 _n_git_set_status(repo, "create branch '%s' failed", branch_name);
952 goto cleanup;
953 }
954
955 _n_git_set_status(repo, "created branch '%s'", branch_name);
956 ret = 0;
957
958cleanup:
959 if (new_branch) git_reference_free(new_branch);
960 if (head_commit) git_commit_free(head_commit);
961 git_reference_free(head_ref);
962 return ret;
963} /* n_git_create_branch */

1021int n_git_current_branch(N_GIT_REPO* repo, char* out, size_t out_size) {
1022 __n_assert(repo, return -1);
1023 __n_assert(repo->repo, return -1);
1024 __n_assert(out, return -1);
1025 __n_assert(out_size > 0, return -1);
1026
1027 git_reference* head_ref = NULL;
1028
1029 int err = git_repository_head(&head_ref, repo->repo);
1030 if (err == 0) {
1031 const char* shorthand = git_reference_shorthand(head_ref);
1032 snprintf(out, out_size, "%s", shorthand ? shorthand : "");
1033 git_reference_free(head_ref);
1034 return 0;
1035 }
1036
1037 /* Handle unborn HEAD (empty repo, no commits yet).
1038 * Read the symbolic HEAD reference to determine the branch name. */
1039 if (err == GIT_EUNBORNBRANCH) {
1040 git_reference* sym_ref = NULL;
1041 int sym_err = git_reference_lookup(&sym_ref, repo->repo, "HEAD");
1042 if (sym_err == 0 && git_reference_type(sym_ref) == GIT_REFERENCE_SYMBOLIC) {
1043 const char* target = git_reference_symbolic_target(sym_ref);
1044 /* target is e.g. "refs/heads/main", extract branch name */
1045 if (target) {
1046 const char* prefix = "refs/heads/";
1047 size_t prefix_len = strlen(prefix);
1048 if (strncmp(target, prefix, prefix_len) == 0) {
1049 snprintf(out, out_size, "%s", target + prefix_len);
1050 } else {
1051 snprintf(out, out_size, "%s", target);
1052 }
1053 git_reference_free(sym_ref);
1054 return 0;
1055 }
1056 }
1057 if (sym_ref) git_reference_free(sym_ref);
1058 }
1059
1060 _n_git_log_error("n_git_current_branch");
1061 return -1;
1062} /* n_git_current_branch */

1099static int _n_git_cred_cb(git_credential** out, const char* url, const char* username_from_url, unsigned int allowed_types, void* payload) {
1100 (void)url;
1101 N_GIT_REMOTE_AUTH* auth = (N_GIT_REMOTE_AUTH*)payload;
1102 if (!auth) return GIT_PASSTHROUGH;
1103
1104 switch (auth->type) {
1105 case N_GIT_AUTH_TOKEN:
1106 if ((allowed_types & GIT_CREDENTIAL_USERPASS_PLAINTEXT) &&
1107 auth->password) {
1108 const char* user = auth->username ? auth->username : "x-access-token";
1109 return git_credential_userpass_plaintext_new(out, user, auth->password);
1110 }
1111 break;
1112
1113 case N_GIT_AUTH_BASIC:
1114 if ((allowed_types & GIT_CREDENTIAL_USERPASS_PLAINTEXT) &&
1115 auth->username && auth->password) {
1116 return git_credential_userpass_plaintext_new(out, auth->username, auth->password);
1117 }
1118 break;
1119
1120 case N_GIT_AUTH_SSH:
1121 if ((allowed_types & GIT_CREDENTIAL_SSH_KEY) &&
1122 auth->ssh_privkey_path) {
1123 const char* user = auth->username ? auth->username : (username_from_url ? username_from_url : "git");
1124 return git_credential_ssh_key_new(out, user,
1125 auth->ssh_pubkey_path, auth->ssh_privkey_path,
1126 auth->ssh_passphrase ? auth->ssh_passphrase : "");
1127 }
1128 /* Try SSH agent if key auth not available */
1129 if (allowed_types & GIT_CREDENTIAL_SSH_KEY) {
1130 const char* user = auth->username ? auth->username : (username_from_url ? username_from_url : "git");
1131 return git_credential_ssh_key_from_agent(out, user);
1132 }
1133 break;
1134
1135 default:
1136 break;
1137 }
1138
1139 return GIT_PASSTHROUGH;
1140} /* _n_git_cred_cb */

1216int n_git_push(N_GIT_REPO* repo) {
1217 __n_assert(repo, return -1);
1218 __n_assert(repo->repo, return -1);
1219
1220 _n_git_ensure_init();
1221
1222 /* Refuse to push an empty repo (no commits yet).
1223 * git_repository_head_unborn() returns 1 if unborn, 0 if not, <0 on error. */
1224 int unborn = git_repository_head_unborn(repo->repo);
1225 if (unborn < 0) {
1226 _n_git_log_error("n_git_push: head_unborn");
1227 _n_git_set_status(repo, "push failed: unable to check HEAD");
1228 return -1;
1229 }
1230 if (unborn == 1) {
1231 n_log(LOG_ERR, "n_git_push: repository has no commits, commit first before pushing");
1232 _n_git_set_status(repo, "push failed: no commits yet, commit first");
1233 return -1;
1234 }
1235
1236 /* Get current branch name */
1237 char branch[128];
1238 if (n_git_current_branch(repo, branch, sizeof(branch)) != 0) {
1239 n_log(LOG_ERR, "n_git_push: cannot determine current branch");
1240 _n_git_set_status(repo, "push failed: cannot determine current branch");
1241 return -1;
1242 }
1243
1244 /* Build refspec: refs/heads/branch:refs/heads/branch */
1245 char refspec[300];
1246 snprintf(refspec, sizeof(refspec), "refs/heads/%s:refs/heads/%s",
1247 branch, branch);
1248
1249 git_remote* remote = NULL;
1250 int err = git_remote_lookup(&remote, repo->repo, "origin");
1251 if (err < 0) {
1252 _n_git_log_error("n_git_push: remote lookup");
1253 _n_git_set_status(repo, "push failed: remote 'origin' not found");
1254 return -1;
1255 }
1256
1257 /* Set up push options with credential callback */
1258 git_push_options opts = GIT_PUSH_OPTIONS_INIT;
1259 if (repo->remote_auth) {
1260 opts.callbacks.credentials = _n_git_cred_cb;
1261 opts.callbacks.payload = repo->remote_auth;
1262 }
1263
1264 const char* refspecs[] = {refspec};
1265 git_strarray refspec_arr = {(char**)refspecs, 1};
1266
1267 err = git_remote_push(remote, &refspec_arr, &opts);
1268 git_remote_free(remote);
1269
1270 if (err < 0) {
1271 const git_error* gerr = git_error_last();
1272 _n_git_log_error("n_git_push");
1273 _n_git_set_status(repo, "push failed: %s",
1274 (gerr && gerr->message) ? gerr->message : "unknown error");
1275 return -1;
1276 }
1277
1278 n_log(LOG_INFO, "n_git_push: pushed %s to origin", branch);
1279 _n_git_set_status(repo, "pushed '%s' to origin", branch);
1280 return 0;
1281} /* n_git_push */

1286int n_git_pull(N_GIT_REPO* repo) {
1287 __n_assert(repo, return -1);
1288 __n_assert(repo->repo, return -1);
1289
1290 _n_git_ensure_init();
1291
1292 /* Fetch from origin */
1293 git_remote* remote = NULL;
1294 int err = git_remote_lookup(&remote, repo->repo, "origin");
1295 if (err < 0) {
1296 _n_git_log_error("n_git_pull: remote lookup");
1297 _n_git_set_status(repo, "pull failed: remote 'origin' not found");
1298 return -1;
1299 }
1300
1301 git_fetch_options fetch_opts = GIT_FETCH_OPTIONS_INIT;
1302 if (repo->remote_auth) {
1303 fetch_opts.callbacks.credentials = _n_git_cred_cb;
1304 fetch_opts.callbacks.payload = repo->remote_auth;
1305 }
1306
1307 err = git_remote_fetch(remote, NULL, &fetch_opts, "pull");
1308 git_remote_free(remote);
1309
1310 if (err < 0) {
1311 const git_error* gerr = git_error_last();
1312 _n_git_log_error("n_git_pull: fetch");
1313 _n_git_set_status(repo, "pull failed: fetch error: %s",
1314 (gerr && gerr->message) ? gerr->message : "unknown error");
1315 return -1;
1316 }
1317
1318 /* Get current branch name */
1319 char branch[128];
1320 if (n_git_current_branch(repo, branch, sizeof(branch)) != 0) {
1321 n_log(LOG_ERR, "n_git_pull: cannot determine current branch");
1322 _n_git_set_status(repo, "pull failed: cannot determine current branch");
1323 return -1;
1324 }
1325
1326 /* Look up the remote tracking ref: refs/remotes/origin/<branch> */
1327 char remote_ref[300];
1328 snprintf(remote_ref, sizeof(remote_ref), "refs/remotes/origin/%s", branch);
1329
1330 git_oid remote_oid;
1331 err = git_reference_name_to_id(&remote_oid, repo->repo, remote_ref);
1332 if (err < 0) {
1333 _n_git_log_error("n_git_pull: remote ref lookup");
1334 _n_git_set_status(repo, "pull failed: no remote tracking branch for '%s'", branch);
1335 return -1;
1336 }
1337
1338 /* Get HEAD commit */
1339 git_oid local_oid;
1340 err = git_reference_name_to_id(&local_oid, repo->repo, "HEAD");
1341 if (err < 0) {
1342 _n_git_log_error("n_git_pull: HEAD lookup");
1343 _n_git_set_status(repo, "pull failed: cannot read HEAD");
1344 return -1;
1345 }
1346
1347 /* Check if already up to date */
1348 if (git_oid_equal(&local_oid, &remote_oid)) {
1349 n_log(LOG_INFO, "n_git_pull: already up to date");
1350 _n_git_set_status(repo, "already up to date");
1351 return 0;
1352 }
1353
1354 /* Check if fast-forward is possible */
1355 git_annotated_commit* remote_commit = NULL;
1356 err = git_annotated_commit_lookup(&remote_commit, repo->repo, &remote_oid);
1357 if (err < 0) {
1358 _n_git_log_error("n_git_pull: annotated commit lookup");
1359 _n_git_set_status(repo, "pull failed: cannot lookup remote commit");
1360 return -1;
1361 }
1362
1363 git_merge_analysis_t analysis;
1364 git_merge_preference_t preference;
1365 const git_annotated_commit* their_heads[] = {remote_commit};
1366 err = git_merge_analysis(&analysis, &preference, repo->repo, their_heads, 1);
1367 if (err < 0) {
1368 git_annotated_commit_free(remote_commit);
1369 _n_git_log_error("n_git_pull: merge analysis");
1370 _n_git_set_status(repo, "pull failed: merge analysis error");
1371 return -1;
1372 }
1373
1374 if (analysis & GIT_MERGE_ANALYSIS_UP_TO_DATE) {
1375 git_annotated_commit_free(remote_commit);
1376 n_log(LOG_INFO, "n_git_pull: already up to date");
1377 _n_git_set_status(repo, "already up to date");
1378 return 0;
1379 }
1380
1381 if (analysis & GIT_MERGE_ANALYSIS_FASTFORWARD) {
1382 /* Fast-forward: move HEAD to remote commit */
1383 git_reference* head_ref = NULL;
1384 err = git_repository_head(&head_ref, repo->repo);
1385 if (err < 0) {
1386 git_annotated_commit_free(remote_commit);
1387 _n_git_log_error("n_git_pull: get HEAD ref");
1388 _n_git_set_status(repo, "pull failed: cannot get HEAD reference");
1389 return -1;
1390 }
1391
1392 git_reference* new_ref = NULL;
1393 err = git_reference_set_target(&new_ref, head_ref, &remote_oid, "pull: fast-forward");
1394 git_reference_free(head_ref);
1395 if (err < 0) {
1396 git_annotated_commit_free(remote_commit);
1397 _n_git_log_error("n_git_pull: set target");
1398 _n_git_set_status(repo, "pull failed: cannot fast-forward HEAD");
1399 return -1;
1400 }
1401 git_reference_free(new_ref);
1402
1403 /* Checkout the new HEAD */
1404 git_object* target = NULL;
1405 git_object_lookup(&target, repo->repo, &remote_oid, GIT_OBJECT_COMMIT);
1406 if (target) {
1407 git_checkout_options co_opts = GIT_CHECKOUT_OPTIONS_INIT;
1408 co_opts.checkout_strategy = GIT_CHECKOUT_SAFE;
1409 git_checkout_tree(repo->repo, target, &co_opts);
1410 git_object_free(target);
1411 }
1412
1413 git_annotated_commit_free(remote_commit);
1414 n_log(LOG_INFO, "n_git_pull: fast-forwarded %s", branch);
1415 _n_git_set_status(repo, "pulled '%s': fast-forwarded to latest", branch);
1416 return 0;
1417 }
1418
1419 /* Not a fast-forward, we don't handle merge conflicts */

1468int n_git_ahead_behind(N_GIT_REPO* repo, size_t* ahead, size_t* behind) {
1469 __n_assert(repo, return -1);
1470 __n_assert(repo->repo, return -1);
1471
1472 _n_git_ensure_init();
1473
1474 if (ahead) *ahead = 0;
1475 if (behind) *behind = 0;
1476
1477 /* Check for unborn HEAD (no commits yet) */
1478 int unborn = git_repository_head_unborn(repo->repo);
1479 if (unborn < 0) {
1480 _n_git_log_error("n_git_ahead_behind: head_unborn");
1481 _n_git_set_status(repo, "ahead/behind: unable to check HEAD");
1482 return -1;
1483 }
1484 if (unborn == 1) {
1485 /* No commits yet, nothing to compare */
1486 _n_git_set_status(repo, "no commits yet");
1487 return 0;
1488 }
1489
1490 /* Get current branch name, 256 bytes covers practical branch names */
1491 char branch[256];
1492 if (n_git_current_branch(repo, branch, sizeof(branch)) != 0) {
1493 _n_git_set_status(repo, "ahead/behind: cannot determine current branch");
1494 return -1;
1495 }
1496
1497 /* Resolve local HEAD */
1498 git_oid local_oid;
1499 int err = git_reference_name_to_id(&local_oid, repo->repo, "HEAD");
1500 if (err < 0) {
1501 _n_git_log_error("n_git_ahead_behind: HEAD lookup");
1502 _n_git_set_status(repo, "ahead/behind: cannot read HEAD");
1503 return -1;
1504 }
1505
1506 /* Resolve remote tracking ref */
1507 char remote_ref[512];
1508 int ref_len = snprintf(remote_ref, sizeof(remote_ref),
1509 "refs/remotes/origin/%s", branch);
1510 if (ref_len < 0 || (size_t)ref_len >= sizeof(remote_ref)) {
1511 _n_git_set_status(repo, "ahead/behind: branch name too long");
1512 return -1;
1513 }
1514 git_oid remote_oid;
1515 err = git_reference_name_to_id(&remote_oid, repo->repo, remote_ref);
1516 if (err < 0) {
1517 /* Distinguish "no tracking branch" from other libgit2 errors */
1518 if (err == GIT_ENOTFOUND || err == GIT_EINVALIDSPEC) {
1519 _n_git_set_status(repo, "ahead/behind: no remote tracking branch for '%s'", branch);
1520 } else {
1521 const git_error* e;
1522 _n_git_log_error("n_git_ahead_behind: remote tracking ref lookup");
1523 e = git_error_last();
1524 if (e && e->message) {
1525 _n_git_set_status(repo, "ahead/behind: %s", e->message);
1526 } else {
1527 _n_git_set_status(repo, "ahead/behind: failed to read remote tracking branch");
1528 }
1529 }
1530 return -1;
1531 }
1532
1533 /* Already equal? */
1534 if (git_oid_equal(&local_oid, &remote_oid)) {
1535 _n_git_set_status(repo, "up to date with origin/%s", branch);
1536 return 0;
1537 }
1538
1539 /* Compute ahead/behind counts */
1540 size_t a = 0, b = 0;
1541 err = git_graph_ahead_behind(&a, &b, repo->repo, &local_oid, &remote_oid);
1542 if (err < 0) {
1543 _n_git_log_error("n_git_ahead_behind: graph_ahead_behind");
1544 _n_git_set_status(repo, "ahead/behind: comparison failed");
1545 return -1;
1546 }
1547

1549 if (behind) *behind = b;
1550
1551 _n_git_set_status(repo, "origin/%s: %zu ahead, %zu behind", branch, a, b);
1552 return 0;
1553} /* n_git_ahead_behind */
