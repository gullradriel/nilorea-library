110/* Drain an eventfd. Reads in 8-byte units (eventfd's contract);
111 * keeps reading until EAGAIN. We don't care about the count, just
112 * the fact that something happened. */
113static long long drain_eventfd(int efd) {
114 long long total = 0;
115 uint64_t buf;
116 for (;;) {
117 ssize_t n = read(efd, &buf, sizeof(buf));
118 if (n == sizeof(buf)) {
119 total += (long long)buf;
120 continue;
121 }
122 if (n < 0 && (errno == EAGAIN || errno == EWOULDBLOCK)) break;
123 /* Either a partial read (impossible per eventfd's contract)
124 * or some other error. Either way, break, the wake event
125 * is "something happened", we don't lose state. */
126 break;
127 }
128 return total;
129}

170/* Pop the next N_STR from netw->send_buf (under sendbolt) and pre-frame
171 * it into a contiguous buffer ready to send: 4-byte state word + 4-byte
172 * payload length + payload bytes, with optional zlib/lz4 compression
173 * applied to the payload (same gating as netw_send_func: compress_mode
174 * != NONE && payload >= NETW_COMPRESS_THRESHOLD && ratio >= 10 %).
175 *
176 * On success populates netw->reactor_send_buf / _len / _off and returns
177 * 1. On nothing-to-send returns 0. On allocation failure returns -1
178 * (caller should treat as fatal).
179 */
180static int reactor_send_state_load_next(NETWORK* netw) {
181 if (netw->reactor_send_buf) return 1; /* already loaded */
182
183 pthread_mutex_lock(&netw->sendbolt);
184 N_STR* msg = list_shift(netw->send_buf, N_STR);
185 pthread_mutex_unlock(&netw->sendbolt);
186 if (!msg) return 0;
187
188 /* Optional compression, same gating as netw_send_func, including
189 * the per-NETWORK compress_mode choice (NONE / ZLIB / LZ4) and
190 * the threshold + ratio macros (NETW_COMPRESS_THRESHOLD,
191 * NETW_COMPRESS_MIN_RATIO) shared with the thread engine via the
192 * public n_network.h. */
193 uint32_t pkt_state = NETW_RUN;
194 if (netw->compress_mode != NETW_COMPRESS_NONE &&
195 msg->written >= (size_t)NETW_COMPRESS_THRESHOLD) {
196 N_STR* zipped = NULL;
197 uint32_t flag_bit = 0;
198 if (netw->compress_mode == NETW_COMPRESS_LZ4) {
199 zipped = zip4_nstr(msg);
200 flag_bit = NETW_COMPRESSED_LZ4;
201 } else {
202 zipped = zip_nstr(msg);
203 flag_bit = NETW_COMPRESSED_ZLIB;
204 }
205 if (zipped && zipped->written > 0 &&
206 zipped->written * 100u <=
207 msg->written * (100u - NETW_COMPRESS_MIN_RATIO)) {
208 /* Saved at least NETW_COMPRESS_MIN_RATIO %, swap. */
209 free_nstr(&msg);
210 msg = zipped;
211 pkt_state |= flag_bit;
212 } else if (zipped) {
213 free_nstr(&zipped);
214 }
215 }
216
217 if (msg->written > UINT32_MAX) {
218 n_log(LOG_ERR, "n_reactor: payload too large (%zu > UINT32_MAX); dropping",
219 msg->written);
220 free_nstr(&msg);
221 return 0;
222 }
223 size_t total = 8 + msg->written;
224 char* buf = NULL;
225 Malloc(buf, char, total);
226 if (!buf) {
227 free_nstr(&msg);
228 return -1;
229 }
230 uint32_t state_be = htonl(pkt_state);
231 uint32_t length_be = htonl((uint32_t)msg->written);
232 memcpy(buf, &state_be, 4);
233 memcpy(buf + 4, &length_be, 4);
234 memcpy(buf + 8, msg->data, msg->written);
235 free_nstr(&msg);
236
237 netw->reactor_send_buf = buf;
238 netw->reactor_send_len = total;
239 netw->reactor_send_off = 0;
240 return 1;
241}

243/* Drain as much of netw->reactor_send_buf as the kernel will accept,
244 * loop to load the next queued frame on completion. Returns:
245 * 1, drained fully (send_buf empty, no in-flight). Caller disarms
246 * EPOLLOUT.
247 * 0, back-pressure (EAGAIN). Caller arms EPOLLOUT.
248 * -1, hard error / connection closed by peer. Caller unregisters
249 * and flags NETW_ERROR.
250 */
251static int reactor_drain_writes(NETWORK* netw, n_reactor* reactor) {
252 for (;;) {
253 if (!netw->reactor_send_buf) {
254 int r = reactor_send_state_load_next(netw);
255 if (r < 0) return -1;
256 if (r == 0) return 1; /* nothing to send */
257 }
258 size_t remaining = netw->reactor_send_len - netw->reactor_send_off;
259 if (remaining == 0) {
260 /* Buffer fully drained, free and try the next queued msg. */
261 Free(netw->reactor_send_buf);
262 netw->reactor_send_buf = NULL;
263 netw->reactor_send_len = 0;
264 netw->reactor_send_off = 0;
265 continue;
266 }
267 /* TLS-over-reactor: route through the NETWORK's
268 * single-attempt send pointer (raw send() on cleartext,
269 * SSL_write on crypto sockets). EINTR is retried inside the
270 * call; partial-write resume rides the same buffer+offset
271 * thanks to SSL_MODE_ENABLE_PARTIAL_WRITE. */
272 netw->reactor_send_wants_read = 0;
273 uint32_t attempt = (remaining > UINT32_MAX) ? UINT32_MAX : (uint32_t)remaining;
274 ssize_t sent = netw->send_data_once((void*)netw,
275 netw->reactor_send_buf + netw->reactor_send_off,
276 attempt);
277 if (sent > 0) {
278 netw->reactor_send_off += (size_t)sent;
279 continue;
280 }
281 if (sent == NETW_IO_WANT_WRITE) {
282 atomic_fetch_add(&reactor->writes_partial, 1);
283 return 0;
284 }
285 if (sent == NETW_IO_WANT_READ) {
286 /* TLS renegotiation: the write needs inbound bytes first.
287 * Flag it so the readable path retries the drain; the
288 * caller's EPOLLOUT arming is harmless (edge-triggered,
289 * fires at most once while we wait). */
290 netw->reactor_send_wants_read = 1;
291 atomic_fetch_add(&reactor->writes_partial, 1);
292 return 0;
293 }
294 /* NETW_SOCKET_DISCONNECTED / NETW_SOCKET_ERROR (already logged
295 * inside the once-helper). */
296 return -1;
297 }
298}

300/* Push a fully-received frame onto the NETWORK's recv_buf. Mirrors
301 * the thread-mode netw_recv_func tail: optionally decompresses
302 * based on the frame's state-word flags, then list_push under the
303 * recvbolt mutex. Frees pkt_payload on failure. */
304static void reactor_recv_dispatch_frame(NETWORK* netw,
305 uint32_t pkt_state,
306 uint32_t pkt_length,
307 char* pkt_payload) {
308 /* Wrap the raw bytes in an N_STR, same shape the thread-mode
309 * recv path produces. */
310 N_STR* msg = NULL;
311 Malloc(msg, N_STR, 1);
312 if (!msg) {
313 Free(pkt_payload);
314 return;
315 }
316 msg->data = pkt_payload;
317 msg->length = (size_t)pkt_length + 1;
318 msg->written = (size_t)pkt_length;
319 /* Ensure NUL-terminator for downstream consumers that treat
320 * the buffer as a C string (decompression's unzip*_nstr
321 * inspects strptr->data[written] in some paths). The buffer
322 * was sized to pkt_length+1 in the caller. */
323 msg->data[pkt_length] = '\0';
324
325 /* Decompression: same logic as netw_recv_func. */
326 int want_zlib = (pkt_state & NETW_COMPRESSED_ZLIB) != 0;
327 int want_lz4 = (pkt_state & NETW_COMPRESSED_LZ4) != 0;
328 if (want_zlib || want_lz4) {
329 N_STR* plain = want_lz4 ? unzip4_nstr(msg) : unzip_nstr(msg);
330 if (plain) {
331 free_nstr(&msg);
332 msg = plain;
333 } else {
334 n_log(LOG_ERR,
335 "n_reactor: failed to decompress payload "
336 "(%" PRIu32 " bytes, codec=%s); dropping",
337 pkt_length, want_lz4 ? "lz4" : "zlib");
338 free_nstr(&msg);
339 return;
340 }
341 }
342
343 pthread_mutex_lock(&netw->recvbolt);
344 if (list_push(netw->recv_buf, msg, free_nstr_ptr) == FALSE) {
345 pthread_mutex_unlock(&netw->recvbolt);
346 n_log(LOG_ERR, "n_reactor: recv_buf list_push failed; dropping frame");
347 free_nstr(&msg);
348 return;
349 }
350 pthread_mutex_unlock(&netw->recvbolt);
351}

353/* Sweep the registered list for NETWORKs whose game thread has set
354 * NETW_EXIT_ASKED, drain pending sends best-effort,
355 * shutdown(SHUT_WR) so the peer observes EOF, unregister from the
356 * epoll set, and signal the close ack so the game thread can close
357 * the fd.
358 *
359 * Snapshots pointers under the lock, then processes them with the
360 * lock released, `n_reactor_unregister` reacquires the same lock,
361 * so we mustn't hold it across the call. The snapshot batch caps at
362 * N_REACTOR_BATCH_SIZE; any additional EXIT_ASKED entries are caught
363 * on the next sweep (the sweep runs at the end of every wake batch
364 * + every heartbeat tick, so the lag is bounded). */
365static void reactor_sweep_exit_asked(n_reactor* reactor) {
366 NETWORK* batch[N_REACTOR_BATCH_SIZE];
367 int count = 0;
368
369 pthread_mutex_lock(&reactor->registered_lock);
370 LIST_NODE* node = reactor->registered->start;
371 while (node && count < N_REACTOR_BATCH_SIZE) {
372 NETWORK* n = (NETWORK*)node->ptr;
373 node = node->next;
374 if (!n || !netw_atomic_read_reactor_mode(n)) continue;
375 uint32_t st = 0;
376 int thr_st = 0;
377 netw_get_state(n, &st, &thr_st);
378 if (st & NETW_EXIT_ASKED) {
379 batch[count++] = n;
380 }
381 }
382 pthread_mutex_unlock(&reactor->registered_lock);
383
384 for (int i = 0; i < count; i++) {
385 NETWORK* n = batch[i];
386 /* Best-effort drain, peer is going away so partial is fine.
387 * Swallow the return value: error / EAGAIN both lead to the
388 * same teardown path next. */
389 if (n->reactor_send_buf || (n->send_buf && n->send_buf->nb_items > 0)) {
390 (void)reactor_drain_writes(n, reactor);
391 }
392 /* Half-close the write side so the peer's read side observes
393 * EOF immediately. The fd close itself happens later in the
394 * caller's netw_close after `reactor_close_acked` flips.
395 * n_reactor_unregister publishes that ack with a release store
396 * as its final action, so nothing may touch `n` afterwards: the
397 * game thread's acquire-load can observe the ack and free the
398 * NETWORK the instant unregister returns. */
399 shutdown(n->link.sock, SHUT_WR);
400 n_reactor_unregister(reactor, n);
401 }
402}

404/* Drain the socket non-blockingly into the NETWORK's recv accumulator.
405 * Parses as many complete frames as the bytes allow; pushes each onto
406 * recv_buf. Edge-triggered: keeps reading until EAGAIN. Returns 1 on
407 * normal progress, 0 if the connection should be torn down (peer
408 * closed cleanly, or a hard error). */
409static int reactor_handle_readable(NETWORK* netw, n_reactor* reactor) {
410 char chunk[4096];
411 int eof = 0;
412
413 /* TLS-over-reactor: route through the NETWORK's
414 * single-attempt recv pointer (raw recv() on cleartext, SSL_read
415 * on crypto sockets). Looping until NETW_IO_WANT_READ both drains
416 * the edge-triggered readiness AND empties OpenSSL's internal
417 * plaintext buffer (SSL_read reports WANT_READ only once that
418 * buffer is dry), so no SSL_pending() poll is needed. */
419 netw->reactor_recv_wants_write = 0;
420
421 for (;;) {
422 ssize_t got = netw->recv_data_once((void*)netw, chunk, (uint32_t)sizeof(chunk));
423 if (got == NETW_SOCKET_DISCONNECTED) {
424 eof = 1;
425 break;
426 }
427 if (got == NETW_IO_WANT_READ) break; /* drained */
428 if (got == NETW_IO_WANT_WRITE) {
429 /* TLS renegotiation: the read needs outbound room first.
430 * Flag it so the run loop arms EPOLLOUT and re-runs this
431 * drain after the socket turns writable. */
432 netw->reactor_recv_wants_write = 1;
433 break;
434 }
435 if (got <= 0) {
436 /* NETW_SOCKET_ERROR: already logged inside the helper. */
437 return 0;
438 }
439
440 /* Feed `got` bytes through the state machine. */
441 const char* p = chunk;
442 size_t rem = (size_t)got;
443 while (rem > 0) {
444 switch (netw->reactor_read_phase) {
445 case 0: /* STATE word */
446 case 1: /* LENGTH word */
447 {
448 size_t need = 4 - (size_t)netw->reactor_read_hdr_have;
449 size_t take = (rem < need) ? rem : need;
450 memcpy(netw->reactor_read_hdr_buf + netw->reactor_read_hdr_have,
451 p, take);
452 netw->reactor_read_hdr_have += (int)take;
453 p += take;
454 rem -= take;
455 if (netw->reactor_read_hdr_have < 4) {
456 /* Need more bytes for this header word. */
457 atomic_fetch_add(&reactor->reads_partial, 1);
458 break;
459 }
460 /* Header word complete. */
461 uint32_t word;
462 memcpy(&word, netw->reactor_read_hdr_buf, sizeof(word));
463 word = ntohl(word);
464 netw->reactor_read_hdr_have = 0;
465 if (netw->reactor_read_phase == 0) {
466 netw->reactor_read_pkt_state = word;
467 if (word == NETW_EXIT_ASKED) {
468 /* Peer requested clean shutdown. Treat as EOF. */
469 eof = 1;
470 rem = 0;
471 break;
472 }
473 netw->reactor_read_phase = 1;
474 } else {
475 netw->reactor_read_pkt_length = word;
476 /* Allocate the payload buffer. +1 for the NUL
477 * the dispatch helper writes after fill. */
478 if (word == 0) {
479 /* Zero-byte payload: dispatch immediately
480 * with an empty N_STR-shaped buffer. */
481 char* empty = NULL;
482 Malloc(empty, char, 1);
483 if (!empty) {
484 n_log(LOG_ERR, "n_reactor: alloc(empty payload) failed");
485 return 0;
486 }
487 reactor_recv_dispatch_frame(netw,
488 netw->reactor_read_pkt_state,
489 0, empty);
490 netw->reactor_read_phase = 0;
491 } else {
492 Malloc(netw->reactor_read_payload, char, word + 1);
493 if (!netw->reactor_read_payload) {
494 n_log(LOG_ERR, "n_reactor: alloc(%u-byte payload) failed",
495 word);
496 return 0;
497 }
498 netw->reactor_read_payload_have = 0;
499 netw->reactor_read_phase = 2;
500 }
501 }
502 break;
503 }
504 case 2: /* PAYLOAD */
505 {
506 size_t need = (size_t)netw->reactor_read_pkt_length - netw->reactor_read_payload_have;
507 size_t take = (rem < need) ? rem : need;
508 memcpy(netw->reactor_read_payload + netw->reactor_read_payload_have,
509 p, take);
510 netw->reactor_read_payload_have += take;
511 p += take;
512 rem -= take;
513 if (netw->reactor_read_payload_have < netw->reactor_read_pkt_length) {
514 /* Partial payload, stay in payload-read state, wait
515 * for more bytes (next read or next epoll event). */
516 atomic_fetch_add(&reactor->reads_partial, 1);
517 break;
518 }
519 /* Complete payload, dispatch. ownership of the
520 * buffer transfers into the dispatcher. */
521 char* payload = netw->reactor_read_payload;
522 netw->reactor_read_payload = NULL;
523 reactor_recv_dispatch_frame(netw,
524 netw->reactor_read_pkt_state,
525 netw->reactor_read_pkt_length,
526 payload);
527 netw->reactor_read_phase = 0;
528 netw->reactor_read_payload_have = 0;
529 break;
530 }
531 }
532 }
533 }
534 if (eof) return 0;
535 return 1;
536}

540n_reactor* n_reactor_new(int max_fds_hint) {
541 (void)max_fds_hint; /* per-fd table grows on demand at register time */
542
543 n_reactor* r = NULL;
544 Malloc(r, n_reactor, 1); /* calloc-backed: r is zero-initialised */
545 __n_assert(r, return NULL);
546 r->epoll_fd = -1;
547 r->stop_efd = -1;
548 r->wake_efd = -1;
549
550 /* CLOEXEC because forking the asset auditor (server SIGHUP path
551 * does this) shouldn't leak the epoll fd to the child. */
552 r->epoll_fd = epoll_create1(EPOLL_CLOEXEC);
553 if (r->epoll_fd < 0) {
554 n_log(LOG_ERR, "n_reactor_new: epoll_create1 failed: %s",
555 strerror(errno));
556 goto fail;
557 }
558
559 /* Both eventfds: nonblocking (so drain doesn't block the run
560 * loop) and CLOEXEC (same fork hygiene). EFD_SEMAPHORE is NOT
561 * set, we want each `write` to add to the count, not sit
562 * one-event-at-a-time. */
563 r->stop_efd = eventfd(0, EFD_NONBLOCK | EFD_CLOEXEC);
564 if (r->stop_efd < 0) {
565 n_log(LOG_ERR, "n_reactor_new: eventfd(stop) failed: %s",
566 strerror(errno));
567 goto fail;
568 }
569 r->wake_efd = eventfd(0, EFD_NONBLOCK | EFD_CLOEXEC);
570 if (r->wake_efd < 0) {
571 n_log(LOG_ERR, "n_reactor_new: eventfd(wake) failed: %s",
572 strerror(errno));
573 goto fail;
574 }
575
576 if (!register_internal_efd(r->epoll_fd, r->stop_efd, N_REACTOR_TAG_STOP) ||
577 !register_internal_efd(r->epoll_fd, r->wake_efd, N_REACTOR_TAG_WAKE)) {
578 goto fail;
579 }
580
581 r->registered = new_generic_list(MAX_LIST_ITEMS);
582 if (!r->registered) {
583 n_log(LOG_ERR, "n_reactor_new: cannot allocate registered list");
584 goto fail;
585 }
586 pthread_mutex_init(&r->registered_lock, NULL);
587
588 atomic_store(&r->events_processed, 0);
589 atomic_store(&r->writes_partial, 0);
590 atomic_store(&r->reads_partial, 0);
591 atomic_store(&r->fds_registered, 0);
592 atomic_store(&r->fds_unregistered, 0);
593 atomic_store(&r->wake_signals, 0);
594 atomic_store(&r->wake_walks, 0);
595 atomic_store(&r->wake_walk_visits, 0);
596 atomic_store(&r->wake_walk_drains, 0);
597 r->stop_requested = 0;
598
599 /* Dirty-list: see n_reactor.h dirty_pending field comment for the
600 * design rationale. */
601 r->dirty_pending = new_generic_list(MAX_LIST_ITEMS);
602 if (!r->dirty_pending) {
603 n_log(LOG_ERR, "n_reactor_new: cannot allocate dirty_pending list");
604 goto fail;
605 }
606 pthread_mutex_init(&r->dirty_lock, NULL);
607
608 n_log(LOG_INFO, "n_reactor_new: created (epoll_fd=%d stop_efd=%d wake_efd=%d)",
609 r->epoll_fd, r->stop_efd, r->wake_efd);
610 return r;
611
612fail:
613 if (r) {
614 if (r->wake_efd >= 0) close(r->wake_efd);
615 if (r->stop_efd >= 0) close(r->stop_efd);
616 if (r->epoll_fd >= 0) close(r->epoll_fd);
617 Free(r);
618 }
619 return NULL;
620}

622void n_reactor_destroy(n_reactor** reactor) {
623 if (!reactor || !*reactor) return;
624 n_reactor* r = *reactor;
625
626 /* If the run loop is still on another thread, the caller
627 * should have called `n_reactor_stop` and joined that thread
628 * already. We don't enforce, best-effort cleanup either way. */
629 if (r->wake_efd >= 0) close(r->wake_efd);
630 if (r->stop_efd >= 0) close(r->stop_efd);
631 if (r->epoll_fd >= 0) close(r->epoll_fd);
632 if (r->registered) {
633 list_destroy(&r->registered); /* nodes hold raw NETWORK *, no destructor */
634 pthread_mutex_destroy(&r->registered_lock);
635 }
636 if (r->dirty_pending) {
637 list_destroy(&r->dirty_pending); /* same: nodes are NETWORK* aliases, owner == registered list */
638 pthread_mutex_destroy(&r->dirty_lock);
639 }
640
641 Free(r);
642 *reactor = NULL;
643}

645void n_reactor_run(n_reactor* reactor) {
646 if (!reactor) return;
647
648 struct epoll_event events[N_REACTOR_BATCH_SIZE];
649
650 while (!reactor->stop_requested) {
651 /* Heartbeat: bound the wait so the EXIT_ASKED sweep runs at
652 * least once per second even if nobody calls
653 * n_reactor_notify_send / n_reactor_close_netw_sync. The
654 * notify path remains the immediate one (sub-millisecond);
655 * the heartbeat is the safety net. */
656 int n = epoll_wait(reactor->epoll_fd, events,
657 N_REACTOR_BATCH_SIZE, 1000);
658 if (n < 0) {
659 if (errno == EINTR) continue;
660 n_log(LOG_ERR, "n_reactor_run: epoll_wait failed: %s",
661 strerror(errno));
662 break;
663 }
664 if (n == 0) {
665 /* Heartbeat tick: nothing on the wire, but a connection
666 * may have been flagged for teardown without an explicit
667 * notify. */
668 reactor_sweep_exit_asked(reactor);
669 continue;
670 }
671
672 for (int i = 0; i < n; i++) {
673 uint64_t tag = events[i].data.u64;
674 atomic_fetch_add(&reactor->events_processed, 1);
675
676 if (tag == N_REACTOR_TAG_STOP) {
677 /* Drain the eventfd so it doesn't refire, then
678 * mark the loop for exit. We don't break
679 * immediately, finish processing this batch so
680 * any concurrent NETWORK events still get drained. */
681 drain_eventfd(reactor->stop_efd);
682 reactor->stop_requested = 1;
683 continue;
684 }
685 if (tag == N_REACTOR_TAG_WAKE) {
686 /* Producer-side wakeup. Drain the eventfd, then scan
687 * registered NETWORKs for any with pending sends and
688 * try to drain them, for the common no-back-pressure
689 * case that finishes the work before the next
690 * epoll_wait, avoiding the EPOLLOUT round trip. */
691 long long drained = drain_eventfd(reactor->wake_efd);
692 atomic_fetch_add(&reactor->wake_signals, drained);
693
694 /* Wake-handler walks ONLY the dirty list. Splice
695 * dirty_pending into a local list under a single lock
696 * pair so producers can continue pushing onto the fresh
697 * list (picked up on the next wake) without contending
698 * with the drain loop. */
699 atomic_fetch_add(&reactor->wake_walks, 1);
700 long long visits_this_walk = 0;
701 long long drains_this_walk = 0;
702
703 LIST* walk_list = NULL;
704 int walk_owned = 0; /* 1 = we allocated walk_list, must list_destroy */
705 pthread_mutex_t* walk_lock = NULL;
706 pthread_mutex_lock(&reactor->dirty_lock);
707 {
708 LIST* old = reactor->dirty_pending;
709 LIST* fresh = new_generic_list(MAX_LIST_ITEMS);
710 if (fresh) {
711 reactor->dirty_pending = fresh;
712 walk_list = old;
713 walk_owned = 1;
714 } else {
715 /* Allocation failure: fall back to walking in-place
716 * under the lock. Suboptimal but correct. */
717 walk_list = old;
718 walk_lock = &reactor->dirty_lock;
719 }
720 }
721 if (walk_owned) {
722 pthread_mutex_unlock(&reactor->dirty_lock);
723 }
724
725 LIST_NODE* node = walk_list ? walk_list->start : NULL;
726 while (node) {
727 NETWORK* netw = (NETWORK*)node->ptr;
728 LIST_NODE* next = node->next;
729 /* Every entry is by construction a NETWORK with
730 * pending data when it was pushed; the reactor_mode
731 * gate stays for safety against concurrent unregister. */
732 if (netw && netw_atomic_read_reactor_mode(netw)) {
733 visits_this_walk++;
734 /* Clear the in-list flag BEFORE draining.
735 * A producer that pushes during the drain succeeds
736 * the CAS and re-adds for the next wake; no
737 * message can be lost. */
738 __atomic_store_n(&netw->in_dirty_list, 0, __ATOMIC_RELEASE);
739 /* Quick check: nothing to do if both buffers
740 * (in-flight + queue) are empty. The recvbolt
741 * isn't taken, we read nb_items as a hint;
742 * the producer may be in the middle of a
743 * push, but the next wake will catch it. */
744 int has_inflight = (netw->reactor_send_buf != NULL);
745 int has_queued = (netw->send_buf->nb_items > 0);
746 if (has_inflight || has_queued) {
747 drains_this_walk++;
748 int rc = reactor_drain_writes(netw, reactor);
749 if (rc == 0) {
750 /* EAGAIN, arm EPOLLOUT for back-pressure relief. */
751 if (!netw->reactor_write_armed) {
752 if (reactor_epoll_mod(reactor, netw,
753 EPOLLIN | EPOLLOUT | EPOLLRDHUP | EPOLLET)) {
754 netw->reactor_write_armed = 1;
755 }
756 }
757 } else if (rc < 0) {
758 /* n_reactor_unregister takes its own
759 * locks (registered_lock + dirty_lock).
760 * Release whatever walk-side lock we
761 * currently hold first to preserve the
762 * documented ordering and avoid
763 * self-deadlock; reacquire after. Set the
764 * state flag BEFORE unregister: unregister
765 * publishes the close ack as its last act,
766 * after which the game thread may free the
767 * NETWORK, so netw must not be touched. */
768 if (walk_lock) pthread_mutex_unlock(walk_lock);
769 netw_set(netw, NETW_ERROR);
770 n_reactor_unregister(reactor, netw);
771 if (walk_lock) pthread_mutex_lock(walk_lock);
772 } else {
773 /* Fully drained, disarm EPOLLOUT if armed. */
774 if (netw->reactor_write_armed) {
775 if (reactor_epoll_mod(reactor, netw,
776 EPOLLIN | EPOLLRDHUP | EPOLLET)) {
777 netw->reactor_write_armed = 0;
778 }
779 }
780 }
781 }
782 }
783 node = next;
784 }
785 /* Release whatever walk-side lock we still hold
786 * (dirty_lock in the fallback path, none for the
787 * spliced-list happy path), then free the spliced list
788 * if we owned it. */
789 if (walk_lock) pthread_mutex_unlock(walk_lock);
790 if (walk_owned && walk_list) {
791 list_destroy(&walk_list); /* nodes are NETWORK* aliases, no destructor */
792 }
793 /* Flush per-walk visit/drain tally into the lifetime
794 * accumulators after dropping the lock. */
795 atomic_fetch_add(&reactor->wake_walk_visits, visits_this_walk);
796 atomic_fetch_add(&reactor->wake_walk_drains, drains_this_walk);
797 continue;
798 }
799
800 /* NETWORK event. data.ptr points at the NETWORK *.
801 * We rely on data.u64 >= N_REACTOR_TAG_INTERNAL_MAX
802 * because every legitimate pointer is far above 16 in
803 * any modern address space. */
804 NETWORK* netw = (NETWORK*)events[i].data.ptr;
805 if (!netw) continue;
806
807 uint32_t evmask = events[i].events;
808 if (evmask & (EPOLLERR | EPOLLHUP | EPOLLRDHUP)) {
809 /* Hard error or peer closed: flag the NETWORK so the
810 * game thread observes it on next netw_get_msg via the
811 * existing state-flag check, THEN unregister. Order
812 * matters: unregister publishes the close ack as its
813 * last act, releasing a game thread that may be spinning
814 * in n_reactor_close_netw_sync and free the NETWORK, so
815 * netw must not be touched after unregister returns. */
816 netw_set(netw, NETW_ERROR);
817 n_reactor_unregister(reactor, netw);
818 continue;
819 }
820 if (evmask & EPOLLIN) {
821 if (!reactor_handle_readable(netw, reactor)) {
822 /* EOF or unrecoverable read error. Flag before
823 * unregister (unregister publishes the close ack
824 * last; netw may be freed the moment it returns). */
825 netw_set(netw, NETW_EXIT_ASKED);
826 n_reactor_unregister(reactor, netw);
827 continue;
828 }
829 /* TLS plumbing: a send that returned
830 * WANT_READ (renegotiation) can progress now that
831 * inbound bytes arrived. */
832 if (netw->reactor_send_wants_read) {
833 int rc = reactor_drain_writes(netw, reactor);
834 if (rc < 0) {
835 /* Flag before unregister: unregister publishes
836 * the close ack last and netw may be freed once
837 * it returns. */
838 netw_set(netw, NETW_ERROR);
839 n_reactor_unregister(reactor, netw);
840 continue;
841 }
842 if (rc > 0 && netw->reactor_write_armed &&
843 !netw->reactor_recv_wants_write) {
844 if (reactor_epoll_mod(reactor, netw,
845 EPOLLIN | EPOLLRDHUP | EPOLLET)) {
846 netw->reactor_write_armed = 0;
847 }
848 } else if (rc == 0 && !netw->reactor_write_armed) {
849 if (reactor_epoll_mod(reactor, netw,
850 EPOLLIN | EPOLLOUT | EPOLLRDHUP | EPOLLET)) {
851 netw->reactor_write_armed = 1;
852 }
853 }
854 }
855 /* TLS plumbing: a recv that returned WANT_WRITE needs
856 * the socket writable: arm EPOLLOUT so the drain
857 * re-runs from that branch. */
858 if (netw->reactor_recv_wants_write && !netw->reactor_write_armed) {
859 if (reactor_epoll_mod(reactor, netw,
860 EPOLLIN | EPOLLOUT | EPOLLRDHUP | EPOLLET)) {
861 netw->reactor_write_armed = 1;
862 }
863 }
864 }
865 if (evmask & EPOLLOUT) {
866 /* TLS plumbing: a recv blocked on WANT_WRITE
867 * retries first, the socket just turned writable. */
868 if (netw->reactor_recv_wants_write) {
869 if (!reactor_handle_readable(netw, reactor)) {
870 /* Flag before unregister: unregister publishes
871 * the close ack last and netw may be freed once
872 * it returns. */
873 netw_set(netw, NETW_EXIT_ASKED);
874 n_reactor_unregister(reactor, netw);
875 continue;
876 }
877 }
878 int rc = reactor_drain_writes(netw, reactor);
879 if (rc < 0) {
880 /* Flag before unregister: unregister publishes the
881 * close ack last and netw may be freed once it
882 * returns. */
883 netw_set(netw, NETW_ERROR);
884 n_reactor_unregister(reactor, netw);
885 continue;
886 }
887 if (rc > 0 && netw->reactor_write_armed &&
888 !netw->reactor_recv_wants_write) {
889 /* Drained; disarm EPOLLOUT to avoid spurious wakeups
890 * (kept armed while a renegotiating recv still
891 * needs the writable signal). */
892 if (reactor_epoll_mod(reactor, netw,
893 EPOLLIN | EPOLLRDHUP | EPOLLET)) {
894 netw->reactor_write_armed = 0;
895 }
896 }
897 /* rc == 0 (back-pressure / WANT) means EPOLLOUT remains
898 * armed, kernel will fire again when buffer has room. */
899 }
900 }
901 /* End-of-batch teardown sweep, pick up any NETWORK whose
902 * game-side has flagged EXIT_ASKED during this batch (or
903 * earlier batches that didn't reach the snapshot cap). */
904 reactor_sweep_exit_asked(reactor);
905 }
906
907 n_log(LOG_INFO, "n_reactor_run: exiting (events=%lld)",
908 (long long)atomic_load(&reactor->events_processed));
909}

911void n_reactor_stop(n_reactor* reactor) {
912 if (!reactor || reactor->stop_efd < 0) return;
913 /* eventfd write semantics: any 8-byte value > 0 increments the
914 * counter and triggers EPOLLIN on the read side. Use 1 for
915 * clarity. */
916 uint64_t one = 1;
917 ssize_t w = write(reactor->stop_efd, &one, sizeof(one));
918 (void)w; /* best-effort; if it fails the loop will exit on next
919 * idle epoll_wait via stop_requested check */
920}

944int n_reactor_register(n_reactor* reactor, NETWORK* netw) {
945 if (!reactor || !netw) return 0;
946 if (netw_atomic_read_reactor_mode(netw)) {
947 n_log(LOG_WARNING, "n_reactor_register: socket %d already in reactor mode",
948 netw->link.sock);
949 return 0;
950 }
951 if (netw->threaded_engine_status == NETW_THR_ENGINE_STARTED) {
952 n_log(LOG_ERR,
953 "n_reactor_register: socket %d has thread engine started; "
954 "reactor and thread mode are mutually exclusive",
955 netw->link.sock);
956 return 0;
957 }
958 if (!set_nonblocking(netw->link.sock)) {
959 n_log(LOG_ERR, "n_reactor_register: O_NONBLOCK on socket %d failed: %s",
960 netw->link.sock, strerror(errno));
961 return 0;
962 }
963 /* The reactor's I/O goes through the single-attempt
964 * pointers (TLS-aware). netw_new seeds them (and the crypto setup
965 * swaps in the SSL variants), but a hand-constructed NETWORK
966 * shell (tests, embedders) may leave them NULL, default to the
967 * raw cleartext variants rather than dereferencing NULL. */
968 if (!netw->send_data_once) netw->send_data_once = &send_data_once;
969 if (!netw->recv_data_once) netw->recv_data_once = &recv_data_once;
970
971 /* Reset accumulator state from any prior use of the slot. */
972 reactor_recv_state_reset(netw);
973 reactor_send_state_reset(netw);
974 /* Clear stale ack from a prior cycle. This runs before the netw is
975 * published to the reactor thread (reactor_mode release, below), so
976 * it cannot race the sweep; use the same atomic accessor for
977 * consistency with the release/acquire pair on this flag. */
978 __atomic_store_n(&netw->reactor_close_acked, 0, __ATOMIC_RELEASE);
979
980 struct epoll_event ev;
981 /* Edge-triggered: drain the socket fully on each event. EPOLLRDHUP
982 * gives us peer-half-close as a separate signal. EPOLLOUT is added
983 * lazily by the send path when there's pending data. */
984 ev.events = EPOLLIN | EPOLLRDHUP | EPOLLET;
985 ev.data.ptr = netw;
986 if (epoll_ctl(reactor->epoll_fd, EPOLL_CTL_ADD, netw->link.sock, &ev) != 0) {
987 n_log(LOG_ERR, "n_reactor_register: epoll_ctl ADD socket %d failed: %s",
988 netw->link.sock, strerror(errno));
989 return 0;
990 }
991 netw_atomic_write_reactor_handle(netw, (void*)reactor);
992 /* Latch that this NETWORK now owes a reactor close-handshake ack.
993 * Set only here, on full registration success, and never cleared by
994 * the reactor; netw_close keys its ack-wait on this rather than on
995 * the live reactor_mode (see the field doc in n_network.h). The
996 * register failure paths above all return 0 before this point, so a
997 * caller that netw_close's a never-registered NETWORK leaves the
998 * latch 0 and avoids an unanswerable wait. */
999 __atomic_store_n(&netw->reactor_registered, 1, __ATOMIC_RELEASE);
1000 /* Publish reactor_mode last (release on the flag pairs with the
1001 * acquire reads in netw_close / netw_add_msg) so any thread that
1002 * sees reactor_mode == 1 also sees a non-NULL reactor_handle. */
1003 netw_atomic_write_reactor_mode(netw, 1);
1004
1005 /* Track in the reactor's registered list so the wake-event
1006 * handler can find this NETWORK without a tree lookup. The
1007 * list_push happens with NULL destructor, nodes hold raw
1008 * NETWORK *, ownership stays with the caller. */
1009 pthread_mutex_lock(&reactor->registered_lock);
1010 list_push(reactor->registered, netw, NULL);
1011 pthread_mutex_unlock(&reactor->registered_lock);
1012
1013 atomic_fetch_add(&reactor->fds_registered, 1);
1014 n_log(LOG_DEBUG, "n_reactor: registered socket %d", netw->link.sock);
1015 return 1;
1016}

1018void n_reactor_unregister(n_reactor* reactor, NETWORK* netw) {
1019 if (!reactor || !netw) return;
1020 if (!netw_atomic_read_reactor_mode(netw)) return;
1021 /* EPOLL_CTL_DEL is best-effort, if the socket is already closed
1022 * the kernel returns EBADF, which we silently ignore. */
1023 if (epoll_ctl(reactor->epoll_fd, EPOLL_CTL_DEL, netw->link.sock, NULL) != 0) {
1024 if (errno != EBADF && errno != ENOENT) {
1025 n_log(LOG_WARNING,
1026 "n_reactor_unregister: epoll_ctl DEL socket %d "
1027 "failed: %s",
1028 netw->link.sock, strerror(errno));
1029 }
1030 }
1031 reactor_recv_state_reset(netw);
1032 reactor_send_state_reset(netw);
1033
1034 /* Remove from registered list. Walk to find, small-N typical. */
1035 pthread_mutex_lock(&reactor->registered_lock);
1036 LIST_NODE* node = reactor->registered->start;
1037 while (node) {
1038 if (node->ptr == (void*)netw) {
1039 remove_list_node(reactor->registered, node, NETWORK);
1040 break;
1041 }
1042 node = node->next;
1043 }
1044 pthread_mutex_unlock(&reactor->registered_lock);
1045
1046 /* Defensive removal from dirty_pending. A racing producer
1047 * could have CAS'd in_dirty_list to 1 between the wake handler's
1048 * splice and our unregister. Walk + remove if present, then clear
1049 * in_dirty_list BEFORE clearing reactor_mode (below) so a producer
1050 * that observes the cleared mode never leaves a stale flag. */
1051 pthread_mutex_lock(&reactor->dirty_lock);
1052 {
1053 LIST_NODE* dn = reactor->dirty_pending->start;
1054 while (dn) {
1055 if (dn->ptr == (void*)netw) {
1056 remove_list_node(reactor->dirty_pending, dn, NETWORK);
1057 break;
1058 }
1059 dn = dn->next;
1060 }
1061 }
1062 pthread_mutex_unlock(&reactor->dirty_lock);
1063 __atomic_store_n(&netw->in_dirty_list, 0, __ATOMIC_RELEASE);
1064
1065 /* Clear reactor_mode first (release) so any thread that observes
1066 * the cleared flag will not subsequently dereference an invalid
1067 * reactor_handle pointer. The handle is then nulled, a brief
1068 * window exists where mode == 0 but handle != NULL, which is
1069 * harmless because every consumer guards on mode first. */
1070 netw_atomic_write_reactor_mode(netw, 0);
1071 netw_atomic_write_reactor_handle(netw, NULL);
1072 atomic_fetch_add(&reactor->fds_unregistered, 1);
1073 n_log(LOG_DEBUG, "n_reactor: unregistered socket %d", netw->link.sock);
1074
1075 /* Publish the close ack as the very last action. Every unregister
1076 * path, the EXIT_ASKED sweep, a peer EOF / EPOLLRDHUP, or a hard
1077 * I/O error, must release a game thread that may already be
1078 * spinning in n_reactor_close_netw_sync; only the sweep used to do
1079 * this, so an event-driven unregister stranded the close (TSan
1080 * timeout). The release store orders every side effect above before
1081 * the game thread's acquire-load, after which it may close(fd) and
1082 * free the NETWORK, so nothing below this line may touch `netw`. */
1083 __atomic_store_n(&netw->reactor_close_acked, 1, __ATOMIC_RELEASE);
1084}

1086void n_reactor_notify_send(NETWORK* netw) {
1087 if (!netw) return;
1088 n_reactor* r = (n_reactor*)netw_atomic_read_reactor_handle(netw);
1089 if (!r) return;
1090 if (r->wake_efd < 0) return;
1091
1092 /* Dirty-list: add this NETWORK to the reactor's pending list via
1093 * a CAS guard so only the first push between drains takes the
1094 * dirty_lock. Multiple producer threads pushing to the SAME
1095 * NETWORK between drains see CAS fail and skip the push (the first
1096 * one already enqueued it). The reactor clears in_dirty_list BEFORE
1097 * draining so any producer that pushes during the drain succeeds
1098 * the CAS and re-adds for the next pass. */
1099 {
1100 int expected = 0;
1101 if (__atomic_compare_exchange_n(&netw->in_dirty_list, &expected, 1,
1102 0, __ATOMIC_ACQ_REL, __ATOMIC_ACQUIRE)) {
1103 pthread_mutex_lock(&r->dirty_lock);
1104 list_push(r->dirty_pending, netw, NULL); /* NULL dtor: alias only */
1105 pthread_mutex_unlock(&r->dirty_lock);
1106 }
1107 }
1108
1109 uint64_t one = 1;
1110 ssize_t w = write(r->wake_efd, &one, sizeof(one));
1111 (void)w; /* best-effort; eventfd write of 8 bytes either succeeds
1112 * fully or returns EAGAIN if the counter is at UINT64_MAX
1113 * minus 1, which is impossible in practice */
1114}
1115
1116void n_reactor_close_netw_sync(NETWORK* netw) {
1117 if (!netw) return;
1118
1119 /* Step 1: if the connection is still registered, prod the reactor to
1120 * tear it down. We must NOT gate the whole function on reactor_mode:
1121 * the reactor can clear reactor_mode mid-unregister (peer EOF / hard
1122 * error) and then keep touching the NETWORK for a few more lines
1123 * (the EPOLL_CTL_DEL, list removal, the unregistered-socket log read
1124 * of link.sock) before publishing the ack. Returning early on
1125 * reactor_mode == 0 would let the caller free the NETWORK under those
1126 * accesses (use-after-free). Instead we always wait for the ack
1127 * below; this prod is only needed when the reactor has not yet begun
1128 * the teardown. The reactor_mode read is racy but only steers the
1129 * (idempotent) prod, not the free, so the raciness is benign.
1130 *
1131 * netw_set serialises under eventbolt; n_reactor_notify_send is a
1132 * no-op once reactor_handle has been nulled, which is fine because in
1133 * that case the ack is already set or imminently will be. */
1134 if (netw_atomic_read_reactor_mode(netw)) {
1135 netw_set(netw, NETW_EXIT_ASKED);
1136 n_reactor_notify_send(netw);
1137 }
1138
1139 /* Step 2: wait for the reactor to ack. Busy-poll with a 1 ms sleep,
1140 * same shape as the existing nb_running_threads spin in netw_close.
1141 * The ack is published by n_reactor_unregister as the very last thing
1142 * the reactor ever does with this NETWORK, regardless of which path
1143 * unregistered it (the EXIT_ASKED sweep in the common case, but also
1144 * a concurrent peer EOF / hard error). reactor_mode is always cleared
1145 * before the ack store, so once we observe reactor_mode == 0 the ack
1146 * is guaranteed to land. The acquire-load pairs with unregister's
1147 * release store so all of its side effects (epoll DEL, cleared
1148 * reactor_mode, freed accumulators, list removal, the final log) are
1149 * visible, and ordered before, the caller's subsequent close(fd) /
1150 * free. */

1152 struct timespec ts = {0, 1000000L}; /* 1 ms */
1153 nanosleep(&ts, NULL);
1154 }
