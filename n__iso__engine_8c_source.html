227int ScreenToMap(int mx, int my, int* Tilex, int* Tiley, ALLEGRO_BITMAP* mousemap) {
228 /* SCREEN TO MAP VARIABLES */
229 int RegionX, RegionY,
230 RegionDX = 0, RegionDY = 0,
231 MouseMapX, MouseMapY,
232 MouseMapWidth, MouseMapHeight;
233
234 MouseMapWidth = al_get_bitmap_width(mousemap);
235 MouseMapHeight = al_get_bitmap_height(mousemap);
236
237 /* First step find out where we are on the mousemap */
238 RegionX = (mx / MouseMapWidth);
239 RegionY = (my / MouseMapHeight) << 1; /* The multiplying by two is very important */
240
241 /* Second Step: Find out WHERE in the mousemap our mouse is, by finding MouseMapX and MouseMapY. */
242
243 MouseMapX = mx % MouseMapWidth;
244 MouseMapY = my % MouseMapHeight;
245
246 /* third step: Find the color in the mouse map */
247 ALLEGRO_COLOR c = al_get_pixel(mousemap, MouseMapX, MouseMapY);
248 unsigned char r, g, b;
249 al_unmap_rgb(c, &r, &g, &b);
250
251 if (r == 255 && g == 0 && b == 0) {
252 /* red: top-left */
253 RegionDX = -1;
254 RegionDY = -1;
255 }
256
257 if (r == 255 && g == 255 && b == 0) {
258 /* yellow: top-right */
259 RegionDX = 0;
260 RegionDY = -1;
261 }
262
263 if (r == 255 && g == 255 && b == 255) {
264 /* white: center diamond */
265 RegionDX = 0;
266 RegionDY = 0;
267 }
268
269 if (r == 0 && g == 255 && b == 0) {
270 /* green: bottom-left */
271 RegionDX = -1;
272 RegionDY = 1;
273 }
274
275 if (r == 0 && g == 0 && b == 255) {
276 /* blue: bottom-right */
277 RegionDX = 0;
278 RegionDY = 1;
279 }
280
281 *Tilex = (RegionDX + RegionX);
282 *Tiley = (RegionDY + RegionY);
283
284 return TRUE;
285} /* ScreenToMap( ... ) */

338int draw_map(MAP* map, ALLEGRO_BITMAP* bmp, int destx, int desty, int mode) {
339 int a, b, /* iterators */
340 x, y, /* temp store of coordinate */
341 mvx, mvy, /* precomputed offset */
342 tw, th, /* number of tile for one screen */
343 TW2, TH2; /* size of TILE_W /2 and TILE_H/2 */
344
345 ALLEGRO_BITMAP* prev_target = al_get_target_bitmap();
346 al_set_target_bitmap(bmp);
347
348 int bmp_w = al_get_bitmap_width(bmp);
349 int bmp_h = al_get_bitmap_height(bmp);
350
351 /* computing tw&th */
352 tw = 1 + ((bmp_w - destx) / map->TILEW);
353 th = 3 + ((bmp_h - desty) / (map->TILEH >> 1));
354
355 TW2 = map->TILEW >> 1;
356 TH2 = map->TILEH >> 1;
357
358 mvx = destx - TW2 - map->X;
359 mvy = desty - TH2 - map->Y;
360
361 for (a = 0; a <= tw; a++) {
362 for (b = 0; b <= th; b++) {
363 x = (a * map->TILEW + ((b & 1) * TW2)) - (map->TILEW >> 1) + mvx;
364 y = (b * TH2) + mvy;
365
366 if ((a + map->ptanchorX) < map->XSIZE && (b + map->ptanchorY) < map->YSIZE) {
367 if (map->grid[a + map->ptanchorX + (b + map->ptanchorY) * map->XSIZE].tilenumber == FALSE) {
368 al_draw_bitmap(map->colortile, (float)x, (float)y, 0);
369 }
370
371 else {
372 /* placeholder: add sprite blit */
373 float fx = (float)x;
374 float fy = (float)y;
375 float ftw = (float)map->TILEW;
376 float fth = (float)map->TILEH;
377
378 al_draw_filled_triangle(fx, fy + fth / 2.0f,
379 fx + ftw, fy + fth / 2.0f,
380 fx + ftw / 2.0f, fy,
381 al_map_rgb(255, 255, 0));
382 al_draw_filled_triangle(fx, fy + fth / 2.0f,
383 fx + ftw, fy + fth / 2.0f,
384 fx + ftw / 2.0f, fy + fth,
385 al_map_rgb(255, 0, 255));
386
387 } /* if( map -> grid... ) ) */
388
389 if (mode == 1) {
390 float fx = (float)x;
391 float fy = (float)y;
392 float ftw = (float)map->TILEW;
393 float fth = (float)map->TILEH;
394
395 al_draw_line(fx + ftw / 2.0f - 2, fy,
396 fx, fy + fth / 2.0f - 1,
397 map->wirecolor, 0);
398 al_draw_line(fx + ftw / 2.0f - 2, fy + fth - 1,
399 fx, fy + fth / 2.0f,
400 map->wirecolor, 0);
401 al_draw_line(fx + ftw / 2.0f + 1, fy + fth - 1,
402 fx + ftw - 1, fy + fth / 2.0f,
403 map->wirecolor, 0);
404 al_draw_line(fx + ftw / 2.0f + 1, fy,
405 fx + ftw - 1, fy + fth / 2.0f - 1,
406 map->wirecolor, 0);
407 }
408
409 /* TODO: Object-library TTL handling is not implemented here.
410 * Intended behavior:
411 * - Check if object is in the library; if yes, refresh its TTL.
412 * - If not, add it with the default TTL.
413 */
414
415 } /* if( (a + ...) ) */
416
417 } /* for( a... ) */
418 } /* for( b... ) */
419
420 /* TODO: sort the object list by X and by Y and blit all the sprites of the active list */
421
422 /* restore previous target */
423 if (prev_target)
424 al_set_target_bitmap(prev_target);
425
426 return TRUE;
427
428} /* draw_map( ... ) */

436int load_map(MAP** map, char* filename) {
437 ALLEGRO_FILE* loaded;
438
439 char temp_name[1024];
440 int temp_XSIZE, temp_YSIZE, temp_TILEW, temp_TILEH;
441 int temp_ptanchorX, temp_ptanchorY, temp_X, temp_Y;
442 uint32_t temp_bgcolor, temp_wirecolor;
443
444 int it = 0;
445
446 loaded = al_fopen(filename, "rb");
447 if (!loaded) {
448 n_log(LOG_ERR, "could not open %s for reading", filename);
449 return FALSE;
450 }
451
452 if (map && *map)
453 free_map(map);
454
455 /* reading name */
456 it = al_fread32le(loaded);
457
458 if (it > 1023) it = 1023;
459 al_fread(loaded, temp_name, (size_t)it);
460 temp_name[it] = '\0';
461
462 /* reading states */
463 temp_XSIZE = al_fread32le(loaded);
464 temp_YSIZE = al_fread32le(loaded);
465 temp_TILEW = al_fread32le(loaded);
466 temp_TILEH = al_fread32le(loaded);
467 temp_ptanchorX = al_fread32le(loaded);
468 temp_ptanchorY = al_fread32le(loaded);
469 temp_X = al_fread32le(loaded);
470 temp_Y = al_fread32le(loaded);
471 temp_bgcolor = (uint32_t)al_fread32le(loaded);
472 temp_wirecolor = (uint32_t)al_fread32le(loaded);
473
474 if (create_empty_map(map, temp_name,
475 temp_XSIZE, temp_YSIZE,
476 temp_TILEW, temp_TILEH,
477 2000, 2000, 2000, 2000, 2000) != TRUE) {
478 al_fclose(loaded);
479 return FALSE;
480 }
481 (*map)->XSIZE = temp_XSIZE;
482 (*map)->YSIZE = temp_YSIZE;
483 (*map)->TILEW = temp_TILEW;
484 (*map)->TILEH = temp_TILEH;
485 (*map)->ptanchorX = temp_ptanchorX;
486 (*map)->ptanchorY = temp_ptanchorY;
487 (*map)->X = temp_X;
488 (*map)->Y = temp_Y;
489 (*map)->bgcolor = al_map_rgba((unsigned char)((temp_bgcolor >> 24) & 0xFF),
490 (unsigned char)((temp_bgcolor >> 16) & 0xFF),
491 (unsigned char)((temp_bgcolor >> 8) & 0xFF),
492 (unsigned char)(temp_bgcolor & 0xFF));
493 (*map)->wirecolor = al_map_rgba((unsigned char)((temp_wirecolor >> 24) & 0xFF),
494 (unsigned char)((temp_wirecolor >> 16) & 0xFF),
495 (unsigned char)((temp_wirecolor >> 8) & 0xFF),
496 (unsigned char)(temp_wirecolor & 0xFF));
497
498 /* loading the grid */
499 for (it = 0; it < (*map)->XSIZE; it++) {
500 for (int it1 = 0; it1 < (*map)->YSIZE; it1++) {
501 (*map)->grid[it + it1 * (*map)->XSIZE].tilenumber = al_fread32le(loaded);
502 (*map)->grid[it + it1 * (*map)->XSIZE].ability = al_fread32le(loaded);
503 (*map)->grid[it + it1 * (*map)->XSIZE].objectnumber = al_fread32le(loaded);
504 (*map)->grid[it + it1 * (*map)->XSIZE].music = al_fread32le(loaded);
505 }
506 }
507
508 /*load_animlib( loaded , & (*map ) -> libtiles );*/
509 /*load_animlib( loaded , & (*map ) -> libanims );*/
510
511 al_fclose(loaded);
512
513 return TRUE;
514} /* load_map( ... ) */

522int save_map(MAP* map, char* filename) {
523 ALLEGRO_FILE* saved;
524
525 int it = 0;
526
527 saved = al_fopen(filename, "wb");
528 if (!saved) {
529 n_log(LOG_ERR, "could not open %s for writing", filename);
530 return FALSE;
531 }
532
533 /* writing name */
534 int name_len = (int)(strlen(map->name) + 1);
535 al_fwrite32le(saved, name_len);
536 al_fwrite(saved, map->name, (size_t)name_len);
537
538 /* writing states */
539 al_fwrite32le(saved, map->XSIZE);
540 al_fwrite32le(saved, map->YSIZE);
541 al_fwrite32le(saved, map->TILEW);
542 al_fwrite32le(saved, map->TILEH);
543 al_fwrite32le(saved, map->ptanchorX);
544 al_fwrite32le(saved, map->ptanchorY);
545 al_fwrite32le(saved, map->X);
546 al_fwrite32le(saved, map->Y);
547
548 /* pack ALLEGRO_COLOR into 32-bit RGBA */
549 unsigned char r, g, b, a;
550 al_unmap_rgba(map->bgcolor, &r, &g, &b, &a);
551 al_fwrite32le(saved, (int32_t)(((uint32_t)r << 24) | ((uint32_t)g << 16) | ((uint32_t)b << 8) | (uint32_t)a));
552 al_unmap_rgba(map->wirecolor, &r, &g, &b, &a);
553 al_fwrite32le(saved, (int32_t)(((uint32_t)r << 24) | ((uint32_t)g << 16) | ((uint32_t)b << 8) | (uint32_t)a));
554
555 /* saving the grid */
556 for (it = 0; it < map->XSIZE; it++) {
557 for (int it1 = 0; it1 < map->YSIZE; it1++) {
558 al_fwrite32le(saved, map->grid[it + it1 * map->XSIZE].tilenumber);
559 al_fwrite32le(saved, map->grid[it + it1 * map->XSIZE].ability);
560 al_fwrite32le(saved, map->grid[it + it1 * map->XSIZE].objectnumber);
561 al_fwrite32le(saved, map->grid[it + it1 * map->XSIZE].music);
562 }
563 }
564
565 /*save_animlib( saved , map -> libtiles );*/
566 /*save_animlib( saved , map -> libanims );*/
567
568 al_fclose(saved);
569
570 return TRUE;
571} /* save_map( ... ) */

608ISO_MAP* iso_map_new(int width, int height, int num_terrains, int max_height) {
609 __n_assert(width > 0, return NULL);
610 __n_assert(height > 0, return NULL);
611
612 ISO_MAP* map = NULL;
613 Malloc(map, ISO_MAP, 1);
614 __n_assert(map, return NULL);
615
616 map->width = width;
617 map->height = height;
618 map->num_terrains = num_terrains;
619 map->max_height = max_height;
620
621 size_t total = (size_t)width * (size_t)height;
622
623 Malloc(map->terrain, int, total);
624 if (!map->terrain) {
625 Free(map);
626 return NULL;
627 }
628
629 Malloc(map->heightmap, int, total);
630 if (!map->heightmap) {
631 Free(map->terrain);
632 Free(map);
633 return NULL;
634 }
635
636 Malloc(map->ability, int, total);
637 if (!map->ability) {
638 Free(map->heightmap);
639 Free(map->terrain);
640 Free(map);
641 return NULL;
642 }
643
644 memset(map->terrain, 0, total * sizeof(int));
645 memset(map->heightmap, 0, total * sizeof(int));
646 for (size_t i = 0; i < total; i++) {
647 map->ability[i] = WALK;
648 }
649
650 /* default classic 2:1 projection for 64px wide tiles */
651 iso_map_set_projection(map, ISO_PROJ_CLASSIC, 64);
652
653 map->segments = NULL; /* allocated on demand by iso_map_set_segments() */
654
655 /* Overlay layer: allocated and zeroed, 0 = no overlay */
656 Malloc(map->overlay, int, total);
657 if (map->overlay) memset(map->overlay, 0, total * sizeof(int));
658
659 /* default rendering flags */
660 map->smooth_height = 0;
661 map->smooth_slope_max = 1;
662 map->show_grid = 0;
663 map->hover_mx = -1;
664 map->hover_my = -1;
665 map->height_tint_intensity = 0.0f; /* disabled by default; game sets this */
666
667 /* Ambient color defaults: full daylight (no tinting) */
668 map->ambient_r = 1.0f;
669 map->ambient_g = 1.0f;
670 map->ambient_b = 1.0f;
671 map->dynamic_light_map = NULL; /* caller-owned, set before draw */
672
673 return map;
674} /* iso_map_new() */

717void iso_map_set_projection(ISO_MAP* map, int preset, float tile_width) {
718 __n_assert(map, return);
719 float hw = tile_width / 2.0f;
720 float angle = _iso_preset_angle(preset);
721 float hh = hw * tanf(angle * (float)M_PI / 180.0f);
722
723 map->proj.half_w = hw;
724 map->proj.half_h = hh;
725 map->proj.angle_deg = angle;
726 map->proj.target_angle = angle;
727 map->proj.lerp_speed = 3.0f;
728 map->proj.tile_lift = hh;
729} /* iso_map_set_projection() */

748void iso_map_lerp_projection(ISO_MAP* map, float dt) {
749 __n_assert(map, return);
750 float diff = map->proj.target_angle - map->proj.angle_deg;
751 if (fabsf(diff) < 0.01f) {
752 map->proj.angle_deg = map->proj.target_angle;
753 } else {
754 map->proj.angle_deg += diff * map->proj.lerp_speed * dt;
755 }
756 float rad = map->proj.angle_deg * (float)M_PI / 180.0f;
757 map->proj.half_h = map->proj.half_w * sinf(rad) / cosf(rad);
758 if (map->proj.half_h < 8.0f) map->proj.half_h = 8.0f;
759 if (map->proj.half_h > map->proj.half_w * 2.0f) map->proj.half_h = map->proj.half_w * 2.0f;
760} /* iso_map_lerp_projection() */

767const char* iso_projection_name(int preset) {
768 switch (preset) {

980int iso_is_in_diamond(int px, int py, int tile_w, int tile_h) {

1021void iso_screen_to_map_height(const ISO_MAP* map, float screen_x, float screen_y, int tile_w, int tile_h, int* mx, int* my) {
1022 __n_assert(map, return);
1023 __n_assert(mx, return);
1024 __n_assert(my, return);
1025
1026 float half_w = map->proj.half_w;
1027 float half_h = map->proj.half_h;
1028 float tile_lift = map->proj.tile_lift;
1029 float tw = half_w * 2.0f;
1030 float th = half_h * 2.0f;
1031
1032 for (int h = map->max_height; h >= 0; h--) {
1033 float adj_sy = screen_y + (float)h * tile_lift;
1034 float fmx = screen_x / tw + adj_sy / th - 1.0f;
1035 float fmy = adj_sy / th - screen_x / tw + 1.0f;
1036 int tmx = (int)floorf(fmx);
1037 int tmy = (int)floorf(fmy);
1038
1039 if (tmx < 0 || tmx >= map->width || tmy < 0 || tmy >= map->height)
1040 continue;
1041
1042 if (iso_map_get_height(map, tmx, tmy) == h) {
1043 float tile_sx, tile_sy;
1044 iso_map_to_screen(map, tmx, tmy, h, &tile_sx, &tile_sy);
1045 float norm_x = (screen_x - tile_sx) / tw * (float)tile_w;
1046 float norm_y = (screen_y - tile_sy) / th * (float)tile_h;
1047 int lpx = (int)norm_x;
1048 int lpy = (int)norm_y;
1049 if (lpx >= 0 && lpx < tile_w && lpy >= 0 && lpy < tile_h &&
1050 iso_is_in_diamond(lpx, lpy, tile_w, tile_h)) {
1051 *mx = tmx;
1052 *my = tmy;
1053 return;
1054 }
1055 }
1056 }
1057
1058 /* Fallback: flat projection (height 0) */
1059 *mx = (int)floorf(screen_x / tw + screen_y / th - 1.0f);
1060 *my = (int)floorf(screen_y / th - screen_x / tw + 1.0f);
1061} /* iso_screen_to_map_height() */

1077void iso_screen_to_map_height_f(const ISO_MAP* map, float screen_x, float screen_y, int tile_w, int tile_h, int* mx, int* my, float* out_fx, float* out_fy) {
1078 __n_assert(map, return);
1079 __n_assert(mx, return);
1080 __n_assert(my, return);
1081
1082 float half_w = map->proj.half_w;
1083 float half_h = map->proj.half_h;
1084 float tile_lift = map->proj.tile_lift;
1085 float tw = half_w * 2.0f;
1086 float th = half_h * 2.0f;
1087
1088 for (int h = map->max_height; h >= 0; h--) {
1089 float adj_sy = screen_y + (float)h * tile_lift;
1090 float fmx = screen_x / tw + adj_sy / th - 1.0f;
1091 float fmy = adj_sy / th - screen_x / tw + 1.0f;
1092 int tmx = (int)floorf(fmx);
1093 int tmy = (int)floorf(fmy);
1094
1095 if (tmx < 0 || tmx >= map->width || tmy < 0 || tmy >= map->height)
1096 continue;
1097
1098 if (iso_map_get_height(map, tmx, tmy) == h) {
1099 float tile_sx, tile_sy;
1100 iso_map_to_screen(map, tmx, tmy, h, &tile_sx, &tile_sy);
1101 float norm_x = (screen_x - tile_sx) / tw * (float)tile_w;
1102 float norm_y = (screen_y - tile_sy) / th * (float)tile_h;
1103 int lpx = (int)norm_x;
1104 int lpy = (int)norm_y;
1105 if (lpx >= 0 && lpx < tile_w && lpy >= 0 && lpy < tile_h &&
1106 iso_is_in_diamond(lpx, lpy, tile_w, tile_h)) {
1107 *mx = tmx;
1108 *my = tmy;
1109 if (out_fx || out_fy) {
1110 /* Refine fractional coordinates using interpolated height
1111 * so that iso_map_to_screen_f(fx, fy, interp_h) returns
1112 * the original screen position. Without this, sub-tile
1113 * positions are off when the bilinear-interpolated height
1114 * differs from the tile's integer height (e.g. at edges
1115 * between tiles of different heights).
1116 *
1117 * Only refine in SMOOTH mode. In CUT mode the rendered
1118 * tile sits at the flat integer height, so the initial
1119 * fmx/fmy (computed at that height) are already correct.
1120 * Refining with bilinear-interpolated heights in CUT mode
1121 * pulls the position toward tile corners and causes
1122 * click-to-move to land at the wrong spot. */
1123 float refined_fx = fmx;
1124 float refined_fy = fmy;
1125 if (map->smooth_height) {
1126 /* Upper bounds: allow the full extent of the last tile.
1127 * Tile tmx occupies [tmx, tmx+1) in map coords, and the
1128 * last valid position must stay < map->width so that
1129 * floor() never yields an out-of-bounds tile index. */
1130 float map_max_fx = (float)map->width - 1e-4f;
1131 float map_max_fy = (float)map->height - 1e-4f;
1132 /* Tile-local bounds to prevent divergence: clamp each
1133 * refinement step to the detected tile so that large
1134 * height differences with neighbours cannot pull the
1135 * position into an adjacent tile. */
1136 float tile_min_fx = (float)tmx;
1137 float tile_min_fy = (float)tmy;
1138 float tile_max_fx = (float)(tmx + 1);
1139 float tile_max_fy = (float)(tmy + 1);
1140 if (tile_max_fx > map_max_fx) tile_max_fx = map_max_fx;
1141 if (tile_max_fy > map_max_fy) tile_max_fy = map_max_fy;
1142 for (int iter = 0; iter < 3; iter++) {
1143 float interp_h = iso_map_interpolate_height(map, refined_fx, refined_fy);
1144 float new_adj_sy = screen_y + interp_h * tile_lift;
1145 refined_fx = screen_x / tw + new_adj_sy / th - 1.0f;
1146 refined_fy = new_adj_sy / th - screen_x / tw + 1.0f;
1147 /* Keep within detected tile */
1148 if (refined_fx < tile_min_fx) refined_fx = tile_min_fx;
1149 if (refined_fy < tile_min_fy) refined_fy = tile_min_fy;
1150 if (refined_fx > tile_max_fx) refined_fx = tile_max_fx;
1151 if (refined_fy > tile_max_fy) refined_fy = tile_max_fy;
1152 }
1153 }
1154 /* Clamp to map bounds so border clicks stay valid */
1155 float map_max_fx2 = (float)map->width - 1e-4f;
1156 float map_max_fy2 = (float)map->height - 1e-4f;
1157 if (refined_fx < 0.0f) refined_fx = 0.0f;
1158 if (refined_fy < 0.0f) refined_fy = 0.0f;
1159 if (refined_fx > map_max_fx2) refined_fx = map_max_fx2;
1160 if (refined_fy > map_max_fy2) refined_fy = map_max_fy2;
1161 if (out_fx) *out_fx = refined_fx;
1162 if (out_fy) *out_fy = refined_fy;
1163 }
1164 return;
1165 }
1166 }
1167 }
1168
1169 /* Fallback: flat projection (height 0) */
1170 float fmx = screen_x / tw + screen_y / th - 1.0f;
1171 float fmy = screen_y / th - screen_x / tw + 1.0f;
1172 *mx = (int)floorf(fmx);
1173 *my = (int)floorf(fmy);
1174 if (out_fx) *out_fx = fmx;
1175 if (out_fy) *out_fy = fmy;
1176} /* iso_screen_to_map_height_f() */

1186float iso_map_interpolate_height(const ISO_MAP* map, float fx, float fy) {
1187 __n_assert(map, return 0.0f);
1188
1189 /* Clamp to valid map range so border tiles interpolate correctly.
1190 * Allow the full extent of the last tile (up to width/height minus
1191 * a tiny epsilon) so that positions in the right half of border
1192 * tiles are not snapped to the tile corner. */
1193 if (fx < 0.0f) fx = 0.0f;
1194 if (fy < 0.0f) fy = 0.0f;
1195 float max_fx = (float)map->width - 1e-4f;
1196 float max_fy = (float)map->height - 1e-4f;
1197 if (fx > max_fx) fx = max_fx;
1198 if (fy > max_fy) fy = max_fy;
1199
1200 int ix = (int)floorf(fx);
1201 int iy = (int)floorf(fy);
1202 float frac_x = fx - (float)ix;
1203 float frac_y = fy - (float)iy;
1204
1205 /* Clamp neighbor indices for edge tiles */
1206 int ix1 = (ix + 1 < map->width) ? ix + 1 : ix;
1207 int iy1 = (iy + 1 < map->height) ? iy + 1 : iy;
1208
1209 float h00 = (float)iso_map_get_height(map, ix, iy);
1210 float h10 = (float)iso_map_get_height(map, ix1, iy);
1211 float h01 = (float)iso_map_get_height(map, ix, iy1);
1212 float h11 = (float)iso_map_get_height(map, ix1, iy1);
1213
1214 float top = h00 + (h10 - h00) * frac_x;
1215 float bot = h01 + (h11 - h01) * frac_x;
1216 return top + (bot - top) * frac_y;
1217} /* iso_map_interpolate_height() */

1231void iso_map_calc_transitions(const ISO_MAP* map, int mx, int my, int* edge_bits, int* corner_bits) {

1234 __n_assert(corner_bits, return);
1235

1237 *edge_bits = 0;
1238 *corner_bits = 0;
1239
1240 /* Edge neighbors: W(-1,0), N(0,-1), E(+1,0), S(0,+1) */

1242 int n = iso_map_get_terrain(map, mx, my - 1);

1467int iso_map_set_segments(ISO_MAP* map, int mx, int my, const ISO_TILE_SEGMENT* segs, int count) {
1468 __n_assert(map, return 0);
1469 if (mx < 0 || mx >= map->width || my < 0 || my >= map->height) return 0;
1470 if (count < 0 || count > ISO_MAX_SEGMENTS_PER_TILE) return 0;
1471
1472 /* Validate invariants */
1473 for (int i = 0; i < count; i++) {
1474 if (segs[i].bottom >= segs[i].top) return 0;
1475 if (i > 0 && segs[i - 1].top > segs[i].bottom) return 0;
1476 }
1477
1478 /* Allocate segment array on first use */
1479 if (!map->segments) {
1480 size_t total = (size_t)map->width * (size_t)map->height;
1481 Malloc(map->segments, ISO_TILE_SEGMENTS, total);
1482 if (!map->segments) return 0;
1483 memset(map->segments, 0, total * sizeof(ISO_TILE_SEGMENTS));
1484 }
1485
1486 int idx = my * map->width + mx;
1487 map->segments[idx].count = count;
1488 for (int i = 0; i < count; i++) map->segments[idx].segs[i] = segs[i];
1489 for (int i = count; i < ISO_MAX_SEGMENTS_PER_TILE; i++) {
1490 map->segments[idx].segs[i].bottom = 0;
1491 map->segments[idx].segs[i].top = 0;
1492 map->segments[idx].segs[i].upper_tile = -1;
1493 map->segments[idx].segs[i].lower_tile = -1;
1494 }
1495
1496 /* Update derived heightmap */
1497 map->heightmap[idx] = (count > 0) ? segs[count - 1].top : 0;
1498 return 1;
1499}

1507static int _cmp_draw_entry(const void* a, const void* b) {
1508 const ISO_DRAW_ENTRY* ea = (const ISO_DRAW_ENTRY*)a;
1509 const ISO_DRAW_ENTRY* eb = (const ISO_DRAW_ENTRY*)b;
1510 /* Primary: isometric depth (back-to-front) */
1511 int depth_a = ea->my + ea->mx;
1512 int depth_b = eb->my + eb->mx;
1513 if (depth_a != depth_b) return depth_a - depth_b;
1514 /* Secondary: segment base height (lower drawn first) */
1515 if (ea->bottom != eb->bottom) return ea->bottom - eb->bottom;
1516 /* Tertiary: row then column */
1517 if (ea->my != eb->my) return ea->my - eb->my;
1518 return ea->mx - eb->mx;
1519}

1521int iso_map_build_draw_order(const ISO_MAP* map, ISO_DRAW_ENTRY* out, int max_entries) {
1522 __n_assert(map, return 0);
1523 __n_assert(out, return 0);
1524
1525 int n = 0;
1526 for (int my = 0; my < map->height; my++) {
1527 for (int mx = 0; mx < map->width; mx++) {
1528 int idx = my * map->width + mx;
1529 if (map->segments && map->segments[idx].count > 0) {
1530 const ISO_TILE_SEGMENTS* ts = &map->segments[idx];
1531 for (int si = 0; si < ts->count && n < max_entries; si++) {
1532 out[n].mx = mx;
1533 out[n].my = my;
1534 out[n].seg_idx = si;
1535 out[n].bottom = ts->segs[si].bottom;
1536 out[n].top = ts->segs[si].top;
1537 /* underside: bottom > 0 and no segment directly below */
1538 out[n].underside = (ts->segs[si].bottom > 0 &&
1539 (si == 0 || ts->segs[si - 1].top < ts->segs[si].bottom));
1540 n++;
1541 }
1542 } else {
1543 /* No segments or count==0: single entry from heightmap */
1544 if (n >= max_entries) continue;
1545 int h = map->heightmap[idx];
1546 out[n].mx = mx;
1547 out[n].my = my;
1548 out[n].seg_idx = 0;
1549 out[n].bottom = 0;
1550 out[n].top = h;
1551 out[n].underside = 0;
1552 n++;
1553 }
1554 }
1555 }
1556
1557 qsort(out, (size_t)n, sizeof(ISO_DRAW_ENTRY), _cmp_draw_entry);
1558 return n;
1559}

1587void iso_map_calc_transitions_full(const ISO_MAP* map, int mx, int my, int* edge_bits, int* corner_bits) {
1588 __n_assert(map, return);
1589 __n_assert(edge_bits, return);
1590 __n_assert(corner_bits, return);
1591
1592 int base = iso_map_get_terrain(map, mx, my);
1593 memset(edge_bits, 0, sizeof(int) * (size_t)map->num_terrains);
1594 memset(corner_bits, 0, sizeof(int) * (size_t)map->num_terrains);
1595
1596 /* Cardinal neighbors: only if height-connected */
1597 int t_w = iso_map_should_transition_smooth(map, mx, my, mx - 1, my)
1598 ? iso_map_get_terrain(map, mx - 1, my)
1599 : base;
1600 int t_n = iso_map_should_transition_smooth(map, mx, my, mx, my - 1)
1601 ? iso_map_get_terrain(map, mx, my - 1)
1602 : base;
1603 int t_e = iso_map_should_transition_smooth(map, mx, my, mx + 1, my)
1604 ? iso_map_get_terrain(map, mx + 1, my)
1605 : base;
1606 int t_s = iso_map_should_transition_smooth(map, mx, my, mx, my + 1)
1607 ? iso_map_get_terrain(map, mx, my + 1)
1608 : base;
1609
1610 /* Diagonal neighbors */
1611 int t_nw = iso_map_should_transition_smooth(map, mx, my, mx - 1, my - 1)
1612 ? iso_map_get_terrain(map, mx - 1, my - 1)
1613 : base;
1614 int t_ne = iso_map_should_transition_smooth(map, mx, my, mx + 1, my - 1)
1615 ? iso_map_get_terrain(map, mx + 1, my - 1)
1616 : base;
1617 int t_se = iso_map_should_transition_smooth(map, mx, my, mx + 1, my + 1)
1618 ? iso_map_get_terrain(map, mx + 1, my + 1)
1619 : base;
1620 int t_sw = iso_map_should_transition_smooth(map, mx, my, mx - 1, my + 1)
1621 ? iso_map_get_terrain(map, mx - 1, my + 1)
1622 : base;
1623
1624 for (int t = base + 1; t < map->num_terrains; t++) {
1625 if (t_w == t) edge_bits[t] |= ISO_EDGE_W;
1626 if (t_n == t) edge_bits[t] |= ISO_EDGE_N;
1627 if (t_e == t) edge_bits[t] |= ISO_EDGE_E;
1628 if (t_s == t) edge_bits[t] |= ISO_EDGE_S;
1629
1630 if (t_nw == t && !(edge_bits[t] & ISO_EDGE_W) && !(edge_bits[t] & ISO_EDGE_N))
1631 corner_bits[t] |= ISO_CORNER_NW;
1632 if (t_ne == t && !(edge_bits[t] & ISO_EDGE_N) && !(edge_bits[t] & ISO_EDGE_E))
1633 corner_bits[t] |= ISO_CORNER_NE;
1634 if (t_se == t && !(edge_bits[t] & ISO_EDGE_E) && !(edge_bits[t] & ISO_EDGE_S))
1635 corner_bits[t] |= ISO_CORNER_SE;
1636 if (t_sw == t && !(edge_bits[t] & ISO_EDGE_S) && !(edge_bits[t] & ISO_EDGE_W))
1637 corner_bits[t] |= ISO_CORNER_SW;
1638 }
1639} /* iso_map_calc_transitions_full() */

1758 if (h == 0) return edges;

1765 edges.draw_nw = 1;
1766 } else {
1767 edges.draw_nw = 1; /* world boundary */
1768 }
1769 } else if (abs(h - iso_map_get_height(map, mx - 1, my)) >= 1) {
1770 edges.draw_nw = 1;
1771 }
1772
1773 /* NE edge: neighbor (mx, my-1) */
1774 if (my <= 0) {
1775 if (map->neighbor_heights_north) {
1776 if (abs(h - map->neighbor_heights_north[mx]) >= 1)
1777 edges.draw_ne = 1;
1778 } else {
1779 edges.draw_ne = 1;
1780 }
1781 } else if (abs(h - iso_map_get_height(map, mx, my - 1)) >= 1) {
1782 edges.draw_ne = 1;
1783 }
1784
1785 /* SE edge: neighbor (mx+1, my) */
1786 if (mx >= map->width - 1) {
1787 if (map->neighbor_heights_east) {
1788 if (abs(h - map->neighbor_heights_east[my]) >= 1)
1789 edges.draw_se = 1;
1790 } else {
1791 edges.draw_se = 1;
1792 }
1793 } else if (abs(h - iso_map_get_height(map, mx + 1, my)) >= 1) {
1794 edges.draw_se = 1;
1795 }

1798 if (my >= map->height - 1) {
1799 if (map->neighbor_heights_south) {

1801 edges.draw_sw = 1;
1802 } else {

1826 if (!map || mx < 0 || mx >= map->width || my < 0 || my >= map->height)
1827 return 0;

1829 if (ts && ts->count > 0) {

1832 return 1;
1833 }
1834 return 0;
1835 }
1836 int h = iso_map_get_height(map, mx, my);
1837 return (h > z + 1) ? 1 : 0;

1864ISO_VISIBLE_EDGES iso_map_get_visible_edges_segment(const ISO_MAP* map, int mx, int my, int seg_idx) {
1865 ISO_VISIBLE_EDGES edges = {0};
1866 if (!map) return edges;
1867
1868 /* Get this tile's segment */
1869 ISO_TILE_SEGMENT seg;
1870 const ISO_TILE_SEGMENTS* ts = iso_map_get_segments(map, mx, my);
1871 if (ts && seg_idx >= 0 && seg_idx < ts->count) {
1872 seg = ts->segs[seg_idx];
1873 } else {
1874 /* Fallback: heightmap as single segment {0, h} */
1875 int h = iso_map_get_height(map, mx, my);
1876 if (h == 0) return edges;
1877 seg.bottom = 0;
1878 seg.top = h;
1879 }
1880
1881 if (seg.top <= seg.bottom) return edges;
1882
1883 /* Direction offsets: NW→(mx-1,my), NE→(mx,my-1), SE→(mx+1,my), SW→(mx,my+1) */
1884 static const int ddx[4] = {-1, 0, 1, 0};
1885 static const int ddy[4] = {0, -1, 0, 1};
1886 int* flags[4];
1887 flags[0] = &edges.draw_nw;
1888 flags[1] = &edges.draw_ne;
1889 flags[2] = &edges.draw_se;
1890 flags[3] = &edges.draw_sw;
1891
1892 int any_side_exposed = 0; /* for underside check */
1893 int ntop[4] = {0, 0, 0, 0}; /* neighbor top height per direction */
1894
1895 for (int d = 0; d < 4; d++) {
1896 int nx = mx + ddx[d];
1897 int ny = my + ddy[d];
1898 int neighbor_matches = 0;
1899 int neighbor_covers_bottom = 0;
1900
1901 if (nx < 0 || nx >= map->width || ny < 0 || ny >= map->height) {
1902 /* Boundary: check neighbor chunk height arrays */
1903 int nh = -1;
1904 if (d == 0 && map->neighbor_heights_west) nh = map->neighbor_heights_west[my];
1905 if (d == 1 && map->neighbor_heights_north) nh = map->neighbor_heights_north[mx];
1906 if (d == 2 && map->neighbor_heights_east) nh = map->neighbor_heights_east[my];
1907 if (d == 3 && map->neighbor_heights_south) nh = map->neighbor_heights_south[mx];
1908
1909 if (nh > 0) {
1910 /* Boundary treated as single segment {0,nh}: match if same top */
1911 neighbor_matches = (nh == seg.top);
1912 neighbor_covers_bottom = (nh >= seg.bottom);
1913 ntop[d] = nh;
1914 }
1915 /* else: world boundary → no match, not covered, ntop stays 0 */
1916 } else {
1917 const ISO_TILE_SEGMENTS* nts = iso_map_get_segments(map, nx, ny);
1918 if (nts && nts->count > 0) {
1919 /* Match if any neighbour segment has the same top height.
1920 * This applies uniformly to ground and floating segments:
1921 * same top = connected surface = no border.
1922 * Different top = visible height step = border drawn. */
1923 for (int si = 0; si < nts->count; si++) {
1924 if (nts->segs[si].top == seg.top) {
1925 neighbor_matches = 1;
1926 }
1927 if (nts->segs[si].top >= seg.bottom) {
1928 neighbor_covers_bottom = 1;
1929 }
1930 }
1931 /* Use heightmap value as neighbor top for vertical lines */
1932 ntop[d] = iso_map_get_height(map, nx, ny);
1933 } else {
1934 /* No segment data: heightmap as single segment {0, h} */
1935 int nh = iso_map_get_height(map, nx, ny);
1936 if (nh > 0) {
1937 neighbor_matches = (nh == seg.top);
1938 neighbor_covers_bottom = (nh >= seg.bottom);
1939 }
1940 ntop[d] = nh;
1941 }
1942 }
1943
1944 if (!neighbor_matches) *flags[d] = 1;
1945 if (!neighbor_covers_bottom) any_side_exposed = 1;
1946 }
1947
1948 /* OCCLUSION: NW/NE edges face away from camera.
1949 * Suppress if the tile drawn later (south in painter order) has
1950 * geometry covering this segment's top face height.
1951 * NW edge: covered by SW neighbour (mx, my+1)
1952 * NE edge: covered by SE neighbour (mx+1, my) */
1953 if (edges.draw_nw && _tile_covers_height(map, mx, my + 1, seg.top))
1954 edges.draw_nw = 0;
1955 if (edges.draw_ne && _tile_covers_height(map, mx + 1, my, seg.top))
1956 edges.draw_ne = 0;
1957
1958 /* Underside border for floating segments */
1959 if (seg.bottom > 0 && any_side_exposed) {
1960 edges.draw_underside = 1;
1961 }
1962
1963 /* Store neighbor top heights for vertical corner line computation */
1964 edges.neighbor_top_nw = ntop[0];
1965 edges.neighbor_top_ne = ntop[1];
1966 edges.neighbor_top_se = ntop[2];
1967 edges.neighbor_top_sw = ntop[3];
1968
1969 return edges;
1970}

2031/* Extended tint: combines height tint with ambient color and per-tile dynamic light.
2032 * ambient_r/g/b: global day/night ambient (1.0 = full daylight).
2033 * dyn_r/g/b: additive dynamic light contribution at this tile (0.0 = no light). */
2034static inline ALLEGRO_COLOR _compute_tile_tint(int seg_top, int max_height, float intensity, float ambient_r, float ambient_g, float ambient_b, float dyn_r, float dyn_g, float dyn_b) {
2035 /* Start with height-based brightness */
2036 float hb = 1.0f;
2037 if (intensity > 0.0f && max_height > 0) {
2038 float ratio = (float)seg_top / (float)max_height;
2039 hb = 1.0f - intensity * (1.0f - ratio);
2040 if (hb < 0.0f) hb = 0.0f;
2041 if (hb > 1.0f) hb = 1.0f;
2042 }
2043
2044 /* Multiply by ambient color */
2045 float r = hb * ambient_r;
2046 float g = hb * ambient_g;
2047 float b = hb * ambient_b;
2048
2049 /* Add dynamic light (additive boost) */
2050 r += dyn_r;
2051 g += dyn_g;
2052 b += dyn_b;
2053
2054 /* Clamp to avoid overexposure */
2055 if (r > 1.0f) r = 1.0f;
2056 if (g > 1.0f) g = 1.0f;
2057 if (b > 1.0f) b = 1.0f;
2058
2059 return al_map_rgba_f(r, g, b, 1.0f);

2258static int _iso_object_check_occlusion(const ISO_MAP* map,
2259 float obj_fx,
2260 float obj_fy,
2261 float obj_fz,
2262 float obj_sx,
2263 float obj_sy,
2264 float sprite_h,
2265 float cam_px,
2266 float cam_py,
2267 float zoom) {
2268 float phw = map->proj.half_w * zoom;
2269 float phh = map->proj.half_h * zoom;
2270 float lift = map->proj.tile_lift * zoom;
2271
2272 /* Entity head screen Y: sprite_h world units above feet */
2273 float head_sy = obj_sy - sprite_h * lift;
2274
2275 int base_mx = (int)floorf(obj_fx);
2276 int base_my = (int)floorf(obj_fy);
2277 int check_range = map->max_height + 2;
2278
2279 for (int dy = -1; dy <= check_range; dy++) {
2280 for (int dx = -1; dx <= check_range; dx++) {
2281 /* only check tiles drawn after the object in the for(my) for(mx) loop */
2282 if (dy < 0 || (dy == 0 && dx <= 0)) continue;
2283
2284 int tmx = base_mx + dx;
2285 int tmy = base_my + dy;
2286 if (tmx < 0 || tmx >= map->width || tmy < 0 || tmy >= map->height) continue;
2287
2288 int th = iso_map_get_height(map, tmx, tmy);
2289 /* tile must reach above the entity's head to occlude */
2290 if ((float)th < obj_fz + sprite_h) continue;
2291
2292 /* compute tile screen position (cam_px already snapped) */
2293 float tsx, tsy;
2294 iso_map_to_screen(map, tmx, tmy, th, &tsx, &tsy);
2295 float tile_dx = tsx * zoom + cam_px;
2296 float tile_dy = tsy * zoom + cam_py;
2297
2298 /* tile must cover the entity's head in screen space */
2299 float tile_left = tile_dx;
2300 float tile_right = tile_dx + 2.0f * phw;
2301 float tile_top = tile_dy;
2302 float tile_bot = tile_dy + 2.0f * phh + (float)th * lift;
2303
2304 if (obj_sx >= tile_left && obj_sx <= tile_right &&
2305 tile_top <= head_sy && obj_sy <= tile_bot) {
2306 return 1;
2307 }
2308 }
2309 }
2310 return 0;

2326static void _draw_height_borders(const ISO_MAP* map,
2327 int mx,
2328 int my,
2329 int seg_idx,
2330 int seg_bottom,
2331 int seg_top,
2332 int has_underside_face,
2333 const float vx[4],
2334 const float vy[4],
2335 float lift) {
2336 ISO_VISIBLE_EDGES edges = iso_map_get_visible_edges_segment(map, mx, my, seg_idx);
2337 if (!edges.draw_nw && !edges.draw_ne && !edges.draw_se && !edges.draw_sw && !(has_underside_face && edges.draw_underside))
2338 return;
2339
2340 ALLEGRO_COLOR border_col = al_map_rgba(0, 0, 0, 200);
2341 float thickness = 1.5f;
2342
2343 /* Top face edges */
2344 if (edges.draw_se)
2345 al_draw_line(vx[1], vy[1], vx[2], vy[2], border_col, thickness);
2346 if (edges.draw_sw)
2347 al_draw_line(vx[2], vy[2], vx[3], vy[3], border_col, thickness);
2348 if (edges.draw_nw)
2349 al_draw_line(vx[0], vy[0], vx[3], vy[3], border_col, thickness);
2350 if (edges.draw_ne)
2351 al_draw_line(vx[0], vy[0], vx[1], vy[1], border_col, thickness);
2352
2353 /* Vertical corner lines.
2354 *
2355 * For each drawn edge where seg_top > neighbor_top (A is higher):
2356 * Draw a vertical line at BOTH endpoints of that edge,
2357 * from tileToScreen(vertex, seg_top) down to
2358 * tileToScreen(vertex, neighbor_top).
2359 * Exception: NEVER draw at the North vertex (vx[0]/vy[0]).
2360 *
2361 * If seg_top <= neighbor_top: no vertical lines (wall belongs to N).
2362 *
2363 * vx[]/vy[] are already at tileToScreen(vertex, seg_top).
2364 * The bottom endpoint is (seg_top - HN) * lift pixels below.
2365 *
2366 * Diamond vertices: N=0, E=1, S=2, W=3.
2367 * Edge endpoints: NE=(N,E) SE=(E,S) SW=(S,W) NW=(W,N)
2368 *
2369 * Multiple edges may want to draw at the same vertex with different
2370 * drops. We draw the longest line (smallest HN) at each vertex
2371 * to avoid duplicate shorter lines.
2372 */
2373 {
2374 /* Per-vertex: track the largest drop (longest line) requested.
2375 *
2376 * CONTINUITY CHECK: a vertical corner line at a vertex is only
2377 * drawn if the wall face ENDS at that vertex — i.e. the adjacent
2378 * tile in the wall's direction is NOT at the same height.
2379 * If the adjacent tile IS at the same height, the wall surface
2380 * continues past the vertex and the line is interior (hidden).
2381 *
2382 * For the east-facing wall (SE edge):
2383 * E vertex: wall continues northward if NE neighbor same height → skip
2384 * S vertex: wall continues southward if SW neighbor same height → skip
2385 * For the south-facing wall (SW edge):
2386 * S vertex: wall continues eastward if SE neighbor same height → skip
2387 * W vertex: wall continues westward if NW neighbor same height → skip
2388 * For NE edge → E vertex: wall continues if SE neighbor same height
2389 * For NW edge → W vertex: wall continues if SW neighbor same height
2390 */
2391 float drop_e = 0, drop_s = 0, drop_w = 0;
2392 int nt_nw = edges.neighbor_top_nw;
2393 int nt_ne = edges.neighbor_top_ne;
2394 int nt_se = edges.neighbor_top_se;
2395 int nt_sw = edges.neighbor_top_sw;
2396
2397 /* Clamp neighbor tops to seg_bottom: vertical lines must not
2398 * extend below the segment's own base (floating segments). */
2399 int bot = seg_bottom;
2400 int cl_nw = nt_nw > bot ? nt_nw : bot;
2401 int cl_ne = nt_ne > bot ? nt_ne : bot;
2402 int cl_se = nt_se > bot ? nt_se : bot;
2403 int cl_sw = nt_sw > bot ? nt_sw : bot;
2404
2405 /* NE edge (N,E): only E vertex (N is never drawn) */
2406 if (edges.draw_ne && seg_top > cl_ne) {
2407 /* E vertex: east wall continues if SE neighbor same height */
2408 if (nt_se != seg_top) {
2409 float d = (float)(seg_top - cl_ne) * lift;
2410 if (d > drop_e) drop_e = d;
2411 }
2412 }
2413 /* SE edge (E,S): E and S vertices */
2414 if (edges.draw_se && seg_top > cl_se) {
2415 float d = (float)(seg_top - cl_se) * lift;
2416 /* E vertex: east wall continues northward if NE neighbor same height */
2417 if (nt_ne != seg_top) {
2418 if (d > drop_e) drop_e = d;
2419 }
2420 /* S vertex: east wall continues southward if SW neighbor same height */
2421 if (nt_sw != seg_top) {
2422 if (d > drop_s) drop_s = d;
2423 }
2424 }
2425 /* SW edge (S,W): S and W vertices */
2426 if (edges.draw_sw && seg_top > cl_sw) {
2427 float d = (float)(seg_top - cl_sw) * lift;
2428 /* S vertex: south wall continues eastward if SE neighbor same height */
2429 if (nt_se != seg_top) {
2430 if (d > drop_s) drop_s = d;
2431 }
2432 /* W vertex: south wall continues westward if NW neighbor same height */
2433 if (nt_nw != seg_top) {
2434 if (d > drop_w) drop_w = d;
2435 }
2436 }
2437 /* NW edge (W,N): only W vertex (N is never drawn) */
2438 if (edges.draw_nw && seg_top > cl_nw) {
2439 /* W vertex: south wall continues if SW neighbor same height */
2440 if (nt_sw != seg_top) {
2441 float d = (float)(seg_top - cl_nw) * lift;
2442 if (d > drop_w) drop_w = d;
2443 }
2444 }
2445
2446 if (drop_e > 0)
2447 al_draw_line(vx[1], vy[1], vx[1], vy[1] + drop_e, border_col, thickness);
2448 if (drop_s > 0)
2449 al_draw_line(vx[2], vy[2], vx[2], vy[2] + drop_s, border_col, thickness);
2450 if (drop_w > 0)
2451 al_draw_line(vx[3], vy[3], vx[3], vy[3] + drop_w, border_col, thickness);
2452 }
2453
2454 /* Underside face border (floating segments only).
2455 * The underside diamond is at seg_bottom, offset below the top face
2456 * by (seg_top - seg_bottom) * lift pixels. Only drawn if the
2457 * underside face itself was rendered (has_underside_face).
2458 * Only SE and SW edges — NW and NE are always hidden behind the
2459 * segment's own side walls from the camera angle. */
2460 if (has_underside_face && edges.draw_underside) {
2461 float dy = (float)(seg_top - seg_bottom) * lift;
2462 /* SE edge */
2463 al_draw_line(vx[1], vy[1] + dy, vx[2], vy[2] + dy, border_col, thickness);
2464 /* SW edge */
2465 al_draw_line(vx[2], vy[2] + dy, vx[3], vy[3] + dy, border_col, thickness);
2466 }
2467}

2488void iso_map_draw(const ISO_MAP* map,
2489 ALLEGRO_BITMAP** tile_bitmaps,
2490 ALLEGRO_BITMAP*** transition_tiles,
2491 int num_masks,
2492 ALLEGRO_BITMAP** overlay_bitmaps,
2493 int num_overlay_tiles,
2494 float cam_px,
2495 float cam_py,
2496 float zoom,
2497 int screen_w,
2498 int screen_h,
2499 int player_mode,
2500 N_ISO_OBJECT* objects,
2501 int num_objects) {
2502 __n_assert(map, return);
2503 __n_assert(tile_bitmaps, return);
2504
2505 al_set_blender(ALLEGRO_ADD, ALLEGRO_ONE, ALLEGRO_INVERSE_ALPHA);
2506
2507 float hw0 = map->proj.half_w;
2508 float hh0 = map->proj.half_h;
2509 float tl0 = map->proj.tile_lift;
2510 float phw = hw0 * zoom;
2511 float phh = hh0 * zoom;
2512 float lift = tl0 * zoom;
2513 float tile_draw_w = 2.0f * phw;
2514 float tile_draw_h = 2.0f * phh;
2515
2516 /* cam_px / cam_py are pre-computed pixel offsets provided by the caller.
2517 * The caller is responsible for integer-snapping the base camera offset
2518 * (floorf(cam_x * zoom)) and adding any chunk pixel offset after the
2519 * snap. This ensures all chunks share the same base snap and adjacent
2520 * chunk seams are pixel-perfect. */
2521
2522 int tile_w = tile_bitmaps[0] ? al_get_bitmap_width(tile_bitmaps[0]) : 64;
2523 int tile_h = tile_bitmaps[0] ? al_get_bitmap_height(tile_bitmaps[0]) : 32;
2524
2525 int num_edge_masks = ISO_NUM_EDGE_MASKS;
2526
2527 /* Build segment draw order */
2528 int max_draw_entries = map->width * map->height * ISO_MAX_SEGMENTS_PER_TILE;
2529 ISO_DRAW_ENTRY* draw_order = (ISO_DRAW_ENTRY*)malloc((size_t)max_draw_entries * sizeof(ISO_DRAW_ENTRY));
2530 if (!draw_order) return;
2531 int num_entries = iso_map_build_draw_order(map, draw_order, max_draw_entries);
2532
2533 /* Pre-process objects: compute screen positions and sort by depth.
2534 * Sort key matches segment draw order: (floor(fz), my, mx). */
2535 int obj_order_buf[64];
2536 int* obj_order = NULL;
2537 float* obj_sx_arr = NULL;
2538 float* obj_sy_arr = NULL;
2539 int nobj = (objects && num_objects > 0) ? num_objects : 0;
2540
2541 if (nobj > 0) {
2542 obj_order = (nobj <= 64) ? obj_order_buf : (int*)malloc((size_t)nobj * sizeof(int));
2543 obj_sx_arr = (float*)malloc((size_t)nobj * sizeof(float));
2544 obj_sy_arr = (float*)malloc((size_t)nobj * sizeof(float));
2545
2546 if (!obj_order || !obj_sx_arr || !obj_sy_arr) {
2547 n_log(LOG_ERR, "iso_map_draw: object allocation failed");
2548 if (obj_order && obj_order != obj_order_buf) free(obj_order);
2549 if (obj_sx_arr) free(obj_sx_arr);
2550 if (obj_sy_arr) free(obj_sy_arr);
2551 nobj = 0;
2552 obj_order = NULL;
2553 obj_sx_arr = NULL;
2554 obj_sy_arr = NULL;
2555 }
2556 }
2557
2558 if (nobj > 0) {
2559 for (int i = 0; i < nobj; i++) {
2560 obj_order[i] = i;
2561 objects[i].is_occluded = 0;
2562 float wsx, wsy;
2563 iso_map_to_screen_f(map, objects[i].fx, objects[i].fy, objects[i].fz, &wsx, &wsy);
2564 obj_sx_arr[i] = wsx * zoom + cam_px;
2565 obj_sy_arr[i] = wsy * zoom + cam_py;
2566 }
2567
2568 /* insertion sort by (depth, floor(fz), my, mx) to match segment draw order */
2569 for (int i = 1; i < nobj; i++) {
2570 int key = obj_order[i];
2571 int key_fz = (int)floorf(objects[key].fz);
2572 int key_my = (int)floorf(objects[key].fy);
2573 int key_mx = (int)floorf(objects[key].fx);
2574 int key_depth = key_my + key_mx;
2575 int j = i - 1;
2576 while (j >= 0) {
2577 int cj = obj_order[j];
2578 int cj_fz = (int)floorf(objects[cj].fz);
2579 int cj_my = (int)floorf(objects[cj].fy);
2580 int cj_mx = (int)floorf(objects[cj].fx);
2581 int cj_depth = cj_my + cj_mx;
2582 if (cj_depth > key_depth ||
2583 (cj_depth == key_depth && cj_fz > key_fz) ||
2584 (cj_depth == key_depth && cj_fz == key_fz && cj_my > key_my) ||
2585 (cj_depth == key_depth && cj_fz == key_fz && cj_my == key_my && cj_mx > key_mx)) {
2586 obj_order[j + 1] = obj_order[j];
2587 j--;
2588 } else {
2589 break;
2590 }
2591 }
2592 obj_order[j + 1] = key;
2593 }
2594 }
2595 /* ===== PASS 1: Ground segments only (bottom == 0), no objects =====
2596 * Draw all ground-level tiles first so they can never overdraw
2597 * elevated content (entities on bridges, floating segments, etc.). */
2598 for (int di = 0; di < num_entries; di++) {
2599 if (draw_order[di].bottom > 0) continue; /* skip elevated */
2600
2601 int mx = draw_order[di].mx;
2602 int my = draw_order[di].my;
2603 int seg_top = draw_order[di].top;
2604 int h = seg_top;
2605 int base = iso_map_get_terrain(map, mx, my);
2606
2607 /* Per-segment tile overrides: upper_tile for top face, lower_tile for walls */
2608 int top_terrain = base;
2609 int wall_terrain = base;
2610 {
2611 int si = draw_order[di].seg_idx;
2612 int tidx = my * map->width + mx;
2613 if (map->segments && tidx >= 0 && tidx < map->width * map->height &&
2614 si >= 0 && si < map->segments[tidx].count) {
2615 int ut = map->segments[tidx].segs[si].upper_tile;
2616 int lt = map->segments[tidx].segs[si].lower_tile;
2617 if (ut >= 0 && ut < map->num_terrains) top_terrain = ut;
2618 if (lt >= 0 && lt < map->num_terrains) wall_terrain = lt;
2619 }
2620 }
2621
2622 float vx[4], vy[4];
2623 float base_dx, base_dy;
2624
2625 if (map->smooth_height) {
2626 float h_n, h_e, h_s, h_w;
2627 iso_map_smooth_corner_heights(map, mx, my, &h_n, &h_e, &h_s, &h_w);
2628 _iso_corner_to_screen(hw0, hh0, tl0, mx, my, h_n, cam_px, cam_py, zoom, &vx[0], &vy[0]);
2629 _iso_corner_to_screen(hw0, hh0, tl0, mx + 1, my, h_e, cam_px, cam_py, zoom, &vx[1], &vy[1]);
2630 _iso_corner_to_screen(hw0, hh0, tl0, mx + 1, my + 1, h_s, cam_px, cam_py, zoom, &vx[2], &vy[2]);
2631 _iso_corner_to_screen(hw0, hh0, tl0, mx, my + 1, h_w, cam_px, cam_py, zoom, &vx[3], &vy[3]);
2632 base_dx = vx[3];
2633 base_dy = vy[0];
2634 } else {
2635 _iso_corner_to_screen(hw0, hh0, tl0, mx, my, (float)h, cam_px, cam_py, zoom, &vx[0], &vy[0]);
2636 _iso_corner_to_screen(hw0, hh0, tl0, mx + 1, my, (float)h, cam_px, cam_py, zoom, &vx[1], &vy[1]);
2637 _iso_corner_to_screen(hw0, hh0, tl0, mx + 1, my + 1, (float)h, cam_px, cam_py, zoom, &vx[2], &vy[2]);
2638 _iso_corner_to_screen(hw0, hh0, tl0, mx, my + 1, (float)h, cam_px, cam_py, zoom, &vx[3], &vy[3]);
2639 base_dx = vx[3];
2640 base_dy = vy[0];
2641 }
2642
2643 /* Culling */
2644 float min_vy = fminf(fminf(vy[0], vy[1]), fminf(vy[2], vy[3]));
2645 float max_vy = fmaxf(fmaxf(vy[0], vy[1]), fmaxf(vy[2], vy[3]));
2646 float max_side = fmaxf((float)h, (float)map->max_height) * lift;
2647 if (vx[1] < 0 || vx[3] > (float)screen_w) continue;
